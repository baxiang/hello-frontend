## 2.1 JSX 深入

### JSX 本质

```jsx
// JSX 代码
const element = (
  <h1 className="greeting">
    Hello, world!
  </h1>
);

// 编译后的 JavaScript
const element = React.createElement(
  'h1',
  {className: 'greeting'},
  'Hello, world!'
);

// React.createElement 返回的对象
const element = {
  type: 'h1',
  props: {
    className: 'greeting',
    children: 'Hello, world!'
  }
};
```

### JSX 规则

```jsx
function App() {
  return (
    <div>
      {/* 1. 单个根元素 */}
      {/* 使用 Fragment 或 div 包裹 */}
      
      {/* 2. 闭合所有标签 */}
      <img src="photo.jpg" alt="照片" />
      <br />
      
      {/* 3. 驼峰命名属性 */}
      <div className="container" tabIndex={1}>
        <svg viewBox="0 0 100 100">
          <path fillRule="evenodd" />
        </svg>
      </div>
      
      {/* 4. JavaScript 表达式用 {} */}
      <p>{1 + 2}</p>
      <p>{'hello'.toUpperCase()}</p>
    </div>
  );
}
```

## 2.2 组件定义

### 函数组件

```jsx
// 基础函数组件
function Welcome() {
  return <h1>Hello</h1>;
}

// 箭头函数组件
const Welcome = () => <h1>Hello</h1>;

// 接收 Props 的组件
function Welcome({ name }) {
  return <h1>Hello, {name}</h1>;
}
```

### 组件组合

```jsx
// 父组件
function App() {
  return (
    <div>
      <Header />
      <Main />
      <Footer />
    </div>
  );
}

// 子组件
function Header() {
  return <header>网站头部</header>;
}

function Main() {
  return <main>主要内容</main>;
}

function Footer() {
  return <footer>网站底部</footer>;
}
```

## 2.3 Props 传递

### 基础 Props

```jsx
// 定义组件接收 Props
function UserCard({ name, age, email }) {
  return (
    <div className="card">
      <h2>{name}</h2>
      <p>年龄：{age}</p>
      <p>邮箱：{email}</p>
    </div>
  );
}

// 使用组件传递 Props
function App() {
  return (
    <div>
      <UserCard name="张三" age={25} email="zhangsan@example.com" />
      <UserCard name="李四" age={30} email="lisi@example.com" />
    </div>
  );
}
```

### 默认 Props

```jsx
function Button({ text = '点击', type = 'primary', disabled = false }) {
  return (
    <button className={`btn btn-${type}`} disabled={disabled}>
      {text}
    </button>
  );
}

// 使用
<Button />                      // text="点击", type="primary"
<Button text="提交" />          // type="primary"
<Button type="danger" />        // text="点击"
```

### Props 解构

```jsx
function UserCard(props) {
  // 解构 props
  const { name, age, email = '未设置' } = props;
  
  return (
    <div>
      <h2>{name}</h2>
      <p>{age} 岁</p>
      <p>{email}</p>
    </div>
  );
}

// 直接在参数中解构
function UserCard({ name, age, email = '未设置' }) {
  // ...
}
```

### Children Props

```jsx
// 接收 children
function Card({ title, children }) {
  return (
    <div className="card">
      <h2 className="card-title">{title}</h2>
      <div className="card-body">
        {children}
      </div>
    </div>
  );
}

// 使用 children
function App() {
  return (
    <Card title="用户信息">
      <p>姓名：张三</p>
      <p>年龄：25</p>
    </Card>
  );
}
```

## 2.4 State 状态

### useState 基础

```jsx
import { useState } from 'react';

function Counter() {
  // 声明状态变量
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>计数：{count}</p>
      <button onClick={() => setCount(count + 1)}>
        +1
      </button>
      <button onClick={() => setCount(0)}>
        重置
      </button>
    </div>
  );
}
```

### 状态更新规则

```jsx
function Counter() {
  const [count, setCount] = useState(0);

  // ✅ 直接更新
  const increment = () => setCount(count + 1);

  // ✅ 函数式更新（推荐用于依赖前值的情况）
  const incrementSafe = () => setCount(prev => prev + 1);

  // ✅ 批量更新时使用函数式更新
  const incrementFive = () => {
    setCount(prev => prev + 1);
    setCount(prev => prev + 1);
    setCount(prev => prev + 1);
    setCount(prev => prev + 1);
    setCount(prev => prev + 1);
  };

  return (
    <div>
      <p>{count}</p>
      <button onClick={incrementFive}>+5</button>
    </div>
  );
}
```

### 对象状态

```jsx
function Form() {
  const [user, setUser] = useState({
    name: '',
    email: '',
    age: 0
  });

  // ✅ 正确：使用展开运算符更新
  const updateName = (e) => {
    setUser({
      ...user,
      name: e.target.value
    });
  };

  // ✅ 更好的方式：使用函数式更新
  const updateEmail = (e) => {
    setUser(prev => ({
      ...prev,
      email: e.target.value
    }));
  };

  return (
    <form>
      <input value={user.name} onChange={updateName} />
      <input value={user.email} onChange={updateEmail} />
    </form>
  );
}
```

### 数组状态

```jsx
function TodoList() {
  const [todos, setTodos] = useState([]);

  // 添加项
  const addTodo = (text) => {
    setTodos(prev => [...prev, { id: Date.now(), text }]);
  };

  // 删除项
  const removeTodo = (id) => {
    setTodos(prev => prev.filter(todo => todo.id !== id));
  };

  // 更新项
  const updateTodo = (id, newText) => {
    setTodos(prev => prev.map(todo =>
      todo.id === id ? { ...todo, text: newText } : todo
    ));
  };

  return (
    <div>
      <button onClick={() => addTodo('新任务')}>添加</button>
      <ul>
        {todos.map(todo => (
          <li key={todo.id}>
            {todo.text}
            <button onClick={() => removeTodo(todo.id)}>删除</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## 2.5 Props vs State

| 特性 | Props | State |
|------|-------|-------|
| 来源 | 父组件传递 | 组件内部定义 |
| 可变性 | 不可变（只读） | 可变（通过 setState） |
| 用途 | 配置组件 | 存储动态数据 |
| 初始化 | 父组件设置 | useState 设置默认值 |

```jsx
function Counter({ initialCount = 0, step = 1 }) {
  // Props：initialCount、step（来自父组件，只读）
  // State：count（组件内部，可变）
  const [count, setCount] = useState(initialCount);

  return (
    <div>
      <p>{count}</p>
      <button onClick={() => setCount(count + step)}>
        +{step}
      </button>
    </div>
  );
}
```

## 2.6 组件通信

### 父传子（Props）

```jsx
// 父组件
function App() {
  const [message, setMessage] = useState('Hello');

  return <Child message={message} />;
}

// 子组件
function Child({ message }) {
  return <p>{message}</p>;
}
```

### 子传父（回调函数）

```jsx
// 父组件
function App() {
  const [data, setData] = useState('');

  const handleChildData = (childData) => {
    setData(childData);
  };

  return (
    <div>
      <p>来自子组件：{data}</p>
      <Child onSendData={handleChildData} />
    </div>
  );
}

// 子组件
function Child({ onSendData }) {
  return (
    <button onClick={() => onSendData('子组件数据')}>
      发送数据
    </button>
  );
}
```

## 2.7 常见问题

### Q1: 为什么不能直接修改 State？

```jsx
// ❌ 错误：直接修改
state.count = state.count + 1;

// ✅ 正确：使用 setState
setCount(count + 1);
```

直接修改 State 不会触发重新渲染，React 无法检测到变化。

### Q2: State 更新后为什么读不到新值？

```jsx
function Counter() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    setCount(count + 1);
    console.log(count); // 还是旧值！
  };

  // ✅ 使用 useEffect 监听变化
  useEffect(() => {
    console.log('count 更新了:', count);
  }, [count]);
}
```

setState 是异步的，更新后的值要在下一次渲染才能获取。

### Q3: 如何在组件间共享 State？

- 提升 State 到共同父组件
- 使用 Context API
- 使用状态管理库（Redux、Zustand）
