## 4.1 深入闭包

### 闭包的形成

```javascript
// 闭包：函数 + 函数创建时的词法环境
function outer() {
    let count = 0;
    return function inner() {
        count++;
        console.log(count);
    };
}

const counter = outer();
counter();  // 1
counter();  // 2
counter();  // 3
```

### 闭包的应用

```javascript
// 1. 数据私有化
function createPrivateData(initialValue) {
    let _value = initialValue;  // 私有变量

    return {
        getValue: () => _value,
        setValue: (val) => {
            if (typeof val === 'number') {
                _value = val;
            }
        }
    };
}

const data = createPrivateData(100);
console.log(data.getValue());  // 100
data.setValue(200);
// console.log(_value);  // 无法访问

// 2. 函数柯里化
function curry(fn) {
    return function curried(...args) {
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        } else {
            return function(...moreArgs) {
                return curried.apply(this, [...args, ...moreArgs]);
            };
        }
    };
}

function add(a, b, c) {
    return a + b + c;
}

const curriedAdd = curry(add);
curriedAdd(1, 2, 3);      // 6
curriedAdd(1, 2)(3);      // 6
curriedAdd(1)(2, 3);      // 6
curriedAdd(1)(2)(3);      // 6

// 3. 函数防抖节流
function debounce(fn, delay) {
    let timer = null;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

function throttle(fn, limit) {
    let inThrottle = false;
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}
```

### 闭包陷阱

```javascript
// 经典问题：循环中的闭包
for (var i = 0; i < 5; i++) {
    setTimeout(() => {
        console.log(i);  // 5, 5, 5, 5, 5
    }, 100);
}

// 解决方案 1：使用 let
for (let i = 0; i < 5; i++) {
    setTimeout(() => {
        console.log(i);  // 0, 1, 2, 3, 4
    }, 100);
}

// 解决方案 2：IIFE
for (var i = 0; i < 5; i++) {
    ((j) => {
        setTimeout(() => {
            console.log(j);  // 0, 1, 2, 3, 4
        }, 100);
    })(i);
}

// 解决方案 3：bind
for (var i = 0; i < 5; i++) {
    setTimeout(console.log.bind(console, i), 100);
}
```

## 4.2 this 绑定

### this 的四种绑定

```javascript
// 1. 默认绑定（严格模式下为 undefined，非严格为 window）
function foo() {
    console.log(this);
}
foo();  // 严格模式：undefined

// 2. 隐式绑定（作为对象方法调用）
const obj = {
    name: '张三',
    getName() {
        console.log(this.name);
    }
};
obj.getName();  // '张三'

// 3. 显式绑定（call/apply/bind）
function greet(greeting) {
    console.log(`${greeting}, ${this.name}`);
}

const person = { name: '李四' };
greet.call(person, 'Hello');     // 'Hello, 李四'
greet.apply(person, ['Hello']);  // 'Hello, 李四'

const boundGreet = greet.bind(person);
boundGreet('Hi');  // 'Hi, 李四'

// 4. new 绑定（构造函数）
function Person(name) {
    this.name = name;
}
const p = new Person('王五');
console.log(p.name);  // '王五'
```

### this 绑定优先级

```javascript
// 优先级：new > 显式 > 隐式 > 默认

const obj = {
    name: '对象名',
    fn: function() {
        console.log(this.name);
    }
};

const name = '全局变量';
obj.fn();  // '对象名'（隐式绑定）

const fn2 = obj.fn;
fn2();  // 严格模式：undefined，非严格：'全局变量'（默认绑定）

// 显式绑定优先级高于隐式
const obj2 = { name: '另一个对象' };
obj.fn.call(obj2);  // '另一个对象'

// new 优先级最高
const Fn3 = obj.fn.bind(obj2);
const instance = new Fn3();
console.log(instance.name);  // undefined（new 创建了新的 this）
```

### 箭头函数的 this

```javascript
// 箭头函数的 this 由定义时的上下文决定
const obj = {
    name: '张三',
    regularFn: function() {
        console.log(this.name);  // '张三'
    },
    arrowFn: () => {
        console.log(this.name);  // undefined（this 指向 window）
    }
};

obj.regularFn();
obj.arrowFn();

// 箭头函数适合回调场景
function Timer() {
    this.seconds = 0;
    setInterval(() => {
        this.seconds++;  // this 指向 Timer 实例
    }, 1000);
}

// 对比普通函数
function Timer2() {
    this.seconds = 0;
    setInterval(function() {
        this.seconds++;  // this 指向 window，错误
    }, 1000);
}
```


### 原型链继承

```javascript
// 寄生组合式继承（最佳实践）
function Parent(name) {
    this.name = name;
    this.colors = ['red', 'blue'];
}

Parent.prototype.getName = function() {
    return this.name;
};

function Child(name, age) {
    // 借用构造函数继承属性
    Parent.call(this, name);
    this.age = age;
}

// 原型链继承方法
Child.prototype = Object.create(Parent.prototype);
Child.prototype.constructor = Child;

// 添加子类方法
Child.prototype.getAge = function() {
    return this.age;
};

const child = new Child('小明', 10);
console.log(child.getName());  // '小明'
console.log(child.getAge());   // 10
```

### ES6 class 继承

```javascript
class Parent {
    constructor(name) {
        this.name = name;
    }

    getName() {
        return this.name;
    }

    // 静态方法
    static create(name) {
        return new Parent(name);
    }
}

class Child extends Parent {
    constructor(name, age) {
        super(name);  // 调用父类构造函数
        this.age = age;
    }

    getAge() {
        return this.age;
    }

    // 重写父类方法
    getName() {
        return `${this.name} (${this.age}岁)`;
    }
}

const child = new Child('小明', 10);
console.log(child.getName());  // '小明 (10 岁)'
console.log(Child.create('张三'));  // 静态方法
```

### 纯函数

```javascript
// 纯函数：相同输入总是得到相同输出，无副作用

// 纯函数示例
function add(a, b) {
    return a + b;
}

// 不纯函数

    counter++;  // 修改外部状态，有副作用

// 纯函数的好处：可测试、可缓存、可组合
const memoize = (fn) => {
    const cache = {};
    return (...args) => {
        const key = JSON.stringify(args);
        if (cache[key]) {
            return cache[key];
        }
        const result = fn(...args);
        cache[key] = result;
        return result;
    };
};

const memoizedAdd = memoize(add);
```

## 4.7 实践练习

### 练习 1：事件发射器

```javascript
class EventEmitter {
    constructor() {
        this.events = new Map();
    }

    // 订阅事件
    on(event, listener) {
        if (!this.events.has(event)) {
            this.events.set(event, []);
        }
        this.events.get(event).push(listener);
        return this;
    }

    // 一次性订阅
    once(event, listener) {
        const wrapper = (...args) => {
            listener(...args);
            this.off(event, wrapper);
        };
        return this.on(event, wrapper);
    }

    // 取消订阅
    off(event, listener) {
        if (!this.events.has(event)) return this;

        const listeners = this.events.get(event);
        const index = listeners.indexOf(listener);
        if (index > -1) {
            listeners.splice(index, 1);
        }
        if (listeners.length === 0) {
            this.events.delete(event);
        }
        return this;
    }

    // 发布事件
    emit(event, ...args) {
        if (!this.events.has(event)) return false;

        const listeners = this.events.get(event);
        listeners.forEach(listener => listener(...args));
        return true;
    }

    // 获取监听器数量
    listenerCount(event) {
        if (!this.events.has(event)) return 0;
        return this.events.get(event).length;
    }
}

// 使用示例
const emitter = new EventEmitter();

emitter.on('message', (msg) => {
    console.log('收到消息:', msg);
});

emitter.once('data', (data) => {
    console.log('一次性数据:', data);
});

emitter.emit('message', 'Hello');
emitter.emit('data', { id: 1 });
emitter.emit('data', { id: 2 });  // 不会触发 once 监听
```

### 练习 2：Observable（可观察对象）

```javascript
class Observable {
    constructor(subscribeFn) {
        this.subscribeFn = subscribeFn;
    }

    subscribe(observer) {
        return this.subscribeFn(observer);
    }

    // 操作符：map
    map(project) {
        return new Observable((observer) => {
            return this.subscribe({
                next: (value) => observer.next(project(value)),
                error: (err) => observer.error(err),
                complete: () => observer.complete()
            });
        });
    }

    // 操作符：filter
    filter(predicate) {
        return new Observable((observer) => {
            return this.subscribe({
                next: (value) => {
                    if (predicate(value)) {
                        observer.next(value);
                    }
                },
                error: (err) => observer.error(err),
                complete: () => observer.complete()
            });
        });
    }

    // 操作符：reduce
    reduce(accumulator, initialValue) {
        return new Observable((observer) => {
            let acc = initialValue;
            return this.subscribe({
                next: (value) => {
                    acc = accumulator(acc, value);
                },
                complete: () => observer.next(acc)
            });
        });
    }
}

// 使用示例
const numbers$ = new Observable((observer) => {
    [1, 2, 3, 4, 5].forEach(n => observer.next(n));
    observer.complete();
});

numbers$
    .filter(n => n % 2 === 0)
    .map(n => n * 10)
    .reduce((acc, n) => acc + n, 0)
    .subscribe({
        next: (result) => console.log('结果:', result),  // 60
        complete: () => console.log('完成')
    });
```

### 练习 3：代理实现数据绑定

```javascript
class DataBinding {
    constructor(initialData = {}) {
        this.data = new Proxy(initialData, {
            get: (target, key) => {
                console.log(`读取 ${key}`);
                return target[key];
            },
            set: (target, key, value) => {
                console.log(`设置 ${key} = ${value}`);
                target[key] = value;
                this.notify(key, value);
                return true;
            }
        });
        this.listeners = new Map();
    }

    watch(key, callback) {
        if (!this.listeners.has(key)) {
            this.listeners.set(key, []);
        }
        this.listeners.get(key).push(callback);
    }

    notify(key, value) {
        if (this.listeners.has(key)) {
            this.listeners.get(key).forEach(cb => cb(value));
        }
    }
}

// 使用示例
const binding = new DataBinding({ count: 0, name: '张三' });

binding.watch('count', (newValue) => {
    console.log(`count 变更为 ${newValue}`);
});

binding.watch('name', (newValue) => {
    console.log(`name 变更为 ${newValue}`);
});

binding.data.count = 10;   // 触发 count 的回调
binding.data.name = '李四'; // 触发 name 的回调
```

### Q1: 什么是提升（hoisting）？

```javascript
// 变量提升
console.log(a);  // undefined
var a = 1;

// 等价于
var a;
console.log(a);

// 函数提升

```

### Q2: 如何正确比较对象？

```javascript
// == 和 === 比较的是引用
const obj1 = { a: 1 };
const obj2 = { a: 1 };
console.log(obj1 === obj2);  // false

// 深度比较
function deepEqual(a, b) {
    if (a === b) return true;
    if (typeof a !== 'object' || typeof b !== 'object') return false;
    if (a === null || b === null) return false;

    const keysA = Object.keys(a);
    const keysB = Object.keys(b);
    if (keysA.length !== keysB.length) return false;

    for (const key of keysA) {
        if (!keysB.includes(key)) return false;
        if (!deepEqual(a[key], b[key])) return false;
    }
    return true;
}
```

### Q3: 什么是尾调用优化？

```javascript
// 非尾递归（会栈溢出）
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);  // 最后一步是乘法
}

// 尾递归（可优化）
function factorialTail(n, acc = 1) {
    if (n <= 1) return acc;
    return factorialTail(n - 1, n * acc);  // 最后一步是函数调用
}
```
