## 4.1 深入闭包 - JavaScript 的灵魂

### 什么是闭包？

**闭包就是"函数记住并访问其创建时的环境"。**

```
生活例子：
你离开家（外部函数执行完），
但你口袋里还带着家里的钥匙（闭包），
随时可以回家（访问外部变量）。
```

```javascript
// 闭包的形成
function outer() {
    let count = 0;  // 外部函数的变量
    
    // 内部函数（闭包）
    return function inner() {
        count++;
        console.log(count);
    };
}

const counter = outer();
counter();  // 1
counter();  // 2
counter();  // 3

// count 变量一直被保留着
```

### 为什么会有闭包？

```
想象一个俄罗斯套娃：
- 小娃娃（内部函数）在大娃娃（外部函数）里面
- 小娃娃可以看到大娃娃的东西
- 即使把大娃娃拿走，小娃娃还记得里面的东西
```

```javascript
function createMessage(name) {
    const greeting = '你好';  // 外部变量
    
    return function() {
        console.log(`${greeting}, ${name}!`);
        // 内部函数可以访问外部变量
    };
}

const sayHello = createMessage('张三');
sayHello();  // '你好，张三!'
```

### 闭包的应用场景

#### 1. 数据私有化

```javascript
// 模拟私有变量
function createBankAccount(initialBalance) {
    let balance = initialBalance;  // 私有变量，外部不能直接访问
    
    return {
        deposit: function(amount) {
            balance += amount;
            return balance;
        },
        withdraw: function(amount) {
            if (amount > balance) {
                throw new Error('余额不足');
            }
            balance -= amount;
            return balance;
        },
        getBalance: function() {
            return balance;
        }
    };
}

const account = createBankAccount(1000);
console.log(account.getBalance());  // 1000
account.deposit(500);
console.log(account.getBalance());  // 1500

// balance 不能被外部直接修改
// account.balance = 999999;  // 无效
```

#### 2. 函数工厂

```javascript
// 创建特定功能的函数
function createMultiplier(multiplier) {
    return function(number) {
        return number * multiplier;
    };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15
```

#### 3. 防抖和节流

```javascript
// 防抖：一段时间内只执行最后一次
function debounce(fn, delay) {
    let timer = null;
    
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

// 使用：搜索框输入
const searchInput = document.getElementById('search');
searchInput.addEventListener('input', debounce((e) => {
    console.log('搜索:', e.target.value);
    // 发送搜索请求
}, 300));
```

## 4.2 作用域链 - 变量的查找规则

### 什么是作用域链？

**作用域链是变量查找的路径，从内向外查找。**

```
生活例子：
在家找东西：
1. 先看自己房间（当前作用域）
2. 再看客厅（父作用域）
3. 最后看整个家（全局作用域）
```

```javascript
let name = '全局';

function outer() {
    let name = '外部';
    
    function inner() {
        let name = '内部';
        console.log(name);  // '内部'
    }
    
    inner();
}

outer();
```

### 词法作用域

```javascript
// 作用域在函数定义时就确定了
const name = '全局';

function greet() {
    console.log(`你好，${name}`);
}

function outer() {
    const name = '外部';
    greet();  // '你好，全局'（不是'外部'）
}

outer();
```

## 4.3 this 绑定 - 最容易混淆的概念

### 什么是 this？

**this 指向"调用者"，谁调用就指向谁。**

```
生活例子：
- 你说"我家" → 指你的家
- 我说"我家" → 指我的家
- "我家"的"我"是谁，取决于谁在说

this 也一样：
- obj.fn() → this 指向 obj
- fn() → this 指向全局（或 undefined）
```

### this 绑定规则

#### 1. 默认绑定

```javascript
// 独立函数调用，this 指向全局（非严格模式）或 undefined（严格模式）
function greet() {
    console.log(this);
}

greet();  // 非严格模式：window，严格模式：undefined
```

#### 2. 隐式绑定

```javascript
// 作为对象方法调用，this 指向对象
const obj = {
    name: '张三',
    greet() {
        console.log(`你好，我是${this.name}`);
    }
};

obj.greet();  // '你好，我是张三'，this 指向 obj
```

#### 3. 显式绑定

```javascript
// 使用 call、apply、bind 指定 this
function greet() {
    console.log(`你好，我是${this.name}`);
}

const person = { name: '张三' };

greet.call(person);     // '你好，我是张三'
greet.apply(person);    // '你好，我是张三'

const boundGreet = greet.bind(person);
boundGreet();           // '你好，我是张三'
```

#### 4. new 绑定

```javascript
// 使用 new 调用，this 指向新创建的对象
function Person(name) {
    this.name = name;
}

const p = new Person('张三');
console.log(p.name);  // '张三'
```

### this 陷阱

```javascript
// 陷阱 1：方法赋值
const obj = {
    name: '张三',
    greet() {
        console.log(`你好，${this.name}`);
    }
};

const fn = obj.greet;
fn();  // undefined（this 指向全局）

// 解决：用 bind
const boundFn = obj.greet.bind(obj);
boundFn();  // '你好，张三'

// 陷阱 2：回调函数中的 this
const obj2 = {
    name: '张三',
    greet() {
        setTimeout(function() {
            console.log(this.name);  // undefined
        }, 1000);
    }
};

// 解决 1：保存 this
const obj3 = {
    name: '张三',
    greet() {
        const self = this;
        setTimeout(function() {
            console.log(self.name);  // '张三'
        }, 1000);
    }
};

// 解决 2：用箭头函数
const obj4 = {
    name: '张三',
    greet() {
        setTimeout(() => {
            console.log(this.name);  // '张三'
        }, 1000);
    }
};
```

## 4.4 原型与原型链 - JavaScript 的继承机制

### 什么是原型？

**每个函数都有一个 prototype 属性，叫做原型。**

```
生活例子：
原型就像"模具"：
- 用同一个模具做出来的产品都有相同的特征
- 产品可以访问模具上的方法
```

```javascript
function Person(name) {
    this.name = name;
}

// 原型上的方法，所有实例共享
Person.prototype.sayHello = function() {
    console.log(`你好，我是${this.name}`);
};

const p1 = new Person('张三');
const p2 = new Person('李四');

p1.sayHello();  // '你好，我是张三'
p2.sayHello();  // '你好，我是李四'

// 方法在原型上，不在实例上
console.log(p1.sayHello === p2.sayHello);  // true
```

### 原型链

```
实例 → 构造函数原型 → 父类原型 → ... → Object.prototype → null

就像家族族谱：
你 → 父母 → 祖父母 → ... → 祖先
```

```javascript
function Animal(name) {
    this.name = name;
}

Animal.prototype.eat = function() {
    console.log('吃东西');
};

function Dog(name) {
    this.name = name;
}

// 继承 Animal
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.bark = function() {
    console.log('汪汪汪');
};

const dog = new Dog('旺财');

dog.bark();  // '汪汪汪'（自己的方法）
dog.eat();   // '吃东西'（原型链上的方法）
```

### 原型链查找

```javascript
const obj = { a: 1 };

// 查找顺序：
obj.a;           // 1（自身属性）
obj.toString();  // [object Object]（原型链上）
obj.xxx;         // undefined（找不到）

// __proto__ 指向原型
console.log(obj.__proto__ === Object.prototype);  // true
```

## 4.5 继承的多种实现

### 原型链继承

```javascript
function Parent() {
    this.name = 'Parent';
}

Parent.prototype.getName = function() {
    return this.name;
};

function Child() {
    this.type = 'Child';
}

// 继承
Child.prototype = Object.create(Parent.prototype);
Child.prototype.constructor = Child;

const child = new Child();
console.log(child.getName());  // 'Parent'
```

### 构造函数继承

```javascript
function Parent(name) {
    this.name = name;
}

function Child(name, age) {
    Parent.call(this, name);  // 调用父类构造函数
    this.age = age;
}

const child = new Child('张三', 10);
console.log(child.name);  // '张三'
console.log(child.age);   // 10
```

### class 继承（ES6）

```javascript
class Parent {
    constructor(name) {
        this.name = name;
    }
    
    getName() {
        return this.name;
    }
}

class Child extends Parent {
    constructor(name, age) {
        super(name);  // 调用父类构造函数
        this.age = age;
    }
    
    getInfo() {
        return `${this.name}, ${this.age}岁`;
    }
}

const child = new Child('张三', 10);
console.log(child.getName());  // '张三'
console.log(child.getInfo());  // '张三，10 岁'
```

## 4.6 函数柯里化 - 固定部分参数

### 什么是柯里化？

**柯里化是把多参数函数变成一系列单参数函数。**

```
生活例子：
点奶茶：
- 原味：选茶底 → 选甜度 → 选温度
- 柯里化：先固定茶底，再固定甜度，最后固定温度
```

```javascript
// 普通函数
function add(a, b, c) {
    return a + b + c;
}
add(1, 2, 3);  // 6

// 柯里化
function curriedAdd(a) {
    return function(b) {
        return function(c) {
            return a + b + c;
        };
    };
}
curriedAdd(1)(2)(3);  // 6
```

### 柯里化应用

```javascript
// 通用柯里化函数
function curry(fn) {
    return function curried(...args) {
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        } else {
            return function(...moreArgs) {
                return curried.apply(this, args.concat(moreArgs));
            };
        }
    };
}

// 使用
function multiply(a, b, c) {
    return a * b * c;
}

const curriedMultiply = curry(multiply);
console.log(curriedMultiply(2)(3)(4));  // 24
```

## 4.7 函数防抖和节流 - 控制执行频率

### 防抖（Debounce）

```javascript
// 防抖：一段时间内只执行最后一次
function debounce(fn, delay) {
    let timer = null;
    
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

// 应用场景：搜索框
const input = document.getElementById('search');
input.addEventListener('input', debounce((e) => {
    console.log('搜索:', e.target.value);
}, 300));

// 生活例子：
// 电梯关门：每次有人进来就重新计时
// 最后一个人进来后，等几秒再关门
```

### 节流（Throttle）

```javascript
// 节流：一段时间内只执行一次
function throttle(fn, limit) {
    let inThrottle = false;
    
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

// 应用场景：滚动加载
window.addEventListener('scroll', throttle(() => {
    console.log('滚动了');
    // 加载更多内容
}, 100));

// 生活例子：
// 地铁进站：每隔固定时间发一班车
// 不管有多少人等
```

## 4.8 深浅拷贝 - 复制对象

### 浅拷贝

```javascript
// 浅拷贝：只复制一层
const obj1 = { a: 1, b: { c: 2 } };

// 展开运算符
const obj2 = { ...obj1 };
obj2.b.c = 999;
console.log(obj1.b.c);  // 999（影响了原对象）

// Object.assign
const obj3 = Object.assign({}, obj1);
```

### 深拷贝

```javascript
// 深拷贝：递归复制所有层

// 方法 1：JSON（有局限）
const obj1 = { a: 1, b: { c: 2 } };
const obj2 = JSON.parse(JSON.stringify(obj1));

// 局限：不能处理函数、undefined、Symbol、Date 等

// 方法 2：递归实现
function deepClone(obj) {
    if (obj === null || typeof obj !== 'object') return obj;
    
    if (Array.isArray(obj)) {
        return obj.map(item => deepClone(item));
    }
    
    const cloned = {};
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            cloned[key] = deepClone(obj[key]);
        }
    }
    return cloned;
}

// 方法 3：structuredClone（现代浏览器）
const obj3 = structuredClone(obj1);
```

## 4.11 实践练习

### 练习 1：实现事件发射器

```javascript
class EventEmitter {
    constructor() {
        this.events = {};
    }
    
    // 订阅事件
    on(event, listener) {
        if (!this.events[event]) {
            this.events[event] = [];
        }
        this.events[event].push(listener);
        return this;
    }
    
    // 一次性订阅
    once(event, listener) {
        const wrapper = (...args) => {
            listener(...args);
            this.off(event, wrapper);
        };
        this.on(event, wrapper);
        return this;
    }
    
    // 取消订阅
    off(event, listener) {
        if (!this.events[event]) return this;
        this.events[event] = this.events[event].filter(
            l => l !== listener
        );
        return this;
    }
    
    // 触发事件
    emit(event, ...args) {
        if (!this.events[event]) return false;
        this.events[event].forEach(listener => listener(...args));
        return true;
    }
}

// 使用
const emitter = new EventEmitter();

emitter.on('data', (data) => {
    console.log('收到数据:', data);
});

emitter.emit('data', { id: 1, name: '张三' });
```

### 练习 2：实现 Promise

```javascript
class MyPromise {
    constructor(executor) {
        this.state = 'pending';
        this.value = undefined;
        this.onFulfilled = [];
        this.onRejected = [];
        
        const resolve = (value) => {
            setTimeout(() => {
                if (this.state === 'pending') {
                    this.state = 'fulfilled';
                    this.value = value;
                    this.onFulfilled.forEach(fn => fn(value));
                }
            });
        };
        
        const reject = (reason) => {
            setTimeout(() => {
                if (this.state === 'pending') {
                    this.state = 'rejected';
                    this.value = reason;
                    this.onRejected.forEach(fn => fn(reason));
                }
            });
        };
        
        try {
            executor(resolve, reject);
        } catch (error) {
            reject(error);
        }
    }
    
    then(onFulfilled, onRejected) {
        return new MyPromise((resolve, reject) => {
            if (this.state === 'fulfilled') {
                try {
                    const result = onFulfilled(this.value);
                    resolve(result);
                } catch (error) {
                    reject(error);
                }
            } else if (this.state === 'rejected') {
                try {
                    const result = onRejected(this.value);
                    resolve(result);
                } catch (error) {
                    reject(error);
                }
            } else {
                this.onFulfilled.push(value => {
                    try {
                        const result = onFulfilled(value);
                        resolve(result);
                    } catch (error) {
                        reject(error);
                    }
                });
                this.onRejected.push(reason => {
                    try {
                        const result = onRejected(reason);
                        resolve(result);
                    } catch (error) {
                        reject(error);
                    }
                });
            }
        });
    }
    
    catch(onRejected) {
        return this.then(undefined, onRejected);
    }
}
```

## 4.12 常见问答

### Q1: 箭头函数和普通函数有什么区别？

**答：**
1. this 绑定不同（箭头函数没有自己的 this）
2. 箭头函数不能作为构造函数
3. 箭头函数没有 arguments
4. 箭头函数不能用作生成器

### Q2: 什么是闭包？有什么副作用？

**答：** 闭包是函数记住并访问其创建时的环境。

副作用：
- 可能导致内存泄漏
- 解决方法：及时解除引用

### Q3: this 到底指向谁？

**答：** 看调用方式：
- 普通函数：全局（或 undefined）
- 对象方法：对象
- call/apply/bind：指定的对象
- 箭头函数：定义时的上下文
- new：新创建的对象
