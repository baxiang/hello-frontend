# 09-错误处理

> 学会捕获和处理 JavaScript 中的错误，编写健壮的代码

## 学习目标

- ✅ 理解错误的类型
- ✅ 掌握 try-catch-finally 语句
- ✅ 学会抛出和处理自定义错误
- ✅ 掌握 console 调试方法
- ✅ 理解常见的错误场景

## 9.1 错误的类型

### JavaScript 内置错误类型

```javascript
// 1. Error - 基础错误
let error = new Error('这是一个错误');

// 2. SyntaxError - 语法错误
// eval('if (true { }');  // 括号不匹配

// 3. ReferenceError - 引用错误
// console.log(notDefined);  // 变量未定义

// 4. TypeError - 类型错误
// null.toString();  // 不能调用 null 的方法
// 'hello'.nonexistent();  // 方法不存在

// 5. RangeError - 范围错误
// new Array(-1);  // 负数长度
// (10).toFixed(1000);  // 小数位过多

// 6. URIError - URI 错误
// decodeURIComponent('%');  // 无效的 URI

// 7. EvalError - eval 错误（现代 JS 很少见）
```

### 错误对象的结构

```javascript
try {
    throw new Error('出错了');
} catch (error) {
    console.log(error.name);     // 'Error'
    console.log(error.message);  // '出错了'
    console.log(error.stack);    // 堆栈跟踪
}
```

## 9.2 try-catch-finally

### 基本语法

```javascript
try {
    // 可能出错的代码
    let result = riskyOperation();
    console.log(result);
} catch (error) {
    // 处理错误
    console.error('出错了:', error.message);
} finally {
    // 无论是否出错都会执行
    console.log('清理工作');
}
```

### 执行流程

```
┌─────────────────┐
│   try 代码块    │
└────────┬────────┘
         │
    ┌────┴────┐
    │ 有错误？ │
    └────┬────┘
         │
    ┌────┴────┐
    │    是   │    否
    ▼         ▼
┌───────┐  ┌───────┐
│ catch │  │ 跳过  │
└───┬───┘  └───┬───┘
    │          │
    └────┬─────┘
         │
         ▼
    ┌───────┐
    │ finally│
    └───────┘
```

### 实际应用场景

```javascript
// 1. API 请求
async function fetchData(url) {
    try {
        let response = await fetch(url);
        let data = await response.json();
        return data;
    } catch (error) {
        console.error('请求失败:', error.message);
        throw error;  // 重新抛出，让调用者处理
    }
}

// 2. JSON 解析
function parseJSON(str) {
    try {
        return JSON.parse(str);
    } catch (error) {
        console.error('JSON 解析失败:', error.message);
        return null;
    }
}

// 3. 文件操作（Node.js）
// try {
//     let data = fs.readFileSync('file.txt', 'utf8');
//     console.log(data);
// } catch (error) {
//     console.error('读取文件失败:', error.message);
// }
```

## 9.3 finally 语句

### 特点

```javascript
// finally 总是执行，即使 try 中有 return
function test() {
    try {
        console.log('try');
        return 'from try';
    } catch (error) {
        console.log('catch');
        return 'from catch';
    } finally {
        console.log('finally');  // 这个也会执行
    }
}

let result = test();
// 输出：
// try
// finally
console.log(result);  // 'from try'
```

### 使用场景

```javascript
// 1. 关闭资源
let file;
try {
    file = openFile('data.txt');
    processFile(file);
} catch (error) {
    console.error('处理失败:', error.message);
} finally {
    if (file) {
        closeFile(file);  // 确保文件被关闭
    }
}

// 2. 隐藏加载状态
function loadData() {
    showLoading();
    try {
        return fetchData();
    } catch (error) {
        console.error(error);
    } finally {
        hideLoading();  // 无论成功失败都隐藏
    }
}
```

## 9.4 抛出错误

### throw 语句

```javascript
// 抛出内置错误
throw new Error('出错了');
throw new TypeError('类型错误');
throw new RangeError('超出范围');

// 抛出字符串（不推荐）
throw '错误信息';

// 抛出数字（不推荐）
throw 404;
```

### 自定义错误类

```javascript
// 继承 Error 类
class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = 'ValidationError';
    }
}

class NotFoundError extends Error {
    constructor(resource) {
        super(`${resource} not found`);
        this.name = 'NotFoundError';
        this.code = 404;
    }
}

// 使用
function validateAge(age) {
    if (age < 0 || age > 150) {
        throw new ValidationError('年龄必须在 0-150 之间');
    }
}

try {
    validateAge(-1);
} catch (error) {
    if (error instanceof ValidationError) {
        console.error('验证失败:', error.message);
    }
}
```

## 9.5 错误处理策略

### 1. 吞掉错误（谨慎使用）

```javascript
// 只在确定错误可以忽略时使用
try {
    localStorage.getItem('key');
} catch (error) {
    // 静默处理（可能 localStorage 不可用）
}

// 更好的做法：记录日志
try {
    localStorage.getItem('key');
} catch (error) {
    console.warn('localStorage 不可用:', error.message);
}
```

### 2. 重新抛出

```javascript
function processData(data) {
    try {
        return transform(data);
    } catch (error) {
        console.error('处理失败:', error.message);
        throw error;  // 重新抛出，让调用者决定如何处理
    }
}
```

### 3. 包装错误

```javascript
function fetchUser(id) {
    try {
        return api.getUser(id);
    } catch (error) {
        throw new Error(`获取用户 ${id} 失败：${error.message}`);
    }
}
```

### 4. 多个 catch（条件处理）

```javascript
try {
    riskyOperation();
} catch (error) {
    if (error instanceof TypeError) {
        handleTypeError(error);
    } else if (error instanceof NetworkError) {
        handleNetworkError(error);
    } else {
        handleUnknownError(error);
    }
}
```

## 9.6 调试技巧

### console 方法

```javascript
// 基本日志
console.log('普通日志');
console.info('信息');
console.warn('警告');
console.error('错误');

// 表格输出
console.table([
    { name: '张三', age: 25 },
    { name: '李四', age: 30 }

// 分组
console.group('用户信息');
console.log('name: 张三');
console.log('age: 25');
console.groupEnd();

// 计时
console.time('耗时');
// ... 执行代码
console.timeEnd('耗时');  // 输出：耗时：123.456ms

// 断言
console.assert(age >= 0, '年龄不能为负数');
// 相当于：
// if (!(age >= 0)) {
//     console.error('年龄不能为负数');
// }

// 堆栈跟踪
console.trace('追踪');
// 输出调用堆栈
```

### debugger 语句

```javascript
function debugMe() {
    debugger;  // 代码执行会在这里暂停
    // 可以在 DevTools 中查看变量、单步调试
}

// 条件断点
function process(value) {
    if (value < 0) {
        debugger;  // 只在 value < 0 时暂停
    }
    return value * 2;
}
```

## 9.7 常见错误场景

### 1. 空值访问

```javascript
// ❌ 容易出错
let city = user.address.city;

// ✅ 安全访问
let city = user?.address?.city ?? '未知城市';

// ✅ 或提前检查
if (user && user.address && user.address.city) {
    let city = user.address.city;
}
```

### 2. 类型转换

```javascript
// ❌ 隐式转换可能导致问题
let result = '5' + 3;  // '53'（字符串拼接）

// ✅ 显式转换
let result = Number('5') + 3;  // 8

// ❌ parseInt 可能返回 NaN
let num = parseInt('abc');  // NaN

// ✅ 检查转换结果
let num = parseInt('123px');
if (isNaN(num)) {
    console.error('无效的数字');
}
```

### 3. 异步错误

```javascript
// ❌ Promise 错误未处理
fetch('/api/data').then(res => res.json());

// ✅ 添加 catch
fetch('/api/data')
    .then(res => res.json())
    .catch(error => console.error(error));

// ✅ 或使用 async/await
async function getData() {
    try {
        let res = await fetch('/api/data');
        let data = await res.json();
        return data;
    } catch (error) {
        console.error(error);
    }
}
```

### 4. 数组越界

```javascript
let arr = [1, 2, 3];

// ❌ 可能 undefined
let item = arr[5];

// ✅ 检查索引
if (index >= 0 && index < arr.length) {
    let item = arr[index];
}

// ✅ 或使用可选链
let item = arr?.[5];
```

## 9.8 实践练习

### 练习 1：错误处理基础

```javascript
// 1. 安全的 JSON 解析
function safeJSONParse(str, defaultValue = null) {
    try {
        return JSON.parse(str);
    } catch (error) {
        console.warn('JSON 解析失败:', error.message);
        return defaultValue;
    }
}

console.log(safeJSONParse('{"a": 1}'));  // { a: 1 }
console.log(safeJSONParse('invalid'));   // null

// 2. 安全的属性访问
function safeGet(obj, path, defaultValue) {
    try {
        return path.split('.').reduce((acc, key) => acc?.[key], obj) ?? defaultValue;
    } catch (error) {
        return defaultValue;
    }
}

let user = { profile: { name: '张三' } };
console.log(safeGet(user, 'profile.name', '未知'));  // '张三'
console.log(safeGet(user, 'profile.age', 0));        // 0
```

### 练习 2：自定义错误

```javascript
// 1. 实现带验证的除数
class DivisionError extends Error {
    constructor(message) {
        super(message);
        this.name = 'DivisionError';
    }
}

function divide(a, b) {
    if (b === 0) {
        throw new DivisionError('除数不能为零');
    }
    return a / b;
}

try {
    console.log(divide(10, 0));
} catch (error) {
    if (error instanceof DivisionError) {
        console.error('除法错误:', error.message);
    }
}

// 2. 实现带超时的请求
class TimeoutError extends Error {
    constructor(ms) {
        super(`请求超时 (${ms}ms)`);
        this.name = 'TimeoutError';
    }
}

async function fetchWithTimeout(url, timeout = 5000) {
    const controller = new AbortController();
    const id = setTimeout(() => controller.abort(), timeout);
    
    try {
        const response = await fetch(url, {
            signal: controller.signal
        });
        return await response.json();
    } catch (error) {
        if (error.name === 'AbortError') {
            throw new TimeoutError(timeout);
        }
        throw error;
    } finally {
        clearTimeout(id);
    }
}
```

### 练习 3：日志工具

```javascript
// 实现一个简单的日志工具
const Logger = {
    level: 'info',  // debug, info, warn, error
    
    log(level, message, ...args) {
        const levels = ['debug', 'info', 'warn', 'error'];
        if (levels.indexOf(level) < levels.indexOf(this.level)) {
            return;
        }
        
        const timestamp = new Date().toISOString();
        const prefix = `[${timestamp}] [${level.toUpperCase()}]`;
        
        switch (level) {
            case 'error':
                console.error(prefix, message, ...args);
                break;
            case 'warn':
                console.warn(prefix, message, ...args);
                break;
            default:
                console.log(prefix, message, ...args);
        }
    },
    
    debug(message, ...args) {
        this.log('debug', message, ...args);
    },
    
    info(message, ...args) {
        this.log('info', message, ...args);
    },
    
    warn(message, ...args) {
        this.log('warn', message, ...args);
    },
    
    error(message, ...args) {
        this.log('error', message, ...args);
    }
};

// 使用
Logger.info('应用启动');
Logger.warn('配置项缺失，使用默认值');
Logger.error('请求失败', { url: '/api/data' });
```

## 9.9 常见问答

### Q1: try-catch 会影响性能吗？

**会，但影响很小**。现代 JS 引擎优化得很好，日常使用无需担心。但不要在循环中滥用。

```javascript
// ❌ 不推荐
for (let i = 0; i < 1000; i++) {
    try {
        // 可能出错的代码
    } catch (e) {}
}

// ✅ 推荐
try {
    for (let i = 0; i < 1000; i++) {
        // 可能出错的代码
    }
} catch (e) {
    // 处理错误
}
```

### Q2: 什么时候用 try-catch，什么时候用条件判断？

```javascript
// 用 try-catch：处理异常情况
try {
    JSON.parse(str);
} catch (e) {
    // 处理解析错误
}

// 用条件判断：处理预期内的情况
if (user && user.name) {
    console.log(user.name);
} else {
    console.log('匿名用户');
}
```

### Q3: 如何捕获全局错误？

```javascript
// 浏览器中
window.addEventListener('error', (event) => {
    console.error('全局错误:', event.error);
});

// Node.js 中
process.on('uncaughtException', (error) => {
    console.error('未捕获的异常:', error);
});
```
