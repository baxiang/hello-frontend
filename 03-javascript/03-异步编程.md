## 3.1 同步与异步

### 概念对比

```javascript
// 同步代码：按顺序执行，后一个任务等待前一个完成
console.log('1');
console.log('2');
console.log('3');
// 输出：1, 2, 3

// 异步代码：不等待任务完成，继续执行后续代码
console.log('1');
setTimeout(() => {
    console.log('2');
}, 1000);
console.log('3');
// 输出：1, 3, 2
```

### 事件循环

```javascript
// 调用栈（Call Stack）- 执行同步代码
// 宏任务队列（Macrotask Queue）- setTimeout, setInterval, I/O
// 微任务队列（Microtask Queue）- Promise.then, MutationObserver

console.log('1 - 同步');

setTimeout(() => {
    console.log('2 - 宏任务');
}, 0);

Promise.resolve().then(() => {
    console.log('3 - 微任务');
});

console.log('4 - 同步');
// 输出顺序：1, 4, 3, 2
```

## 3.2 回调函数

### 基础用法

```javascript
// 回调函数示例
function fetchData(callback) {
    setTimeout(() => {
        const data = { name: '张三' };
        callback(data);
    }, 1000);
}

fetchData((data) => {
    console.log('获取到数据:', data);
});

// 带错误处理的回调
function readFile(path, callback) {
    setTimeout(() => {
        if (path === 'error') {
            callback(new Error('文件不存在'), null);
        } else {
            callback(null, '文件内容');
        }
    }, 1000);
}

readFile('file.txt', (error, content) => {
    if (error) {
        console.error('读取失败:', error);
    } else {
        console.log('读取成功:', content);
    }
});
```

### 回调地狱

```javascript
// 回调地狱示例（不推荐）
getUser(userId, (user) => {
    getOrders(user.id, (orders) => {
        getOrderDetails(orders[0].id, (details) => {
            getProduct(details.productId, (product) => {
                console.log(product);
            });
        });
    });
});

// 使用 Promise 改进
getUser(userId)
    .then(user => getOrders(user.id))
    .then(orders => getOrderDetails(orders[0].id))
    .then(details => getProduct(details.productId))
    .then(product => console.log(product));
```

## 3.3 Promise 基础

### 创建 Promise

```javascript
// Promise 的三种状态
// pending - 进行中
// fulfilled - 已成功
// rejected - 已失败

const promise = new Promise((resolve, reject) => {
    // 异步操作
    setTimeout(() => {
        const success = true;
        if (success) {
            resolve('操作成功');
        } else {
            reject(new Error('操作失败'));
        }
    }, 1000);
});

// 使用 Promise
promise
    .then(result => {
        console.log(result);
    })
    .catch(error => {
        console.error(error);
    })
    .finally(() => {
        console.log('无论成功失败都会执行');
    });
```

### Promise 状态

```javascript
// 状态只能改变一次
const promise = new Promise((resolve, reject) => {
    resolve('成功');
    reject('失败');  // 不会执行
});

// 链式调用
Promise.resolve(1)
    .then(x => x + 1)      // 2
    .then(x => x * 2)      // 4
    .then(x => {
        return Promise.resolve(x + 1);  // 5
    })
    .then(x => console.log(x));  // 5

// 错误传播
Promise.resolve()
    .then(() => {
        throw new Error('错误');
    })
    .catch(error => {
        console.error(error);
        return '恢复值';
    })
    .then(x => console.log(x));  // '恢复值'
```

## 3.4 Promise 组合

### Promise.all

```javascript
// 等待所有 Promise 完成
Promise.all([
    fetch('/api/user'),
    fetch('/api/posts'),
    fetch('/api/comments')
])
.then(([userRes, postsRes, commentsRes]) => {
    return Promise.all([
        userRes.json(),
        postsRes.json(),
        commentsRes.json()
    ]);
})
.then(([user, posts, comments]) => {
    console.log('所有数据加载完成');
})
.catch(error => {
    console.error('任一请求失败:', error);
});

// 带超时的 Promise.all
function promiseWithTimeout(promise, timeout) {
    return Promise.race([
        promise,
        new Promise((_, reject) =>
            setTimeout(() => reject(new Error('超时')), timeout)
        )
    ]);
}
```

### Promise.allSettled

```javascript
// 等待所有 Promise 完成（不论成功失败）
Promise.allSettled([
    Promise.resolve('成功 1'),
    Promise.reject('失败 1'),
    Promise.resolve('成功 2')
])
.then(results => {
    results.forEach(result => {
        if (result.status === 'fulfilled') {
            console.log('成功:', result.value);
        } else {
            console.log('失败:', result.reason);
        }
    });
});
```

### Promise.race

```javascript
// 第一个完成的 Promise
Promise.race([
    fetch('/api/fast'),
    fetch('/api/slow')
])
.then(response => {
    console.log('最快的响应:', response);
});

// 超时控制
function fetchWithTimeout(url, timeout = 5000) {
    const timeoutPromise = new Promise((_, reject) => {
        setTimeout(() => reject(new Error('请求超时')), timeout);
    });

    return Promise.race([
        fetch(url),
        timeoutPromise
    ]);
}
```

### Promise.any

```javascript
// 第一个成功的 Promise
Promise.any([
    fetch('/api/primary'),
    fetch('/api/backup1'),
    fetch('/api/backup2')
])
.then(response => {
    console.log('第一个成功的响应:', response);
})
.catch(error => {
    // 所有都失败时的聚合错误
    console.error('所有请求都失败了:', error);
});
```

## 3.5 async/await

### 基础语法

```javascript
// async 函数总是返回 Promise
async function fetchData() {
    return '数据';
    // 等价于 return Promise.resolve('数据');
}

// await 只能在 async 函数中使用
async function getUser() {
    const response = await fetch('/api/user');
    const user = await response.json();
    return user;
}

// 错误处理
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}`);
        }
        return await response.json();
    } catch (error) {
        console.error('请求失败:', error);
        throw error;  // 重新抛出
    }
}
```

### 并行执行

```javascript
// 不推荐：串行执行（慢）
async function getData() {
    const user = await fetch('/api/user').then(r => r.json());
    const posts = await fetch('/api/posts').then(r => r.json());
    const comments = await fetch('/api/comments').then(r => r.json());
    return { user, posts, comments };
}

// 推荐：并行执行（快）
async function getData() {
    const [user, posts, comments] = await Promise.all([
        fetch('/api/user').then(r => r.json()),
        fetch('/api/posts').then(r => r.json()),
        fetch('/api/comments').then(r => r.json())
    ]);
    return { user, posts, comments };
}
```

### 顶层 await

```javascript
// 在支持顶层 await 的环境中
const data = await fetch('/api/data').then(r => r.json());
console.log(data);

// 模块中使用
export const config = await loadConfig();
```

### Fetch API

```javascript
// GET 请求
fetch('/api/users')
    .then(response => {
        if (!response.ok) {

        return response.json();
    })
    .then(data => console.log(data))
    .catch(error => console.error(error));
```

### Fetch 封装

```javascript
// 通用请求封装
const request = {
    base: '/api',

    async get(url, params = {}) {
        const queryString = new URLSearchParams(params).toString();
        const fullUrl = `${this.base}/${url}?${queryString}`;
        const response = await fetch(fullUrl);
        return this.handleResponse(response);
    },

    async post(url, data = {}) {
        const response = await fetch(`${this.base}/${url}`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });
        return this.handleResponse(response);
    },

    async put(url, data = {}) {
        const response = await fetch(`${this.base}/${url}`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });
        return this.handleResponse(response);
    },

    async delete(url) {
        const response = await fetch(`${this.base}/${url}`, {
            method: 'DELETE'
        });
        return this.handleResponse(response);
    },

    async handleResponse(response) {
        if (!response.ok) {
            const error = await response.json().catch(() => ({}));
            throw new Error(error.message || `HTTP ${response.status}`);
        }
        return response.json();
    }
};

// 使用示例
request.get('/users', { page: 1, limit: 10 });
request.post('/users', { name: '张三' });
```

## 3.7 定时器

### setTimeout / setInterval

```javascript
// setTimeout - 延迟执行
const timeoutId = setTimeout(() => {
    console.log('3 秒后执行');
}, 3000);

// 清除定时器
clearTimeout(timeoutId);

// setInterval - 周期性执行
let count = 0;
const intervalId = setInterval(() => {
    console.log(`第 ${++count} 次`);
    if (count >= 5) {
        clearInterval(intervalId);
    }
}, 1000);

// 使用 setTimeout 模拟 setInterval（推荐）
function repeatableSetInterval(callback, delay) {
    let timer;
    function tick() {
        callback();
        timer = setTimeout(tick, delay);
    }
    tick();
    return () => clearTimeout(timer);
}

const stop = repeatableSetInterval(() => {
    console.log('执行');
}, 1000);

// 5 秒后停止
setTimeout(stop, 5000);
```

### 定时器封装

```javascript
// 防抖（debounce）
function debounce(fn, delay = 300) {
    let timer = null;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

// 节流（throttle）
function throttle(fn, limit = 300) {
    let inThrottle = false;
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

// 使用示例
const handleSearch = debounce((query) => {
    console.log('搜索:', query);
}, 300);

const handleScroll = throttle(() => {
    console.log('滚动');
}, 100);
```

## 3.8 实践练习

### 练习 1：Promise 版图片加载

```javascript
// 加载单张图片
function loadImage(src) {
    return new Promise((resolve, reject) => {
        const img = new Image();
        img.onload = () => resolve(img);
        img.onerror = () => reject(new Error(`加载失败：${src}`));
        img.src = src;
    });
}

// 使用示例
loadImage('image.jpg')
    .then(img => {
        document.body.appendChild(img);
    })
    .catch(error => {
        console.error(error);
    });

// 加载多张图片
async function loadImages(sources) {
    const promises = sources.map(src => loadImage(src));
    const results = await Promise.allSettled(promises);

    return results.map((result, index) => ({
        src: sources[index],
        success: result.status === 'fulfilled',
        img: result.value,
        error: result.reason
    }));
}

// 批量加载
const images = await loadImages([
    'img1.jpg',
    'img2.jpg',
    'img3.jpg'
]);
images.forEach(item => {
    if (item.success) {
        console.log(`${item.src} 加载成功`);
    } else {
        console.error(`${item.src} 加载失败`);
    }
});
```

### 练习 2：请求重试机制

```javascript
// 带重试的请求
async function fetchWithRetry(url, options = {}, maxRetries = 3) {
    let lastError;

    for (let i = 0; i < maxRetries; i++) {
        try {
            const response = await fetch(url, options);
            if (!response.ok) {
                throw new Error(`HTTP ${response.status}`);
            }
            return await response.json();
        } catch (error) {
            lastError = error;
            console.warn(`第 ${i + 1} 次请求失败:`, error);

            if (i < maxRetries - 1) {
                // 指数退避
                const delay = Math.pow(2, i) * 1000;
                await new Promise(resolve => setTimeout(resolve, delay));
            }
        }
    }

    throw lastError;
}

// 使用示例
try {
    const data = await fetchWithRetry('/api/unstable', {}, 5);
    console.log('获取成功:', data);
} catch (error) {
    console.error('所有重试都失败了:', error);
}

// 带条件的重试
async function fetchWithRetry(url, options = {}, config = {}) {
    const {
        maxRetries = 3,
        retryDelay = 1000,
        shouldRetry = (error) => true,
        onRetry = (error, attempt) => {}
    } = config;

    for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            const response = await fetch(url, options);
            if (!response.ok) {
                throw new Error(`HTTP ${response.status}`);
            }
            return await response.json();
        } catch (error) {
            if (!shouldRetry(error) || attempt === maxRetries) {
                throw error;
            }
            onRetry(error, attempt);
            await new Promise(resolve =>
                setTimeout(resolve, retryDelay * attempt)
            );
        }
    }
}
```

### 练习 3：任务队列

```javascript
class TaskQueue {
    constructor(concurrency = 1) {
        this.concurrency = concurrency;
        this.queue = [];
        this.running = 0;
    }

    add(task) {
        return new Promise((resolve, reject) => {
            this.queue.push({ task, resolve, reject });
            this.run();
        });
    }

    async run() {
        while (this.running < this.concurrency && this.queue.length > 0) {
            const { task, resolve, reject } = this.queue.shift();
            this.running++;

            try {
                const result = await task();
                resolve(result);
            } catch (error) {
                reject(error);
            } finally {
                this.running--;
                this.run();
            }
        }
    }
}

// 使用示例
const queue = new TaskQueue(2);  // 最多并发 2 个任务

async function downloadFile(id) {
    console.log(`开始下载 ${id}`);
    await new Promise(resolve => setTimeout(resolve, 2000));
    console.log(`完成下载 ${id}`);
    return `文件 ${id}`;
}

// 添加任务
queue.add(() => downloadFile(1));
queue.add(() => downloadFile(2));
queue.add(() => downloadFile(3));
queue.add(() => downloadFile(4));
queue.add(() => downloadFile(5));
```

### 练习 4：请求取消

```javascript
// 使用 AbortController 取消请求
function createCancelableFetch() {
    const controller = new AbortController();

    return {
        fetch(url, options = {}) {
            return fetch(url, {
                ...options,
                signal: controller.signal
            });
        },
        cancel() {
            controller.abort();
        }
    };
}

// 使用示例
const request = createCancelableFetch();

request.fetch('/api/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => {
        if (error.name === 'AbortError') {
            console.log('请求已取消');
        } else {
            console.error('请求失败:', error);
        }
    });

// 5 秒后取消
setTimeout(() => request.cancel(), 5000);

// 搜索框防抖取消
class SearchBox {
    constructor(input, onSearch) {
        this.input = input;
        this.onSearch = onSearch;
        this.controller = null;
        this.debounceTimer = null;

        this.init();
    }

    init() {
        this.input.addEventListener('input', (e) => {
            const query = e.target.value.trim();

            // 清除之前的定时器
            clearTimeout(this.debounceTimer);

            // 取消之前的请求
            if (this.controller) {
                this.controller.abort();
            }

            if (!query) return;

            // 防抖延迟
            this.debounceTimer = setTimeout(async () => {
                this.controller = new AbortController();
                try {
                    const results = await this.onSearch(query, this.controller.signal);
                    console.log('搜索结果:', results);
                } catch (error) {
                    if (error.name !== 'AbortError') {
                        console.error('搜索失败:', error);
                    }
                }
            }, 300);
        });
    }
}
```

## 3.9 常见问题

### Q1: Promise 和 async/await 如何选择？

```javascript
// 简单场景：async/await 更直观
async function getData() {
    const user = await fetch('/api/user').then(r => r.json());
    return user;
}

// 并发场景：Promise.all 更简洁
async function getAllData() {
    const [user, posts, comments] = await Promise.all([
        fetch('/api/user').then(r => r.json()),
        fetch('/api/posts').then(r => r.json()),
        fetch('/api/comments').then(r => r.json())
    ]);
}
```

### Q2: 如何处理 Promise 链中的错误？

```javascript
// 方式 1：每个 then 都处理
promise
    .then(result => handle(result))
    .catch(error => console.error(error));

// 方式 2：链尾统一处理
promise
    .then(result => handle(result))
    .then(() => doSomething())
    .catch(error => console.error('任何错误都在这里处理', error));

// 方式 3：分别处理
promise
    .then(result => handle(result))
    .catch(error => {
        if (error.type === 'NetworkError') {
            // 网络错误
        } else {
            // 其他错误
        }
    });
```

### Q3: async/await 中的错误如何处理？

```javascript
// try-catch 方式
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('请求失败:', error);
        throw error;
    }
}

// .catch() 方式
fetchData().catch(error => {
    console.error('处理错误:', error);
});
```

### Q4: 如何实现 Promise 串行执行？

```javascript
// 方式 1：reduce
const tasks = [task1, task2, task3];
tasks.reduce((promise, task) => {
    return promise.then(() => task());
}, Promise.resolve());

// 方式 2：for...of
async function runSerial(tasks) {
    for (const task of tasks) {
        await task();
    }
}

// 方式 3：递归
function runSerial(tasks) {
    if (tasks.length === 0) return Promise.resolve();
    const [first, ...rest] = tasks;
    return first().then(() => runSerial(rest));
}
```
