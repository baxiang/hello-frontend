## 1.1 变量与常量

### 声明方式

```javascript
// var - 函数作用域（不推荐使用）
var name = '张三';
var name = '李四';  // 可以重复声明

// let - 块级作用域（推荐）
let age = 25;
age = 26;  // 可以重新赋值
// let age = 30;  // 报错：重复声明

// const - 块级作用域，常量（推荐）
const PI = 3.14159;
// PI = 3;  // 报错：常量不能重新赋值

const user = { name: '张三' };
user.name = '李四';  // 可以修改对象属性
// user = {};  // 报错：不能重新赋值
```

### 命名规则

```javascript
// 合法变量名
let userName;
let $element;
let _private;
let $123;

// 非法变量名
// let 123abc;  // 数字开头
// let user-name;  // 包含连字符
// let class;  // 保留字

// 驼峰命名（推荐）
let firstName;
let getUserInfo;

// 常量：大写下划线
const MAX_COUNT = 100;
const API_URL = 'https://api.example.com';
```

## 1.3 运算符

### 算术运算符

```javascript
let a = 10, b = 3;

a + b      // 13  加
a - b      // 7   减
a * b      // 30  乘
a / b      // 3.333...  除
a % b      // 1   取余
a ** b     // 1000  幂运算

a++        // 后置递增
++a        // 前置递增
a--        // 后置递减
--a        // 前置递减
```

### 比较运算符

```javascript
// 相等比较
5 == '5'       // true  宽松相等（会类型转换）
5 === '5'      // false 严格相等（不会类型转换）
5 != '5'       // false
5 !== '5'      // true

// 大小比较
5 > 3          // true
5 < 3          // false
5 >= 5         // true
5 <= 4         // false
```

### 逻辑运算符

```javascript
// 与或非
true && false      // false
true || false      // true
!true              // false

// 短路求值
let name = userInput || '默认用户';
let admin = isLoggedIn && isAdmin;

// 空值合并运算符（ES2020）
let count = inputCount ?? 10;  // 只有 null/undefined 时使用默认值
```

### 位运算符

```javascript
5 & 3        // 1   按位与
5 | 3        // 7   按位或
5 ^ 3        // 6   按位异或
~5           // -6  按位非
5 << 1       // 10  左移
5 >> 1       // 2   右移
5 >>> 1      // 2   无符号右移
```

## 1.4 控制流

### 条件语句

```javascript
// if-else
let score = 85;

if (score >= 90) {
    console.log('优秀');
} else if (score >= 80) {
    console.log('良好');
} else if (score >= 60) {
    console.log('及格');
} else {
    console.log('不及格');
}

// 三元运算符
let result = score >= 60 ? '及格' : '不及格';

// switch
let day = 3;
switch (day) {
    case 1:
        console.log('周一');
        break;
    case 2:
        console.log('周二');
        break;
    default:
        console.log('其他');
}
```

### 循环语句

```javascript
// for 循环
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// while 循环
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

// do-while 循环
let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

// for...in 遍历对象属性
let obj = { a: 1, b: 2, c: 3 };
for (let key in obj) {
    console.log(key, obj[key]);
}

// for...of 遍历可迭代对象（ES6）
let arr = [1, 2, 3, 4, 5];
for (let value of arr) {
    console.log(value);
}

// break 和 continue
for (let i = 0; i < 10; i++) {
    if (i === 3) continue;  // 跳过本次
    if (i === 7) break;     // 跳出循环
    console.log(i);
}
```

## 1.5 函数

### 函数声明

```javascript
// 函数声明（会提升）
function add(a, b) {
    return a + b;
}

// 函数表达式（不会提升）
const subtract = function(a, b) {
    return a - b;
};

// 箭头函数（ES6）
const multiply = (a, b) => a * b;

// 多语句箭头函数
const divide = (a, b) => {
    if (b === 0) {
        throw new Error('除数不能为零');
    }
    return a / b;
};
```

### 参数

```javascript
// 默认参数
function greet(name = '访客') {
    console.log(`你好，${name}`);
}

// 剩余参数
function sum(...numbers) {
    return numbers.reduce((acc, cur) => acc + cur, 0);
}
sum(1, 2, 3, 4);  // 10

// 解构参数
function printUser({ name, age }) {
    console.log(`${name}, ${age}岁`);
}

// 参数验证
function setUser(user) {
    if (!user || typeof user.name !== 'string') {
        throw new Error('Invalid user');
    }
}
```

### 作用域与闭包

```javascript
// 全局作用域
let globalVar = '全局';

function outer() {
    // 函数作用域
    let outerVar = '外部';

    function inner() {
        // 嵌套作用域
        let innerVar = '内部';
        console.log(outerVar);  // 可以访问
        console.log(globalVar); // 可以访问
    }

    // console.log(innerVar);  // 报错：无法访问
}

// 块级作用域（let/const）
if (true) {
    let blockVar = '块级';
    console.log(blockVar);  // '块级'
}
// console.log(blockVar);  // 报错：无法访问

// 闭包示例
function createCounter() {
    let count = 0;
    return {
        increment: () => ++count,
        decrement: () => --count,
        getCount: () => count
    };
}

const counter = createCounter();
counter.increment();  // 1
counter.increment();  // 2
console.log(counter.getCount());  // 2
```

## 1.7 数组

### 创建数组

```javascript
let arr = [1, 2, 3, 4, 5];
let arr2 = new Array(5);        // [empty × 5]
let arr3 = Array.from('hello'); // ['h', 'e', 'l', 'l', 'o']
```

### 数组方法

```javascript
let arr = [1, 2, 3, 4, 5];

// 增删
arr.push(6);           // 末尾添加，返回新长度
arr.pop();             // 末尾删除，返回删除的元素
arr.unshift(0);        // 开头添加
arr.shift();           // 开头删除

// 切片
arr.slice(1, 3);       // [2, 3]  不修改原数组
arr.splice(1, 2);      // 删除 2 个元素，返回删除的
arr.splice(1, 0, 'a'); // 在位置 1 插入'a'

// 查找
arr.indexOf(3);        // 2
arr.includes(3);       // true
arr.find(x => x > 3);  // 4
arr.findIndex(x => x > 3); // 3

// 转换
arr.join('-');         // '1-2-3-4-5'
arr.reverse();         // 反转
arr.sort((a, b) => a - b); // 排序

// 映射/过滤/归约
arr.map(x => x * 2);           // [2, 4, 6, 8, 10]
arr.filter(x => x > 2);        // [3, 4, 5]
arr.reduce((sum, x) => sum + x, 0); // 15

// 其他
arr.forEach(x => console.log(x));
arr.some(x => x > 3);          // true
arr.every(x => x > 0);         // true
```

### 数组解构

```javascript
let [a, b, c] = [1, 2, 3];
let [x, ...rest] = [1, 2, 3, 4];  // x=1, rest=[2,3,4]

// 交换变量
let m = 1, n = 2;
[m, n] = [n, m];  // m=2, n=1
```

## 1.8 错误处理

### try-catch

```javascript
try {
    // 可能出错的代码
    let result = riskyOperation();
    console.log(result);
} catch (error) {
    // 处理错误
    console.error('出错了:', error.message);
} finally {

    console.log('清理工作');
}
```

## 1.9 实践练习

### 练习 1：简易计算器

```javascript
class Calculator {
    constructor() {
        this.history = [];
    }

    add(a, b) {
        return this._record(a + b, `${a} + ${b}`);
    }

    subtract(a, b) {
        return this._record(a - b, `${a} - ${b}`);
    }

    multiply(a, b) {
        return this._record(a * b, `${a} × ${b}`);
    }

    divide(a, b) {
        if (b === 0) {
            throw new Error('除数不能为零');
        }
        return this._record(a / b, `${a} ÷ ${b}`);
    }

    _record(result, expression) {
        this.history.push({ expression, result });
        return result;
    }

    getHistory() {
        return this.history;
    }

    clearHistory() {
        this.history = [];
    }
}

// 使用示例
const calc = new Calculator();
console.log(calc.add(5, 3));      // 8
console.log(calc.multiply(4, 7)); // 28
console.log(calc.getHistory());
```

### 练习 2：待办事项管理

```javascript
class TodoManager {
    constructor() {
        this.todos = [];
        this.nextId = 1;
    }

    add(text, priority = 'medium') {
        const todo = {
            id: this.nextId++,
            text,
            priority,
            completed: false,
            createdAt: new Date()
        };
        this.todos.push(todo);
        return todo;
    }

    remove(id) {
        const index = this.todos.findIndex(t => t.id === id);
        if (index === -1) {
            throw new Error(`未找到 ID 为 ${id} 的待办事项`);
        }
        return this.todos.splice(index, 1)[0];
    }

    toggle(id) {
        const todo = this.todos.find(t => t.id === id);
        if (!todo) {
            throw new Error(`未找到 ID 为 ${id} 的待办事项`);
        }
        todo.completed = !todo.completed;
        return todo;
    }

    list(options = {}) {
        let result = [...this.todos];

        if (options.completed !== undefined) {
            result = result.filter(t => t.completed === options.completed);
        }

        if (options.priority) {
            result = result.filter(t => t.priority === options.priority);
        }

        if (options.sortBy === 'priority') {
            const order = { high: 0, medium: 1, low: 2 };
            result.sort((a, b) => order[a.priority] - order[b.priority]);
        } else if (options.sortBy === 'date') {
            result.sort((a, b) => b.createdAt - a.createdAt);
        }

        return result;
    }

    getStats() {
        const total = this.todos.length;
        const completed = this.todos.filter(t => t.completed).length;
        return {
            total,
            completed,
            pending: total - completed,
            completionRate: total === 0 ? 0 : (completed / total * 100).toFixed(1) + '%'
        };
    }
}

// 使用示例
const todo = new TodoManager();
todo.add('学习 JavaScript', 'high');
todo.add('完成练习', 'medium');
todo.add('复习 ES6', 'low');
todo.toggle(1);
console.log(todo.list());
console.log(todo.getStats());
```

### 练习 3：数据格式化

```javascript
const formatter = {
    // 格式化货币
    formatCurrency(amount, currency = 'CNY') {
        return new Intl.NumberFormat('zh-CN', {
            style: 'currency',
            currency
        }).format(amount);
    },

    // 格式化日期
    formatDate(date, options = {}) {
        const defaultOptions = {
            year: 'numeric',
            month: '2-digit',
            day: '2-digit'
        };
        return new Intl.DateTimeFormat('zh-CN', { ...defaultOptions, ...options })
            .format(new Date(date));
    },

    // 格式化数字（千分位）
    formatNumber(num) {
        return new Intl.NumberFormat('zh-CN').format(num);
    },

    // 格式化百分比
    formatPercent(value, decimals = 0) {
        return new Intl.NumberFormat('zh-CN', {
            style: 'percent',
            minimumFractionDigits: decimals,
            maximumFractionDigits: decimals
        }).format(value / 100);
    },

    // 格式化文件大小
    formatFileSize(bytes) {
        const units = ['B', 'KB', 'MB', 'GB', 'TB'];
        let unitIndex = 0;
        let size = bytes;

        while (size >= 1024 && unitIndex < units.length - 1) {
            size /= 1024;
            unitIndex++;
        }

        return `${size.toFixed(2)} ${units[unitIndex]}`;
    },

    // 格式化相对时间
    formatRelativeTime(date) {
        const now = new Date();
        const diff = now - new Date(date);
        const seconds = Math.floor(diff / 1000);
        const minutes = Math.floor(seconds / 60);
        const hours = Math.floor(minutes / 60);
        const days = Math.floor(hours / 24);

        if (days > 0) return `${days}天前`;
        if (hours > 0) return `${hours}小时前`;
        if (minutes > 0) return `${minutes}分钟前`;
        return '刚刚';
    }
};

// 使用示例
console.log(formatter.formatCurrency(1234567.89));  // ¥1,234,567.89
console.log(formatter.formatDate('2024-01-15'));    // 2024/01/15
console.log(formatter.formatFileSize(1536000));     // 1.46 MB
console.log(formatter.formatRelativeTime(new Date())); // 刚刚
```

## 1.10 常见问题

### Q1: var、let、const 有什么区别？

| 特性 | var | let | const |
|------|-----|-----|-------|
| 作用域 | 函数 | 块级 | 块级 |
| 变量提升 | 是 | 否 | 否 |
| 重复声明 | 允许 | 不允许 | 不允许 |
| 重新赋值 | 允许 | 允许 | 不允许 |

### Q2: == 和 === 有什么区别？

```javascript
// == 会进行类型转换
5 == '5'      // true
0 == false    // true
'' == false   // true
null == undefined  // true

// === 不会进行类型转换
5 === '5'    // false
0 === false  // false
```

### Q3: 什么是闭包？有什么用途？

闭包是指函数能够记住并访问其创建时的词法作用域。

```javascript
// 闭包应用：数据私有化
function createCounter() {
    let count = 0;  // 私有变量
    return {
        increment: () => ++count,
        getCount: () => count
    };
}
```
