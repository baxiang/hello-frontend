## 6.1 什么是作用域？

### 概念讲解

**作用域就是变量的"活动范围"**，规定了在哪里可以访问某个变量。

```
生活例子：
- 全局作用域 = 广场（谁都能访问）
- 函数作用域 = 房间（只有房间里的人能访问）
- 块级作用域 = 保险箱（只有特定代码能访问）
```

### 作用域的类型

```javascript
// 1. 全局作用域 - 任何地方都能访问
let globalVar = '全局变量';

function func() {
    console.log(globalVar);  // ✅ 可以访问
}

// 2. 函数作用域 - 只有函数内能访问
function outer() {
    let outerVar = '外部变量';
    console.log(outerVar);  // ✅
}
// console.log(outerVar);  // ❌ 报错

// 3. 块级作用域 - 只有块内能访问
if (true) {
    let blockVar = '块级变量';
    console.log(blockVar);  // ✅
}
// console.log(blockVar);  // ❌ 报错
```

## 6.2 全局作用域

### 特点

```javascript
// 在任何地方都能访问
let globalVar = '全局变量';
const GLOBAL_CONST = '全局常量';

function func1() {
    console.log(globalVar);  // ✅
}

function func2() {
    console.log(GLOBAL_CONST);  // ✅
}

// 浏览器中，全局变量会挂载到 window
console.log(window.globalVar);  // '全局变量'
```

### 注意事项

```javascript
// ❌ 避免过多全局变量
// 会污染全局命名空间，可能导致冲突

// ✅ 推荐：使用模块或 IIFE 封装
(function() {
    // 内部变量不会污染全局
    let privateVar = '私有';
})();
```

## 6.3 函数作用域（var）

### var 的函数作用域

```javascript
function test() {
    var functionVar = '函数内';
    console.log(functionVar);  // ✅
}
// console.log(functionVar);  // ❌ 报错

// var 没有块级作用域
function test2() {
    if (true) {
        var blockVar = '块内';
    }
    console.log(blockVar);  // ✅ 可以访问（var 没有块级作用域）
}
```

### 变量提升

```javascript
// var 存在变量提升
console.log(x);  // undefined
var x = 10;

// 实际执行顺序：
// var x;           // 声明提升
// console.log(x);  // undefined
// x = 10;          // 赋值留在原地
```

## 6.4 块级作用域（let/const）

### let/const 的块级作用域

```javascript
// if 语句
if (true) {
    let a = 1;
    const b = 2;
    console.log(a);  // ✅
}
// console.log(a);  // ❌ 报错

// for 循环
for (let i = 0; i < 5; i++) {
    console.log(i);
}
// console.log(i);  // ❌ 报错

// 任意代码块
{
    let blockVar = '块级变量';
    console.log(blockVar);  // ✅
}
// console.log(blockVar);  // ❌ 报错
```

### 经典问题：循环中的 var

```javascript
// ❌ var 的问题
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// 输出：3, 3, 3
// 原因：i 是同一个变量，循环结束时 i=3

// ✅ let 的解决方案
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// 输出：0, 1, 2
// 原因：每次循环 i 都是新的块级变量
```

## 6.5 词法作用域

### 什么是词法作用域？

**词法作用域**意味着函数的作用域在定义时就确定了，而不是在执行时。

```javascript
let name = '全局';

function outer() {
    let name = '外部';
    
    function inner() {
        console.log(name);
    }
    
    return inner;
}

let fn = outer();
fn();  // '外部'（不是'全局'）
// 函数 inner 的作用域在定义时就确定了
```

### 作用域链

```javascript
// 作用域链：从内向外查找变量
let level1 = '第一层';

function outer() {
    let level2 = '第二层';
    
    function inner() {
        let level3 = '第三层';
        console.log(level3);  // 第三层
        console.log(level2);  // 第二层
        console.log(level1);  // 第一层
    }
    
    inner();
}
```

## 6.6 什么是闭包？

### 概念讲解

**闭包就是函数记住并访问其创建时的环境**。

```javascript
// 生活例子：
// 你离开家（外部函数执行完），
// 但你口袋里还带着家里的钥匙（闭包），
// 随时可以回家（访问外部变量）

function createCounter() {
    let count = 0;  // 外部函数的变量

    // 内部函数（闭包）
    return function() {
        count++;
        return count;
    };
}

const counter = createCounter();
console.log(counter());  // 1
console.log(counter());  // 2
console.log(counter());  // 3

// count 变量一直被保留着
```

### 闭包的原理

```javascript
function outer() {
    let x = 10;
    
    function inner() {
        console.log(x);  // 访问外部的 x
    }
    
    return inner;  // 返回内部函数
}

let fn = outer();  // outer 执行完毕
fn();  // 10 - 仍然可以访问 x

// 为什么可以访问？
// 因为 inner 函数在创建时"记住"了它的作用域
// 这个"记忆"就是闭包
```

## 6.7 闭包的实际应用

### 1. 数据私有化

```javascript
function createBankAccount(initialBalance) {
    let balance = initialBalance;  // 私有变量

    return {
        deposit: function(amount) {
            balance += amount;
        },
        withdraw: function(amount) {
            if (amount > balance) {
                throw new Error('余额不足');
            }
            balance -= amount;
        },
        getBalance: function() {
            return balance;
        }
    };
}

const account = createBankAccount(1000);
account.deposit(500);
console.log(account.getBalance());  // 1500

// 不能直接访问 balance，只能通过方法操作
// console.log(account.balance);  // undefined
```

### 2. 函数工厂

```javascript
// 创建乘法器工厂
function createMultiplier(factor) {
    return function(x) {
        return x * factor;
    };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15
```

### 3. 防抖函数

```javascript
// 防抖：延迟执行，期间被触发则重新计时
function debounce(fn, delay) {
    let timer = null;  // 闭包保存 timer
    
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

// 使用：搜索框输入
// const search = debounce((query) => {
//     fetchResults(query);
// }, 300);
```

### 4. 节流函数

```javascript
// 节流：固定时间内只执行一次
function throttle(fn, limit) {
    let inThrottle = false;
    
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

// 使用：滚动事件
// window.addEventListener('scroll', throttle(() => {
//     handleScroll();
// }, 100));
```

### 5. 迭代器

```javascript
function createIterator(arr) {
    let index = 0;
    
    return {
        next: function() {
            if (index < arr.length) {
                return { value: arr[index++], done: false };
            }
            return { value: undefined, done: true };
        }
    };
}

const iter = createIterator([1, 2, 3]);
console.log(iter.next());  // { value: 1, done: false }
console.log(iter.next());  // { value: 2, done: false }
console.log(iter.next());  // { value: 3, done: false }
console.log(iter.next());  // { value: undefined, done: true }
```

## 6.8 闭包的注意事项

### 内存泄漏

```javascript
// ❌ 闭包可能导致内存泄漏
function createLargeClosure() {
    let largeData = new Array(1000000).fill('data');
    
    return function() {
        console.log('small');
        // 即使不使用 largeData，它也不会被回收
    };
}

// ✅ 优化：只保留需要的数据
function createOptimizedClosure() {
    let largeData = new Array(1000000).fill('data');
    let needed = largeData[0];  // 只保留需要的
    
    return function() {
        console.log(needed);
    };
}
```

### 循环中的闭包

```javascript
// ❌ 经典问题
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// 输出：3, 3, 3

// ✅ 解决方案 1：用 let
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// 输出：0, 1, 2

// ✅ 解决方案 2：IIFE 创建新作用域
for (var i = 0; i < 3; i++) {
    ((j) => {
        setTimeout(() => console.log(j), 100);
    })(i);
}
// 输出：0, 1, 2
```

## 6.9 实践练习

### 练习 1：作用域理解

```javascript
// 1. 预测输出
let x = 1;
function outer() {
    let x = 2;
    function inner() {
        let x = 3;
        console.log(x);  // 3
    }
    inner();

outer();

// 2. 预测输出
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log('var:', i), 0);
}
for (let j = 0; j < 3; j++) {
    setTimeout(() => console.log('let:', j), 0);
}
// var: 3, 3, 3
// let: 0, 1, 2
```

### 练习 2：闭包应用

```javascript
// 1. 实现计数器
function createCounter() {
    let count = 0;
    return {
        increment: () => ++count,
        decrement: () => --count,
        getCount: () => count
    };
}

const counter = createCounter();
console.log(counter.increment());  // 1
console.log(counter.increment());  // 2
console.log(counter.getCount());   // 2

// 2. 实现 once 函数（只执行一次）
function once(fn) {
    let called = false;
    let result;
    
    return function(...args) {
        if (!called) {
            called = true;
            result = fn.apply(this, args);
        }
        return result;
    };
}

const init = once(() => {
    console.log('初始化');
    return 'done';
});
init();  // '初始化'
init();  // 无输出，返回 'done'
```

### 练习 3：模块模式

```javascript
// 使用闭包实现模块
const TodoModule = (function() {
    let todos = [];
    let nextId = 1;
    
    return {
        add: function(text) {
            todos.push({ id: nextId++, text, completed: false });
        },
        remove: function(id) {
            todos = todos.filter(t => t.id !== id);
        },
        list: function() {
            return [...todos];
        },
        count: function() {
            return todos.length;
        }
    };
})();

TodoModule.add('学习 JavaScript');
TodoModule.add('完成练习');
console.log(TodoModule.list());
console.log(TodoModule.count());  // 2
```

## 6.10 常见问答

### Q1: 闭包会导致内存泄漏吗？

**不一定**。只有当闭包持有大对象且不再需要时，才可能造成内存问题。

```javascript
// ❌ 可能的问题
function createClosure() {
    let largeData = new Array(1000000);
    return function() {};  // 持有 largeData
}

// ✅ 正确做法
function createClosure() {
    let largeData = new Array(1000000);
    let needed = largeData[0];
    largeData = null;  // 释放
    return function() {
        console.log(needed);
    };
}
```

### Q2: 如何销毁闭包？

```javascript
let fn = createCounter();
fn();  // 使用

// 解除引用，让垃圾回收
fn = null;
```

### Q3: 闭包的性能如何？

闭包比普通函数稍慢，但现代 JS 引擎优化得很好，日常使用无需担心。
