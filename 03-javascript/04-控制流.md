## 4.1 控制流概述

### 什么是控制流？

**控制流**决定了代码的执行顺序。

```
顺序执行 → 按顺序一行行执行
    ↓
条件分支 → 根据条件选择不同路径
    ↓
循环     → 重复执行某段代码
```

### 三种基本结构

```javascript
// 1. 顺序结构 - 从上到下依次执行
let a = 1;
let b = 2;
let c = a + b;

// 2. 分支结构 - 根据条件选择执行
if (a > 0) {
    console.log('正数');
}

// 3. 循环结构 - 重复执行
for (let i = 0; i < 5; i++) {
    console.log(i);
}
```

## 4.2 if-else 条件语句

### 基本语法

```javascript
// 单一条件
let age = 20;

if (age >= 18) {
    console.log('成年人');
}

// if-else
if (age >= 18) {
    console.log('成年人');
} else {
    console.log('未成年人');
}

// if-else if-else
let score = 85;

if (score >= 90) {
    console.log('优秀');
} else if (score >= 80) {
    console.log('良好');
} else if (score >= 60) {
    console.log('及格');
} else {
    console.log('不及格');
}
```

### 实际应用场景

```javascript
// 1. 用户登录状态
let isLoggedIn = true;

if (isLoggedIn) {
    showDashboard();
} else {
    showLoginPage();
}

// 2. 权限检查
let role = 'admin';

if (role === 'admin') {
    showAdminPanel();
} else if (role === 'user') {
    showUserPanel();
} else {
    showGuestPanel();
}

// 3. 表单验证
let email = 'test@example.com';

if (!email) {
    showError('邮箱不能为空');
} else if (!email.includes('@')) {
    showError('邮箱格式不正确');
} else {
    submitForm();
}
```

### 条件简化

```javascript
// 冗余写法
if (isLoggedIn === true) { }
if (name === undefined) { }

// 简洁写法
if (isLoggedIn) { }
if (!name) { }

// 冗余写法
if (score > 60) {
    return true;
} else {
    return false;
}

// 简洁写法
return score > 60;
```

## 4.3 三元运算符 - 简化 if-else

### 基本语法

```javascript
// 语法：条件 ? 真值 : 假值

// 传统 if-else
let result;
if (score >= 60) {
    result = '及格';
} else {
    result = '不及格';
}

// 三元运算符
let result = score >= 60 ? '及格' : '不及格';
```

### 嵌套三元

```javascript
let score = 85;

// 多个条件
let grade = score >= 90 ? 'A' 
              : score >= 80 ? 'B' 
              : score >= 60 ? 'C' 
              : 'D';

console.log(grade);  // 'B'
```

### 实际应用场景

```javascript
// 1. 条件渲染
let isLoggedIn = true;
let buttonText = isLoggedIn ? '退出登录' : '登录';

// 2. 条件样式
let isActive = false;
let className = isActive ? 'btn-active' : 'btn-inactive';

// 3. 默认值
let count = userCount > 0 ? userCount : 10;
```

## 4.4 switch 语句 - 多分支选择

### 基本语法

```javascript
// 生活例子：星期几
let day = 3;

switch (day) {
    case 1:
        console.log('周一');
        break;  // 重要：跳出 switch
    case 2:
        console.log('周二');
        break;
    case 3:
        console.log('周三');
        break;
    default:
        console.log('其他');
}
```

### 实际应用场景

```javascript
// 1. 订单状态处理
let orderStatus = 'paid';

switch (orderStatus) {
    case 'pending':
        showPendingUI();
        break;
    case 'paid':
        showPaidUI();
        break;
    case 'shipped':
        showShippedUI();
        break;
    case 'delivered':
        showDeliveredUI();
        break;
    default:
        showUnknownUI();
}

// 2. 按键处理
let key = 'Enter';

switch (key) {
    case 'Enter':
        submitForm();
        break;
    case 'Escape':
        closeModal();
        break;
    case 'ArrowUp':
        scrollUp();
        break;
    case 'ArrowDown':
        scrollDown();
        break;
}
```

### 多个 case 合并

```javascript
// 多个值执行相同逻辑
let month = 3;
let season;

switch (month) {
    case 3:
    case 4:
    case 5:
        season = '春';
        break;
    case 6:
    case 7:
    case 8:
        season = '夏';
        break;
    case 9:
    case 10:
    case 11:
        season = '秋';
        break;
    default:
        season = '冬';
}
```

### switch vs if-else

```javascript
// 适合用 switch：
// - 多个固定值的比较
// - 等于判断（===）

// 适合用 if-else：
// - 范围判断
if (score >= 90) { }

// - 复杂条件
if (isLoggedIn && hasPermission) { }

// - 类型检查
if (typeof value === 'string') { }
```

## 4.5 for 循环 - 最常用的循环

### 基本语法

```javascript
// 语法：for (初始值; 条件; 更新)
for (let i = 0; i < 5; i++) {
    console.log(i);  // 0, 1, 2, 3, 4
}

// 执行流程：
// 1. 执行初始值：let i = 0
// 2. 检查条件：i < 5 ?
// 3. 执行循环体
// 4. 执行更新：i++
// 5. 回到步骤 2
```

### 生活例子

```javascript
// 跑圈：for (let 圈数 = 0; 圈数 < 10; 圈数++)
for (let lap = 0; lap < 10; lap++) {
    console.log(`第${lap + 1}圈`);
}

// 爬楼梯：for (let 台阶 = 1; 台阶 <= 20; 台阶++)
for (let step = 1; step <= 20; step++) {
    console.log(`爬第${step}阶`);
}
```

### 实际应用场景

```javascript
// 1. 遍历数组
let fruits = ['苹果', '香蕉', '橙子'];
for (let i = 0; i < fruits.length; i++) {
    console.log(fruits[i]);
}

// 2. 累加求和
let numbers = [1, 2, 3, 4, 5];
let sum = 0;
for (let i = 0; i < numbers.length; i++) {
    sum += numbers[i];
}
console.log(sum);  // 15

// 3. 查找元素
let target = 3;
let found = false;
for (let i = 0; i < numbers.length; i++) {
    if (numbers[i] === target) {
        found = true;
        console.log(`找到 ${target}，位置：${i}`);
        break;
    }
}
```

### 倒序循环

```javascript
// 从后往前遍历
let arr = [1, 2, 3, 4, 5];
for (let i = arr.length - 1; i >= 0; i--) {
    console.log(arr[i]);  // 5, 4, 3, 2, 1
}

// 删除数组元素（倒序避免索引问题）
for (let i = arr.length - 1; i >= 0; i--) {
    if (arr[i] % 2 === 0) {
        arr.splice(i, 1);  // 删除偶数
    }
}
```

## 4.6 while 循环 - 条件满足就继续

### 基本语法

```javascript
// 语法：while (条件)
let i = 0;
while (i < 5) {
    console.log(i);  // 0, 1, 2, 3, 4
    i++;
}

// 执行流程：
// 1. 检查条件
// 2. 如果为真，执行循环体
// 3. 回到步骤 1
```

### 实际应用场景

```javascript
// 1. 游戏主循环
let gameOver = false;
while (!gameOver) {
    updateGame();
    render();
    if (checkGameOver()) {
        gameOver = true;
    }
}

// 2. 读取数据直到完成
let hasMoreData = true;
while (hasMoreData) {
    let data = fetchData();
    processData(data);
    hasMoreData = data.hasMore;
}

// 3. 用户输入验证
let userInput = '';
while (!isValidInput(userInput)) {
    userInput = prompt('请输入有效内容：');
}
```

## 4.7 do-while 循环 - 至少执行一次

### 基本语法

```javascript
// 语法：do { ... } while (条件)
let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

// 和 while 的区别：
// while：先判断，再执行（可能一次都不执行）
// do-while：先执行，再判断（至少执行一次）
```

### 实际应用场景

```javascript
// 1. 菜单选择
let choice;
do {
    choice = showMenu();
    handleChoice(choice);
} while (choice !== 'exit');

// 2. 密码输入
let password;
do {
    password = prompt('请输入密码：');
} while (password !== 'correct');
```

## 4.8 for...in 和 for...of

### for...in - 遍历对象的键

```javascript
// 遍历对象属性
let obj = { a: 1, b: 2, c: 3 };
for (let key in obj) {
    console.log(key, obj[key]);
    // 'a' 1
    // 'b' 2
    // 'c' 3
}

// 遍历数组索引（不推荐）
let arr = [1, 2, 3];
for (let index in arr) {
    console.log(index, arr[index]);
    // '0' 1
    // '1' 2
    // '2' 3
}
```

### for...of - 遍历数组的值（ES6）

```javascript
// 遍历数组
let arr = [1, 2, 3, 4, 5];
for (let value of arr) {
    console.log(value);
    // 1, 2, 3, 4, 5
}

// 遍历字符串
let str = 'hello';
for (let char of str) {
    console.log(char);
    // 'h', 'e', 'l', 'l', 'o'
}

// 遍历 Set
let set = new Set([1, 2, 3]);
for (let value of set) {
    console.log(value);
}

// 遍历 Map
let map = new Map([['a', 1], ['b', 2]]);
for (let [key, value] of map) {
    console.log(key, value);
}
```

### 对比总结

| 特性 | for...in | for...of |
|------|----------|----------|
| 遍历对象 | ✅ 是 | ❌ 否 |
| 遍历数组 | ⚠️ 得到索引 | ✅ 得到值 |
| 遍历字符串 | ⚠️ 得到索引 | ✅ 得到字符 |
| ES 版本 | ES3 | ES6 |

## 4.9 break 和 continue

### break - 跳出循环

```javascript
// 找到第一个符合条件的就停止
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break;  // 直接跳出循环
    }
    console.log(i);  // 0, 1, 2, 3, 4
}
```

### continue - 跳过本次

```javascript
// 跳过某些迭代
for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) {
        continue;  // 跳过偶数，继续下一次
    }
    console.log(i);  // 1, 3, 5, 7, 9
}
```

### 综合使用

```javascript
for (let i = 0; i < 10; i++) {
    if (i === 3) continue;  // 跳过 3
    if (i === 7) break;     // 在 7 处停止
    console.log(i);
}
// 输出：0, 1, 2, 4, 5, 6
```

## 4.10 循环嵌套

### 基本示例

```javascript
// 打印九九乘法表
for (let i = 1; i <= 9; i++) {
    let row = '';
    for (let j = 1; j <= i; j++) {
        row += `${j}×${i}=${i * j} `;
    }
    console.log(row);
}
```

### 二维数组遍历

```javascript
// 二维数组
let matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];

// 遍历
for (let i = 0; i < matrix.length; i++) {
    for (let j = 0; j < matrix[i].length; j++) {
        console.log(matrix[i][j]);
    }
}

// 或用 for...of
for (let row of matrix) {
    for (let cell of row) {
        console.log(cell);
    }
}
```

## 4.11 实践练习

### 练习 1：条件判断

```javascript
// 1. 判断闰年
function isLeapYear(year) {
    return (year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0);
}
console.log(isLeapYear(2024));  // true

// 2. 成绩等级
function getGrade(score) {
    if (score >= 90) return 'A';
    if (score >= 80) return 'B';
    if (score >= 60) return 'C';
    return 'D';
}
console.log(getGrade(85));  // 'B'
```

### 练习 2：循环应用

```javascript
// 1. 计算阶乘
function factorial(n) {
    let result = 1;
    for (let i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}
console.log(factorial(5));  // 120

// 2. 斐波那契数列
function fibonacci(n) {
    let a = 0, b = 1;
    for (let i = 0; i < n; i++) {
        console.log(a);
        [a, b] = [b, a + b];
    }
}
fibonacci(7);  // 0, 1, 1, 2, 3, 5, 8

// 3. 找出数组最大值
function findMax(arr) {
    let max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}
console.log(findMax([3, 7, 2, 9, 1]));  // 9
```

## 4.12 常见问答

### Q1: for...in 和 for...of 有什么区别？

```javascript
let arr = [1, 2, 3];

// for...in 得到索引（字符串）
for (let i in arr) {
    console.log(typeof i);  // 'string'
}

// for...of 得到值
for (let v of arr) {
    console.log(v);  // 1, 2, 3
}
```

### Q2: while 和 do-while 应该用哪个？

**看是否需要至少执行一次**

```javascript
// 可能一次都不执行 → while
while (hasNext()) {
    processNext();
}

// 至少要执行一次 → do-while
do {
    retry();
} while (shouldRetry());
```

### Q3: 如何跳出多层嵌套循环？

```javascript
// 使用标签
outer: for (let i = 0; i < 3; i++) {
    for (let j = 0; j < 3; j++) {
        if (i === 1 && j === 1) {
            break outer;  // 跳出外层循环
        }
    }
}

// 或用函数返回
function findElement() {
    for (let i = 0; i < 3; i++) {
        for (let j = 0; j < 3; j++) {
            if (condition) {
                return;  // 直接返回
            }
        }
    }
}
```
