## 7.1 什么是对象？

### 概念讲解

**对象是键值对的集合**，用于表示现实世界中的事物。

```
对象 = {
    属性 1: 值 1,
    属性 2: 值 2,
    方法 1: 函数
}

生活例子：
汽车对象 = {
    品牌：'特斯拉',
    型号：'Model 3',
    颜色：'白色',
    启动：function() { ... },
    加速：function() { ... }
}
```

## 7.2 创建对象的方式

### 1. 对象字面量（最常用）

```javascript
// 语法：{ key: value, ... }
let user = {
    name: '张三',
    age: 25,
    email: 'zhang@example.com',
    sayHi() {
        console.log(`Hi, I'm ${this.name}`);
    }
};

// 空对象
let empty = {};
```

### 2. 构造函数

```javascript
// 自定义构造函数
function Person(name, age) {
    this.name = name;
    this.age = age;
    this.sayHi = function() {
        console.log(`Hi, I'm ${this.name}`);
    };
}

// 使用 new 创建实例
let p1 = new Person('张三', 25);
let p2 = new Person('李四', 30);

// new 的执行过程：
// 1. 创建空对象 {}
// 2. 将 this 绑定到这个对象
// 3. 执行构造函数
// 4. 返回这个对象
```

### 3. Object.create

```javascript
// 基于已有对象创建新对象
let proto = {
    greet() {
        console.log('Hello');
    }
};

let obj = Object.create(proto);
obj.greet();  // 'Hello'

// 可以设置第二个参数（属性描述符）
let obj2 = Object.create(proto, {
    name: {
        value: '张三',
        writable: true,
        enumerable: true,
        configurable: true
    }
});
```

### 4. class 语法（ES6）

```javascript
class Animal {
    constructor(name) {
        this.name = name;
    }
    
    speak() {
        console.log(`${this.name} makes a sound`);
    }
}

class Dog extends Animal {
    speak() {
        console.log(`${this.name} barks`);
    }
}

let dog = new Dog('旺财');
dog.speak();  // '旺财 barks'
```

### 5. Object.assign 合并对象

```javascript
let defaults = { theme: 'light', lang: 'zh' };
let userPrefs = { theme: 'dark' };

let config = Object.assign({}, defaults, userPrefs);
console.log(config);  // { theme: 'dark', lang: 'zh' }
```

### 6. 展开运算符（ES6）

```javascript
let defaults = { theme: 'light', lang: 'zh' };
let userPrefs = { theme: 'dark' };

// 更简洁的合并方式
let config = { ...defaults, ...userPrefs };
console.log(config);  // { theme: 'dark', lang: 'zh' }

// 浅拷贝
let copy = { ...user };
```

## 7.3 访问属性

### 点表示法 vs 方括号表示法

```javascript
let user = { name: '张三', age: 25 };

// 点表示法 - 属性名固定
console.log(user.name);  // '张三'
user.age = 26;

// 方括号表示法 - 属性名可以是变量
let key = 'name';
console.log(user[key]);  // '张三'

let propName = 'age';
user[propName] = 26;

// 动态属性名
let dynamicKey = 'email';
user[dynamicKey] = 'zhang@example.com';
```

### 什么时候用方括号？

```javascript
// 1. 属性名包含特殊字符
let obj = {
    'first-name': '张三',
    '123abc': '值'
};
console.log(obj['first-name']);

// 2. 属性名是变量
let key = 'name';
console.log(obj[key]);

// 3. 属性名在运行时确定
function getProperty(obj, prop) {
    return obj[prop];
}
```

## 7.4 修改对象

### 添加/修改属性

```javascript
let user = { name: '张三' };

// 添加属性
user.age = 25;
user['email'] = 'zhang@example.com';

// 修改属性
user.name = '李四';
user['age'] = 26;
```

### 删除属性

```javascript
let user = { name: '张三', age: 25 };

// delete 操作符
delete user.age;
delete user['name'];

// 检查是否删除成功
console.log('name' in user);  // false
```

### 检查属性

```javascript
let user = { name: '张三', age: 25 };

// in 操作符 - 检查自身和原型链
'name' in user;  // true

// hasOwnProperty - 只检查自身
user.hasOwnProperty('name');  // true

// 检查方法
typeof user.sayHi === 'function';  // true
```

## 7.5 对象方法

### 方法的定义方式

```javascript
let obj = {
    // 1. 传统方法
    method1: function() {
        console.log('方法 1');
    },

    // 2. 简写（ES6）
    method2() {
        console.log('方法 2');
    },

    // 3. 箭头函数（注意 this）
    method3: () => {
        console.log('方法 3');
    }
};
```

### this 关键字

```javascript
// this 指向调用者
let person = {
    name: '张三',
    greet() {
        console.log(`Hi, I'm ${this.name}`);
    }
};

person.greet();  // 'Hi, I'm 张三'

// this 的绑定问题
let greet = person.greet;
greet();  // 'Hi, I'm undefined'（this 指向全局）

// 解决方案：bind
let boundGreet = greet.bind(person);
boundGreet();  // 'Hi, I'm 张三'
```

### 箭头函数中的 this

```javascript
// 箭头函数的 this 指向定义时的上下文
let obj = {
    name: '张三',
    regularFunc: function() {
        console.log(this.name);  // '张三'
    },
    arrowFunc: () => {
        console.log(this.name);  // undefined（this 指向全局）
    }
};

obj.regularFunc();  // '张三'
obj.arrowFunc();    // undefined
```

## 7.6 对象操作 API

### 获取属性列表

```javascript
let user = { name: '张三', age: 25, email: 'zhang@example.com' };

// Object.keys - 获取所有键
Object.keys(user);  // ['name', 'age', 'email']

// Object.values - 获取所有值
Object.values(user);  // ['张三', 25, 'zhang@example.com']

// Object.entries - 获取键值对
Object.entries(user);
// [['name', '张三'], ['age', 25], ['email', 'zhang@example.com']]
```

### 遍历对象

```javascript
let user = { name: '张三', age: 25 };

// for...in
for (let key in user) {
    console.log(key, user[key]);
}

// Object.keys + forEach
Object.keys(user).forEach(key => {
    console.log(key, user[key]);
});

// Object.entries + for...of
for (let [key, value] of Object.entries(user)) {
    console.log(key, value);
}
```

### 对象合并

```javascript
let defaults = { theme: 'light', lang: 'zh' };
let userPrefs = { theme: 'dark' };

// 方法 1：Object.assign
let config1 = Object.assign({}, defaults, userPrefs);

// 方法 2：展开运算符（推荐）
let config2 = { ...defaults, ...userPrefs };

console.log(config2);  // { theme: 'dark', lang: 'zh' }
```

### 对象拷贝

```javascript
let user = { name: '张三', address: { city: '北京' } };

// 浅拷贝
let shallowCopy = { ...user };
let shallowCopy2 = Object.assign({}, user);

shallowCopy.address.city = '上海';
console.log(user.address.city);  // '上海'（原对象也被修改）

// 深拷贝
// 方法 1：JSON（简单对象）
let deepCopy1 = JSON.parse(JSON.stringify(user));

// 方法 2：structuredClone（现代浏览器）
let deepCopy2 = structuredClone(user);

// 方法 3：lodash cloneDeep
// let deepCopy3 = _.cloneDeep(user);
```

## 7.7 属性描述符

### 获取描述符

```javascript
let obj = { name: '张三' };

let descriptor = Object.getOwnPropertyDescriptor(obj, 'name');
console.log(descriptor);
// {
//     value: '张三',
//     writable: true,
//     enumerable: true,
//     configurable: true
// }
```

### 设置描述符

```javascript
let obj = { name: '张三' };

Object.defineProperty(obj, 'name', {
    writable: false,      // 不可修改
    enumerable: false,    // 不可遍历
    configurable: false   // 不可删除
});

obj.name = '李四';  // 无效（严格模式报错）
delete obj.name;    // 无效（严格模式报错）
```

### 描述符选项

```javascript
Object.defineProperty(obj, 'prop', {
    value: '值',
    writable: false,      // 是否可写
    enumerable: false,    // 是否可枚举（for...in）
    configurable: false,  // 是否可配置（删除、修改描述符）
    get() {               // getter
        return this._value;
    },
    set(val) {            // setter
        this._value = val;
    }
});
```

## 7.8 原型和继承

### 原型基础

```javascript
// 每个对象都有 __proto__ 指向原型
let obj = { name: '张三' };
console.log(obj.__proto__ === Object.prototype);  // true

// 原型链查找
obj.toString();  // '[object Object]'
// 查找过程：obj → Object.prototype → null
```

### 构造函数原型

```javascript
function Person(name) {
    this.name = name;
}

// 在原型上添加方法
Person.prototype.sayHi = function() {
    console.log(`Hi, I'm ${this.name}`);
};

let p1 = new Person('张三');
let p2 = new Person('李四');

p1.sayHi();  // 'Hi, I'm 张三'
p2.sayHi();  // 'Hi, I'm 李四'
// 两个实例共享同一个 sayHi 方法
```

### 原型链继承

```javascript
function Animal(name) {
    this.name = name;
}

Animal.prototype.speak = function() {
    console.log(`${this.name} makes a sound`);
};

function Dog(name) {
    Animal.call(this, name);  // 调用父类构造函数
}

// 继承原型
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

// 添加子类方法
Dog.prototype.speak = function() {
    console.log(`${this.name} barks`);
};

let dog = new Dog('旺财');
dog.speak();  // '旺财 barks'
```

## 7.9 实践练习

### 练习 1：对象基础

```javascript
// 1. 创建用户对象
let user = {
    name: '张三',
    age: 25,
    email: 'zhang@example.com',
    introduce() {
        console.log(`我叫${this.name}，今年${this.age}岁`);
    }
};

user.introduce();  // '我叫张三，今年 25 岁'

// 2. 动态属性
function createPerson(name, age) {
    return {
        name,
        age,
        getBirthYear() {
            return new Date().getFullYear() - this.age;
        }
    };
}

let p = createPerson('李四', 30);
console.log(p.getBirthYear());  // 1996
```

### 练习 2：对象操作

```javascript
// 1. 对象合并
function mergeObjects(...objects) {
    return Object.assign({}, ...objects);
}

let result = mergeObjects({ a: 1 }, { b: 2 }, { c: 3 });
console.log(result);  // { a: 1, b: 2, c: 3 }

// 2. 对象过滤
function pick(obj, keys) {
    let result = {};
    for (let key of keys) {
        if (key in obj) {
            result[key] = obj[key];
        }
    }
    return result;
}

let user = { name: '张三', age: 25, email: 'zhang@example.com' };
console.log(pick(user, ['name', 'email']));
// { name: '张三', email: 'zhang@example.com' }
```

### 练习 3：实现类

```javascript
// 实现一个简单的栈
class Stack {
    constructor() {
        this.items = [];
    }

    push(item) {
        this.items.push(item);
    }

    pop() {
        return this.items.pop();
    }

    peek() {
        return this.items[this.items.length - 1];
    }

    isEmpty() {
        return this.items.length === 0;
    }

    size() {
        return this.items.length;
    }
}

let stack = new Stack();
stack.push(1);
stack.push(2);
stack.push(3);
console.log(stack.pop());    // 3
console.log(stack.peek());   // 2
console.log(stack.size());   // 2
```

## 7.10 常见问答

### Q1: 对象和数组有什么区别？

```javascript
// 对象：键值对，键是字符串
let obj = { name: '张三', age: 25 };

// 数组：有序列表，索引是数字
let arr = [1, 2, 3];

// 数组也是对象
typeof [];  // 'object'
```

### Q2: 如何判断一个值是对象？

```javascript
// typeof 不准确
typeof {};   // 'object'
typeof [];   // 'object'
typeof null; // 'object'（bug）

// 推荐方法
function isObject(value) {
    return value !== null && typeof value === 'object';
}

// 更准确
Object.prototype.toString.call({});  // '[object Object]'
```

### Q3: 浅拷贝和深拷贝有什么区别？

```javascript
let obj = { a: 1, b: { c: 2 } };

// 浅拷贝 - 只复制一层
let shallow = { ...obj };
shallow.b.c = 3;
console.log(obj.b.c);  // 3（原对象被修改）

// 深拷贝 - 递归复制所有层
let deep = JSON.parse(JSON.stringify(obj));
deep.b.c = 4;
console.log(obj.b.c);  // 3（原对象不受影响）
```
