## 3.2 算术运算符

### 基本算术运算

```javascript
let a = 10, b = 3;

// 加法
a + b;      // 13

// 减法
a - b;      // 7

// 乘法
a * b;      // 30

// 除法
a / b;      // 3.3333333333333335

// 取余（求余数）
a % b;      // 1

// 幂运算（ES2016）
a ** b;     // 1000（10 的 3 次方）
// 等同于 Math.pow(10, 3)
```

### 自增自减

```javascript
let x = 5;

// 后置递增：先用后加
x++;        // 返回 5，然后 x 变成 6
console.log(x);  // 6

// 前置递增：先加后用
++x;        // x 先变成 7，然后返回 7

// 后置递减：先用后减
x--;        // 返回 7，然后 x 变成 6

// 前置递减：先减后用
--x;        // x 先变成 5，然后返回 5
```

### 实际应用

```javascript
// 1. 判断奇偶数
let num = 7;
if (num % 2 === 0) {
    console.log('偶数');
} else {
    console.log('奇数');
}

// 2. 计算折扣
let price = 100;
let discount = 0.8;
let finalPrice = price * discount;  // 80

// 3. 幂运算的实际应用
let square = 5 ** 2;    // 25（平方）
let cube = 5 ** 3;      // 125（立方）
```

## 3.3 比较运算符

### 相等比较

```javascript
// 宽松相等（会类型转换）
5 == '5';       // true
0 == false;     // true
'' == false;    // true
null == undefined;  // true

// 严格相等（不会类型转换）- 推荐
5 === '5';      // false（类型不同）
0 === false;    // false（类型不同）
'' === false;   // false（类型不同）
null === undefined;  // false

// 不相等
5 != '5';       // false
5 !== '5';      // true
```

### 大小比较

```javascript
5 > 3;          // true
5 < 3;          // false
5 >= 5;         // true（大于或等于）
5 <= 4;         // false

// 字符串比较（按 Unicode 编码）
'a' < 'b';      // true
'abc' < 'abd';  // true
'A' < 'a';      // true（大写字母编码小）
```

### 为什么推荐 ===？

```javascript
// == 的问题：会进行隐式类型转换
'' == 0;        // true（空字符串转 0）
'0' == 0;       // true（字符串转数字）
[] == 0;        // true（数组转数字）
[] == '';       // true（数组转字符串）
[] == ![];      // true（？？？）

// === 更可靠
'' === 0;       // false（类型不同）
'0' === 0;      // false（类型不同）
[] === 0;       // false（类型不同）
```

## 3.4 逻辑运算符

### 与（&&）- 两个都真才真

```javascript
true && true;   // true
true && false;  // false
false && true;  // false
false && false; // false

// 实际应用
let isLoggedIn = true;
let isAdmin = true;
let canDelete = isLoggedIn && isAdmin;  // true
```

### 或（||）- 有一个真就真

```javascript
true || true;   // true
true || false;  // true
false || true;  // true
false || false; // false

// 实际应用
let role = 'user';
let canView = role === 'admin' || role === 'user';  // true
```

### 非（!）- 取反

```javascript
!true;          // false
!false;         // true
!!true;         // true（双重否定 = 肯定）

// 转换为布尔值
!!'hello';      // true
!!0;            // false
!!{};           // true
```

### 短路求值

```javascript
// && 的短路：第一个为假，直接返回第一个值
false && 'anything';  // false
0 && 'anything';      // 0

// || 的短路：第一个为真，直接返回第一个值
true || 'anything';   // true
'hello' || 'world';   // 'hello'

// 实际应用：设置默认值（老方法）
let name = userInput || '默认用户';
// 如果 userInput 有值，用 userInput
// 如果 userInput 是空，用 '默认用户'
```

## 3.5 赋值运算符

### 基本赋值

```javascript
let x = 10;
```

### 复合赋值

```javascript
let a = 10;

// 加法赋值
a += 5;       // 等同于 a = a + 5，结果 15

// 减法赋值
a -= 3;       // 等同于 a = a - 3，结果 12

// 乘法赋值
a *= 2;       // 等同于 a = a * 2，结果 20

// 除法赋值
a /= 4;       // 等同于 a = a / 4，结果 5

// 取余赋值
a %= 3;       // 等同于 a = a % 3，结果 2

// 幂赋值（ES2016）
a **= 3;      // 等同于 a = a ** 3，结果 8
```

## 3.6 三元运算符 - 简化 if-else

### 基本语法

```javascript
// 语法：条件 ? 真值 : 假值

// 传统 if-else
let result;
if (score >= 60) {
    result = '及格';
} else {
    result = '不及格';
}

// 三元运算符 - 简洁写法
let result = score >= 60 ? '及格' : '不及格';
```

### 嵌套三元

```javascript
// 多个条件判断
let score = 85;
let grade = score >= 90 ? 'A' 
              : score >= 80 ? 'B' 
              : score >= 60 ? 'C' 
              : 'D';

console.log(grade);  // 'B'
```

### 实际应用场景

```javascript
// 1. 条件渲染
let isLoggedIn = true;
let button = isLoggedIn ? '退出登录' : '登录';

// 2. 条件样式
let isActive = false;
let className = isActive ? 'active' : 'inactive';

// 3. 参数默认值
let count = userCount > 0 ? userCount : 10;
```

## 3.7 空值合并运算符（??）- ES2020

### 为什么需要 ??？

```javascript
// || 的问题：0、空字符串也会被当作默认值
let count = userCount || 10;
// 如果 userCount 是 0，会得到 10（可能不是你想要的）

// ?? 只有 null 或 undefined 才用默认值
let count2 = userCount ?? 10;
// 如果 userCount 是 0，会得到 0（正确）
```

### 对比演示

```javascript
// || 的行为
'' || 'default';      // 'default'
0 || 10;              // 10
false || true;        // true
null || 'default';    // 'default'
undefined || 10;      // 10

// ?? 的行为
'' ?? 'default';      // ''（空字符串保留）
0 ?? 10;              // 0（0 保留）
false ?? true;        // false
null ?? 'default';    // 'default'
undefined ?? 10;      // 10
```

### 实际应用场景

```javascript
// 1. 表单默认值
let username = userInput ?? '匿名用户';
// 只有 null/undefined 时才用默认值

// 2. API 响应默认值
let data = response.data ?? [];
// 如果 data 是空数组，保留空数组

// 3. 配置默认值
let timeout = config.timeout ?? 5000;
// 只有未设置时才用默认超时
```

## 3.8 可选链运算符（?.）- ES2020

### 安全访问嵌套属性

```javascript
// 传统写法 - 容易报错
let userName = user && user.profile && user.profile.name;

// 可选链 - 简洁安全
let userName = user?.profile?.name;
// 如果 user 或 profile 是 null/undefined，返回 undefined
```

### 实际应用场景

```javascript
// 1. 访问可能不存在的属性
let city = user?.address?.city;

// 2. 调用可能不存在的方法
let result = obj?.method?.();

// 3. 数组访问
let firstItem = items?.[0];

// 4. 结合空值合并
let name = user?.profile?.name ?? '匿名用户';
```

### 注意事项

```javascript
// 可选链不能用于赋值
// user?.name = '张三';  // ❌ 报错

// 正确写法
if (user) {
    user.name = '张三';
}
```

## 3.9 运算符优先级

### 优先级从高到低

```
1. 括号 () []
2. 一元运算符 ! ++ -- typeof
3. 算术运算符 * / % + -
4. 比较运算符 > < >= <=
5. 相等运算符 == === != !==
6. 逻辑运算符 && ||
7. 三元运算符 ?:
8. 赋值运算符 = += -=
```

### 实际例子

```javascript
// 加法的优先级高于比较
5 + 3 > 7;      // true（先算 5+3=8，再比较 8>7）

// && 的优先级高于 ||
true || false && false;  // true（先算&&，再算||）
// 等同于：true || (false && false)

// 使用括号明确优先级
(1 + 2) * 3;    // 9
1 + (2 * 3);    // 7
```

## 3.10 实践练习

### 练习 1：算术运算

```javascript
// 1. 计算圆的面积
let radius = 5;
let area = Math.PI * radius ** 2;
console.log(area);  // 78.53981633974483

// 2. 判断奇偶数
let num = 17;
let isEven = num % 2 === 0;
console.log(isEven);  // false

// 3. 计算折扣
let price = 200;
let discount = 0.75;
let finalPrice = price * discount;
console.log(finalPrice);  // 150
```

### 练习 2：比较和逻辑运算

```javascript
// 1. 判断年龄范围
let age = 25;
let isAdult = age >= 18 && age <= 60;
console.log(isAdult);  // true

// 2. 判断闰年
let year = 2024;
let isLeap = (year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0);
console.log(isLeap);  // true

// 3. 成绩评定
let score = 85;
let grade = score >= 90 ? 'A' 
              : score >= 80 ? 'B' 
              : score >= 60 ? 'C' 
              : 'D';
console.log(grade);  // 'B'
```

### 练习 3：可选链和空值合并

```javascript
// 1. 安全访问嵌套对象
let user = {
    profile: {
        name: '张三',
        address: {
            city: '北京'
        }
    }
};

let city = user?.profile?.address?.city ?? '未知城市';
console.log(city);  // '北京'

// 2. 处理可能缺失的数据
let config = {};
let timeout = config.timeout ?? 5000;
console.log(timeout);  // 5000
```

## 3.11 常见问答

### Q1: == 和 === 应该用哪个？

**推荐总是用 ===**，除非你明确需要类型转换。

```javascript
// 推荐
if (count === 0) { }
if (name === null) { }

// 特殊情况可以用 ==
if (value == null) { }  // 同时检查 null 和 undefined
```

### Q2: || 和 ?? 有什么区别？

```javascript
// || 会把 0、''、false 当作假值
0 || 10;        // 10
'' || 'default'; // 'default'

// ?? 只把 null/undefined 当作空值
0 ?? 10;        // 0
'' ?? 'default'; // ''
```

### Q3: 如何记住运算符优先级？

**建议：多用括号，不要依赖优先级**

```javascript
// 不推荐 - 依赖优先级
let result = a + b * c > d && e || f;

// 推荐 - 明确优先级
let result = ((a + (b * c)) > d && e) || f;
```
