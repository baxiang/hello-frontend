## L1 理解层：会用

### 函数声明

**语法结构图：**

```
三种声明方式：

函数声明（有提升）:
  function name(params) { return value; }

函数表达式（无提升）:
  const fn = function(params) { return value; };

箭头函数（ES6，无提升）:
  const fn = (params) => value;
  const fn = x => x * 2;        ← 单参数省略括号
  const fn = () => { };         ← 无参数需要括号
  const fn = x => ({ key: x }); ← 返回对象需要括号
```

**最简示例（1-3行）：**

```javascript
function add(a, b) { return a + b; }
const double = x => x * 2;
add(3, 5);  // 8
```

**详细示例：**

```javascript
// 函数声明 — 可在声明前调用
console.log(greet('张三'));  // ✅
function greet(name) { return `你好，${name}！`; }

// 函数表达式 — 不能提前调用
const greet2 = function(name) { return `你好，${name}！`; };

// 箭头函数 — 简洁
const add = (a, b) => a + b;
const double = x => x * 2;
const sayHi = () => console.log('Hi!');
```

### 参数

**语法结构图：**

```
参数类型：

普通参数:      function fn(a, b) { }
默认参数:      function fn(name = '访客') { }
剩余参数:      function fn(...args) { }  ← 收集为数组
解构参数:      function fn({ name, age }) { }
```

**最简示例：**

```javascript
function greet(name = '访客') { console.log(`你好，${name}`); }
function sum(...nums) { return nums.reduce((a, b) => a + b, 0); }
```

**详细示例：**

```javascript
// 默认参数
function createUser(name = '匿名用户', age = 0) {
    return { name, age };
}
createUser('张三', 25);  // { name: '张三', age: 25 }
createUser();            // { name: '匿名用户', age: 0 }

// 剩余参数
function average(...numbers) {
    if (numbers.length === 0) return 0;
    return numbers.reduce((sum, n) => sum + n, 0) / numbers.length;
}
average(1, 2, 3, 4, 5);  // 3

// 解构参数
function printUser({ name, age, city = '未知' }) {
    console.log(`${name}, ${age}岁，来自${city}`);
}
printUser({ name: '张三', age: 25, city: '北京' });
```

### 返回值

**最简示例：**

```javascript
function add(a, b) { return a + b; }
function sayHi() { console.log('Hi'); }  // 返回 undefined
```

**详细示例：**

```javascript
// 多个 return
function getSign(num) {
    if (num > 0) return '正数';
    if (num < 0) return '负数';
    return '零';
}

// 提前返回（守卫子句）
function divide(a, b) {
    if (b === 0) return '除数不能为零';
    return a / b;
}

// 箭头函数返回对象（需要括号）
const createUser = (name, age) => ({ name, age });
```

## L2 实践层：用好

### 三种声明方式对比

| 特性 | 函数声明 | 函数表达式 | 箭头函数 |
|------|:---:|:---:|:---:|
| 提升 | ✅ | ❌ | ❌ |
| this 绑定 | 动态 | 动态 | 静态（词法） |
| 可作构造函数 | ✅ | ✅ | ❌ |
| arguments 对象 | ✅ | ✅ | ❌ |
| 推荐场景 | 顶层函数 | 条件赋值 | 回调、简工具函数 |

### 反模式：不要这样做

```javascript
// ❌ 错误：对象方法用箭头函数（this 不对）
const obj = {
    name: '张三',
    greet: () => console.log(this.name)  // undefined
};

// ✅ 正确：用普通方法
const obj = {
    name: '张三',
    greet() { console.log(this.name); }  // '张三'
};
```

```javascript
// ❌ 错误：用 || 给可能为 0 的参数设默认值
function scale(value, factor) {
    factor = factor || 2;  // 如果 factor 为 0，会变成 2
}

// ✅ 正确：用默认参数
function scale(value, factor = 2) { }
```

```javascript
// ❌ 错误：回调地狱
loadFile('a.txt', (a) => {
    loadFile('b.txt', (b) => {
        loadFile('c.txt', (c) => { });
    });
});

// ✅ 正确：用 async/await
const a = await loadFile('a.txt');
const b = await loadFile('b.txt');
const c = await loadFile('c.txt');
```

### 适用场景

| 场景 | 推荐方案 | 原因 |
|------|---------|------|
| 顶层命名函数 | 函数声明 | 有提升，可读性好 |
| 回调函数 | 箭头函数 | 简洁 |
| 对象方法 | 简写方法 | `greet() { }` |
| 不依赖 this 的工具函数 | 箭头函数 | 无 this 混淆 |
| 构造函数 | 函数声明/class | 箭头函数不能 new |

## L3 专家层：深入

### 箭头函数的 this

```
this 绑定对比：

普通函数 — this 由调用方式决定：
  obj.fn()      → this = obj
  fn()          → this = 全局（严格模式 undefined）
  fn.call(ctx)  → this = ctx

箭头函数 — this 在定义时就确定：
  继承外层作用域的 this，永远不变
```

```javascript
const obj = {
    name: '张三',
    // 普通方法：this 指向 obj
    greet() { console.log(this.name); },       // '张三'
    // 箭头函数：this 继承外层（全局）
    arrowGreet: () => console.log(this.name),  // undefined
};
```

### 高阶函数与闭包

```javascript
// 接受函数作为参数
function forEach(arr, callback) {
    for (const item of arr) callback(item);
}

// 返回函数（函数工厂）
function createMultiplier(factor) {
    return x => x * factor;  // 闭包记住 factor
}
const double = createMultiplier(2);
const triple = createMultiplier(3);
double(5);  // 10
```

### 性能考量

| 操作 | 性能影响 | 说明 |
|------|---------|------|
| 函数调用 | 极低 | 现代引擎内联优化 |
| 箭头函数 vs 普通函数 | 无差异 | 编译后等价 |
| 闭包 | 低 | 持有外部引用，稍多内存 |
| arguments 对象 | 中等 | 阻止引擎优化，用剩余参数替代 |

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **函数** | 可复用的代码块，接受参数，返回值 | `function fn(x) { return x * 2; }` |
| **函数提升** | 函数声明被移到作用域顶部 | 可在声明前调用 |
| **默认参数** | 参数未传入时使用的默认值 | `function fn(x = 1) {}` |
| **剩余参数** | 将多余参数收集为数组 | `function fn(...args) {}` |
| **箭头函数** | ES6 简洁函数语法，词法 this | `const fn = x => x * 2` |
| **高阶函数** | 接受函数作为参数或返回函数的函数 | `map`, `filter`, `reduce` |
| **回调函数** | 作为参数传递，在适当时机被调用的函数 | `setTimeout(callback, 1000)` |
| **闭包** | 函数记住并访问其定义时的词法环境 | 函数工厂、数据私有化 |
| **守卫子句** | 在函数开头提前返回，减少嵌套 | `if (!valid) return;` |

## 实践练习

### 练习：函数综合应用

```javascript
// 练习 1：实现 once（只执行一次）
function once(fn) {
    let called = false, result;
    return function(...args) {
        if (!called) { called = true; result = fn.apply(this, args); }
        return result;
    };
}
const init = once(() => { console.log('初始化'); return 'done'; });
init();  // '初始化'
init();  // 无输出，返回 'done'

// 练习 2：实现防抖
function debounce(fn, delay) {
    let timer;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => fn.apply(this, args), delay);
    };
}

// 练习 3：数据处理管道
function pipe(...fns) {
    return x => fns.reduce((v, fn) => fn(v), x);
}
const process = pipe(
    arr => arr.filter(x => x > 0),
    arr => arr.map(x => x * 2),
    arr => arr.reduce((sum, x) => sum + x, 0)
);
console.log(process([-1, 2, 3, -4, 5]));  // 20
```

## 常见问题

### Q1：箭头函数和普通函数有什么区别？

**核心区别是 this 绑定：**

```javascript
// 普通函数：this 指向调用者
// 箭头函数：this 继承定义时的外层

// 此外：箭头函数不能 new、没有 arguments、不能用作生成器
```

### Q2：什么时候用箭头函数？

```javascript
// ✅ 适合：回调、不依赖 this 的工具函数
arr.map(x => x * 2);
const add = (a, b) => a + b;

// ❌ 不适合：对象方法、构造函数、事件处理器（需要 this）
const obj = { name: '张三', greet() { console.log(this.name); } };
```

### Q3：默认参数和剩余参数能一起用吗？

```javascript
function fn(a = 1, b = 2, ...rest) {
    console.log(a, b, rest);
}
fn();           // 1, 2, []
fn(10, 20, 30); // 10, 20, [30]
```
