# HTML5 表单详解

## 学习目标
- 掌握表单的基本结构和常用控件
- 理解 HTML5 新增的输入类型
- 学会表单验证和无障碍设计

---

## 2.1 表单基础

### 基本结构

```html
<form action="/submit" method="POST" enctype="multipart/form-data">
    <!-- 表单控件 -->
    <input type="text" name="username">

    <!-- 提交按钮 -->
    <button type="submit">提交</button>
</form>
```

### form 属性详解

```html
<!-- action: 提交地址 -->
<form action="/api/register">

<!-- method: 提交方法 -->
<form method="GET">      <!-- 参数拼接到 URL -->
<form method="POST">     <!-- 参数在请求体中 -->
<form method="PUT">      <!-- 需要_hidden input -->

<!-- enctype: 编码类型 -->
<form enctype="application/x-www-form-urlencoded">  <!-- 默认 -->
<form enctype="multipart/form-data">                 <!-- 文件上传 -->
<form enctype="text/plain">                          <!-- 纯文本 -->

<!-- target: 打开方式 -->
<form target="_blank">    <!-- 新窗口 -->
<form target="_self">     <!-- 当前窗口（默认） -->
<form target="_parent">   <!-- 父框架 -->
<form target="_top">      <!-- 顶级窗口 -->

<!-- novalidate: 禁用浏览器验证 -->
<form novalidate>

<!-- autocomplete: 自动补全 -->
<form autocomplete="on">
<form autocomplete="off">
```

## 2.2 输入类型（input types）

### 文本输入

```html
<!-- 单行文本 -->
<input type="text" name="username">

<!-- 密码（字符会隐藏） -->
<input type="password" name="password">

<!-- 邮箱（自动验证格式） -->
<input type="email" name="email">

<!-- 电话 -->
<input type="tel" name="phone">

<!-- URL -->
<input type="url" name="website">

<!-- 搜索（移动端可能显示搜索键盘） -->
<input type="search" name="keyword">
```

### 数值输入

```html
<!-- 数字（带上下箭头） -->
<input type="number" name="age" min="0" max="100" step="1">

<!-- 滑块 -->
<input type="range" name="volume" min="0" max="100" value="50">

<!-- 属性说明 -->
<input type="number"
       min="0"           <!-- 最小值 -->
       max="100"         <!-- 最大值 -->
       step="5"          <!-- 步进 -->
       value="25">       <!-- 默认值 -->
```

### 日期时间

```html
<!-- 日期选择器 -->
<input type="date" name="birthday">

<!-- 年月 -->
<input type="month" name="start-month">

<!-- 周 -->
<input type="week" name="week">

<!-- 时间 -->
<input type="time" name="meeting-time">

<!-- 日期时间（无时区） -->
<input type="datetime-local" name="event-time">

<!-- 设置范围 -->
<input type="date"
       min="2024-01-01"
       max="2024-12-31"
       value="2024-06-15">
```

### 选择类控件

```html
<!-- 复选框（多选） -->
<input type="checkbox" name="hobby" value="reading" id="hobby-reading">
<label for="hobby-reading">阅读</label>

<input type="checkbox" name="hobby" value="sports" id="hobby-sports">
<label for="hobby-sports">运动</label>

<input type="checkbox" name="hobby" value="music" id="hobby-music" checked>
<label for="hobby-music">音乐（默认选中）</label>

<!-- 单选按钮（单选） -->
<input type="radio" name="gender" value="male" id="male">
<label for="male">男</label>

<input type="radio" name="gender" value="female" id="female">
<label for="female">女</label>

<input type="radio" name="gender" value="other" id="other" checked>
<label for="other">其他</label>
```

### 其他类型

```html
<!-- 颜色选择器 -->
<input type="color" name="theme-color" value="#ff0000">

<!-- 文件上传 -->
<input type="file" name="avatar" accept="image/*">
<input type="file" name="document" accept=".pdf,.doc,.docx">
<input type="file" name="photo" accept="image/png,image/jpeg">

<!-- 多文件上传 -->
<input type="file" name="photos" multiple accept="image/*">

<!-- 隐藏字段 -->
<input type="hidden" name="csrf_token" value="abc123">
```

## 2.3 输入属性

### 验证属性

```html
<!-- 必填 -->
<input type="text" name="username" required>

<!-- 最小/最大长度 -->
<input type="text" name="username" minlength="3" maxlength="20">

<!-- 模式匹配（正则表达式） -->
<input type="text"
       name="phone"
       pattern="^1[3-9]\d{9}$"
       title="请输入 11 位手机号">

<!-- 多重验证 -->
<input type="email"
       name="email"
       required
       minlength="5"
       maxlength="100">
```

### 用户体验属性

```html
<!-- 占位符 -->
<input type="text" name="username" placeholder="请输入用户名">

<!-- 自动聚焦 -->
<input type="text" name="search" autofocus>

<!-- 只读 -->
<input type="text" name="username" value="admin" readonly>

<!-- 禁用（不提交值） -->
<input type="text" name="disabled" disabled>

<!-- 自动补全 -->
<input type="text" name="username" autocomplete="username">
<input type="text" name="email" autocomplete="email">
<input type="tel" name="phone" autocomplete="tel">

<!-- 常见 autocomplete 值 -->
<!-- username, new-password, current-password -->
<!-- name, email, tel, street-address, country -->
<!-- cc-number, cc-exp, cc-csc (信用卡) -->

<!-- 只读但提交值 -->
<input type="text" name="code" value="123" readonly>
```

### 关联属性

```html
<!-- 关联 label -->
<label for="email">邮箱：</label>
<input type="email" id="email" name="email">

<!-- 点击 label 触发 input -->
<label>
    <input type="checkbox" name="agree"> 同意条款
</label>

<!-- 描述文本 -->
<input type="password" name="pwd" aria-describedby="pwd-help">
<span id="pwd-help">密码至少 8 位，包含字母和数字</span>

<!-- 错误提示 -->
<input type="email" name="email" aria-invalid="true" aria-describedby="email-error">
<span id="email-error" role="alert">请输入有效的邮箱地址</span>
```

## 2.6 表单验证

### 浏览器内置验证

```html
<form>
    <!-- 必填验证 -->
    <input type="text" name="username" required>

    <!-- 类型验证 -->
    <input type="email" name="email" required>
    <input type="url" name="website">
    <input type="number" name="age">

    <!-- 长度验证 -->
    <input type="text" name="username" minlength="3" maxlength="20">

    <!-- 正则验证 -->
    <input type="text" name="phone" pattern="^1[3-9]\d{9}$">

    <!-- 范围验证 -->
    <input type="number" name="age" min="18" max="100">

    <button type="submit">提交</button>
</form>
```

### 自定义验证消息

```html
<input type="email"
       name="email"
       required
       oninvalid="this.setCustomValidity('请输入有效的邮箱地址')"
       oninput="this.setCustomValidity('')">

<input type="password"
       name="password"
       minlength="8"
       pattern="^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).+$"
       oninvalid="this.setCustomValidity('密码至少 8 位，包含大小写字母和数字')"
       oninput="this.setCustomValidity('')">
```

### JavaScript 验证

```html
<form id="myForm">
    <input type="email" name="email" required id="email">
    <input type="password" name="password" required id="password">
    <input type="password" name="confirm" required id="confirm">
    <button type="submit">注册</button>
</form>

<script>
const form = document.getElementById('myForm');

form.addEventListener('submit', function(e) {
    e.preventDefault();

    const email = document.getElementById('email');
    const password = document.getElementById('password');
    const confirm = document.getElementById('confirm');

    // 检查密码匹配
    if (password.value !== confirm.value) {
        confirm.setCustomValidity('两次密码输入不一致');
        confirm.reportValidity();
        return;
    } else {
        confirm.setCustomValidity('');
    }

    // 手动验证
    if (!form.checkValidity()) {
        form.reportValidity();
        return;
    }

    // 提交表单
    console.log('验证通过，提交表单');
    form.submit();
});

// 实时验证
password.addEventListener('input', function() {
    const strength = calculateStrength(this.value);
    console.log('密码强度:', strength);
});

function calculateStrength(pwd) {
    let score = 0;
    if (pwd.length >= 8) score++;
    if (/[a-z]/.test(pwd)) score++;
    if (/[A-Z]/.test(pwd)) score++;
    if (/\d/.test(pwd)) score++;
    if (/[^a-zA-Z0-9]/.test(pwd)) score++;
    return score;
}
</script>
```

```html
    } catch (error) {
        alert('注册失败：' + error.message);
    } finally {
        submitBtn.disabled = false;
        submitBtn.textContent = '立即注册';
    }
});
</script>
```

## 2.9 常见问答

### Q1: GET 和 POST 有什么区别？

| 特性 | GET | POST |
|------|-----|------|
| 数据位置 | URL 参数 | 请求体 |
| 数据可见 |  URL 中可见 | 不可见 |
| 长度限制 | 有（约 2000 字符） | 无 |
| 可缓存 | 是 | 否 |
| 可收藏 | 是 | 否 |
| 适用场景 | 查询、搜索 | 提交、创建 |

### Q2: 为什么要用 label？

- 提高可访问性（点击 label 也能聚焦 input）
- 屏幕阅读器会朗读 label 文本
- 更大的点击区域（移动端友好）

### Q3: 如何处理文件上传？

```html
<form enctype="multipart/form-data">
    <input type="file" name="avatar" accept="image/*">
</form>
```
后端需要使用相应的库来处理 multipart 数据。

### Q4: 如何禁止表单自动填充？

```html
<form autocomplete="off">
    <input type="text" name="username" autocomplete="off">
</form>
```
但注意浏览器可能忽略此设置。
