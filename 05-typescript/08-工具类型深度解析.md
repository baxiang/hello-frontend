## 8.1 内置工具类型概览

TypeScript 内置了丰富的工具类型，位于 `lib.es5.d.ts` 中。

```typescript
// 类型变换
Partial<T>        // 所有属性可选
Required<T>       // 所有属性必填
Readonly<T>       // 所有属性只读
Record<K, T>      // 键值对类型

// 属性选择
Pick<T, K>        // 选取指定属性
Omit<T, K>        // 排除指定属性

// 类型排除
Exclude<T, U>     // 排除可赋值给 U 的类型
Extract<T, U>     // 提取可赋值给 U 的类型
NonNullable<T>    // 排除 null 和 undefined

// 函数相关
Parameters<T>     // 获取函数参数类型
ReturnType<T>     // 获取函数返回类型
InstanceType<T>   // 获取构造函数实例类型
ThisType<T>       // 获取 this 类型

// ES2020+
Awaited<T>        // 解包 Promise
```

## 8.2 Partial / Required / Readonly

### Partial<T>

```typescript
// 源码实现
type MyPartial<T> = {
    [P in keyof T]?: T[P];
};

// 使用
interface Todo {
    id: number;
    title: string;
    completed: boolean;
}

type PartialTodo = Partial<Todo>;
// { id?: number; title?: string; completed?: boolean; }

// 实际应用：更新操作
function updateTodo(id: number, updates: Partial<Todo>) {
    // 只需要提供要更新的字段
}

updateTodo(1, { completed: true });
```

### Required<T>

```typescript
// 源码实现
type MyRequired<T> = {
    [P in keyof T]-?: T[P];
};

// 使用
type RequiredTodo = Required<PartialTodo>;
// { id: number; title: string; completed: boolean; }

// 实际应用：配置验证
interface Config {
    port?: number;
    host?: string;
    timeout?: number;
}

function createConfig(config: Config): Required<Config> {
    return {
        port: config.port ?? 3000,
        host: config.host ?? 'localhost',
        timeout: config.timeout ?? 5000
    };
}
```

### Readonly<T>

```typescript
// 源码实现
type MyReadonly<T> = {
    readonly [P in keyof T]: T[P];
};

// 使用
type ReadonlyTodo = Readonly<Todo>;
// { readonly id: number; readonly title: string; readonly completed: boolean; }

// 实际应用：不可变数据
const todo: ReadonlyTodo = {
    id: 1,
    title: '学习 TS',
    completed: false
};

// todo.completed = true;  // ❌ 错误
```

## 8.3 Pick / Omit

### Pick<T, K>

```typescript
// 源码实现
type MyPick<T, K extends keyof T> = {
    [P in K]: T[P];
};

// 使用
interface User {
    id: number;
    name: string;
    email: string;
    password: string;
}

type UserPublic = Pick<User, 'id' | 'name'>;
// { id: number; name: string; }

// 实际应用：API 响应
type UserResponse = Pick<User, 'id' | 'name' | 'email'>;
```

### Omit<T, K>

```typescript
// 源码实现
type MyOmit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

// 使用
type UserSafe = Omit<User, 'password'>;
// { id: number; name: string; email: string; }

type UserCreate = Omit<User, 'id'>;
// { name: string; email: string; password: string; }

// 实际应用：表单数据
type LoginForm = Omit<User, 'id' | 'email'>;
// { name: string; password: string; }
```

## 8.4 Record<K, T>

### 基础用法

```typescript
// 源码实现
type MyRecord<K extends keyof any, T> = {
    [P in K]: T;
};

// 使用
// 定义枚举映射
enum Status {
    Pending = 'pending',
    Success = 'success',
    Error = 'error'
}

const statusLabels: Record<Status, string> = {
    [Status.Pending]: '待处理',
    [Status.Success]: '成功',
    [Status.Error]: '错误'
};

// 定义配置对象
type RouteConfig = Record<string, { path: string; component: string }>;

const routes: RouteConfig = {
    home: { path: '/', component: 'HomePage' },
    about: { path: '/about', component: 'AboutPage' }
};
```

### 实际应用

```typescript
// 权限映射
type Role = 'admin' | 'user' | 'guest';
type Permissions = Record<Role, string[]>;

const permissions: Permissions = {
    admin: ['read', 'write', 'delete'],
    user: ['read'],
    guest: ['read']
};

// 翻译字典
type TranslationKey = 'hello' | 'goodbye' | 'thank_you';
type Translations = Record<TranslationKey, string>;

const translations: Translations = {
    hello: '你好',
    goodbye: '再见',
    thank_you: '谢谢'
};
```

## 8.5 Exclude / Extract / NonNullable

### Exclude<T, U>

```typescript
// 源码实现
type MyExclude<T, U> = T extends U ? never : T;

// 使用
type T0 = Exclude<'a' | 'b' | 'c', 'a'>;           // 'b' | 'c'
type T1 = Exclude<string | number | null, null>;   // string | number

// 实际应用
type EventTypes = 'click' | 'scroll' | 'mousemove';
type MouseEventTypes = Exclude<EventTypes, 'scroll'>;
// 'click' | 'mousemove'
```

### Extract<T, U>

```typescript
// 源码实现
type MyExtract<T, U> = T extends U ? T : never;

// 使用
type T0 = Extract<'a' | 'b' | 'c', 'a' | 'b'>;     // 'a' | 'b'
type T1 = Extract<string | number | null, number>; // number

// 实际应用
type Status = 'success' | 'error' | 'pending' | null;
type ValidStatus = Extract<Status, string>;
// 'success' | 'error' | 'pending'
```

### NonNullable<T>

```typescript
// 源码实现
type MyNonNullable<T> = T extends null | undefined ? never : T;

// 使用
type T0 = NonNullable<string | null | undefined>;  // string
type T1 = NonNullable<string | number | null>;     // string | number

// 实际应用
function getValue<T>(value: T | null | undefined): NonNullable<T> {
    if (value === null || value === undefined) {
        throw new Error('Value is null or undefined');
    }
    return value as NonNullable<T>;
}
```

## 8.6 Parameters / ReturnType

### Parameters<T>

```typescript
// 源码实现
type MyParameters<T extends (...args: any) => any> = 
    T extends (...args: infer P) => any ? P : never;

// 使用
type T0 = Parameters<(a: string, b: number) => void>;
// [string, number]

// 实际应用
function createUser(name: string, email: string) {
    return { id: 1, name, email };
}

type CreateUserData = Parameters<typeof createUser>;
// [string, string]
```

### ReturnType<T>

```typescript
// 源码实现
type MyReturnType<T extends (...args: any) => any> = 
    T extends (...args: any) => infer R ? R : any;

// 使用
type T0 = ReturnType<() => { id: number; name: string }>;
// { id: number; name: string; }

// 实际应用
type CreateUserResult = ReturnType<typeof createUser>;
// { id: number; name: string; email: string; }
```

## 8.7 自定义工具类型

### DeepPartial

```typescript
type DeepPartial<T> = {
    [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

// 测试
interface User {
    profile: {
        address: {
            city: string;
        };
    };
}

type UpdateUser = DeepPartial<User>;
// { profile?: { address?: { city?: string; }; }; }
```

### DeepReadonly

```typescript
type DeepReadonly<T> = {
    readonly [P in keyof T]: T[P] extends object ? DeepReadonly<T[P]> : T[P];
};
```

### Nullable

```typescript
type Nullable<T> = T | null;

type T0 = Nullable<string>;  // string | null
```

### ValueOf

```typescript
type ValueOf<T> = T[keyof T];

const colors = {
    red: '#ff0000',
    green: '#00ff00',
    blue: '#0000ff'
} as const;

type ColorValue = ValueOf<typeof colors>;
// '#ff0000' | '#00ff00' | '#0000ff'
```

### Mutable

```typescript
type Mutable<T> = {
    -readonly [P in keyof T]: T[P];
};

interface ReadonlyUser {
    readonly id: number;
    readonly name: string;
}

type User = Mutable<ReadonlyUser>;
// { id: number; name: string; }
```

## 8.8 实践练习：工具类型实现

```typescript
// 1. 实现 CapitalizeKeys
type CapitalizeKeys<T> = {
    [K in keyof T as K extends string ? Capitalize<K> : K]: T[K];
};

// 2. 实现 LowercaseKeys
type LowercaseKeys<T> = {
    [K in keyof T as K extends string ? Lowercase<K> : K]: T[K];
};

// 3. 实现 FilterByType
type FilterByType<T, U> = {
    [K in keyof T as T[K] extends U ? K : never]: T[K];
};

interface Mixed {
    name: string;
    age: number;
    email: string;
    active: boolean;
}

type StringFields = FilterByType<Mixed, string>;
// { name: string; email: string; }

// 4. 实现 FunctionParams
type FunctionParams<T> = T extends (...args: infer P) => any ? P : never;

// 5. 实现 AsyncReturnType
type AsyncReturnType<T> = T extends (...args: any) => Promise<infer U> ? U : never;
```

    id: number;
    name: string;
    email: string;
    password: string;
}

type UserUpdate = Partial<Omit<User, 'id' | 'password'>>;
// { name?: string; email?: string; }

type UserCreate = Required<Omit<User, 'id'>>;
// { name: string; email: string; password: string; }
