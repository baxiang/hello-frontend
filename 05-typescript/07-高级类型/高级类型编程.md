## L1 理解层：会用

### 递归类型

**最简示例（1-3行）：**

```typescript
type DeepReadonly<T> = T extends object ? { readonly [K in keyof T]: DeepReadonly<T[K]> } : T;
type DeepPartial<T> = T extends object ? { [K in keyof T]?: DeepPartial<T[K]> } : T;
```

**详细示例：**

```typescript
// 深层嵌套类型
interface NestedConfig {
    database: { host: string; port: number };
    cache: { enabled: boolean; ttl: number };
}

// DeepPartial — 所有层级都可选
type DeepPartialConfig = DeepPartial<NestedConfig>;
// { database?: { host?: string; port?: number }; cache?: { ... } }

// DeepRequired — 所有层级都必填
type DeepRequired<T> = T extends object
    ? { [K in keyof T]-?: DeepRequired<T[K]> }
    : T;
```

### 模板字面量类型

```typescript
// 事件名称生成
type EventName<T extends string> = `on${Capitalize<T>}`;
type ClickEvent = EventName<'click'>;  // 'onClick'

// CSS 类名
type BEM<Block extends string, Element extends string, Modifier extends string> =
    `${Block}__${Element}--${Modifier}`;

type ButtonPrimary = BEM<'btn', 'icon', 'primary'>;  // 'btn__icon--primary'

// API 路径
type APIVersion = 'v1' | 'v2';
type Endpoint = 'users' | 'posts' | 'comments';
type APIPath = `/api/${APIVersion}/${Endpoint}`;
// '/api/v1/users' | '/api/v1/posts' | ...
```

## L2 实践层：用好

### 字符串工具类型

| 类型 | 说明 | 示例 |
|------|------|------|
| `Uppercase<S>` | 转大写 | `Uppercase<'hello'>` → `'HELLO'` |
| `Lowercase<S>` | 转小写 | `Lowercase<'HELLO'>` → `'hello'` |
| `Capitalize<S>` | 首字母大写 | `Capitalize<'hello'>` → `'Hello'` |
| `Uncapitalize<S>` | 首字母小写 | `Uncapitalize<'Hello'>` → `'hello'` |

### 类型编程挑战

```typescript
// Trim — 去除首尾空格
type Trim<S extends string> = S extends ` ${infer R}` ? Trim<R>
    : S extends `${infer L} ` ? Trim<L>
    : S;

type Result = Trim<'  hello  '>;  // 'hello'

// CamelCase — 转驼峰
type CamelCase<S extends string> = S extends `${infer First}-${infer Rest}`
    ? `${First}${Capitalize<CamelCase<Rest>>}`
    : S;

type Result = CamelCase<'background-color'>;  // 'backgroundColor'

// TupleToUnion — 元组转联合
type TupleToUnion<T extends any[]> = T[number];
type Result = TupleToUnion<[1, 2, 3]>;  // 1 | 2 | 3
```

## L3 专家层：深入

### 类型编程技巧

```typescript
// 1. 用 infer 提取嵌套类型
type First<T extends any[]> = T extends [infer F, ...any[]] ? F : never;
type Last<T extends any[]> = T extends [...any[], infer L] ? L : never;

// 2. 用条件类型过滤
type Filter<T, V> = T extends V ? T : never;
type Strings = Filter<string | number | boolean, string>;  // string

// 3. 用递归处理元组
type Reverse<T extends any[]> = T extends [infer F, ...infer R]
    ? [...Reverse<R>, F]
    : T;

type Result = Reverse<[1, 2, 3]>;  // [3, 2, 1]
```

## 实践练习

```typescript
// 练习 1：实现 SnakeCase
type SnakeCase<S extends string> = S extends `${infer First}${infer Rest}`
    ? Rest extends Uncapitalize<Rest>
        ? `${Lowercase<First>}${SnakeCase<Rest>}`
        : `${Lowercase<First>}_${SnakeCase<Rest>}`
    : S;

type Result = SnakeCase<'camelCase'>;  // 'camel_case'

// 练习 2：实现 OptionalKeys
type OptionalKeys<T> = {
    [K in keyof T]: T extends Record<K, T[K]> ? never : K;
}[keyof T];

type User = { id: number; name?: string };
type Result = OptionalKeys<User>;  // 'name'

// 练习 3：实现 PromiseAll 类型
type PromiseAll<T extends Promise<any>[]> =
    Promise<{ [K in keyof T]: T[K] extends Promise<infer U> ? U : T[K] }>;
```
