## 9.1 装饰器基础

### 什么是装饰器

```
装饰器是一个函数，用于修改类、方法、属性或参数的行为

就像：
- 给礼物包装：不改变礼物本身，但增加了外观
- 给手机戴壳：不改变手机功能，但增加了保护
- 给代码加日志：不改变逻辑，但增加了调试能力
```

### 启用装饰器

```json
// tsconfig.json
{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}
```

### 基本示例

```typescript
// 装饰器是一个函数，用于修改类、方法、属性或参数的行为
function log(target: any, key: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;

    descriptor.value = function(...args: any[]) {
        console.log(`调用 ${key}，参数：`, args);
        const result = originalMethod.apply(this, args);
        console.log(`返回：`, result);
        return result;
    };

    return descriptor;
}

class Calculator {
    @log
    add(a: number, b: number): number {
        return a + b;
    }
}

const calc = new Calculator();
calc.add(1, 2);
// 输出:
// 调用 add，参数：[1, 2]
// 返回：3
```

## 9.2 类装饰器

### 基础用法

```typescript
// 类装饰器接收构造函数作为参数
function LogConstructor(target: Function) {
    console.log(`创建类：${target.name}`);
}

@LogConstructor
class Person {
    constructor(public name: string) {}
}
```

### 修改类

```typescript
// 添加属性
function AddProperties<T extends { new(...args: any[]): {} }>(constructor: T) {
    return class extends constructor {
        createdAt = new Date();
        version = '1.0.0';
    };
}

@AddProperties
class Product {
    constructor(public name: string, public price: number) {}
}

const product = new Product('手机', 5999);
console.log(product.createdAt);
console.log(product.version);
```

### 装饰器工厂

```typescript
function color(color: string) {
    return function (constructor: Function) {
        constructor.prototype.color = color;
    };
}

@color('red')
class Car {
    brand: string;
}
```

## 9.3 方法装饰器

### 基础用法

```typescript
function Log(
    target: any,
    key: string,
    descriptor: PropertyDescriptor
) {
    const original = descriptor.value;

    descriptor.value = function(...args: any[]) {
        console.log(`调用 ${key}`);
        return original.apply(this, args);
    };

    return descriptor;
}

class Math {
    @Log
    add(a: number, b: number): number {
        return a + b;
    }
}
```

### 只读装饰器

```typescript
function readonly(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    descriptor.writable = false;
}

class User {
    @readonly
    name: string = 'Tom';
}

const user = new User();
user.name = 'Jerry';  // ❌ 错误：只读
```

### 权限验证装饰器

```typescript
function RequireAuth(roles: string[]) {
    return function(
        target: any,
        key: string,
        descriptor: PropertyDescriptor
    ) {
        const original = descriptor.value;

        descriptor.value = function(user: any, ...args: any[]) {
            if (!user || !roles.includes(user.role)) {
                throw new Error('没有权限');
            }
            return original.apply(this, [user, ...args]);
        };

        return descriptor;
    };
}

class AdminController {
    @RequireAuth(['admin'])
    deleteUser(id: number) {
        console.log(`删除用户 ${id}`);
    }
}
```

### 缓存装饰器

```typescript
function Cache(ttl: number = 5000) {
    const cache = new Map<string, { value: any; expires: number }>();

    return function(
        target: any,
        key: string,
        descriptor: PropertyDescriptor
    ) {
        const original = descriptor.value;

        descriptor.value = async function(...args: any[]) {
            const cacheKey = `${key}:${JSON.stringify(args)}`;
            const cached = cache.get(cacheKey);

            if (cached && cached.expires > Date.now()) {
                return cached.value;
            }

            const result = await original.apply(this, args);
            cache.set(cacheKey, {
                value: result,
                expires: Date.now() + ttl
            });

            return result;
        };

        return descriptor;
    };
}

class ApiService {
    @Cache(10000)
    async getUser(id: number): Promise<any> {
        console.log('请求 API');
        return { id, name: `用户${id}` };
    }
}
```

### 防抖装饰器

```typescript
function debounce(ms: number) {
    return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        let timeout: number;
        const original = descriptor.value;

        descriptor.value = function (...args: any[]) {
            clearTimeout(timeout);
            timeout = setTimeout(() => {
                original.apply(this, args);
            }, ms);
        };
    };
}

class Api {
    @debounce(300)
    fetchData() {
        console.log('Fetching data...');
    }
}
```

### 方法执行时间装饰器

```typescript
function measure(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function (...args: any[]) {
        const start = performance.now();
        const result = original.apply(this, args);
        const end = performance.now();
        console.log(`${propertyKey} took ${end - start}ms`);
        return result;
    };
}

class DataProcessor {
    @measure
    process(data: number[]): number {
        return data.reduce((a, b) => a + b, 0);
    }
}
```

## 9.4 属性装饰器

### 基础用法

```typescript
function LogProperty(target: any, key: string) {
    console.log(`${key} 被装饰`);
}

class User {
    @LogProperty
    name: string;

    @LogProperty
    age: number;

    constructor(name: string, age: number) {
        this.name = name;
        this.age = age;
    }
}
```

### 可观察属性

```typescript
function Observable(target: any, key: string) {
    let value = target[key];
    const listeners = new Set<(newValue: any) => void>();

    const getter = function() {
        return value;
    };

    const setter = function(newValue: any) {
        const oldValue = value;
        value = newValue;
        listeners.forEach(listener => listener(newValue));
        console.log(`${key} 从 ${oldValue} 变更为 ${newValue}`);
    };

    Object.defineProperty(target, key, {
        get: getter,
        set: setter,
        enumerable: true,
        configurable: true
    });
}

class State {
    @Observable
    count = 0;

    @Observable
    name = '';
}

const state = new State();
state.count = 1;
state.name = 'TS';
```

### 格式化属性

```typescript
function format(formatFn: (value: any) => string) {
    return function (target: any, propertyKey: string) {
        let value: string;

        Object.defineProperty(target, propertyKey, {
            get: () => value,
            set: (v) => {
                value = formatFn(v);
            }
        });
    };
}

class User {
    @format(v => v.toUpperCase())
    name: string;
}

const user = new User();
user.name = 'tom';
console.log(user.name);  // TOM
```

## 9.6 装饰器在框架中的应用

### NestJS 示例

```typescript
import { Controller, Get, Post, Body, Param, UseGuards } from '@nestjs/common';

@Controller('users')
export class UsersController {
    @Get()
    findAll() {
        return this.userService.findAll();
    }

    @Get(':id')
    findOne(@Param('id') id: string) {
        return this.userService.findOne(id);
    }

    @Post()
    @UseGuards(AuthGuard)
    create(@Body() createUserDto: CreateUserDto) {
        return this.userService.create(createUserDto);
    }
}
```

### TypeORM 示例

```typescript
import { Entity, Column, PrimaryGeneratedColumn, ManyToOne } from 'typeorm';

@Entity('users')
export class User {
    @PrimaryGeneratedColumn()
    id: number;

    @Column()
    name: string;

    @Column({ unique: true })
    email: string;

    @Column({ select: false })
    password: string;

    @ManyToOne(() => Post, post => post.author)
    posts: Post[];
}
```
