## 5.1 为什么使用泛型

### 5.1.1 什么是泛型？

```
┌─────────────────────────────────────────────────────────────┐
│  泛型（Generics）= 类型参数化 = 把类型当作参数传递           │
│                                                             │
│  概念：                                                     │
│  - 函数参数：传递值                                          │
│  - 泛型参数：传递类型                                       │
│                                                             │
│  就像：                                                     │
│  - 模具：可以生产不同尺寸的产品                              │
│  - 模板：可以填写不同内容                                   │
│  - 公式：可以代入不同数值                                   │
└─────────────────────────────────────────────────────────────┘
```

### 5.1.2 没有泛型的问题

```typescript
// 不使用泛型 - 需要为每种类型定义函数
function firstNumber(arr: number[]): number | undefined {
    return arr[0];
}

function firstString(arr: string[]): string | undefined {
    return arr[0];
}

function firstBoolean(arr: boolean[]): boolean | undefined {
    return arr[0];
}

// 问题：
// 1. 代码重复
// 2. 返回类型是 any，丢失类型信息
```

### 5.1.3 使用泛型的优势

```typescript
// 使用泛型 - 一个函数搞定所有类型
function firstElement<T>(arr: T[]): T | undefined {
    return arr[0];
}

// 使用
const num = firstElement([1, 2, 3]);      // number
const str = firstElement(['a', 'b', 'c']); // string
const bool = firstElement([true, false]);   // boolean

// 优势：
// 1. 代码复用
// 2. 类型安全
// 3. 类型推断
```

## 5.2 泛型函数

### 5.2.1 基本语法

```typescript
// 定义泛型函数
function identity<T>(arg: T): T {
    return arg;
}

// 使用方式 1：显式指定类型
const result1 = identity<string>('hello');  // string

// 使用方式 2：类型推断（推荐）
const result2 = identity('hello');  // string
const result3 = identity(42);       // number
const result4 = identity({ a: 1 }); // { a: number }
```

### 5.2.2 多个泛型参数

```typescript
// 多个泛型参数
function pair<T, U>(first: T, second: U): [T, U] {
    return [first, second];
}

const p = pair('hello', 42);  // [string, number]
```

### 5.2.3 泛型约束

```
┌─────────────────────────────────────────────────────────────┐
│  泛型约束 = 限制泛型必须满足的条件                           │
│                                                             │
│  语法：                                                     │
│  - extends 关键字                                          │
│  - keyof 关键字                                            │
│                                                             │
│  就像：                                                     │
│  - 模具限制：只能生产特定形状的产品                         │
│  - 参数限制：只能传入特定类型的值                           │
└─────────────────────────────────────────────────────────────┘
```

```typescript
// 约束 1：继承接口
interface Lengthwise {
    length: number;
}

function logLength<T extends Lengthwise>(arg: T): void {
    console.log(arg.length);
}

logLength('hello');     // OK
logLength([1, 2, 3]); // OK
logLength({ length: 5 }); // OK
// logLength(123);    // ❌ 错误

// 约束 2：使用 keyof
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

const user = { name: '张三', age: 25 };
getProperty(user, 'name');  // string
getProperty(user, 'age');  // number
// getProperty(user, 'xxx'); // ❌ 错误
```

## 5.3 泛型接口

### 5.3.1 基本用法

```typescript
// 泛型接口
interface Container<T> {
    value: T;
    getValue(): T;
}

const stringContainer: Container<string> = {
    value: 'hello',
    getValue() {
        return this.value;
    }
};

const numberContainer: Container<number> = {
    value: 42,
    getValue() {
        return this.value;
    }
};
```

### 5.3.2 泛型接口作为函数类型

```typescript
// 泛型函数接口
interface Handler<T> {
    (input: T): void;
}

const handleString: Handler<string> = (input) => {
    console.log(input.toUpperCase());
};

handleString('hello');  // HELLO
```

## 5.4 泛型类

### 5.4.1 基本用法

```typescript
// 泛型类
class Box<T> {
    private content: T;

    constructor(content: T) {
        this.content = content;
    }

    getContent(): T {
        return this.content;
    }

    setContent(value: T): void {
        this.content = value;
    }
}

const stringBox = new Box<string>('hello');
const numberBox = new Box<number>(42);

stringBox.getContent();  // hello
numberBox.getContent();  // 42
```

## 5.5 常用泛型工具类型

### 5.5.1 Partial<T> - 全部可选

```typescript
interface User {
    id: number;
    name: string;
    email: string;
}

// 所有属性变为可选
type PartialUser = Partial<User>;

// 等同于：
// {
//     id?: number;
//     name?: string;
//     email?: string;
// }
```

### 5.5.2 Required<T> - 全部必填

```typescript
interface Config {
    host?: string;
    port?: number;
}

// 所有属性变为必填
type RequiredConfig = Required<Config>;
```

### 5.5.3 Readonly<T> - 全部只读

```typescript
interface User {
    name: string;
    age: number;
}

// 所有属性变为只读
type ReadonlyUser = Readonly<User>;
```

### 5.5.4 Pick<T, K> - 选择属性

```typescript
interface User {
    id: number;
    name: string;
    email: string;
    password: string;
}

// 只选择部分属性
type UserPreview = Pick<User, 'id' | 'name'>;

// 等同于：
// {
//     id: number;
//     name: string;
// }
```

### 5.5.5 Omit<T, K> - 排除属性

```typescript
interface User {
    id: number;
    name: string;
    email: string;
    password: string;
}

// 排除部分属性
type PublicUser = Omit<User, 'password'>;

// 等同于：
// {
//     id: number;
//     name: string;
//     email: string;
// }
```

## 5.6 实践练习

### 练习 1：泛型函数

```typescript
// 1.1 数组去重
function unique<T>(arr: T[]): T[] {
    return [...new Set(arr)];
}

console.log(unique([1, 2, 2, 3]));  // [1, 2, 3]
console.log(unique(['a', 'b', 'a'])); // ['a', 'b']

// 1.2 查找元素
function find<T>(arr: T[], predicate: (item: T) => boolean): T | undefined {
    return arr.find(predicate);
}

const numbers = [1, 2, 3, 4, 5];
console.log(find(numbers, x => x > 3));  // 4
```

### 练习 2：泛型接口

interface ApiResponse<T> {

const response: ApiResponse<User> = {

## 5.7 常见问答

### Q1: 泛型和 any 有什么区别？

```typescript
// any：关闭类型检查
function withAny(value: any): any {
    return value.toUpperCase();  // 不报错
}

// 泛型：保持类型安全
function withGeneric<T extends string>(value: T): T {
    return value.toUpperCase();  // 类型安全
}
```
