# TypeScript 泛型编程

> 掌握泛型的概念和应用，编写可复用的类型安全代码

## 本章学习目标

- 理解泛型的概念和使用场景

- 能够编写通用的泛型工具函数
- 学会在 React 和 API 中使用泛型

## 5.1 为什么使用泛型

### 没有泛型的问题

```typescript
// 不使用泛型 - 需要定义多个重载
function firstElement(arr: number[]): number | undefined;
function firstElement(arr: string[]): string | undefined;
function firstElement(arr: boolean[]): boolean | undefined;
function firstElement(arr: any[]): any | undefined {
    return arr[0];
}

// 问题：
// 1. 需要多个函数签名
// 2. 返回 any 失去类型安全
// 3. 无法覆盖所有类型
```

### 使用泛型的优势

```typescript
// 使用泛型 - 一个定义搞定
function firstElement<T>(arr: T[]): T | undefined {
    return arr[0];
}

// 使用
const num = firstElement([1, 2, 3]);      // num: number
const str = firstElement(['a', 'b', 'c']); // str: string
const bool = firstElement([true, false]);  // bool: boolean

// 优势：
// 1. 类型安全
// 2. 代码复用
// 3. 类型推断
```

### 泛型的核心思想

> 泛型就是**类型参数化**，将类型作为参数传递，在调用时确定具体类型。

```typescript
// 类比函数参数
function identity(arg: number): number {
    return arg;
}
// 只能处理 number

// 泛型版本
function identity<T>(arg: T): T {
    return arg;
}
// 可以处理任何类型
```

## 5.2 泛型函数

### 定义泛型函数

```typescript
// 定义泛型函数
function identity<T>(arg: T): T {
    return arg;
}

// 使用 - 方式 1：显式指定类型
const result1 = identity<string>('hello');  // string

// 使用 - 方式 2：类型推断（推荐）
const result2 = identity(42);  // number
const result3 = identity({ name: '张三' });  // { name: string }
```

### 多个泛型参数

```typescript
// 多个泛型参数
function merge<T, U>(obj1: T, obj2: U): T & U {
    return { ...obj1, ...obj2 };
}

const merged = merge({ a: 1 }, { b: 2 });
// merged: { a: number; b: number; }

// 使用示例
const user = merge(
    { id: 1, name: '张三' },
    { email: 'zhang@example.com', age: 25 }
);
// user: { id: number; name: string; email: string; age: number; }
```

### 泛型箭头函数

```typescript
// 泛型箭头函数
const identity = <T>(arg: T): T => arg;

// 多个参数
const merge = <T, U>(obj1: T, obj2: U): T & U => ({ ...obj1, ...obj2 });

// 带约束的泛型箭头函数
const getLength = <T extends { length: number }>(arg: T): number => arg.length;

// 使用
console.log(getLength('hello'));  // 5
console.log(getLength([1, 2, 3]));  // 3
```

### 泛型工具函数

```typescript
// 创建对象
function createObject<T>(key: string, value: T): Record<string, T> {
    return { [key]: value };
}

const obj = createObject('name', '张三');
// obj: Record<string, string>

// 数组操作
function unique<T>(arr: T[]): T[] {
    return [...new Set(arr)];
}

const nums = unique([1, 2, 2, 3, 3, 3]);  // [1, 2, 3]

// 类型转换
function parseJSON<T>(str: string): T | null {
    try {
        return JSON.parse(str) as T;
    } catch {
        return null;
    }
}

const data = parseJSON<{ id: number; name: string }>('{"id": 1, "name": "张三"}');
```

## 5.3 泛型接口

### 定义泛型接口

```typescript
// 泛型接口
interface Box<T> {
    value: T;
}

const stringBox: Box<string> = { value: 'hello' };
const numberBox: Box<number> = { value: 42 };

// 多个泛型参数
interface Pair<T, U> {
    first: T;
    second: U;
}

const pair: Pair<string, number> = {
    first: 'hello',
    second: 42
};
```

### 泛型接口定义函数

```typescript
// 泛型函数接口
interface SearchFunc {
    <T>(arr: T[], target: T): boolean;
}

// 实现
const find: SearchFunc = <T>(arr: T[], target: T): boolean => {
    return arr.includes(target);
};

// 使用
const found = find([1, 2, 3], 2);  // true
```

### 接口方法泛型

```typescript
interface Logger {
    log<T>(message: T): void;
    logMany<T>(...messages: T[]): void;
}

class ConsoleLogger implements Logger {
    log<T>(message: T) {
        console.log(message);
    }

    logMany<T>(...messages: T[]) {
        messages.forEach(m => console.log(m));
    }
}

// 使用
const logger = new ConsoleLogger();
logger.log('hello');
logger.log(42);
logger.logMany(1, 'two', true);
```

### 泛型接口实际应用

```typescript
// API 响应接口
interface APIResponse<T> {
    code: number;
    message: string;
    data: T;
}

// 用户列表
type UserListResponse = APIResponse<{
    users: Array<{ id: number; name: string }>;
    total: number;
}>;

// 分页接口
interface PaginatedResult<T> {
    data: T[];
    page: number;
    pageSize: number;
    total: number;
    totalPages: number;
}

// 使用
const result: PaginatedResult<{ id: number; title: string }> = {
    data: [{ id: 1, title: '标题' }],
    page: 1,
    pageSize: 10,
    total: 100,
    totalPages: 10
};
```

## 5.4 泛型类

### 定义泛型类

```typescript
// 泛型类
class Stack<T> {
    private items: T[] = [];

    push(item: T): void {
        this.items.push(item);
    }

    pop(): T | undefined {
        return this.items.pop();
    }

    peek(): T | undefined {
        return this.items[this.items.length - 1];
    }

    isEmpty(): boolean {
        return this.items.length === 0;
    }

    size(): number {
        return this.items.length;
    }
}

// 使用
const numberStack = new Stack<number>();
numberStack.push(1);
numberStack.push(2);
console.log(numberStack.pop());  // 2

const stringStack = new Stack<string>();
stringStack.push('hello');
stringStack.push('world');
console.log(stringStack.pop());  // 'world'
```

### 泛型类实现接口

```typescript
interface Collection<T> {
    add(item: T): void;
    remove(item: T): boolean;
    has(item: T): boolean;
    size(): number;
}

class SetCollection<T> implements Collection<T> {
    private items = new Set<T>();

    add(item: T): void {
        this.items.add(item);
    }

    remove(item: T): boolean {
        return this.items.delete(item);
    }

    has(item: T): boolean {
        return this.items.has(item);
    }

    size(): number {
        return this.items.size;
    }
}

// 使用
const numSet = new SetCollection<number>();
numSet.add(1);
numSet.add(2);
console.log(numSet.has(1));  // true
```

### 泛型单例模式

```typescript
class Singleton<T> {
    private static instance: Singleton<any>;
    private data: T;

    private constructor(data: T) {
        this.data = data;
    }

    static getInstance<T>(data: T): Singleton<T> {
        if (!Singleton.instance) {
            Singleton.instance = new Singleton(data);
        }
        return Singleton.instance;
    }

    getData(): T {
        return this.data;
    }
}

// 使用
const config = Singleton.getInstance({ port: 3000, host: 'localhost' });
console.log(config.getData());
```

## 5.5 泛型约束（extends）

### extends 关键字

```typescript
// 约束 T 必须有 length 属性
interface Lengthwise {
    length: number;
}

function logLength<T extends Lengthwise>(arg: T): T {
    console.log(arg.length);
    return arg;
}

// OK
logLength('hello');      // string 有 length
logLength([1, 2, 3]);    // array 有 length
logLength({ length: 10 });

// 错误
// logLength(42);        // number 没有 length
// logLength(undefined); // undefined 没有 length
```

### 对象属性约束

```typescript
// 约束 T 必须有 id 属性
interface Identifiable {
    id: number | string;
}

function getId<T extends Identifiable>(obj: T): T['id'] {
    return obj.id;
}

const user = { id: 1, name: '张三' };
const post = { id: 'abc', title: '标题' };

console.log(getId(user));  // 1
console.log(getId(post));  // 'abc'
```

### 多个约束

```typescript
// 多个约束
function merge<T extends object, U extends object>(
    obj1: T,
    obj2: U
): T & U {
    return { ...obj1, ...obj2 };
}

const result = merge({ a: 1 }, { b: 2 });
// result: { a: number; b: number; }

// 约束继承
interface HasName {
    name: string;
}

interface HasAge {
    age: number;
}

function printInfo<T extends HasName & HasAge>(person: T): void {
    console.log(`${person.name}, ${person.age}岁`);
}

printInfo({ name: '张三', age: 25 });
```

## 5.6 keyof 约束

### 基础用法

```typescript
// 约束 K 必须是 T 的键
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

const user = { id: 1, name: '张三', email: 'zhang@example.com' };

console.log(getProperty(user, 'name'));  // '张三'
console.log(getProperty(user, 'id'));    // 1

// 错误
// console.log(getProperty(user, 'age'));  // 'age' 不是 user 的键
```

### keyof 实际应用

```typescript
// 提取对象键
interface User {
    id: number;
    name: string;
    email: string;
}

type UserKeys = keyof User;  // 'id' | 'name' | 'email'

// 动态属性访问
function pick<T, K extends keyof T>(obj: T, keys: K[]): Pick<T, K> {
    const result = {} as Pick<T, K>;
    for (const key of keys) {
        result[key] = obj[key];
    }
    return result;
}

const user = { id: 1, name: '张三', email: 'zhang@example.com', age: 25 };
const picked = pick(user, ['name', 'email']);
// picked: { name: string; email: string; }
```

### 映射类型中的 keyof

```typescript
// 使用 keyof 遍历属性
type Readonly<T> = {
    readonly [P in keyof T]: T[P];
};

type Partial<T> = {
    [P in keyof T]?: T[P];
};

// 自定义映射类型
type Nullable<T> = {
    [P in keyof T]: T[P] | null;
};

interface User {
    id: number;
    name: string;
}

type NullableUser = Nullable<User>;
// { id: number | null; name: string | null; }
```

## 5.7 默认泛型参数

### 基础用法

```typescript
// 提供默认类型
interface Response<T = any> {
    code: number;
    data: T;
    message: string;
}

// 使用
const res1: Response = { code: 200, data: 'anything', message: 'OK' };
const res2: Response<string> = { code: 200, data: 'string', message: 'OK' };

// 泛型函数默认值
function createArray<T = string>(length: number, value: T): T[] {

```

## 5.8 泛型在 React 中的应用

### 泛型组件

```typescript
import React, { useState, useCallback } from 'react';

// 泛型函数组件
interface ListProps<T> {
    items: T[];
    renderItem: (item: T) => React.ReactNode;
    keyExtractor: (item: T) => string;
}

function List<T>({ items, renderItem, keyExtractor }: ListProps<T>) {
    return (
        <ul>
            {items.map(item => (
                <li key={keyExtractor(item)}>
                    {renderItem(item)}
                </li>
            ))}
        </ul>
    );
}

// 使用
interface User {
    id: number;
    name: string;
}

function UserList() {
    const users: User[] = [
        { id: 1, name: '张三' },
        { id: 2, name: '李四' }
    ];

    return (
        <List
            items={users}
            renderItem={(user) => <span>{user.name}</span>}
            keyExtractor={(user) => String(user.id)}
        />
    );
}
```

### 泛型 Hooks

```typescript
// 泛型 useState
function useLocalStorage<T>(
    key: string,
    initialValue: T
): [T, (value: T) => void] {
    const [storedValue, setStoredValue] = useState<T>(() => {
        const item = localStorage.getItem(key);
        return item ? JSON.parse(item) : initialValue;
    });

    const setValue = (value: T) => {
        setStoredValue(value);
        localStorage.setItem(key, JSON.stringify(value));
    };

    return [storedValue, setValue];
}

// 使用
function UserProfile() {
    const [user, setUser] = useLocalStorage<User>('user', {
        id: 1,
        name: '张三'
    });

    return <div>{user.name}</div>;
}

// 泛型 useReducer
interface State<T> {
    data: T | null;
    loading: boolean;
    error: Error | null;
}

type Action<T> =
    | { type: 'loading' }
    | { type: 'success'; payload: T }
    | { type: 'error'; payload: Error };

function useFetch<T>(url: string) {
    const [state, dispatch] = useReducer(
        (state: State<T>, action: Action<T>): State<T> => {
            switch (action.type) {
                case 'loading':
                    return { ...state, loading: true };
                case 'success':
                    return { data: action.payload, loading: false, error: null };
                case 'error':
                    return { ...state, loading: false, error: action.payload };
                default:
                    return state;
            }
        },
        { data: null, loading: false, error: null }
    );

    useEffect(() => {
        dispatch({ type: 'loading' });
        fetch(url)
            .then(res => res.json())
            .then(data => dispatch({ type: 'success', payload: data }))
            .catch(err => dispatch({ type: 'error', payload: err }));
    }, [url]);

    return state;
}
```

## 5.9 实践练习：通用缓存

```typescript
interface CacheOptions {
    maxAge?: number;      // 缓存过期时间（毫秒）
    maxSize?: number;     // 最大缓存数量
}

class Cache<T> {
    private cache = new Map<string, { value: T; timestamp: number }>();
    private options: Required<CacheOptions>;

    constructor(options: CacheOptions = {}) {
        this.options = {
            maxAge: options.maxAge ?? 60000,
            maxSize: options.maxSize ?? 100
        };
    }

    get(key: string): T | undefined {
        const item = this.cache.get(key);
        if (!item) return undefined;

        // 检查是否过期
        if (Date.now() - item.timestamp > this.options.maxAge) {
            this.cache.delete(key);
            return undefined;
        }

        return item.value;
    }

    set(key: string, value: T): void {
        // 检查大小限制
        if (this.cache.size >= this.options.maxSize) {
            const firstKey = this.cache.keys().next().value;
            if (firstKey) this.cache.delete(firstKey);
        }

        this.cache.set(key, {
            value,
            timestamp: Date.now()
        });
    }

    delete(key: string): boolean {
        return this.cache.delete(key);
    }

    clear(): void {
        this.cache.clear();
    }

    has(key: string): boolean {
        return this.get(key) !== undefined;
    }

    size(): number {
        return this.cache.size;
    }
}

// 使用
const userCache = new Cache<{ id: number; name: string }>({
    maxAge: 300000,  // 5 分钟
    maxSize: 50
});

userCache.set('user1', { id: 1, name: '张三' });
const user = userCache.get('user1');
```

## 5.10 实践练习：API 请求封装

```typescript
// 基础响应类型
interface APIResponse<T> {
    code: number;
    message: string;
    data: T;
}

// 请求配置
interface RequestConfig extends RequestInit {
    baseURL?: string;
    timeout?: number;
}

// 泛型 API 类
class APIClient {
    private baseURL: string;
    private defaultConfig: RequestConfig;

    constructor(baseURL: string, config: RequestConfig = {}) {
        this.baseURL = baseURL;
        this.defaultConfig = {
            timeout: 10000,
            headers: {
                'Content-Type': 'application/json'
            },
            ...config
        };
    }

    private async request<T>(
        endpoint: string,
        config: RequestConfig = {}
    ): Promise<T> {
        const url = `${this.baseURL}${endpoint}`;
        const mergedConfig = { ...this.defaultConfig, ...config };

        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), mergedConfig.timeout);

        try {
            const response = await fetch(url, {
                ...mergedConfig,
                signal: controller.signal
            });

            const result: APIResponse<T> = await response.json();

            if (result.code !== 200) {
                throw new Error(result.message);
            }

            return result.data;
        } finally {
            clearTimeout(timeoutId);
        }
    }

    get<T>(endpoint: string, params?: Record<string, any>): Promise<T> {
        const query = params ? '?' + new URLSearchParams(params).toString() : '';
        return this.request<T>(endpoint + query, { method: 'GET' });
    }

    post<T>(endpoint: string, data?: any): Promise<T> {
        return this.request<T>(endpoint, {
            method: 'POST',
            body: JSON.stringify(data)
        });
    }

    put<T>(endpoint: string, data?: any): Promise<T> {
        return this.request<T>(endpoint, {
            method: 'PUT',
            body: JSON.stringify(data)
        });
    }

    delete<T>(endpoint: string): Promise<T> {
        return this.request<T>(endpoint, { method: 'DELETE' });
    }
}

// 使用
interface User {
    id: number;
    name: string;
    email: string;
}

interface Post {
    id: number;
    title: string;
    content: string;
}

const api = new APIClient('https://api.example.com');

// 类型安全的请求
const users = api.get<User[]>('/users');
const user = api.get<User>(`/users/1`);
const newPost = api.post<Post>('/posts', { title: '标题', content: '内容' });
```

## 5.11 常见问答

### Q1: 泛型约束和类型断言有什么区别？

**答：**
- 泛型约束：在编译时约束类型，更类型安全
- 类型断言：告诉编译器相信你的判断，有风险

```typescript
// 泛型约束（推荐）
function getProperty<T, K extends keyof T>(obj: T, key: K) {
    return obj[key];
}

// 类型断言（谨慎使用）
function getProperty<T>(obj: any, key: string) {
    return (obj as any)[key];
}
```

### Q2: 什么时候使用泛型默认值？

```typescript
// 提供默认类型
interface Response<T = any> {
    code: number;
    data: T;
}

// 使用
const res1: Response = { code: 200, data: 'anything' };
const res2: Response<string> = { code: 200, data: 'string' };
```

### Q3: 泛型如何递归？

```typescript
// 深度 Partial
type DeepPartial<T> = {
    [P in keyof T]?: T[P] extends object
        ? DeepPartial<T[P]>
        : T[P];
};

// 使用
interface User {
    profile: {
        address: {
            city: string;
        };
    };
}

type UpdateUser = DeepPartial<User>;
// { profile?: { address?: { city?: string; }; }; }
```
