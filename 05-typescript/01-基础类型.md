# TypeScript 基础类型

## 学习目标
- 掌握 TypeScript 的基本数据类型
- 理解 any、unknown、never 的区别
- 学会类型推断和类型断言
- 能够正确选择和使用类型

---

## 1.1 布尔类型

```typescript
// 声明布尔类型
let isDone: boolean = false;
let isActive: boolean = true;

// 布尔表达式
let isAdult: boolean = age >= 18;

// 函数返回布尔值
function isValid(value: string): boolean {
    return value.length > 0;
}
```

## 1.2 数字类型

```typescript
// 十进制
let decimal: number = 10;
let negative: number = -5;

// 二进制（以 0b 开头）
let binary: number = 0b1010;  // 10

// 八进制（以 0o 开头）
let octal: number = 0o744;  // 484

// 十六进制（以 0x 开头）
let hex: number = 0xf00d;  // 61453

// 特殊值
let infinityVal: number = Infinity;
let nanVal: number = NaN;

// 数字运算
let sum: number = 1 + 2;
let average: number = (1 + 2 + 3) / 3;
```

## 1.3 字符串类型

```typescript
// 声明字符串
let name: string = "张三";
let greeting: string = 'Hello';

// 模板字符串（反引号）
let message: string = `你好，我叫${name}，今年${25}岁`;

// 多行字符串
let multiLine: string = `
    第一行
    第二行
    第三行
`;

// 字符串方法
let str: string = "hello world";
str.toUpperCase();      // "HELLO WORLD"
str.toLowerCase();      // "hello world"
str.includes("world");  // true
str.split(" ");         // ["hello", "world"]
```

## 1.4 数组

```typescript
// 方式 1：类型 + 方括号
let numbers: number[] = [1, 2, 3];
let names: string[] = ['张三', '李四'];

// 方式 2：泛型语法
let list: Array<number> = [1, 2, 3];
let items: Array<string> = ['a', 'b'];

// 只读数组
const readonlyArr: readonly number[] = [1, 2, 3];
// readonlyArr.push(4);  // 错误

// 数组方法
const arr = [1, 2, 3, 4, 5];
arr.map(x => x * 2);        // [2, 4, 6, 8, 10]
arr.filter(x => x > 2);     // [3, 4, 5]
arr.reduce((sum, x) => sum + x, 0);  // 15
```

## 1.5 元组（Tuple）

```typescript
// 固定长度和类型的数组
let tuple: [string, number] = ["张三", 25];

// 访问元素
let name: string = tuple[0];
let age: number = tuple[1];

// 解构
const [name, age] = tuple;

// 可选元素
let optionalTuple: [string, number?]: ["张三"];

// 剩余元素
let restTuple: [string, ...number[]] = ["张三", 1, 2, 3];

// 命名元组（TS 4.0+）
let namedTuple: [name: string, age: number] = ["张三", 25];
```

## 1.6 枚举（Enum）

### 数字枚举

```typescript
// 默认从 0 开始
enum Direction {
    Up,      // 0
    Down,    // 1
    Left,    // 2
    Right    // 3
}

let dir: Direction = Direction.Up;

// 自定义起始值
enum Status {
    Pending = 10,
    Success = 11,
    Error = 12
}

// 反向映射
const directionName = Direction[0];  // "Up"
```

### 字符串枚举

```typescript
enum Status {
    Pending = "pending",
    Success = "success",
    Error = "error"
}

let status: Status = Status.Success;

// 字符串枚举没有反向映射
// const statusName = Status["pending"];  // 错误
```

### 常量枚举

```typescript
// 编译时会被移除，直接替换为值
const enum Color {
    Red = "red",
    Green = "green",
    Blue = "blue"
}

let c: Color = Color.Red;
// 编译后：let c = "red";
```

## 1.7 any - 任意类型

```typescript
// 可以接受任何类型
let notSure: any = 4;
notSure = "string";
notSure = false;
notSure = { name: '张三' };

// 可以调用任何方法
notSure.toFixed();
notSure.toUpperCase();

// 数组
let list: any[] = [1, "two", false, { name: '张三' }];

// 使用场景
// 1. 第三方库没有类型定义
// 2. 迁移旧代码
// 3. 确实无法确定类型时

// 不推荐：失去类型检查
// function log(value: any) {
//     console.log(value);
// }
```

## 1.8 unknown - 未知类型

```typescript
// 类型安全的 any
let value: unknown;

value = 42;      // OK
value = "hello"; // OK
value = true;    // OK

// 不能直接使用，需要类型检查或断言
// let str: string = value;  // 错误

// 方式 1：类型守卫
if (typeof value === "string") {
    let str: string = value;  // OK
}

// 方式 2：类型断言
let str: string = value as string;

// 方式 3：非空断言
let str: string = value!;
```

### unknown vs any

```typescript
// any 可以直接使用
let anyVal: any = "hello";
console.log(anyVal.length);  // OK

// unknown 需要检查
let unknownVal: unknown = "hello";
// console.log(unknownVal.length);  // 错误

if (typeof unknownVal === "string") {
    console.log(unknownVal.length);  // OK
}

// 函数参数
function safeLog(value: unknown) {
    console.log(String(value));
}

function unsafeLog(value: any) {
    console.log(value);
}
```

## 1.9 void - 空值

```typescript
// 函数没有返回值
function warnUser(): void {
    console.log("警告");
}

// 或者
function warnUser() {
    console.log("警告");
}

// void 类型的变量（很少用）
let unusable: void = undefined;
// let unusable: void = null;  // 需要 --strictNullChecks

// 箭头函数
const log = (): void => {
    console.log("日志");
};
```

## 1.10 null 和 undefined

```typescript
// 默认情况下是所有类型的子类型
let u: undefined = undefined;
let n: null = null;

// 开启 strictNullChecks 后
// let str: string = null;  // 错误
// let str: string = undefined;  // 错误

// 联合类型
let maybeString: string | null = "hello";
maybeString = null;  // OK
maybeString = undefined;  // 错误

// 可选链
const length = maybeString?.length;
```

## 1.11 never - 永不存在的值

```typescript
// 抛出错误的函数
function error(message: string): never {
    throw new Error(message);
}

// 无限循环
function infiniteLoop(): never {
    while (true) {}
}

//  unreachable code
function check(x: number | string) {
    if (typeof x === "number") {
        return x;
    } else if (typeof x === "string") {
        return x;
    }
    // x 的类型是 never（已经被穷尽）
}

// never 是任何类型的子类型
// 但没有类型是 never 的子类型（除了 never 自己）
```

## 1.13 类型推断

```typescript
// 自动推断类型
let x = 3;      // number
let y = "hello"; // string
let z = true;   // boolean

// 函数返回值推断
function add(a: number, b: number) {
    return a + b;  // 推断为 number
}

// 最佳共同类型
let arr = [1, null, "hello"];
// 推断为 (number | null | string)[]

// 上下文归类
window.onmousedown = function(event) {
    // event 推断为 MouseEvent
    console.log(event.button);
};

// 箭头函数
const handler = (event: MouseEvent) => {
    console.log(event.button);
};
```

## 1.14 类型断言

```typescript
// 语法 1：as 语法（推荐）
let someValue: unknown = "hello";
let strLength: number = (someValue as string).length;

// 语法 2：尖括号语法（与 JSX 冲突）
let strLength2: number = (<string>someValue).length;

// DOM 元素断言
const canvas = document.getElementById("myCanvas") as HTMLCanvasElement;
const input = document.querySelector("input") as HTMLInputElement;

// 非空断言
let value: string | null = null;
let length: number = value!.length;  // 告诉 TS value 不为 null

// 常量断言
const obj = { x: 1, y: 2 } as const;
// obj.x 的类型是字面量类型 1，而不是 number

const arr = [1, 2, 3] as const;
// arr 的类型是 readonly [1, 2, 3]
```

## 1.15 实践练习

### 练习 1：类型选择

```typescript
// 选择合适的类型

// 1. 用户 ID（数字或字符串）
type UserID = string | number;

// 2. 可选的电子邮件
type Email = string | null;

// 3. API 响应状态
type APIStatus = "pending" | "success" | "error";

// 4. 坐标点
type Point = [number, number];

// 5. 不可配置的对象
const config = {
    apiUrl: "https://api.example.com",
    timeout: 5000
} as const;
```

### 练习 2：类型转换

```typescript
// 实现类型转换函数

// 1. 字符串转数字
function toNumber(str: string): number {
    return Number(str);
}

// 2. 数字转字符串
function toString(num: number): string {
    return String(num);
}

// 3. 安全的 JSON 解析
function safeJsonParse<T>(str: string): T | null {
    try {
        return JSON.parse(str) as T;
    } catch {
        return null;
    }
}

// 4. 数组去重
function unique<T>(arr: T[]): T[] {
    return [...new Set(arr)];
}
```

### 练习 3：类型守卫

```typescript
// 实现类型检查函数

// 1. 检查是否为字符串
function isString(value: unknown): value is string {
    return typeof value === "string";
}

// 2. 检查是否为数字
function isNumber(value: unknown): value is number {
    return typeof value === "number" && !isNaN(value);
}

// 3. 检查是否为对象
function isObject(value: unknown): value is Record<string, unknown> {
    return typeof value === "object" && value !== null && !Array.isArray(value);
}

// 使用示例
function process(value: unknown) {
    if (isString(value)) {
        console.log(value.toUpperCase());
    } else if (isNumber(value)) {
        console.log(value.toFixed(2));
    } else if (isObject(value)) {
        console.log(Object.keys(value));
    }
}
```

## 1.16 常见问答

### Q1: any 和 unknown 有什么区别？

| 特性 | any | unknown |
|------|-----|---------|
| 类型安全 | ❌ | ✅ |
| 直接使用 | ✅ | ❌ |
| 类型检查 | 不需要 | 需要 |
| 推荐度 | 不推荐 | 推荐 |

### Q2: void 和 undefined 有什么区别？

**答：** void 表示函数没有返回值，undefined 是一个值。

```typescript
function fn(): void {
    // 没有 return
}

const val: undefined = undefined;
```

### Q3: never 和 void 有什么区别？

**答：** never 表示永远不会发生，void 表示没有返回值。

```typescript
function throwError(): never {
    throw new Error();  // 不会返回
}

function log(): void {
    console.log();  // 返回 undefined
}
```

### Q4: 什么时候使用类型断言？

**答：**
1. 第三方库没有类型定义
2. DOM 操作
3. JSON 解析
4. 测试代码
