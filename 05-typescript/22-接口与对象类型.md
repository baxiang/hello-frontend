## 6.1 接口基础

### 什么是接口

接口用于定义对象的结构：

```typescript
interface User {
    id: number;
    name: string;
    email: string;
}

// 使用接口
const user: User = {
    id: 1,
    name: 'Tom',
    email: 'tom@example.com'
};
```

### 接口 vs 类型别名

```typescript
// 接口
interface User {
    name: string;
    age: number;
}

// 类型别名
type User = {
    name: string;
    age: number;
};
```

两者功能相似，接口可以继承和实现。

## 6.2 可选属性

### 使用 ?

```typescript
interface User {
    id: number;
    name: string;
    email?: string;  // 可选属性
}

// 两种都合法
const user1: User = { id: 1, name: 'Tom' };
const user2: User = { id: 1, name: 'Tom', email: 'tom@example.com' };
```

### 实际应用

```typescript
interface SearchParams {
    keyword: string;
    page?: number;
    limit?: number;
}

function search(params: SearchParams) {
    const page = params.page || 1;
    const limit = params.limit || 10;
    // ...
}

search({ keyword: 'TypeScript' });
search({ keyword: 'TypeScript', page: 2 });
```

## 6.3 只读属性

### 使用 readonly

```typescript
interface User {
    readonly id: number;
    name: string;
}

const user: User = { id: 1, name: 'Tom' };
user.id = 2;      // ❌ 错误：id 是只读的
user.name = 'Jerry';  // ✅
```

### 实际应用

```typescript
interface Config {
    readonly apiUrl: string;
    readonly apiKey: string;
}

const config: Config = {
    apiUrl: 'https://api.example.com',
    apiKey: 'secret-key'
};

// 防止配置被修改
```

## 6.4 接口继承

### 单继承

```typescript
interface Person {
    name: string;
    age: number;
}

interface User extends Person {
    email: string;
}

const user: User = {
    name: 'Tom',
    age: 25,
    email: 'tom@example.com'
};
```

### 多继承

```typescript
interface A {
    a: string;
}

interface B {
    b: number;
}

interface C extends A, B {
    c: boolean;
}

const value: C = {
    a: 'hello',
    b: 123,
    c: true
};
```

## 6.5 实现接口

### 类实现接口

```typescript
interface Printable {
    print(): void;
}

class Document implements Printable {
    print() {
        console.log('Printing document...');
    }
}

class Image implements Printable {
    print() {
        console.log('Printing image...');
    }
}
```

### 多接口实现

```typescript
interface Flyable {
    fly(): void;
}

interface Swimmable {
    swim(): void;
}

class Duck implements Flyable, Swimmable {
    fly() {
        console.log('Duck flying');
    }
    swim() {
        console.log('Duck swimming');
    }
}
```

## 6.6 索引签名

### 什么是索引签名

允许动态添加属性：

```typescript
interface StringMap {
    [key: string]: string;
}

const map: StringMap = {
    name: 'Tom',
    city: 'Beijing'
};
```

### 实际应用

```typescript
interface UserScores {
    [subject: string]: number;
}

const scores: UserScores = {
    math: 90,
    english: 85,
    chinese: 92
};
```

### 混合索引签名

```typescript
interface User {
    name: string;
    [key: string]: string | number;
}

const user: User = {
    name: 'Tom',
    age: 25,
    city: 'Beijing'
};
```

## 6.7 函数类型接口

### 接口定义函数

```typescript
interface AddFn {
    (a: number, b: number): number;
}

const add: AddFn = (a, b) => a + b;
```

### 完整示例

```typescript
interface Callback {
    (error: Error | null, result?: string): void;
}

function fetchData(callback: Callback) {
    // 模拟异步操作
    callback(null, 'Data loaded');
}

fetchData((err, result) => {
    if (err) {
        console.error(err);
    } else {
        console.log(result);
    }
});
```

## 6.8 泛型接口

### 基本用法

```typescript
interface Response<T> {
    code: number;
    message: string;
    data: T;
}

interface User {
    id: number;
    name: string;
}

const userResponse: Response<User> = {
    code: 200,
    message: 'success',
    data: { id: 1, name: 'Tom' }
};
```

### 实际应用

```typescript
interface ApiResponse<T> {
    data: T;
    status: number;
    timestamp: Date;
}

interface Product {
    id: number;
    name: string;
    price: number;
}

const productResponse: ApiResponse<Product[]> = {
    data: [
        { id: 1, name: 'Apple', price: 5 },
        { id: 2, name: 'Banana', price: 3 }
    ],
    status: 200,
    timestamp: new Date()
};
```

## 6.8.5 综合示例：接口继承 + 可选属性

### 完整语法解析

```typescript
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}
```

这个语法包含三个知识点：

#### 1. `extends` - 接口继承

```
AuthRequest 继承 Request 的所有属性
```

```typescript
// 原始接口
interface Request {
    method: string;
    url: string;
    body?: any;
}

// 继承后
interface AuthRequest extends Request {
    // 自动拥有: method, url, body
    // 再加上自己的: user?
}
```

#### 2. `?` - 可选属性

```
user? = user 可以不存在
```

```typescript
interface AuthRequest {
    user?: { ... }
}

// 使用
const req1: AuthRequest = {
    method: 'GET',
    url: '/api',
    user: { id: 1, name: 'Tom', role: 'admin' }  // 有 user
};

const req2: AuthRequest = {
    method: 'GET',
    url: '/api'
    // 没有 user 也可以
};
```

#### 3. 嵌套对象类型

```
user?: { id: number; name: string; role: string }
```

```typescript
// 完整展开
interface AuthRequest {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 等价于
interface UserInfo {
    id: number;
    name: string;
    role: string;
}

interface AuthRequest {
    user?: UserInfo;
}
```

### 实际应用场景

```typescript
// 场景：Express 中间件添加用户信息

// 1. 扩展 Request 接口
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 2. 中间件设置用户
function authMiddleware(req: AuthRequest, res: Response, next: NextFunction) {
    const token = req.headers.authorization;
    
    if (token) {
        // 解析 token 后设置用户信息
        req.user = {
            id: 1,
            name: 'Tom',
            role: 'admin'
        };
    }
    
    next();
}

// 3. 路由中使用
app.get('/profile', authMiddleware, (req: AuthRequest, res: Response) => {
    // req.user 可能存在也可能不存在
    if (req.user) {
        res.json({ 
            message: `欢迎，${req.user.name}`,
            role: req.user.role 
        });
    } else {
        res.json({ message: '未登录' });
    }
});
```

### 逐步拆解

```typescript
// 原始写法
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 拆解步骤：

// 步骤 1：定义内部类型
interface UserInfo {
    id: number;
    name: string;
    role: string;
}

// 步骤 2：使用可选属性
interface AuthRequest {
    user?: UserInfo;
}

// 步骤 3：添加继承
interface AuthRequest extends Request {
    user?: UserInfo;
}
```

### 对比：可选对象 vs 必填对象

```typescript
// 必填对象
interface User1 {
    info: {
        name: string;
        age: number;
    };
}

// 可选对象
interface User2 {
    info?: {
        name: string;
        age: number;
    };
}

// 使用
const u1: User1 = { info: { name: 'Tom', age: 25 } };  // ✅ 必须有 info
const u2: User2 = { info: { name: 'Tom', age: 25 } };  // ✅ 可以有
const u3: User2 = { };                                  // ✅ 也可以没有
```

```typescript
interface User {
```

```typescript
interface Logger {
```

## 本章小结

| 特性 | 语法 | 说明 |
|------|------|------|
| 可选属性 | `prop?: type` | 属性可能不存在 |
| 只读属性 | `readonly prop: type` | 创建后不能修改 |
| 接口继承 | `extends` | 继承其他接口 |
| 实现接口 | `implements` | 类实现接口 |
| 索引签名 | `[key: string]: type` | 动态属性 |
| 泛型接口 | `interface<T>` | 支持多种类型 |
