## 6.1 接口基础

### 什么是接口

接口用于定义对象的结构：

```typescript
interface User {
    id: number;
    name: string;
    email: string;
}

// 使用接口
const user: User = {
    id: 1,
    name: 'Tom',
    email: 'tom@example.com'
};
```

### 接口 vs 类型别名

```typescript
// 接口
interface User {
    name: string;
    age: number;
}

// 类型别名
type User = {
    name: string;
    age: number;
};
```

两者功能相似，接口可以继承和实现。

## 6.2 可选属性

### 使用 ?

```typescript
interface User {
    id: number;
    name: string;
    email?: string;  // 可选属性
}

// 两种都合法

```

### 实际应用

```typescript
interface SearchParams {
    keyword: string;
    page?: number;
    limit?: number;
}

function search(params: SearchParams) {
    const page = params.page || 1;
    const limit = params.limit || 10;
    // ...
}

search({ keyword: 'TypeScript' });
search({ keyword: 'TypeScript', page: 2 });
```

## 6.6 索引签名（重点）

### 什么是索引签名？

```
索引签名 = 允许对象有任意数量的属性
         = 动态属性
```

```typescript
// 普通接口：只能有固定的属性
interface User1 {
    name: string;
    age: number;
}

// 索引签名：可以有任意属性
interface User2 {
    [key: string]: string;  // 任意数量的 string 属性
}
```

### 语法规范

```typescript
// 语法：[key: 类型]: 值类型
// key 可以是任意名字，只是占位符
// 常用 key 名称：key, prop, field, name, subject, index

// 这些写法完全等价：
interface A { [key: string]: number }
interface B { [prop: string]: number }
interface C { [anything: string]: number }
```

**注意：key 只是占位符，不是实际属性名！**

```typescript
// 正确理解：
interface StringMap {
    [key: string]: string;  // key 是"任意字符串"的意思
}

// 使用时：
const map: StringMap = {
    name: 'Tom',     // name 是实际属性名
    city: 'Beijing'  // city 也是实际属性名
};
```

### 索引签名的几种写法

```typescript
// 1. 字符串索引 - 键是字符串
interface StringDict {
    [key: string]: string;
}

// 2. 数字索引 - 键是数字
interface NumberArray {
    [index: number]: string;
}
const arr: NumberArray = ['a', 'b', 'c'];
console.log(arr[0]);  // 'a'

// 3. 符号索引（不常用）
interface SymbolDict {
    [key: symbol]: string;
}
```

### 实际应用场景

#### 场景1：动态属性

```typescript
// 不知道具体有哪些属性，但知道类型
interface UserScores {
    [subject: string]: number;
}

const scores: UserScores = {
    math: 90,
    english: 85,
    chinese: 92,
    physics: 88  // 可以随意添加
};
```

#### 场景2：API 返回的动态数据

```typescript
// 不知道具体字段，数据是动态的
interface ApiResponse {
    [key: string]: any;
}

const data: ApiResponse = {
    id: 1,
    name: 'Tom',
    extra: { a: 1 },
    list: [1, 2, 3]
};
```

#### 场景3：混合固定属性和动态属性

```typescript
// 固定属性 + 动态属性
interface User {
    name: string;           // 固定属性
    age: number;            // 固定属性
    [key: string]: any;    // 动态属性
}

const user: User = {
    name: 'Tom',
    age: 25,
    city: 'Beijing',   // 动态属性
    email: 'tom@xx.com'  // 动态属性
};
```

### 为什么用不同的 key 名称？

```typescript
// 这些写法完全等价：
interface A { [key: string]: number }
interface B { [subject: string]: number }
interface C { [prop: string]: number }

// 命名建议：
// - [key: string] - 最常用，表示"任意字符串键"
// - [index: number] - 用于数组
// - [subject: string] - 当描述"主题"时更语义化
// - [prop: string] - 当描述"属性"时更语义化
```

### 索引签名 vs 普通属性

```typescript
// 普通接口 - 固定属性
interface User1 {
    name: string;
    age: number;
}
// 必须有 name 和 age

// 索引签名 - 任意属性
interface User2 {
    [key: string]: string;
}
// 可以有任何 string 类型的属性

// 混合 - 固定 + 任意
interface User3 {
    name: string;
    [key: string]: string;
}
// 必须有 name，其他随意
```

### 注意事项

```typescript
// ❌ 错误：索引签名和其他属性类型冲突
interface Bad {
    [key: string]: string;
    name: number;  // ❌ 冲突：string vs number
}

// ✅ 正确：索引签名类型要兼容其他属性
interface Good {
    name: string;
    [key: string]: string | number;  // 包含 string
}
```

## 6.8.5 综合示例：接口继承 + 可选属性

### 完整语法解析

```typescript
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}
```

这个语法包含三个知识点：

#### 1. `extends` - 接口继承

```
AuthRequest 继承 Request 的所有属性
```

```typescript
// 原始接口
interface Request {
    method: string;
    url: string;
    body?: any;
}

// 继承后
interface AuthRequest extends Request {
    // 自动拥有: method, url, body
    // 再加上自己的: user?
}
```

#### 2. `?` - 可选属性

```
user? = user 可以不存在
```

```typescript
interface AuthRequest {
    user?: { ... }
}

// 使用
const req1: AuthRequest = {
    method: 'GET',
    url: '/api',
    user: { id: 1, name: 'Tom', role: 'admin' }  // 有 user
};

const req2: AuthRequest = {
    method: 'GET',
    url: '/api'
    // 没有 user 也可以
};
```

#### 3. 嵌套对象类型

```
user?: { id: number; name: string; role: string }
```

```typescript
// 完整展开
interface AuthRequest {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 等价于
interface UserInfo {
    id: number;
    name: string;
    role: string;
}

interface AuthRequest {
    user?: UserInfo;
}
```

### 实际应用场景

```typescript
// 场景：Express 中间件添加用户信息

// 1. 扩展 Request 接口
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 2. 中间件设置用户
function authMiddleware(req: AuthRequest, res: Response, next: NextFunction) {
    const token = req.headers.authorization;
    
    if (token) {
        // 解析 token 后设置用户信息
        req.user = {
            id: 1,
            name: 'Tom',
            role: 'admin'
        };
    }
    
    next();
}

// 3. 路由中使用
app.get('/profile', authMiddleware, (req: AuthRequest, res: Response) => {
    // req.user 可能存在也可能不存在
    if (req.user) {
        res.json({ 
            message: `欢迎，${req.user.name}`,
            role: req.user.role 
        });
    } else {
        res.json({ message: '未登录' });
    }
});
```

### 逐步拆解

```typescript
// 原始写法
interface AuthRequest extends Request {
    user?: {
        id: number;
        name: string;
        role: string;
    };
}

// 拆解步骤：

// 步骤 1：定义内部类型
interface UserInfo {
    id: number;
    name: string;
    role: string;
}

// 步骤 2：使用可选属性
interface AuthRequest {
    user?: UserInfo;
}

// 步骤 3：添加继承
interface AuthRequest extends Request {
    user?: UserInfo;
}
```

### 对比：可选对象 vs 必填对象

```typescript
// 必填对象
interface User1 {
    info: {
        name: string;
        age: number;
    };
}

// 可选对象
interface User2 {
    info?: {
        name: string;
        age: number;
    };
}

// 使用
const u1: User1 = { info: { name: 'Tom', age: 25 } };  // ✅ 必须有 info
const u2: User2 = { info: { name: 'Tom', age: 25 } };  // ✅ 可以有
const u3: User2 = { };                                  // ✅ 也可以没有
```

## 6.10 interface vs type vs class（深入理解）

### 6.10.1 什么是"编译后删除"？

#### TypeScript 的编译过程

```
TypeScript (.ts)  →  编译  →  JavaScript (.js)
```

**关键点：JavaScript 没有 interface 和 type！**

```typescript
// TypeScript 源代码
interface User {
    name: string;
    age: number;
}

class UserClass {
    name: string;
    age: number;
}

type UserType = {
    name: string;
    age: number;
};
```

编译后：

```javascript
// JavaScript（编译后）
class UserClass {
    constructor() {
        // 保留
    }
}

// interface 和 type 完全消失了！
// 没有任何 User、UserType 的痕迹
```

#### 为什么会删除？

```
TypeScript 的类型系统是为了"开发时"服务的

- 开发时：IDE 提供类型提示、报错
- 运行时：只需要 JavaScript 代码

interface 和 type 只在开发时有用
编译成 JavaScript 后就不需要了
所以会被删除
```

#### 实际例子

```typescript
// 源代码
interface User {
    name: string;
}

const user: User = { name: 'Tom' };

// 编译后（JavaScript）
const user = { name: 'Tom' };
// interface User 被删除了！
// : User 类型注解也被删除了！
```

### 6.10.2 三者的区别

| 特性 | interface | type | class |
|------|-----------|------|-------|
| **定义对象** | ✅ | ✅ | ✅ |
| **编译后保留** | ❌ 删除 | ❌ 删除 | ✅ 保留 |
| **可实例化** | ❌ | ❌ | ✅ |
| **可继承** | extends | extends & 交叉 | extends |
| **可实现** | implements | ❌ | - |
| **声明合并** | ✅ | ❌ | ❌ |

### 6.10.3 详细对比

#### 1. 定义对象结构

```typescript
// interface
interface Point1 {
    x: number;
    y: number;
}

// type
type Point2 = {
    x: number;
    y: number;
};

// class
class Point3 {
    x: number;
    y: number;
}

// 使用
const p1: Point1 = { x: 1, y: 2 };
const p2: Point2 = { x: 1, y: 2 };
const p3: Point3 = { x: 1, y: 2 };
```

#### 2. 编译后情况（重点！）

```typescript
// ====== TypeScript 源代码 ======
interface User { 
    name: string; 
    age: number;
}

type UserType = { 
    name: string; 
    age: number;
};

const user: User = { name: 'Tom', age: 25 };
const user2: UserType = { name: 'Tom', age: 25 };
```

编译后变成：

```javascript
// ====== JavaScript（编译后） ======
// interface User 完全消失！
// type UserType 完全消失！

const user = { name: 'Tom', age: 25 };
const user2 = { name: 'Tom', age: 25 };

// 完全一样！没有任何区别！
```

**结论：interface 和 type 在编译后完全消失，只剩下纯 JavaScript 对象！**

#### 完整对比

```typescript
// TypeScript
interface User {
    name: string;
    age: number;
    greet(): string;
}

class UserClass {
    name: string;
    age: number;
    
    greet() {
        return 'Hello';
    }
}
```

编译后：

```javascript
// JavaScript
class UserClass {
    greet() {
        return 'Hello';
    }
}

// interface User 消失得无影无踪！
// 只剩下 class 的代码
```

#### class 编译后保留

```typescript
// TypeScript
class User {
    constructor(public name: string) {}
    
    greet(): string {
        return `Hello, ${this.name}`;
    }
}
```

编译后：

```javascript
// JavaScript - 保留！
class User {
    constructor(name) {
        this.name = name;
    }
    
    greet() {
        return `Hello, ${this.name}`;
    }
}
```

#### 3. 实例化

```typescript
// interface - 不能实例化
interface User {
    name: string;
}
const user = new User();  // ❌ 错误

// type - 不能实例化
type User2 = { name: string; };
const user2 = new User2();  // ❌ 错误

// class - 可以实例化
class User3 {
    name: string;
}
const user3 = new User3();  // ✅ 可以
```

#### 4. 声明合并

```typescript
// interface - 可以声明合并
interface User {
    name: string;
}

interface User {
    age: number;
}

// 合并后：{ name: string; age: number }
const user: User = { name: 'Tom', age: 25 };

// type - 不能声明合并（会冲突）
type User = { name: string; };
type User = { age: number };  // ❌ 错误：重复声明
```

#### 5. 继承

```typescript
// interface 继承
interface Animal {
    name: string;
}
interface Dog extends Animal {
    breed: string;
}

// type 继承（用 & 交叉类型）
type Animal2 = { name: string };
type Dog2 = Animal2 & { breed: string };

// class 继承
class Animal3 {
    name: string;
}
class Dog3 extends Animal3 {
    breed: string;
}
```

### 6.10.4 工业级开发中的用法

#### 什么时候用 interface？

```typescript
// 1. 定义对象结构（最常用）
interface User {
    id: number;
    name: string;
    email: string;
}

// 2. 定义函数参数和返回值
interface ApiResponse<T> {
    code: number;
    data: T;
    message: string;
}

// 3. 需要声明合并时
interface Window {
    myPlugin: any;
}

// 4. 类实现接口
class User implements User {
    name: string;
    age: number;
}
```

#### 什么时候用 type？

```typescript
// 1. 联合类型
type Status = 'pending' | 'success' | 'error';

// 2. 交叉类型
type ExtendedUser = User & { age: number };

// 3. 元组类型
type Point = [number, number];

// 4. 映射类型
type Readonly<T> = {
    readonly [P in keyof T]: T[P];
};

// 5. 工具类型
type Partial<T> = { [P in keyof T]?: T[P] };
```

#### 什么时候用 class？

```typescript
// 1. 需要实例化时
class User {
    constructor(public name: string) {}
}
const user = new User('Tom');

// 2. 需要继承时
class Animal {
    eat() { }
}
class Dog extends Animal {
    bark() { }
}

// 3. 需要私有属性时
class User {
    private secret: string;
    public name: string;
}

// 4. 需要 Singleton（单例）模式时
class Singleton {
    private static instance: Singleton;
    private constructor() {}
    static getInstance() { ... }
}
```

### 6.10.5 最佳实践总结

```
┌─────────────────────────────────────────────────────────┐
│                    选择指南                              │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  定义对象结构 → interface                               │
│  - interface User { name: string }                     │
│                                                         │
│  定义联合/交叉/元组 → type                              │
│  - type Status = 'a' | 'b'                             │
│  - type Point = [number, number]                        │
│                                                         │
│  需要实例化 → class                                     │
│  - class User { }                                      │
│                                                         │
│  类需要实现的"契约" → interface                         │
│  - interface Serializable { toJSON() }                 │
│  - class User implements Serializable                   │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

```
interface = "对象的形状"（蓝图）
type = "类型的别名"（联合、交叉等）
class = "可以实例化的对象"（运行时存在）
```

## 本章小结

| 特性 | 语法 | 说明 |
|------|------|------|
| 可选属性 | `prop?: type` | 属性可能不存在 |
