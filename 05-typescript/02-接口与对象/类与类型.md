```
├──────────────────────────────────────────────────────┤
│                                                      │
│    class = 模具图纸                                  │
│    ─────────────                                     │
│    规定产品有什么属性、能做什么                      │
│    图纸本身不是产品，但可以生产产品                  │
│                                                      │
│    new = 用模具生产                                  │
│    ─────────────                                     │
│    new User('张三') → 生产一个用户实例               │
│                                                      │
│    public = 所有人都能用的功能                        │
│    ─────────────                                     │
│    产品的公开接口，谁都能操作                        │
│                                                      │
│    private = 内部结构                                │
│    ─────────────                                     │
│    产品的内部零件，外部看不到也碰不到                │
│    #password → 只有类内部能访问                      │
│                                                      │
│    protected = 家族内部功能                           │
│    ─────────────                                     │
│    自己和子类能用，外人不行                          │
│                                                      │
│    abstract = 半成品模具                             │
│    ─────────────                                     │
│    不能直接生产产品，必须先补充完整                  │
│    子类必须实现抽象方法                              │
│                                                      │
│    implements = 符合某个标准                          │
│    ─────────────                                     │
│    承诺这个类满足接口的所有要求                        │
│                                                      │
└──────────────────────────────────────────────────────┘
```

## L1 理解层：会用

### 基础类定义

**最简示例（1-3行）：**

```typescript
class User { constructor(public name: string) { } greet() { return `Hi, ${this.name}`; } }
const user = new User('张三'); user.greet();
```

**详细示例：**

```typescript
// 基础定义
class User {
    // 属性
    id: number;
    name: string;
    private password: string;

    // 构造函数
    constructor(id: number, name: string, password: string) {
        this.id = id;
        this.name = name;
        this.password = password;
    }

    // 方法
    greet(): string {
        return `你好，我是${this.name}`;
    }

    // getter
    get displayName(): string {
        return this.name.toUpperCase();
    }

    // setter
    set displayName(value: string) {
        this.name = value.trim();
    }
}

// 参数属性简写（推荐）
class User2 {
    constructor(
        public id: number,
        public name: string,
        #password: string  // ES2022 私有字段
    ) {}
}
```

### 访问修饰符

```typescript
class BankAccount {
    public readonly id: number;        // 公开，只读
    private balance: number;           // 私有
    protected createdAt: Date;         // 受保护

    constructor(id: number, initialBalance: number) {
        this.id = id;
        this.balance = initialBalance;
        this.createdAt = new Date();
    }

    public deposit(amount: number): void {
        if (amount > 0) this.balance += amount;
    }

    public getBalance(): number {
        return this.balance;
    }
}

class SavingsAccount extends BankAccount {
    constructor(id: number, balance: number, public interestRate: number) {
        super(id, balance);
        // this.balance;  // ❌ private，子类也不能访问
        // this.createdAt; // ✅ protected，子类可以访问
    }

    addInterest(): void {
        // 可以访问 protected 属性
        const interest = this.createdAt;
        // 但不能访问 private 属性
    }
}
```

### 抽象类

```typescript
// 抽象类 — 不能直接实例化
abstract class Shape {
    constructor(protected color: string) {}

    // 抽象方法 — 子类必须实现
    abstract area(): number;
    abstract perimeter(): number;

    // 具体方法 — 子类可以直接用
    describe(): string {
        return `这是一个${this.color}的形状，面积=${this.area()}`;
    }
}

class Circle extends Shape {
    constructor(color: string, private radius: number) {
        super(color);
    }

    area(): number { return Math.PI * this.radius ** 2; }
    perimeter(): number { return 2 * Math.PI * this.radius; }
}

class Rectangle extends Shape {
    constructor(color: string, private width: number, private height: number) {
        super(color);
    }

    area(): number { return this.width * this.height; }
    perimeter(): number { return 2 * (this.width + this.height); }
}

// const shape = new Shape('red');  // ❌ 抽象类不能实例化
const circle = new Circle('红色', 5);
console.log(circle.describe());  // "这是一个红色的形状，面积=78.5..."
```

### implements

```typescript
interface Printable {
    print(): void;
}

interface Serializable {
    serialize(): string;
}

class Document implements Printable, Serializable {
    constructor(public content: string) {}

    print(): void {
        console.log(`打印: ${this.content}`);
    }

    serialize(): string {
        return JSON.stringify({ content: this.content });
    }
}
```

## L2 实践层：用好

### 访问修饰符对比

| 修饰符 | 类内 | 子类 | 外部 |
|--------|:---:|:---:|:---:|
| `public` | ✅ | ✅ | ✅ |
| `protected` | ✅ | ✅ | ❌ |
| `private` | ✅ | ❌ | ❌ |
| `readonly` | ✅（只读） | ✅（只读） | ✅（只读） |

### 反模式

```typescript
// ❌ 错误：用 public 属性暴露内部状态
class User {
    public password: string;  // 不应该公开
}

// ✅ 正确：用 private 保护敏感数据
class User {
    #password: string;
    verifyPassword(input: string): boolean {
        return input === this.#password;
    }
}
```

### 类作为类型

```typescript
class User {
    constructor(public name: string) {}
}

// 类名既是构造函数也是类型
const user: User = new User('张三');

// 类类型 vs 实例类型
class Person {}
const person: Person = new Person();        // 实例类型
const ctor: typeof Person = Person;         // 构造函数类型
```

## L3 专家层：深入

### 类与接口的区别

```
接口 = 契约（只定义形状，不实现）
类 = 实现（既有形状，也有逻辑）

interface 可以 extends interface
class 可以 extends class
class 可以 implements interface
interface 不能 implements class
```

### 知识关联

```
类知识关联：

┌─────────────┐     ┌─────────────┐     ┌─────────────┐
│  类定义     │────→│  访问修饰符 │────→│  继承       │
│  属性/      │     │  public/    │     │  extends/   │
│  方法/      │     │  private/   │     │  abstract/  │
│  构造函数   │     │  protected  │     │  implements │
└─────────────┘     └─────────────┘     └─────────────┘
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **类** | 创建对象的蓝图 | `class User {}` |
| **构造函数** | 初始化实例的方法 | `constructor(name: string)` |
| **访问修饰符** | 控制属性/方法可见性 | public/private/protected |
| **只读属性** | 初始化后不可修改 | `readonly id: number` |
| **抽象类** | 不能被实例化的基类 | `abstract class Shape` |
| **参数属性** | 在构造函数参数中声明属性 | `constructor(public name: string)` |
| **implements** | 类实现接口 | `class A implements B` |

## 实践练习

```typescript
// 练习 1：实现 EventEmitter 类
class EventEmitter {
    private events = new Map<string, Array<(...args: any[]) => void>>();

    on(event: string, listener: (...args: any[]) => void): this {
        if (!this.events.has(event)) this.events.set(event, []);
        this.events.get(event)!.push(listener);
        return this;
    }

    emit(event: string, ...args: any[]): boolean {
        const listeners = this.events.get(event);
        if (!listeners) return false;
        listeners.forEach(fn => fn(...args));
        return true;
    }

    off(event: string, listener: (...args: any[]) => void): this {
        const listeners = this.events.get(event);
        if (listeners) {
            this.events.set(event, listeners.filter(fn => fn !== listener));
        }
        return this;
    }
}

// 练习 2：实现链表
class ListNode<T> {
    constructor(
        public value: T,
        public next: ListNode<T> | null = null
    ) {}
}

class LinkedList<T> {
    private head: ListNode<T> | null = null;

    append(value: T): void {
        const newNode = new ListNode(value);
        if (!this.head) { this.head = newNode; return; }
        let current = this.head;
        while (current.next) current = current.next;
        current.next = newNode;
    }

    toArray(): T[] {
        const result: T[] = [];
        let current = this.head;
        while (current) { result.push(current.value); current = current.next; }
        return result;
    }
}
```

## 常见问题

### Q1：private 和 # 私有字段有什么区别？

**`private` 是 TypeScript 编译时检查，`#` 是 JavaScript 运行时私有：**

```typescript
class User {
    private tsPrivate: string;  // TS 编译时私有，编译后仍是普通属性
    #jsPrivate: string;         // JS 运行时私有，编译后仍是私有
}
```

### Q2：abstract 和 interface 选哪个？

**需要共享实现用 abstract，只需要契约用 interface：**

```typescript
// 需要共享代码 → abstract
abstract class BaseRepository {
    protected db: Database;
    findById(id: number) { return this.db.query(id); }  // 共享实现
}

// 只需要契约 → interface
interface Repository<T> {
    findById(id: number): T | undefined;
    save(item: T): void;
}
```
