## 7.1 条件类型深入

### 分布式条件类型

```typescript
// 分布式条件类型
type ToArray<T> = T extends any ? T[] : never;

type T0 = ToArray<string>;                        // string[]
type T1 = ToArray<string | number>;               // string[] | number[]
type T2 = ToArray<string | number | boolean>;     // string[] | number[] | boolean[]

// 非分布式（使用元组包装）
type ToArrayNonDistributed<T> = [T] extends [any] ? T[] : never;

type T3 = ToArrayNonDistributed<string | number>;  // (string | number)[]
```

### 条件类型中的 infer

```typescript
// 推断返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type T0 = MyReturnType<() => string>;              // string
type T1 = MyReturnType<(a: number) => Promise<number>>;  // Promise<number>

// 推断数组元素
type MyElementType<T> = T extends (infer U)[] ? U : never;

type T2 = MyElementType<string[]>;        // string
type T3 = MyElementType<(string | number)[]>;  // string | number

// 推断 Promise 值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

type T4 = UnwrapPromise<Promise<string>>;           // string
type T5 = UnwrapPromise<Promise<Promise<number>>>;  // number

// 推断函数参数
type FirstParam<T> = T extends (arg: infer P, ...args: any[]) => any ? P : never;

type T6 = FirstParam<(name: string, age: number) => void>;  // string
```

## 7.2 递归类型

### 基础递归

```typescript
// 递归类型定义
type Json =
    | string
    | number
    | boolean
    | null
    | { [key: string]: Json }
    | Json[];

// 使用
const data: Json = {
    name: '张三',
    age: 25,
    hobbies: ['reading', 'coding'],
    address: {
        city: '北京',
        zipcode: '100000'
    }
};
```

### 深度 Partial

```typescript
// 深度可选
type DeepPartial<T> = {
    [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

interface User {
    id: number;
    profile: {
        name: string;
        address: {
            city: string;
            street: string;
        };
    };
}

type UpdateUserInput = DeepPartial<User>;
// {
//     id?: number;
//     profile?: {
//         name?: string;
//         address?: {
//             city?: string;
//             street?: string;
//         };
//     };
// }
```

### 深度只读

```typescript
// 深度只读
type DeepReadonly<T> = {
    readonly [P in keyof T]: T[P] extends object ? DeepReadonly<T[P]> : T[P];
};

interface Config {
    api: {
        url: string;
        timeout: number;
    };
}

type ReadonlyConfig = DeepReadonly<Config>;
// { readonly api: { readonly url: string; readonly timeout: number; } }
```

## 7.3 模板字面量类型进阶

### 字符串模式匹配

```typescript
// 判断是否以某后缀结尾
type EndsWith<T extends string, U extends string> = 
    T extends `${string}${U}` ? true : false;

type T0 = EndsWith<'hello.ts', '.ts'>;   // true
type T1 = EndsWith<'hello.js', '.ts'>;   // false

// 去除前缀
type RemovePrefix<T extends string, U extends string> = 
    T extends `${U}${infer Rest}` ? Rest : T;

type T2 = RemovePrefix<'@vue/cli', '@'>;  // 'vue/cli'

// 去除后缀
type RemoveSuffix<T extends string, U extends string> = 
    T extends `${infer Rest}${U}` ? Rest : T;

type T3 = RemoveSuffix<'hello.ts', '.ts'>;  // 'hello'
```

### 字符串转换

```typescript
// 首字母大写
type CapitalizeFirst<T extends string> = 
    T extends `${infer First}${infer Rest}` 
        ? `${Uppercase<First>}${Rest}` 
        : T;

type T0 = CapitalizeFirst<'hello'>;  // 'Hello'

// 转 camelCase
type CamelCase<T extends string> = 
    T extends `${infer First}_${infer Rest}` 
        ? `${First}${Capitalize<CamelCase<Rest>>}` 
        : T;

type T1 = CamelCase<'user_name'>;      // 'userName'
type T2 = CamelCase<'first_last_name'>; // 'firstNameName'
```

## 7.4 数组类型编程

### 数组操作类型

```typescript
// 获取数组第一个元素类型
type First<T extends any[]> = T extends [infer Head, ...any[]] ? Head : never;

type T0 = First<[1, 2, 3]>;  // 1

// 获取数组最后一个元素类型
type Last<T extends any[]> = T extends [...any[], infer Tail] ? Tail : never;

type T1 = Last<[1, 2, 3]>;  // 3

// 数组长度
type Length<T extends readonly any[]> = T['length'];

type T2 = Length<[1, 2, 3]>;  // 3

// 元组转联合
type TupleToUnion<T extends any[]> = T[number];

type T3 = TupleToUnion<[string, number, boolean]>;  // string | number | boolean
```

### 数组映射

```typescript
// 数组所有元素变为可选
type ArrayPartial<T extends any[]> = {
    [K in keyof T]?: T[K];
};

type T0 = ArrayPartial<[string, number]>;  // [string?, number?]

// 数组元素包装为 Promise
type PromiseArray<T extends any[]> = {
    [K in keyof T]: Promise<T[K]>;
};

type T1 = PromiseArray<[string, number]>;  // [Promise<string>, Promise<number>]
```

## 7.5 类型挑战练习

### 挑战 1：实现 Pick

```typescript
type MyPick<T, K extends keyof T> = {
    [P in K]: T[P];
};

// 测试
interface Todo {
    title: string;
    description: string;
    completed: boolean;
}

type TodoPreview = MyPick<Todo, 'title' | 'completed'>;
// { title: string; completed: boolean; }
```

### 挑战 2：实现 Omit

```typescript
type MyOmit<T, K extends keyof any> = {
    [P in Exclude<keyof T, K>]: T[P];
};

// 测试
type TodoInfo = MyOmit<Todo, 'completed' | 'description'>;
// { title: string; }
```

### 挑战 3：实现 Readonly

```typescript
type MyReadonly<T> = {
    readonly [P in keyof T]: T[P];
};

// 测试
type ReadonlyTodo = MyReadonly<Todo>;
// { readonly title: string; readonly description: string; readonly completed: boolean; }
```

### 挑战 4：实现 DeepReadonly

```typescript
type DeepReadonly<T> = {
    readonly [P in keyof T]: T[P] extends Function 
        ? T[P] 
        : T[P] extends object 
            ? DeepReadonly<T[P]> 
            : T[P];
};

// 测试
interface Config {
    api: {
        url: string;
        timeout: number;
    };
}

type ReadonlyConfig = DeepReadonly<Config>;
```

### 挑战 5：实现 Parameters

```typescript
type MyParameters<T extends (...args: any) => any> = 
    T extends (...args: infer P) => any ? P : never;

// 测试
type T0 = MyParameters<(a: string, b: number) => void>;
// [string, number]
```

## 7.6 常见问答

### Q1: 如何调试复杂类型？

```typescript
// 使用条件类型打印类型
type Debug<T> = T extends infer U ? { [K in keyof U]: U[K] } : never;

type Result = Debug<{ a: string; b: number }>;
```

### Q2: 如何避免类型递归过深？

```typescript
// 设置递归深度限制
type LimitDepth<T, D extends number = 5> = 
    D extends 0 
        ? T 
        : T extends object 
            ? DeepPartial<T> 
            : T;
```
