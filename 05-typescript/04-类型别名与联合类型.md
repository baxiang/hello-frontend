## 4.1 type 基础用法

### 基本类型别名

```typescript
// 基础类型别名
type Name = string;
type Age = number;
type IsDone = boolean;

// 使用
const name: Name = '张三';
const age: Age = 25;
const isDone: IsDone = true;

// 类型别名可以有注释
/** 用户 ID 类型 */
type UserID = string | number;
```

### 对象类型别名

```typescript
// 对象类型
type User = {
    id: number;
    name: string;
    email: string;
};

const user: User = {
    id: 1,
    name: '张三',
    email: 'zhangsan@example.com'
};

// 嵌套对象
type Address = {
    street: string;
    city: string;
    zipcode: string;
    country?: string;
};

type Person = {
    name: string;
    age: number;
    address: Address;
    contacts: Address[];
};

const person: Person = {
    name: '李四',
    age: 30,
    address: {
        street: '朝阳路',
        city: '北京',
        zipcode: '100000'
    },
    contacts: []
};
```

### 泛型类型别名

```typescript
// 泛型类型别名
type Container<T> = {
    value: T;
};

const stringContainer: Container<string> = {
    value: 'hello'
};

const numberContainer: Container<number> = {
    value: 42
};

// 多个泛型参数
type Pair<T, U> = {
    first: T;
    second: U;
};

const pair: Pair<string, number> = {
    first: 'hello',
    second: 42
};

// 泛型约束
type Maybe<T> = T | null | undefined;

const value1: Maybe<string> = 'hello';
const value2: Maybe<number> = null;
const value3: Maybe<boolean> = undefined;
```

### 类型别名 vs 接口

```typescript
// 类型别名可以定义
type StringOrNumber = string | number;
type Maybe<T> = T | null;
type Point = [number, number];
type Callback = (data: string) => void;

// 接口只能定义对象类型
interface Config {
    name: string;
    value: number;
}

// 类型别名可以定义接口能定义的一切
type Config2 = {
    name: string;
    value: number;
};
```

## 4.2 联合类型

### 基础用法

```typescript
// 可以是多种类型之一
type ID = string | number;

const id1: ID = 'abc123';
const id2: ID = 123456;

// 错误：不能是其他类型
// const id3: ID = true;  // ❌

// 多个类型
type Status = 'pending' | 'success' | 'error';

const status1: Status = 'pending';
const status2: Status = 'success';

// 错误
// const status3: Status = 'failed';  // ❌
```

### 联合类型的属性访问

```typescript
interface Bird {
    name: string;
    fly(): void;
}

interface Fish {
    name: string;
    swim(): void;
}

type Animal = Bird | Fish;

function move(animal: Animal) {
    // 只能访问共有的属性和方法
    console.log(animal.name);  // ✅ OK

    // animal.fly();  // ❌ 错误：Fish 没有 fly
    // animal.swim();  // ❌ 错误：Bird 没有 swim
}
```

### 联合类型与类型守卫

```typescript
type Animal = Bird | Fish;

// 使用 typeof
function isBird(animal: Animal): animal is Bird {
    return 'fly' in animal;
}

function move(animal: Animal) {
    if (isBird(animal)) {
        animal.fly();  // ✅ OK，类型收窄为 Bird
    } else {
        animal.swim();  // ✅ OK，类型收窄为 Fish
    }
}

// 使用 instanceof
class BirdClass {
    fly() {
        console.log('鸟在飞');
    }
}

class FishClass {
    swim() {
        console.log('鱼在游');
    }
}

function move2(animal: BirdClass | FishClass) {
    if (animal instanceof BirdClass) {
        animal.fly();
    } else {
        animal.swim();
    }
}
```

## 4.3 交叉类型

### 基础用法

```typescript
// 合并多个类型
interface A {
    a: string;
}

interface B {
    b: number;
}

type C = A & B;

const obj: C = {
    a: 'hello',
    b: 42
};
```

### 实用场景

```typescript
// 组件 Props 合并
interface BaseProps {
    className?: string;
    style?: React.CSSProperties;
}

interface ButtonProps extends BaseProps {
    label: string;
    onClick: () => void;
    disabled?: boolean;
}

// 使用交叉类型添加额外属性
type ExtendedButtonProps = ButtonProps & {
    icon: string;
    loading?: boolean;
};

// 多个接口交叉
interface Timestamp {
    createdAt: number;
    updatedAt: number;
}

interface SoftDelete {
    deletedAt?: number;
}

type BaseEntity = { id: number } & Timestamp & SoftDelete;

const entity: BaseEntity = {
    id: 1,
    createdAt: Date.now(),
    updatedAt: Date.now(),
    deletedAt: undefined
};
```

### 交叉类型的冲突处理

```typescript
// 同名属性，类型相同
interface A {
    name: string;
}

interface B {
    name: string;
}

type C = A & B;  // name: string

// 同名属性，类型不同
interface X {
    name: string;
}

interface Y {
    name: number;
}

type Z = X & Y;  // name: string & number = never

const obj: Z = {
    // name: 'hello'  // ❌ 错误
    // name: 42  // ❌ 错误
};

// 对象类型的交叉
interface Person {
    name: string;
}

interface Contact {
    email: string;
}

type Employee = Person & Contact;  // { name: string; email: string; }

const employee: Employee = {
    name: '张三',
    email: 'zhang@example.com'
};
```

## 4.4 字面量联合类型

### 字符串字面量联合

```typescript
// 定义字面量联合
type Method = 'GET' | 'POST' | 'PUT' | 'DELETE';
type Status = 'pending' | 'loading' | 'success' | 'error';

// 使用
function request(method: Method, url: string) {
    console.log(`${method} ${url}`);
}

request('GET', '/api/users');
request('POST', '/api/users');

// 状态机
type State = {
    status: 'pending';
} | {
    status: 'loading';
    progress: number;
} | {
    status: 'success';
    data: string;
} | {
    status: 'error';
    error: Error;
};

function handleState(state: State) {
    switch (state.status) {
        case 'pending':
            console.log('等待中');
            break;
        case 'loading':
            console.log(`加载中 ${state.progress}%`);
            break;
        case 'success':
            console.log(`成功：${state.data}`);
            break;
        case 'error':
            console.error(`错误：${state.error.message}`);
            break;
    }
}
```

### 数字字面量联合

```typescript
// 数字字面量
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;
type HttpStatus = 200 | 201 | 400 | 401 | 403 | 404 | 500;

function rollDice(): DiceRoll {
    return (Math.floor(Math.random() * 6) + 1) as DiceRoll;
}

function handleResponse(status: HttpStatus): string {
    switch (status) {
        case 200:
            return 'OK';
        case 201:
            return 'Created';
        case 400:
            return 'Bad Request';
        case 404:
            return 'Not Found';
        case 500:
            return 'Internal Server Error';
        default:
            return 'Unknown';
    }
}
```

## 4.5 映射类型基础

### 基础语法

```typescript
// 基于已有类型创建新类型
type Readonly<T> = {
    readonly [P in keyof T]: T[P];
};

type Partial<T> = {
    [P in keyof T]?: T[P];
};

type Required<T> = {
    [P in keyof T]-?: T[P];
};

// 使用
interface User {
    id: number;
    name: string;
    email: string;
}

type ReadonlyUser = Readonly<User>;
// { readonly id: number; readonly name: string; readonly email: string; }

type PartialUser = Partial<User>;
// { id?: number; name?: string; email?: string; }

type RequiredUser = Required<PartialUser>;
// { id: number; name: string; email: string; }
```

### 修改修饰符

```typescript
// 添加可选
type Nullable<T> = {
    [P in keyof T]: T[P] | null;
};

// 添加只读
type DeepReadonly<T> = {
    readonly [P in keyof T]: T[P];
};

// 移除可选
type Required<T> = {
    [P in keyof T]-?: T[P];
};

// 移除只读
type Mutable<T> = {
    -readonly [P in keyof T]: T[P];
};

// 使用
interface Config {
    readonly timeout: number;
    retries?: number;
}

type MutableConfig = Mutable<Config>;
// { timeout: number; retries?: number; }

type StrictConfig = Required<Config>;
// { timeout: number; retries: number; }

type NullableConfig = Nullable<Config>;
// { timeout: number | null; retries: number | null; }
```

## 4.6 模板字面量类型

### 基础用法

```typescript
// 字符串字面量联合
type EmailTemplate = 'welcome' | 'reset-password' | 'verify-email';
type EmailEvent = `${EmailTemplate}-email`;
// 'welcome-email' | 'reset-password-email' | 'verify-email-email'

// 多个联合类型组合
type Method = 'get' | 'post' | 'put' | 'delete';
type Resource = 'users' | 'posts' | 'comments';
type Endpoint = `${Method} /api/${Resource}`;
// 24 种组合：'get /api/users' | 'get /api/posts' | ...
```

### 实用场景

```typescript
// CSS 属性
type CSSProperties = {
    [K in `margin${'' | 'Top' | 'Right' | 'Bottom' | 'Left'}`]?: string | number;
};

const style: CSSProperties = {
    margin: '10px',
    marginTop: '20px',
    marginRight: '15px'
};

// 事件处理器
type Event = 'click' | 'focus' | 'blur' | 'change';
type EventHandler = {
    [K in `on${Capitalize<Event>}`]: () => void;
};

interface ButtonProps extends EventHandler {
    label: string;
}

const props: ButtonProps = {
    label: 'Submit',
    onClick: () => {},
    onFocus: () => {},
    onBlur: () => {},
    onChange: () => {}
};

// API 路由
type APIVersion = 'v1' | 'v2' | 'v3';
type APIResource = 'users' | 'posts' | 'comments';
type APIRoute = `/api/${APIVersion}/${APIResource}`;
// '/api/v1/users' | '/api/v1/posts' | ...
```

### 内置字符串工具类型

```typescript
// Capitalize - 首字母大写
type T0 = Capitalize<'hello'>;  // 'Hello'

// Lowercase - 转小写
type T1 = Lowercase<'HELLO'>;  // 'hello'

// Uppercase - 转大写
type T2 = Uppercase<'hello'>;  // 'HELLO'

// Uncapitalize - 首字母小写
type T3 = Uncapitalize<'Hello'>;  // 'hello'

// 实用示例
type Events = 'click' | 'focus' | 'blur';
type ReactEvents = {
    [K in `on${Capitalize<Events>}`]: () => void;
};
// { onClick: () => void; onFocus: () => void; onBlur: () => void; }
```

## 4.7 条件类型

### 基础语法

```typescript
// T extends U ? X : Y
type IsString<T> = T extends string ? true : false;

type T0 = IsString<string>;  // true
type T1 = IsString<number>;  // false
type T2 = IsString<'hello'>;  // true（字面量类型也是字符串）

// 判断是否为空对象
type IsEmptyObject<T> = keyof T extends never ? true : false;

type T3 = IsEmptyObject<{}>;           // true
type T4 = IsEmptyObject<{ a: 1 }>;     // false
```

### 分布式条件类型

```typescript
// 当条件类型作用于联合类型时，会分布地应用
type ToArray<T> = T extends any ? T[] : never;

type T0 = ToArray<string>;                        // string[]
type T1 = ToArray<string | number>;               // string[] | number[]
type T2 = ToArray<string | number | boolean>;     // string[] | number[] | boolean[]

// 非分布式（使用元组包装）
type ToArrayNonDistributed<T> = [T] extends [any] ? T[] : never;

type T3 = ToArrayNonDistributed<string | number>;  // (string | number)[]
```

### infer 关键字

```typescript
// 推断返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type T0 = MyReturnType<() => string>;              // string
type T1 = MyReturnType<(a: number) => Promise<number>>;  // Promise<number>

// 推断数组元素类型
type MyElementType<T> = T extends (infer U)[] ? U : never;

type T2 = MyElementType<string[]>;        // string
type T3 = MyElementType<(string | number)[]>;  // string | number

// 推断 Promise 值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

type T4 = UnwrapPromise<Promise<string>>;           // string
type T5 = UnwrapPromise<Promise<Promise<number>>>;  // number
type T6 = UnwrapPromise<string>;                    // string

// 推断函数第一个参数类型
type FirstParam<T> = T extends (arg: infer P, ...args: any[]) => any ? P : never;

type T7 = FirstParam<(name: string, age: number) => void>;  // string

// 推断函数参数元组
type ParamsTuple<T> = T extends (...args: infer P) => any ? P : never;

type T8 = ParamsTuple<(a: string, b: number) => void>;  // [string, number]
```

## 4.8 内置条件工具类型

### Exclude / Extract

```typescript
// Exclude<T, U> - 从 T 中排除可以赋值给 U 的类型
type T0 = Exclude<'a' | 'b' | 'c', 'a'>;           // 'b' | 'c'
type T1 = Exclude<string | number | null, null>;   // string | number

// 源码实现
type MyExclude<T, U> = T extends U ? never : T;

// Extract<T, U> - 从 T 中提取可以赋值给 U 的类型
type T2 = Extract<'a' | 'b' | 'c', 'a' | 'b'>;     // 'a' | 'b'
type T3 = Extract<string | number | null, number>; // number

// 源码实现
type MyExtract<T, U> = T extends U ? T : never;

// 实用示例
type EventTypes = 'click' | 'scroll' | 'mousemove' | 'touchstart';
type MouseEventTypes = Exclude<EventTypes, 'scroll'>;
// 'click' | 'mousemove' | 'touchstart'
```

### NonNullable

```typescript
// NonNullable<T> - 从 T 中排除 null 和 undefined
type T0 = NonNullable<string | null | undefined>;  // string
type T1 = NonNullable<string | number | null>;     // string | number

// 源码实现
type MyNonNullable<T> = T extends null | undefined ? never : T;

// 实用示例
function getValue<T>(value: T | null | undefined): NonNullable<T> {
    if (value === null || value === undefined) {
        throw new Error('Value is null or undefined');
    }
    return value as NonNullable<T>;
}
```

### Parameters / ReturnType

```typescript
// Parameters<T> - 获取函数参数类型
type T0 = Parameters<(a: string, b: number) => void>;
// [string, number]

// ReturnType<T> - 获取函数返回类型
type T1 = ReturnType<() => { id: number; name: string }>;
// { id: number; name: string; }

// 源码实现
type MyParameters<T extends (...args: any) => any> = 
    T extends (...args: infer P) => any ? P : never;

type MyReturnType<T extends (...args: any) => any> = 
    T extends (...args: any) => infer R ? R : any;

// 实用示例
function createUser(name: string, email: string) {
    return { id: 1, name, email };
}

type CreateUserData = Parameters<typeof createUser>;
// [string, string]

type CreateUserResult = ReturnType<typeof createUser>;
// { id: number; name: string; email: string; }
```

### InstanceType / ThisType

```typescript
// InstanceType<T> - 获取构造函数实例类型
class Person {
    constructor(public name: string, public age: number) {}
}

type T0 = InstanceType<typeof Person>;
// Person

// 源码实现
type MyInstanceType<T extends new (...args: any) => any> = 
    T extends new (...args: any) => infer R ? R : any;

// ThisType<T> - 获取 this 类型
interface DB {
    log(): void;
}

interface User {
    id: number;
    name: string;
}

interface UserDB extends DB {
    getUser: (id: number) => {
        id: number;
        name: string;
    };
}
```

## 4.9 键的重映射（as 子句）

### 基础用法（TS 4.1+）

```typescript
// 将 snake_case 转换为 camelCase
type CamelizeKeys<T> = {
    [K in keyof T as K extends `${infer A}_${infer B}`
        ? `${A}${Capitalize<B>}`
        : K]: T[K];
};

interface SnakeCaseObject {
    user_name: string;
    user_age: number;
}

type T0 = CamelizeKeys<SnakeCaseObject>;
// {
//     userName: string;
//     userAge: number;
// }

// 将 camelCase 转换为 snake_case
type SnakeifyKeys<T> = {
    [K in keyof T as K extends `${infer A}${Capitalize<infer B>}`
        ? `${A}_${Lowercase<B>}`
        : K]: T[K];
};

interface CamelCaseObject {
    userName: string;
    userAge: number;
}

type T1 = SnakeifyKeys<CamelCaseObject>;
// {
//     user_name: string;
//     user_age: number;
// }
```

### 过滤属性

```typescript
// 只保留非函数属性
type MutableProperties<T> = {

// {
//     id: number;
//     name: string;

type StringProperties<T> = {

interface Mixed {
    name: string;
    age: number;

    active: boolean;
}

type T1 = StringProperties<Mixed>;
// {
//     name: string;
//     email: string;
// }
```

### 状态管理类型

```typescript
// Action 类型
type Action<T extends string, P = void> = P extends void
    ? { type: T }
    : { type: T; payload: P };

// 组合 Actions
type UserActions =
    | Action<'LOGIN', { username: string; token: string }>
    | Action<'LOGOUT'>
    | Action<'UPDATE_PROFILE', Partial<{ username: string; avatar: string }>>;

// Reducer 类型
type Reducer<S, A> = (state: S, action: A) => S;

// State 类型
type UserState = {
    user: {
        id: number;
        username: string;
        avatar?: string;
    } | null;
    token: string | null;
    isAuthenticated: boolean;
    loading: boolean;
};

// 使用示例
const userReducer: Reducer<UserState, UserActions> = (state, action) => {
    switch (action.type) {
        case 'LOGIN':
            return {
                ...state,
                user: { id: 1, username: action.payload.username },
                token: action.payload.token,
                isAuthenticated: true
            };
        case 'LOGOUT':
            return {
                ...state,
                user: null,
                token: null,
                isAuthenticated: false
            };
        case 'UPDATE_PROFILE':
            return {
                ...state,
                user: state.user ? { ...state.user, ...action.payload } : null
            };
        default:
            return state;
    }
};

// Dispatch 类型
type Dispatch<A> = (action: A) => void;

// Store 类型
type Store<S, A> = {
    state: S;
    dispatch: Dispatch<A>;
    subscribe: (listener: (state: S) => void) => () => void;
    getState: () => S;
};
```

## 4.11 常见问答

### Q1: interface 和 type 如何选择？

**答：**
- 需要声明合并、被类实现 → interface
- 联合类型、映射类型、复杂类型转换 → type

### Q2: 如何创建不可变类型？

```typescript
// 使用 Readonly
type Immutable<T> = Readonly<T>;

// 深度不可变
type DeepImmutable<T> = {
    readonly [P in keyof T]: T[P] extends object
        ? DeepImmutable<T[P]>
        : T[P];
};
```

### Q3: 如何从联合类型中排除某些类型？

```typescript
// 使用 Exclude
type Status = 'success' | 'error' | 'pending';
type FinalStatus = Exclude<Status, 'pending'>;  // 'success' | 'error'

// 自定义排除
type ExcludeNull<T> = Exclude<T, null>;
type T0 = ExcludeNull<string | null>;  // string
```

### Q4: satisfies 操作符有什么用？

```typescript
// satisfies 验证类型，同时保留字面量类型
const config = {
    port: 3000,
    host: 'localhost'
} satisfies { port: number; host: string };

// config.port 的类型是 3000（字面量），而不是 number

// 对比 as
const config2 = {
    port: 3000,
    host: 'localhost'
} as { port: number; host: string };

// config2.port 的类型是 number
```
