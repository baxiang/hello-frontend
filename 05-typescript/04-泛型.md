# TypeScript 泛型

## 学习目标
- 掌握泛型函数、接口和类的定义
- 理解泛型约束和 extends 关键字
- 学会使用泛型工具类型
- 能够在实际项目中应用泛型

---

## 4.1 泛型基础

### 为什么使用泛型

```typescript
// 不使用泛型 - 需要定义多个重载
function firstElement(arr: number[]): number | undefined;
function firstElement(arr: string[]): string | undefined;
function firstElement(arr: any[]): any | undefined {
    return arr[0];
}

// 使用泛型 - 一个定义搞定
function firstElement<T>(arr: T[]): T | undefined {
    return arr[0];
}

// 使用
const num = firstElement([1, 2, 3]);      // num: number
const str = firstElement(['a', 'b', 'c']); // str: string
```

### 泛型函数

```typescript
// 定义泛型函数
function identity<T>(arg: T): T {
    return arg;
}

// 使用 - 方式 1：显式指定类型
const result1 = identity<string>('hello');  // string

// 使用 - 方式 2：类型推断
const result2 = identity(42);  // number
const result3 = identity({ name: '张三' });  // { name: string }

// 多个泛型参数
function merge<T, U>(obj1: T, obj2: U): T & U {
    return { ...obj1, ...obj2 };
}

const merged = merge({ a: 1 }, { b: 2 });
// merged: { a: number; b: number; }
```

### 泛型箭头函数

```typescript
// 泛型箭头函数
const identity = <T>(arg: T): T => arg;

// 多个参数
const merge = <T, U>(obj1: T, obj2: U): T & U => ({ ...obj1, ...obj2 });

// 带约束的泛型箭头函数
const getLength = <T extends { length: number }>(arg: T): number => arg.length;
```

## 4.2 泛型接口

### 定义泛型接口

```typescript
// 泛型接口
interface Box<T> {
    value: T;
}

const stringBox: Box<string> = { value: 'hello' };
const numberBox: Box<number> = { value: 42 };

// 多个泛型参数
interface Pair<T, U> {
    first: T;
    second: U;
}

const pair: Pair<string, number> = {
    first: 'hello',
    second: 42
};
```

### 泛型接口定义函数

```typescript
// 泛型函数接口
interface SearchFunc {
    <T>(arr: T[], target: T): boolean;
}

// 实现
const find: SearchFunc = <T>(arr: T[], target: T): boolean => {
    return arr.includes(target);
};

// 使用
const found = find([1, 2, 3], 2);  // true
```

### 接口方法泛型

```typescript
interface Logger {
    log<T>(message: T): void;
    logMany<T>(...messages: T[]): void;
}

class ConsoleLogger implements Logger {
    log<T>(message: T) {
        console.log(message);
    }

    logMany<T>(...messages: T[]) {
        messages.forEach(m => console.log(m));
    }
}
```

## 4.3 泛型类

### 定义泛型类

```typescript
// 泛型类
class Stack<T> {
    private items: T[] = [];

    push(item: T): void {
        this.items.push(item);
    }

    pop(): T | undefined {
        return this.items.pop();
    }

    peek(): T | undefined {
        return this.items[this.items.length - 1];
    }

    isEmpty(): boolean {
        return this.items.length === 0;
    }

    size(): number {
        return this.items.length;
    }
}

// 使用
const numberStack = new Stack<number>();
numberStack.push(1);
numberStack.push(2);
console.log(numberStack.pop());  // 2

const stringStack = new Stack<string>();
stringStack.push('hello');
stringStack.push('world');
console.log(stringStack.pop());  // 'world'
```

### 泛型类实现接口

```typescript
interface Collection<T> {
    add(item: T): void;
    remove(item: T): boolean;
    has(item: T): boolean;
    size(): number;
}

class SetCollection<T> implements Collection<T> {
    private items = new Set<T>();

    add(item: T): void {
        this.items.add(item);
    }

    remove(item: T): boolean {
        return this.items.delete(item);
    }

    has(item: T): boolean {
        return this.items.has(item);
    }

    size(): number {
        return this.items.size;
    }
}

// 使用
const numSet = new SetCollection<number>();
numSet.add(1);
numSet.add(2);
console.log(numSet.has(1));  // true
```

## 4.4 泛型约束

### extends 关键字

```typescript
// 约束 T 必须有 length 属性
interface Lengthwise {
    length: number;
}

function logLength<T extends Lengthwise>(arg: T): T {
    console.log(arg.length);
    return arg;
}

// OK
logLength('hello');      // string 有 length
logLength([1, 2, 3]);    // array 有 length
logLength({ length: 10 });

// 错误
// logLength(42);        // number 没有 length
// logLength(undefined); // undefined 没有 length
```

### 对象属性约束

```typescript
// 约束 T 必须有 id 属性
interface Identifiable {
    id: number | string;
}

function getId<T extends Identifiable>(obj: T): T['id'] {
    return obj.id;
}

const user = { id: 1, name: '张三' };
const post = { id: 'abc', title: '标题' };

console.log(getId(user));  // 1
console.log(getId(post));  // 'abc'
```

### 多个约束

```typescript
// 多个约束
function merge<T extends object, U extends object>(
    obj1: T,
    obj2: U
): T & U {
    return { ...obj1, ...obj2 };
}

const result = merge({ a: 1 }, { b: 2 });
// result: { a: number; b: number; }

// 约束继承
interface HasName {
    name: string;
}

interface HasAge {
    age: number;
}

function printInfo<T extends HasName & HasAge>(person: T): void {
    console.log(`${person.name}, ${person.age}岁`);
}

printInfo({ name: '张三', age: 25 });
```

### keyof 约束

```typescript
// 约束 K 必须是 T 的键
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

const user = { id: 1, name: '张三', email: 'zhang@example.com' };

console.log(getProperty(user, 'name'));  // '张三'
console.log(getProperty(user, 'id'));    // 1

// 错误
// console.log(getProperty(user, 'age'));  // 'age' 不是 user 的键
```

## 4.5 泛型工具类型

### Partial / Required / Readonly

```typescript
// Partial<T> - 所有属性变为可选
type PartialUser = Partial<{ id: number; name: string }>;
// { id?: number; name?: string; }

// Required<T> - 所有属性变为必需
type RequiredUser = Required<{ id?: number; name?: string }>;
// { id: number; name: string; }

// Readonly<T> - 所有属性变为只读
type ReadonlyUser = Readonly<{ id: number; name: string }>;
// { readonly id: number; readonly name: string; }

// 实际应用
class UserStore {
    private users: Array<{ id: number; name: string }> = [];

    updateUser(id: number, updates: Partial<{ id: number; name: string }>) {
        const index = this.users.findIndex(u => u.id === id);
        if (index !== -1) {
            this.users[index] = { ...this.users[index], ...updates };
        }
    }
}
```

### Pick / Omit

```typescript
// Pick<T, K> - 选取 K 属性
interface User {
    id: number;
    name: string;
    email: string;
    password: string;
}

type UserPublic = Pick<User, 'id' | 'name'>;
// { id: number; name: string; }

// Omit<T, K> - 省略 K 属性
type UserSafe = Omit<User, 'password'>;
// { id: number; name: string; email: string; }

type UserCreate = Omit<User, 'id'>;
// { name: string; email: string; password: string; }
```

### Record / Map / Set

```typescript
// Record<K, T> - 键值对类型
type UserRole = 'admin' | 'user' | 'guest';
type Permissions = Record<UserRole, string[]>;

const permissions: Permissions = {
    admin: ['read', 'write', 'delete'],
    user: ['read'],
    guest: ['read']
};

// 配置对象
type Env = 'development' | 'production' | 'test';
type DBConfig = Record<Env, { host: string; port: number }>;

const dbConfig: DBConfig = {
    development: { host: 'localhost', port: 3306 },
    production: { host: 'prod.db.com', port: 3306 },
    test: { host: 'test.db.com', port: 3306 }
};
```

### Exclude / Extract / NonNullable

```typescript
// Exclude<T, U> - 排除 U
type T0 = Exclude<'a' | 'b' | 'c', 'a'>;  // 'b' | 'c'

// Extract<T, U> - 提取 U
type T1 = Extract<'a' | 'b' | 'c', 'a' | 'b'>;  // 'a' | 'b'

// NonNullable<T> - 排除 null 和 undefined
type T2 = NonNullable<string | null | undefined>;  // string

// 实际应用
type Status = 'pending' | 'success' | 'error' | null;
type ValidStatus = Exclude<Status, null>;  // 'pending' | 'success' | 'error'
```

### Parameters / ReturnType / InstanceType

```typescript
// Parameters<T> - 函数参数类型
type T0 = Parameters<(a: string, b: number) => void>;
// [string, number]

// ReturnType<T> - 函数返回类型
type T1 = ReturnType<() => { id: number }>;
// { id: number; }

// InstanceType<T> - 构造函数实例类型
class Person {
    constructor(public name: string) {}
}
type T2 = InstanceType<typeof Person>;  // Person

// 实际应用
function createUser(name: string, email: string) {
    return { id: 1, name, email, createdAt: Date.now() };
}

type CreateParams = Parameters<typeof createUser>;
type CreateResult = ReturnType<typeof createUser>;
```

## 4.6 高级泛型技巧

### 条件类型

```typescript
// 基础条件类型
type IsString<T> = T extends string ? true : false;

type T0 = IsString<string>;  // true
type T1 = IsString<number>;  // false

// 分布式条件类型
type ToArray<T> = T extends any ? T[] : never;

type T2 = ToArray<string | number>;  // string[] | number[]

// infer 推断
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

type T3 = UnwrapPromise<Promise<string>>;  // string
type T4 = UnwrapPromise<number>;  // number
```

### 映射类型

```typescript
// 基础映射类型
type Readonly<T> = {
    readonly [P in keyof T]: T[P];
};

type Partial<T> = {
    [P in keyof T]?: T[P];
};

// 自定义映射类型
type Nullable<T> = {
    [P in keyof T]: T[P] | null;
};

interface User {
    id: number;
    name: string;
}

type NullableUser = Nullable<User>;
// { id: number | null; name: string | null; }

// 添加/移除修饰符
type Mutable<T> = {
    -readonly [P in keyof T]: T[P];
};

type Required<T> = {
    [P in keyof T]-?: T[P];
};
```

### 模板字面量类型

```typescript
// 基础用法
type Event = 'click' | 'focus' | 'blur';
type Handler = {
    [K in `on${Capitalize<Event>}`]: () => void;
};
// { onClick: () => void; onFocus: () => void; onBlur: () => void; }

// CSS 属性
type CSSProperties = {
    [K in `margin${'' | 'Top' | 'Right' | 'Bottom' | 'Left'}`]?: string;
};

const style: CSSProperties = {
    margin: '10px',
    marginTop: '20px'
};
```

## 4.7 实践练习

### 练习 1：通用缓存

```typescript
interface CacheOptions {
    maxAge?: number;      // 缓存过期时间（毫秒）
    maxSize?: number;     // 最大缓存数量
}

class Cache<T> {
    private cache = new Map<string, { value: T; timestamp: number }>();
    private options: Required<CacheOptions>;

    constructor(options: CacheOptions = {}) {
        this.options = {
            maxAge: options.maxAge ?? 60000,
            maxSize: options.maxSize ?? 100
        };
    }

    get(key: string): T | undefined {
        const item = this.cache.get(key);
        if (!item) return undefined;

        // 检查是否过期
        if (Date.now() - item.timestamp > this.options.maxAge) {
            this.cache.delete(key);
            return undefined;
        }

        return item.value;
    }

    set(key: string, value: T): void {
        // 检查大小限制
        if (this.cache.size >= this.options.maxSize) {
            const firstKey = this.cache.keys().next().value;
            if (firstKey) this.cache.delete(firstKey);
        }

        this.cache.set(key, {
            value,
            timestamp: Date.now()
        });
    }

    delete(key: string): boolean {
        return this.cache.delete(key);
    }

    clear(): void {
        this.cache.clear();
    }

    has(key: string): boolean {
        return this.get(key) !== undefined;
    }
}

// 使用
const userCache = new Cache<{ id: number; name: string }>({
    maxAge: 300000,  // 5 分钟
    maxSize: 50
});

userCache.set('user1', { id: 1, name: '张三' });
const user = userCache.get('user1');
```

### 练习 2：通用数据列表

```typescript
interface ListParams {
    page: number;
    pageSize: number;
}

interface ListResult<T> {
    data: T[];
    total: number;
    page: number;
    pageSize: number;
}

interface ListConfig<T> {
    fetcher: (params: ListParams) => Promise<ListResult<T>>;
    onLoad?: (data: T[]) => void;
    onError?: (error: Error) => void;
}

class VirtualList<T> {
    private items: T[] = [];
    private total = 0;
    private currentPage = 1;
    private loading = false;
    private config: ListConfig<T>;

    constructor(config: ListConfig<T>) {
        this.config = config;
    }

    async loadMore(pageSize = 20): Promise<void> {
        if (this.loading) return;
        if (this.items.length >= this.total && this.total > 0) return;

        this.loading = true;
        try {
            const result = await this.config.fetcher({
                page: this.currentPage,
                pageSize
            });

            this.items = [...this.items, ...result.data];
            this.total = result.total;
            this.currentPage++;

            this.config.onLoad?.(result.data);
        } catch (error) {
            this.config.onError?.(error as Error);
        } finally {
            this.loading = false;
        }
    }

    getData(): T[] {
        return this.items;
    }

    getTotal(): number {
        return this.total;
    }

    hasMore(): boolean {
        return this.items.length < this.total;
    }

    refresh(): void {
        this.items = [];
        this.total = 0;
        this.currentPage = 1;
    }
}

// 使用
interface User {
    id: number;
    name: string;
}

const userList = new VirtualList<User>({
    fetcher: async (params) => {
        const res = await fetch(`/api/users?page=${params.page}&size=${params.pageSize}`);
        return res.json();
    },
    onLoad: (data) => console.log('加载了', data.length, '条数据')
});

// 加载更多
userList.loadMore();
```

### 练习 3：类型安全的事件总线

```typescript
// 定义事件映射
interface EventMap {
    'user:login': { userId: number; token: string };
    'user:logout': { userId: number };
    'data:update': { key: string; value: any };
    'error': { message: string; code: number };
}

// 事件处理器类型
type Handler<T> = (data: T) => void;

// 泛型事件总线
class EventBus<Events extends Record<string, any>> {
    private handlers = new Map<keyof Events, Set<Handler<any>>>();

    on<K extends keyof Events>(event: K, handler: Handler<Events[K]>): void {
        if (!this.handlers.has(event)) {
            this.handlers.set(event, new Set());
        }
        this.handlers.get(event)!.add(handler);
    }

    off<K extends keyof Events>(event: K, handler: Handler<Events[K]>): void {
        const set = this.handlers.get(event);
        if (set) {
            set.delete(handler);
        }
    }

    emit<K extends keyof Events>(event: K, data: Events[K]): void {
        const set = this.handlers.get(event);
        if (set) {
            set.forEach(handler => handler(data));
        }
    }

    once<K extends keyof Events>(event: K, handler: Handler<Events[K]>): void {
        const onceHandler = (data: Events[K]) => {
            handler(data);
            this.off(event, onceHandler);
        };
        this.on(event, onceHandler);
    }
}

// 使用
const bus = new EventBus<EventMap>();

bus.on('user:login', ({ userId, token }) => {
    console.log(`用户 ${userId} 登录，token: ${token}`);
});

bus.emit('user:login', { userId: 1, token: 'abc123' });
```

### 练习 4：API 请求封装

```typescript
// 基础响应类型
interface APIResponse<T> {
    code: number;
    message: string;
    data: T;
}

// 请求配置
interface RequestConfig extends RequestInit {
    baseURL?: string;
    timeout?: number;
}

// 泛型 API 类
class APIClient {
    private baseURL: string;
    private defaultConfig: RequestConfig;

    constructor(baseURL: string, config: RequestConfig = {}) {
        this.baseURL = baseURL;
        this.defaultConfig = {
            timeout: 10000,
            headers: {
                'Content-Type': 'application/json'
            },
            ...config
        };
    }

    private async request<T>(
        endpoint: string,
        config: RequestConfig = {}
    ): Promise<T> {
        const url = `${this.baseURL}${endpoint}`;
        const mergedConfig = { ...this.defaultConfig, ...config };

        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), mergedConfig.timeout);

        try {
            const response = await fetch(url, {
                ...mergedConfig,
                signal: controller.signal
            });

            const result: APIResponse<T> = await response.json();

            if (result.code !== 200) {
                throw new Error(result.message);
            }

            return result.data;
        } finally {
            clearTimeout(timeoutId);
        }
    }

    get<T>(endpoint: string, params?: Record<string, any>): Promise<T> {
        const query = params ? '?' + new URLSearchParams(params).toString() : '';
        return this.request<T>(endpoint + query, { method: 'GET' });
    }

    post<T>(endpoint: string, data?: any): Promise<T> {
        return this.request<T>(endpoint, {
            method: 'POST',
            body: JSON.stringify(data)
        });
    }

    put<T>(endpoint: string, data?: any): Promise<T> {
        return this.request<T>(endpoint, {
            method: 'PUT',
            body: JSON.stringify(data)
        });
    }

    delete<T>(endpoint: string): Promise<T> {
        return this.request<T>(endpoint, { method: 'DELETE' });
    }
}

// 使用
interface User {
    id: number;
    name: string;
    email: string;
}

interface Post {
    id: number;
    title: string;
    content: string;
}

const api = new APIClient('https://api.example.com');

// 类型安全的请求
const users = api.get<User[]>('/users');
const user = api.get<User>(`/users/1`);
const newPost = api.post<Post>('/posts', { title: '标题', content: '内容' });
```

## 4.8 常见问答

### Q1: 泛型约束和类型断言有什么区别？

**答：**
- 泛型约束：在编译时约束类型，更类型安全
- 类型断言：告诉编译器相信你的判断，有风险

```typescript
// 泛型约束（推荐）
function getProperty<T, K extends keyof T>(obj: T, key: K) {
    return obj[key];
}

// 类型断言（谨慎使用）
function getProperty<T>(obj: any, key: string) {
    return (obj as any)[key];
}
```

### Q2: 什么时候使用泛型默认值？

```typescript
// 提供默认类型
interface Response<T = any> {
    code: number;
    data: T;
}

// 使用
const res1: Response = { code: 200, data: 'anything' };
const res2: Response<string> = { code: 200, data: 'string' };
```

### Q3: 泛型如何递归？

```typescript
// 深度 Partial
type DeepPartial<T> = {
    [P in keyof T]?: T[P] extends object
        ? DeepPartial<T[P]>
        : T[P];
};

// 使用
interface User {
    profile: {
        address: {
            city: string;
        };
    };
}

type UpdateUser = DeepPartial<User>;
// { profile?: { address?: { city?: string; }; }; }
```
