## 第一阶段：入门（3小时）

**目标**：理解 TypeScript 是什么，为什么需要它，能写出第一个 TS 程序

### 1.1 什么是 TypeScript（30分钟）

#### 核心概念
- **TypeScript 定义**：JavaScript 的超集，添加了类型系统
- **与 JavaScript 的关系**：TS 最终编译为 JS 在浏览器/Node.js 中运行
- **为什么需要 TypeScript**：静态类型检查，减少运行时错误

#### 生活化类比
```
JavaScript 就像：
- 写完作文直接交卷
- 老师批改时才发现错别字
- 运行代码时才报错

TypeScript 就像：
- 有一个智能助手帮你检查
- 写的时候就会提示"这里可能有错"
- 编译时就发现问题
```

#### 代码对比

```javascript
// JavaScript - 运行时报错
function add(a, b) {
    return a + b;
}
add(1, '2');  // 返回 "12"，不是数字 3
```

```typescript
// TypeScript - 编译时警告
function add(a: number, b: number): number {
    return a + b;
}
add(1, '2');  // ❌ 编译错误：类型 'string' 不能赋值给类型 'number'
```

**练习**：在 [TypeScript Playground](https://www.typescriptlang.org/play/) 中体验类型检查

| strict | 严格模式 | true |
| outDir | 输出目录 | ./dist |
| rootDir | 源码目录 | ./src |

#### 1.3.4 为什么要开 strict 模式？

- 避免 null/undefined 错误

#### 2.1.1 基础类型一览
```typescript
// 布尔
let isDone: boolean = false;

// 数字（整数、浮点数、进制）
let decimal: number = 6;
let hex: number = 0xf00d;
let binary: number = 0b1010;
let octal: number = 0o744;

// 字符串
let color: string = 'blue';
let fullName: string = `Bob Bobbington`;
let sentence: string = `Hello, my name is ${fullName}.`;

// 数组
let list: number[] = [1, 2, 3];
let list2: Array<number> = [1, 2, 3];

// 元组 - 固定长度、固定类型
let x: [string, number];
x = ['hello', 10];  // ✅
x = [10, 'hello'];  // ❌

// 枚举
enum Color { Red, Green, Blue }
let c: Color = Color.Green;

// any - 任意类型
let notSure: any = 4;
notSure = 'maybe a string';
notSure = true;

// unknown - 未知类型（安全版 any）
let value: unknown = 'hello';
// value.toUpperCase()  // ❌ 需要先检查类型
if (typeof value === 'string') {
    value.toUpperCase();  // ✅
}

// void - 无返回值
function warnUser(): void {
    console.log('Warning!');
}

// never - 永不返回
function error(message: string): never {
    throw new Error(message);
}
function infiniteLoop(): never {
    while (true) {}
}

// object - 非原始类型
let obj: object = { name: 'Tom' };
```

#### 2.1.2 any vs unknown vs never 对比
| 类型 | 说明 | 使用场景 |
|------|------|----------|
| any | 任意类型，关闭类型检查 | 快速迁移 JS 代码 |
| unknown | 未知类型，安全版 any | 处理未知数据 |
| never | 永不返回 | 错误处理、类型穷尽 |

```typescript
// any - 绕过所有检查
let a: any = 'hello';
a.toFixed(2);  // ❌ 运行时不报错，但无意义

// unknown - 必须先检查类型
let u: unknown = 'hello';
// u.toFixed(2);  // ❌ 编译错误
if (typeof u === 'number') {
    u.toFixed(2);  // ✅
}

// never - 用于类型穷尽
function handleValue(value: string | number) {
    if (typeof value === 'string') {
        // string 处理
    } else if (typeof value === 'number') {
        // number 处理
    } else {
        // value 是 never - 所有情况已处理
        const _exhaustive: never = value;
    }
}
```

### 3.2 泛型编程（3.5小时）

#### 2.2.1 为什么使用泛型
```typescript
// 问题：每个类型都要写一个函数
function identityNumber(arg: number): number {
    return arg;
}

function identityString(arg: string): string {
    return arg;
}

// 泛型解决方案
function identity<T>(arg: T): T {
    return arg;
}

identity<number>(123);  // number
identity<string>('hello');  // string
identity(123);  // 类型推断为 number
```

#### 2.2.2 泛型约束

```typescript
    return arg;
}

logLength('hello');  // ✅ 字符串有 length
logLength([1, 2, 3]);  // ✅ 数组有 length

logLength(123);  // ❌ 数字没有 length
```

```typescript
function process(value: string | number): string | number {
```

#### 3.3.2 instanceof 守卫
```typescript
class Dog {
    bark(): void {
        console.log('Woof!');
    }
}

class Cat {
    meow(): void {
        console.log('Meow!');
    }
}

function makeSound(animal: Dog | Cat): void {
    if (animal instanceof Dog) {
        animal.bark();
    } else {
        animal.meow();
    }
}
```

#### 3.3.3 in 操作符守卫
```typescript
interface Fish {
    swim(): void;
}

interface Bird {
    fly(): void;
}

function move(animal: Fish | Bird): void {
    if ('swim' in animal) {
        animal.swim();
    } else {
        animal.fly();
    }
}
```

#### 3.3.4 类型谓词（is）
```typescript
interface Fish {
    swim(): void;
}

interface Bird {
    fly(): void;
}

function isFish(animal: Fish | Bird): animal is Fish {
    return (animal as Fish).swim !== undefined;
}

function move(animal: Fish | Bird): void {
    if (isFish(animal)) {
        animal.swim();
    } else {
        animal.fly();
    }
}
```

#### 3.3.5 完整性检查
```typescript
type Shape = 
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; size: number }
    | { kind: 'triangle'; base: number; height: number };

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'square':
            return shape.size ** 2;
        case 'triangle':
            return 0.5 * shape.base * shape.height;
        default:
            // 完整性检查：如果漏掉某个类型，TS 会报错
            const _exhaustive: never = shape;
            throw new Error('Unknown shape');
    }
}
```

#### 4.2.1 映射类型基础

### 4.3 装饰器（2.5小时）

#### 4.3.1 装饰器基础
```typescript
// 启用装饰器需要在 tsconfig.json 中设置
// "experimentalDecorators": true

function log(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function (...args: any[]) {
        console.log(`Calling ${propertyKey} with`, args);
        return original.apply(this, args);
    };
}

class Calculator {
    @log
    add(a: number, b: number): number {
        return a + b;
    }
}

const calc = new Calculator();
calc.add(1, 2);
// 输出：Calling add with [ 1, 2 ]
// 返回：3
```

#### 4.3.2 装饰器工厂
```typescript
function delay(ms: number) {
    return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        const original = descriptor.value;
        descriptor.value = function (...args: any[]) {
            setTimeout(() => {
                return original.apply(this, args);
            }, ms);
        };
    };
}

class Api {
    @delay(1000)
    fetchData() {
        console.log('Data fetched');
    }
}
```

#### 4.3.3 类装饰器
```typescript
function sealed(constructor: Function) {
    Object.seal(constructor);
    Object.seal(constructor.prototype);

@sealed
class User {
    name: string;
    constructor(name: string) {
        this.name = name;
    }
}

// User = sealed(User);
```

```typescript
function readonly(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    descriptor.writable = false;
}
```

## 常见错误与解决方案

### 错误 1：Object is possibly 'null'
```typescript
// ❌
const name = user.name.toUpperCase();

// ✅ 使用可选链
const name = user.name?.toUpperCase();

// ✅ 使用非空断言
const name = user.name!.toUpperCase();

// ✅ 先检查
if (user.name) {
    const name = user.name.toUpperCase();
}
```

### 错误 2：Type 'string' is not assignable to type 'number'
```typescript
// ❌
let age: number = '25';

// ✅
let age: number = 25;
let age: number = Number('25');
```

### 错误 3：Property 'xxx' does not exist on type 'yyy'
```typescript
// ❌ 访问不存在的属性
const name = (user as any).name;

// ✅ 使用类型守卫
if ('name' in user) {
    const name = user.name;
}

// ✅ 使用可选链
const name = user.name;
```

### 错误 4：Generic type 'xxx' requires 1 type argument(s)
```typescript
// ❌
const arr: Array = [1, 2, 3];

// ✅
const arr: Array<number> = [1, 2, 3];
// 或
const arr: number[] = [1, 2, 3];
```
