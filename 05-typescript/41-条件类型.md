## 14.0 通俗理解条件类型

### 什么是条件类型？

```
条件类型 = 类型的"if-else"

就像 JavaScript 的三元运算符：
condition ? value1 : value2

条件类型：
T extends U ? X : Y
```

### 通俗比喻

```
场景：判断一个人是不是学生

JavaScript 三元运算符：
isStudent ? "是学生" : "不是学生"

条件类型：
T extends Student ? true : false
```

### 实际例子

```typescript
// 判断 T 是不是 string
type IsString<T> = T extends string ? true : false;

// 使用
type A = IsString<string>;  // true
type B = IsString<number>;  // false
type C = IsString<'hello'>; // true（'hello' 是字符串字面量）
```

## 14.1 条件类型基础

### 语法

```typescript
T extends U ? X : Y
```

**解读：**
- `T extends U` - 如果 T 是 U 的子类型
- `? X` - 返回 X
- `: Y` - 否则返回 Y

### 示例

```typescript
// 基础用法
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;  // true
type B = IsString<number>;  // false

// 判断是否是 any
type IsAny<T> = 0 extends (T extends any ? 1 : 0) ? true : false;

// 判断是否是 never
type IsNever<T> = [T] extends [never] ? true : false;
```

### 实际应用

```typescript
// 场景：去除数组类型中的 void
type RemoveVoid<T> = T extends void ? never : T;

type Result = RemoveVoid<string | void | number>;
// 结果：string | number

// 场景：判断函数
type IsFunction<T> = T extends (...args: any[]) => any ? true : false;

type A = IsFunction<() => void>;  // true
type B = IsFunction<string>;      // false
```

## 14.2 infer 推断（重点！）

### 什么是 infer？

```
infer = "推断" = 从已知类型中"提取"部分类型

就像从口袋里掏出东西：
- 你知道口袋里有一封信
- infer 就像把手伸进去拿出来
```

### 通俗理解

```typescript
// 场景：从函数类型中提取返回类型

// 不用 infer（麻烦）
type GetReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

// 解读：
// T extends (...args: any[]) => infer R
// 如果 T 是一个函数
// infer R = 推断出返回类型，用 R 表示
// ? R : never = 返回推断出的类型
```

### 示例：从函数提取返回类型

```typescript
// 提取函数返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

// 使用
function getUser() {
    return { id: 1, name: 'Tom' };
}

type User = MyReturnType<typeof getUser>;
// 结果：{ id: number; name: string }

// 另一个例子
function getName() {
    return 'Tom';
}

type Name = MyReturnType<typeof getName>;
// 结果：string
```

### 示例：从函数提取参数类型

```typescript
// 提取第一个参数类型
type FirstParam<T> = T extends (first: infer P, ...args: any[]) => any ? P : never;

function greet(name: string, age: number) {
    return `Hello, ${name}`;
}

type Param = FirstParam<typeof greet>;
// 结果：string
```

### 示例：从 Promise 提取类型

```typescript
// 提取 Promise 内部类型
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

// 使用
type A = UnwrapPromise<Promise<string>>;  // string
type B = UnwrapPromise<number>;             // number（不是 Promise，直接返回）
type C = UnwrapPromise<Promise<{ id: number }>>;  // { id: number }
```

### infer 的本质

```typescript
// infer 就像一个"变量"，用来接收推断出的类型
// 可以用在任何位置

// 1. 推断返回类型
type F1<T> = T extends () => infer R ? R : never;

// 2. 推断参数类型
type F2<T> = T extends (infer P) => void ? P : never;

// 3. 推断数组元素
type F3<T> = T extends (infer E)[] ? E : never;

// 4. 推断对象属性
type F4<T> = T extends { value: infer V } ? V : never;
```

## 14.3 分布式条件类型

### 什么是分布式？

```
分布式 = 联合类型会自动"分发"处理
```

### 通俗理解

```typescript
// 场景：把每个类型都变成数组

type ToArray<T> = T extends any ? T[] : never;

type Result = ToArray<string | number>;
// 分布式后：string[] | number[]
// 不是 (string | number)[]
```

### 为什么需要分布式？

```typescript
// 不用分布式
type Bad<T> = T[];  // 整个联合变成数组

type A = Bad<string | number>;
// 结果：(string | number)[]  - 错误！

// 用分布式
type Good<T> = T extends any ? T[] : never;

type B = Good<string | number>;
// 结果：string[] | number[]  - 正确！
```

### 实际应用

```typescript
// 场景：排除 null 和 undefined
type NonNullable<T> = T extends null | undefined ? never : T;

type A = NonNullable<string | null | undefined>;
// 结果：string

// 场景：提取函数类型
type FunctionKeys<T> = {
    [K in keyof T]: T[K] extends (...args: any[]) => any ? K : never;
}[keyof T];
```

## 14.4 条件类型映射

### 实际示例

```typescript
// 1. 判断是否是字符串
type IsString<T> = T extends string ? true : false;

// 2. 获取数组元素类型
type ArrayElement<T> = T extends (infer E)[] ? E : never;

// 3. 获取函数参数
type Parameters<T> = T extends (...args: infer P) => any ? P : never;

// 4. 获取构造函数参数
type ConstructorParameters<T> = T extends new (...args: infer P) => any ? P : never;
```

## 14.5 完整示例

```typescript
// 综合示例：实现一个工具类型

// 1. 提取函数返回类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

// 2. 提取函数参数
type Parameters<T> = T extends (...args: infer P) => any ? P : never;

// 3. 提取 Promise 内部类型
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

// 测试
function getUser() {
    return { id: 1, name: 'Tom' };
}

function saveUser(name: string, age: number) {
    return { success: true };
}

// 返回类型
type R1 = ReturnType<typeof getUser>;
// { id: number; name: string }

type R2 = ReturnType<typeof saveUser>;
// { success: boolean }

// 参数类型
type P1 = Parameters<typeof saveUser>;
// [string, number]

// Promise
type U1 = UnwrapPromise<Promise<string>>;
// string

type U2 = UnwrapPromise<Promise<{ id: number }>>;
// { id: number }
```
