## 14.1 条件类型基础

### 语法

```typescript
T extends U ? X : Y
```

如果类型 T 是 U 的子类型，则返回 X，否则返回 Y。

### 示例

```typescript
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;  // true
type B = IsString<number>;  // false
```

## 14.2 infer 推断

### 什么是 infer

infer 用于从类型中推断出部分类型：

```typescript
// 获取函数返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

function getUser() {
    return { id: 1, name: 'Tom' };
}

type User = MyReturnType<typeof getUser>;
// { id: number; name: string }
```

### 推断参数类型

```typescript
// 获取函数第一个参数类型
type FirstParam<T> = T extends (first: infer P, ...args: any[]) => any ? P : never;

function greet(name: string, age: number) {
    return `Hello, ${name}!`;
}

type Param = FirstParam<typeof greet>;  // string
```

### 推断数组元素

```typescript
type ArrayElement<T> = T extends (infer E)[] ? E : never;

type Num = ArrayElement<number[]>;  // number
type Str = ArrayElement<string[]>;  // string
```

## 14.3 分布式条件类型

### 什么是分布式

当条件类型的左边是联合类型时，会分发处理：

```typescript
type ToArray<T> = T extends any ? T[] : never;

type StrOrNum = ToArray<string | number>;
// string[] | number[]
```

### 实际应用

```typescript
// 移除 null 和 undefined
type NonNullable<T> = T extends null | undefined ? never : T;

type A = NonNullable<string | null | undefined>;  // string
```

## 14.4 条件类型映射

### 实际示例

```typescript
// 提取函数类型
type FnReturn<T> = T extends (...args: any[]) => infer R ? R : never;

// 提取数组元素类型
type ArrayElement<T> = T extends (infer E)[] ? E : never;

// 提取 Promise 内部类型
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

type P = Promise<string>;
type S = UnwrapPromise<P>;  // string
```

## 14.5 实践练习

### 练习 1：实现 ReturnType

```typescript
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

function getData() {
    return { id: 1, name: 'Tom' };
}

type Data = MyReturnType<typeof getData>;
// { id: number; name: string }
```

### 练习 2：实现 Parameters

```typescript
type MyParameters<T> = T extends (...args: infer P) => any ? P : never;

function createUser(name: string, age: number) {
    return { name, age };
}

type Params = MyParameters<typeof createUser>;
// [string, number]
```

### 练习 3：实现 UnwrapPromise

```typescript
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;

type A = UnwrapPromise<Promise<string>>;  // string
type B = UnwrapPromise<Promise<number[]>>;  // number[]
type C = UnwrapPromise<string>;  // string
```
