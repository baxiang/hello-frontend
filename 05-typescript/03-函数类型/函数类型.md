## L1 理解层：会用

### 基础函数类型

**最简示例（1-3行）：**

```typescript
function add(a: number, b: number): number { return a + b; }
const mul = (a: number, b: number): number => a * b;
```

**详细示例：**

```typescript
// 参数类型
function greet(name: string, age: number): string {
    return `${name}, ${age}岁`;
}

// 可选参数
function createUser(name: string, age?: number) {
    return age ? { name, age } : { name };
}
createUser('张三');       // ✅
createUser('张三', 25);   // ✅

// 默认参数
function paginate(page: number, limit: number = 10) {
    return { page, limit };
}
paginate(1);       // { page: 1, limit: 10 }
paginate(1, 20);   // { page: 1, limit: 20 }

// 剩余参数
function sum(...numbers: number[]): number {
    return numbers.reduce((total, n) => total + n, 0);
}
sum(1, 2, 3);       // 6
sum(10, 20, 30, 40); // 100

// 函数类型表达式
type MathFn = (a: number, b: number) => number;
const add: MathFn = (a, b) => a + b;
const sub: MathFn = (a, b) => a - b;

// void — 无返回值
function log(message: string): void {
    console.log(message);
}

// never — 永不返回
function throwError(msg: string): never {
    throw new Error(msg);
}
```

### 函数重载

**最简示例：**

```typescript
function parse(value: string): number;
function parse(value: number): string;
function parse(value: string | number): string | number {
    return typeof value === 'string' ? Number(value) : String(value);
}
```

**详细示例：**

```typescript
// 函数重载 — 不同参数类型返回不同结果
function getElement(element: string): HTMLElement;
function getElement(element: number): HTMLElement | null;
function getElement(element: string | number): HTMLElement | null {
    if (typeof element === 'string') {
        return document.querySelector(element)!;
    }
    return document.querySelectorAll('*')[element] || null;
}

// 使用
const header = getElement('header');    // 类型：HTMLElement
const third = getElement(2);            // 类型：HTMLElement | null

// 重载与可选参数
function formatDate(date: Date): string;
function formatDate(date: Date, format: 'short' | 'long'): string;
function formatDate(date: Date, format: 'short' | 'long' = 'long'): string {
    return format === 'short'
        ? date.toLocaleDateString()
        : date.toLocaleString();
}
```

## L2 实践层：用好

### 高阶函数类型

```typescript
// 回调函数
function forEach<T>(arr: T[], callback: (item: T, index: number) => void): void {
    for (let i = 0; i < arr.length; i++) {
        callback(arr[i], i);
    }
}

// 返回函数
function createMultiplier(factor: number): (n: number) => number {
    return (n: number) => n * factor;
}
const double = createMultiplier(2);
double(5);  // 10

// 泛型高阶函数
function map<T, U>(arr: T[], fn: (item: T) => U): U[] {
    return arr.map(fn);
}
map([1, 2, 3], n => String(n));  // string[]
```

### 反模式：不要这样做

```typescript
// ❌ 错误：可选参数放在必填参数前面
function bad(name?: string, age: number) { }  // 编译错误

// ✅ 正确：必填参数在前
function good(name: string, age?: number) { }
```

```typescript
// ❌ 错误：重载签名与实现不匹配
function process(x: string): number;
function process(x: number): string;
function process(x: string | number): string | number {
    return x;  // 实现返回类型不匹配签名
}

// ✅ 正确：实现签名必须兼容所有重载
function process(x: string): number;
function process(x: number): string;
function process(x: string | number): string | number {
    return typeof x === 'string' ? Number(x) : String(x);
}
```

### 适用场景

| 场景 | 推荐方案 | 示例 |
|------|---------|------|
| 简单函数 | 直接标注 | `function fn(a: number): number` |
| 函数作为参数 | 函数类型 | `callback: (x: T) => void` |
| 多种输入输出 | 函数重载 | `function fn(x: string): number` |
| 通用处理逻辑 | 泛型函数 | `function map<T, U>(arr: T[], fn: ...)` |

## L3 专家层：深入

### 函数类型兼容性

```typescript
// 参数协变 vs 逆变
type Handler = (event: { type: string }) => void;

// ✅ 参数可以是父类型（逆变）
const handler: Handler = (event: MouseEvent) => { };

// ✅ 返回类型可以是子类型（协变）
type Factory = () => { type: string };
const factory: Factory = () => ({ type: 'click', x: 0, y: 0 });

// ❌ 返回类型不能是父类型
// const bad: Factory = () => ({ type: 'click' });  // 缺少 x, y
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **函数类型** | 描述函数的参数和返回类型 | `(a: number) => string` |
| **可选参数** | 可以省略的参数 | `name?: string` |
| **剩余参数** | 收集多余参数为数组 | `...args: number[]` |
| **函数重载** | 同一个函数有多个类型签名 | `function fn(x: string): number` |
| **void** | 函数没有返回值 | `function fn(): void` |
| **never** | 函数永远不会正常返回 | `function err(): never` |
| **高阶函数** | 接受函数作为参数或返回函数的函数 | `map`, `filter` |

## 实践练习

```typescript
// 练习 1：实现 filter
function filter<T>(arr: T[], predicate: (item: T) => boolean): T[] {
    const result: T[] = [];
    for (const item of arr) {
        if (predicate(item)) result.push(item);
    }
    return result;
}

const evens = filter([1, 2, 3, 4, 5], n => n % 2 === 0);  // [2, 4]

// 练习 2：实现 debounce 类型
function debounce<T extends (...args: any[]) => void>(
    fn: T,
    delay: number
): T {
    let timer: ReturnType<typeof setTimeout>;
    return ((...args: Parameters<T>) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn(...args), delay);
    }) as T;
}

// 练习 3：函数重载 — 安全的 JSON 解析
function parseJSON(str: string): unknown;
function parseJSON<T>(str: string, schema: (data: unknown) => T): T;
function parseJSON<T>(str: string, schema?: (data: unknown) => T): T | unknown {
    const data = JSON.parse(str);
    return schema ? schema(data) : data;
}

const user = parseJSON<{ name: string }>('{"name": "张三"}', (d: any) => d);
// user 类型：{ name: string }
```

## 常见问题

### Q1：可选参数和默认参数有什么区别？

```typescript
// 可选参数 — 值可能是 undefined
function fn1(name?: string) {
    console.log(name);  // string | undefined
}

// 默认参数 — 一定有值
function fn2(name: string = '匿名') {
    console.log(name);  // string
}
```

### Q2：函数重载和联合类型有什么区别？

**重载用于"不同输入对应不同输出"的场景，联合类型用于"可能返回多种类型之一"：**

```typescript
// 重载 — 编译器知道确切返回类型
function parse(x: string): number;
function parse(x: number): string;
const a = parse('123');  // 类型是 number

// 联合 — 编译器只知道可能是其中一种
function parse(x: string | number): string | number { }
const b = parse('123');  // 类型是 string | number
```

### Q3：什么时候用 void，什么时候用 undefined？

**void 用于函数返回值（表示不关心返回值），undefined 用于变量类型：**

```typescript
function log(msg: string): void { console.log(msg); }  // ✅
function log(msg: string): undefined { console.log(msg); }  // ❌ 不推荐
```
