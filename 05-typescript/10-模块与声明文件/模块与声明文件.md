## 17.1 模块导出导入

## 17.2 声明文件 .d.ts

### 什么是声明文件

声明文件为 JavaScript 代码提供类型信息，以 `.d.ts` 结尾。

### 基本用法

```typescript
// utils.js
function formatDate(date) {
    return date.toISOString().split('T')[0];
}

// utils.d.ts
declare function formatDate(date: Date): string;

export { formatDate };
```

## 17.3 为 JS 库编写声明

### 完整示例

```typescript
// my-lib.js
function createUser(name, age) {
    return { name, age, createdAt: new Date() };
}

function getUsers() {
    return [];
}

module.exports = { createUser, getUsers };

// my-lib.d.ts
interface User {
    name: string;
    age: number;
    createdAt: Date;
}

declare function createUser(name: string, age: number): User;
declare function getUsers(): User[];

export { createUser, getUsers };
```

## 17.5 全局声明

### 全局变量

```typescript
// global.d.ts
declare const API_URL: string;
declare const DEBUG: boolean;

// 使用
console.log(API_URL);
```

### 全局接口

```typescript
// global.d.ts
interface Window {
    myPlugin: {
        init(): void;
    };
}

// 使用
window.myPlugin.init();
```

## 17.6 命名空间

### 声明命名空间

```typescript
declare namespace MathUtils {
    function add(a: number, b: number): number;
    function subtract(a: number, b: number): number;
    
    namespace Internal {
        function helper(): void;
    }
}

// 使用
MathUtils.add(1, 2);
MathUtils.Internal.helper();
```

## 17.7 实践练习

### 练习 1：编写声明文件

```typescript
// math.js
function add(a, b) {
    return a + b;
}

function multiply(a, b) {
    return a * b;
}

module.exports = { add, multiply };

// math.d.ts
declare function add(a: number, b: number): number;
declare function multiply(a: number, b: number): number;

export { add, multiply };
```

### 练习 2：全局声明

```typescript
// global.d.ts
declare global {
    interface String {
        capitalize(): string;
        trim(): string;
    }
}

export {};
```
