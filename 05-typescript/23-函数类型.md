## 7.1 函数基础类型

### 参数类型注解

```typescript
function add(a: number, b: number): number {
    return a + b;
}

add(1, 2);      // ✅
add('1', '2');  // ❌
```

### 返回值类型注解

```typescript
function greet(name: string): string {
    return `Hello, ${name}!`;
}

// void 表示无返回值
function log(message: string): void {
    console.log(message);
}
```

### 类型推断

```typescript
// 返回类型会被推断
function add(a: number, b: number) {
    return a + b;  // 推断为 number
}
```

## 7.2 可选参数

### 使用 ?

```typescript
function greet(name: string, greeting?: string): string {
    if (greeting) {
        return `${greeting}, ${name}!`;
    }
    return `Hello, ${name}!`;
}

greet('Tom');              // "Hello, Tom!"
greet('Tom', 'Hi');       // "Hi, Tom!"
```

### 多个可选参数

```typescript
function createUser(name: string, age?: number, email?: string) {
    // ...
}

createUser('Tom');
createUser('Tom', 25);
createUser('Tom', 25, 'tom@example.com');
```

## 7.3 默认参数

```typescript
function greet(name: string, greeting: string = 'Hello'): string {
    return `${greeting}, ${name}!`;
}

greet('Tom');              // "Hello, Tom!"
greet('Tom', 'Hi');       // "Hi, Tom!"
```

### 默认参数 vs 可选参数

```typescript
// 可选参数：可能是 undefined
function fn1(name: string, age?: number) { }

// 默认参数：一定有值
function fn2(name: string, age: number = 18) { }
```

## 7.4 剩余参数

### 使用 ...

```typescript
function sum(...numbers: number[]): number {
    return numbers.reduce((a, b) => a + b, 0);
}

sum(1, 2);         // 3
sum(1, 2, 3, 4);   // 10
sum();             // 0
```

### 组合使用

```typescript
function greet(greeting: string, ...names: string[]): string {
    return `${greeting}, ${names.join(', ')}!`;
}

greet('Hello', 'Tom', 'Jerry', 'Mike');
// "Hello, Tom, Jerry, Mike!"
```

## 7.5 函数重载

### 什么是函数重载

同一个函数名，不同的参数类型或参数数量：

```typescript
// 重载签名
function reverse(x: string): string;
function reverse(x: number): number;

// 实现
function reverse(x: string | number): string | number {
    if (typeof x === 'string') {
        return x.split('').reverse().join('');
    }
    return Number(x.toString().split('').reverse().join(''));
}

reverse('hello');  // "olleh"
reverse(123);      // 321
```

### 实际应用

```typescript
// 根据参数类型返回不同结果
function parse(input: string): string[];
function parse(input: number): number[];
function parse(input: string | number): string[] | number[] {
    return typeof input === 'string' 
        ? input.split('') 
        : [input];
}
```

### 重载示例：用户查询

```typescript
function getUser(id: number): User;
function getUser(name: string): User;
function getUser(idOrName: number | string): User {
    if (typeof idOrName === 'number') {
        return { id: idOrName, name: 'User' + idOrName };
    }
    return { id: 1, name: idOrName };
}

getUser(1);
getUser('Tom');
```

## 7.6 this 类型

### 显式 this 类型

```typescript
interface User {
    name: string;
    greet(this: User): string;
}

const user: User = {
    name: 'Tom',
    greet(this: User) {
        return `Hello, I'm ${this.name}`;
    }
};
```

### 实际应用

```typescript
class Counter {
    count: number = 0;
    
    increment(this: Counter): void {
        this.count++;
    }
}
```

## 7.7 函数类型

### 作为变量类型

```typescript
// 函数类型注解
let add: (a: number, b: number) => number;

add = (a, b) => a + b;
add = function(a, b) {
    return a + b;
};
```

### 作为参数类型

```typescript
function calculate(a: number, b: number, fn: (a: number, b: number) => number): number {
    return fn(a, b);
}

calculate(10, 5, (a, b) => a + b);  // 15
calculate(10, 5, (a, b) => a - b);  // 5
```

### 作为返回类型

```typescript
function getAdder(): (a: number, b: number) => number {
    return (a, b) => a + b;
}

const add = getAdder();
add(1, 2);  // 3
```

## 7.8 实践练习

### 练习 1：基础函数

```typescript
// 实现计算器函数
function calculate(a: number, b: number, operator: string): number {
    switch (operator) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/': return b !== 0 ? a / b : 0;
        default: return 0;
    }
}
```

### 练习 2：可选参数和默认参数

```typescript
// 创建用户函数
function createUser(name: string, age: number = 18, city?: string): string {
    let info = `${name}, ${age}岁`;
    if (city) {
        info += `, 来自${city}`;
    }
    return info;
}

createUser('Tom');              // "Tom, 18岁"
createUser('Tom', 25);          // "Tom, 25岁"
createUser('Tom', 25, 'Beijing'); // "Tom, 25岁, 来自Beijing"
```

### 练习 3：函数重载

```typescript
// 获取用户
function getUser(id: number): { id: number; name: string };
function getUser(name: string): { id: number; name: string };
function getUser(idOrName: number | string): { id: number; name: string } {
    if (typeof idOrName === 'number') {
        return { id: idOrName, name: 'User' + idOrName };
    }
    return { id: Math.floor(Math.random() * 1000), name: idOrName };
}
```

### 练习 4：回调函数

```typescript
// 数组处理函数
function processArray<T>(arr: T[], callback: (item: T, index: number) => void): void {
    arr.forEach((item, index) => callback(item, index));
}

processArray([1, 2, 3], (item, index) => {
    console.log(`Index: ${index}, Value: ${item}`);
});
```

## 本章小结

| 特性 | 语法 | 说明 |
|------|------|------|
| 参数类型 | `(param: type)` | 标注参数类型 |
| 返回类型 | `(): type` | 标注返回值类型 |
| 可选参数 | `param?: type` | 参数可能不存在 |
| 默认参数 | `param = value` | 参数默认值 |
| 剩余参数 | `...params: type[]` | 收集多余参数 |
| 函数重载 | `fn(): type` | 多个函数签名 |
| this 类型 | `fn(this: Type)` | 显式 this 类型 |
