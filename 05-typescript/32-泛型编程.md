## 11.1 为什么使用泛型

### 问题：代码重复

```typescript
// 为每种类型写一个函数
function identityNumber(arg: number): number {
    return arg;
}

function identityString(arg: string): string {
    return arg;
}

function identityBoolean(arg: boolean): boolean {
    return arg;
}
```

### 解决方案：泛型

```typescript
// 一个函数，支持所有类型
function identity<T>(arg: T): T {
    return arg;
}

identity<number>(123);    // number
identity<string>('hello'); // string
identity<boolean>(true);  // boolean
identity(123);            // 类型推断：number
```

## 11.2 泛型函数

### 基本语法

```typescript
function identity<T>(arg: T): T {
    return arg;
}
```

### 使用方式

```typescript
// 方式一：显式指定类型
const result1 = identity<string>('hello');

// 方式二：类型推断
const result2 = identity('hello');
```

### 多个类型参数

```typescript
function pair<K, V>(key: K, value: V): [K, V] {
    return [key, value];
}

const p = pair('name', 'Tom');  // [string, string]
const p2 = pair(1, { id: 1 });  // [number, { id: number }]
```

## 11.3 泛型接口

### 基本用法

```typescript
interface Container<T> {
    value: T;
    getValue(): T;
}

const stringContainer: Container<string> = {
    value: 'hello',
    getValue() {
        return this.value;
    }
};

const numberContainer: Container<number> = {
    value: 123,
    getValue() {
        return this.value;
    }
};
```

### 实际应用：API 响应

```typescript
interface ApiResponse<T> {
    code: number;
    message: string;
    data: T;
}

interface User {
    id: number;
    name: string;
}

const userResponse: ApiResponse<User> = {
    code: 200,
    message: 'success',
    data: { id: 1, name: 'Tom' }
};

const listResponse: ApiResponse<User[]> = {
    code: 200,
    message: 'success',
    data: [
        { id: 1, name: 'Tom' },
        { id: 2, name: 'Jerry' }
    ]
};
```

## 11.4 泛型类

### 基本用法

```typescript
class Container<T> {
    private value: T;
    
    constructor(value: T) {
        this.value = value;
    }
    
    getValue(): T {
        return this.value;
    }
}

const stringContainer = new Container('hello');
const numberContainer = new Container(123);
```

### 实际应用：栈

```typescript
class Stack<T> {
    private items: T[] = [];
    
    push(item: T): void {
        this.items.push(item);
    }
    
    pop(): T | undefined {
        return this.items.pop();
    }
    
    peek(): T | undefined {
        return this.items[this.items.length - 1];
    }
}

const numberStack = new Stack<number>();
numberStack.push(1);
numberStack.push(2);
console.log(numberStack.pop());  // 2
```

## 11.5 泛型约束

### extends 关键字

限制泛型必须满足某些条件：

```typescript
interface Lengthwise {
    length: number;
}

function logLength<T extends Lengthwise>(arg: T): T {
    console.log(arg.length);
    return arg;
}

logLength('hello');      // ✅ 字符串有 length
logLength([1, 2, 3]);    // ✅ 数组有 length
logLength({ length: 10 }); // ✅ 对象有 length
// logLength(123);       // ❌ 数字没有 length
```

### 约束属性

```typescript
interface HasId {
    id: number;
}

function getId<T extends HasId>(obj: T): number {
    return obj.id;
}

getId({ id: 1, name: 'Tom' });  // ✅
getId({ name: 'Tom' });         // ❌ 缺少 id
```

## 11.6 默认泛型参数

```typescript
interface ApiResponse<T = string, E = Error> {
    data: T;
    error: E | null;
}

// 使用默认参数
const response1: ApiResponse = {
    data: 'ok',
    error: null
};

// 覆盖默认参数
const response2: ApiResponse<number> = {
    data: 200,
    error: null
};

// 覆盖多个参数
const response3: ApiResponse<number[], CustomError> = {
    data: [1, 2, 3],
    error: new CustomError()
};
```

## 11.7 实践练习

### 练习 1：泛型函数

```typescript
// 通用数组去重函数
function unique<T>(arr: T[]): T[] {
    return Array.from(new Set(arr));
}

unique([1, 2, 2, 3, 3, 3]);  // [1, 2, 3]
unique(['a', 'b', 'a']);      // ['a', 'b']
```

### 练习 2：泛型接口

```typescript
interface Cache<T> {
    get(key: string): T | undefined;
    set(key: string, value: T): void;
    delete(key: string): void;
}

class MapCache<T> implements Cache<T> {
    private cache = new Map<string, T>();
    
    get(key: string): T | undefined {
        return this.cache.get(key);
    }
    
    set(key: string, value: T): void {
        this.cache.set(key, value);
    }
    
    delete(key: string): void {
        this.cache.delete(key);
    }
}
```

### 练习 3：泛型约束

```typescript
interface HasName {
    name: string;
}

function greet<T extends HasName>(obj: T): string {
    return `Hello, ${obj.name}!`;
}

greet({ name: 'Tom', age: 25 });  // ✅
greet({ name: 'Jerry' });         // ✅
// greet({ age: 25 });            // ❌
```

### 练习 4：泛型类

```typescript
class EventEmitter<T> {
    private listeners: ((data: T) => void)[] = [];
    
    on(listener: (data: T) => void): void {
        this.listeners.push(listener);
    }
    
    emit(data: T): void {
        this.listeners.forEach(listener => listener(data));
    }
}

const emitter = new EventEmitter<string>();
emitter.on((msg) => console.log(msg));
emitter.emit('Hello!');  // 输出: Hello!
```

## 本章小结

| 概念 | 语法 | 说明 |
|------|------|------|
| 泛型函数 | `fn<T>(arg: T): T` | 参数化类型 |
| 泛型接口 | `interface<T>` | 参数化接口 |
| 泛型类 | `class<T>` | 参数化类 |
| 泛型约束 | `extends` | 限制类型范围 |
| 默认参数 | `T = string` | 默认类型 |
