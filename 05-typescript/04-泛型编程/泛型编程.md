## L1 理解层：会用

### 为什么需要泛型

**没有泛型的问题：**

```typescript
// ❌ 问题 1：用 any，丢失类型信息
function first(arr: any[]): any {
    return arr[0];
}
const result = first([1, 2, 3]);  // result 是 any，没有提示

// ❌ 问题 2：为每种类型写一个函数
function firstNumber(arr: number[]): number | undefined { return arr[0]; }
function firstString(arr: string[]): string | undefined { return arr[0]; }
function firstBoolean(arr: boolean[]): boolean | undefined { return arr[0]; }
// 代码重复，无法维护
```

**最简示例（1-3行）：**

```typescript
function first<T>(arr: T[]): T | undefined { return arr[0]; }
first([1, 2, 3]);       // T = number，返回 number | undefined
first(['a', 'b']);      // T = string，返回 string | undefined
```

**详细示例：**

```typescript
// 泛型函数
function identity<T>(value: T): T {
    return value;
}
identity(42);           // T 推断为 number
identity('hello');      // T 推断为 string

// 多类型参数
function pair<A, B>(first: A, second: B): [A, B] {
    return [first, second];
}
pair(1, 'hello');       // [number, string]

// 泛型接口
interface Repository<T> {
    findById(id: number): T | undefined;
    findAll(): T[];
    save(item: T): void;
    delete(id: number): boolean;
}

// 泛型类
class Stack<T> {
    private items: T[] = [];

    push(item: T): void { this.items.push(item); }
    pop(): T | undefined { return this.items.pop(); }
    peek(): T | undefined { return this.items[this.items.length - 1]; }
    get size(): number { return this.items.length; }
    isEmpty(): boolean { return this.items.length === 0; }
}

const numStack = new Stack<number>();
numStack.push(1);
numStack.push(2);
console.log(numStack.pop());  // 2

// 泛型约束 — 限制类型必须满足条件
function logLength<T extends { length: number }>(value: T): void {
    console.log(value.length);
}
logLength('hello');     // ✅ 5
logLength([1, 2, 3]);   // ✅ 3
// logLength(42);       // ❌ number 没有 length 属性
```

### keyof 与索引访问

**最简示例：**

```typescript
type Keys = keyof { name: string; age: number };  // 'name' | 'age'
```

**详细示例：**

```typescript
// keyof — 获取对象键的联合类型
interface User {
    id: number;
    name: string;
    email: string;
}
type UserKeys = keyof User;  // 'id' | 'name' | 'email'

// 泛型 + keyof — 类型安全的属性访问
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}
const user: User = { id: 1, name: '张三', email: 'zhang@example.com' };
getProperty(user, 'name');   // ✅ string
getProperty(user, 'age');    // ❌ 'age' 不在 keyof User 中

// T[K] — 索引访问类型
type UserName = User['name'];        // string
type UserFields = User[keyof User];  // number | string
```

## L2 实践层：用好

### 泛型速查

| 语法 | 说明 | 示例 |
|------|------|------|
| `<T>` | 类型参数 | `function fn<T>(x: T): T` |
| `<T, U>` | 多类型参数 | `function pair<T, U>(a: T, b: U)` |
| `extends` | 泛型约束 | `<T extends { id: number }>` |
| `keyof T` | 键的联合类型 | `keyof User` → `'id' \| 'name'` |
| `T[K]` | 索引访问类型 | `User['name']` → `string` |
| `Partial<T>` | 所有属性可选 | `{ name?: string }` |
| `Pick<T, K>` | 挑选属性 | `Pick<User, 'name'>` |

### 反模式：不要这样做

```typescript
// ❌ 错误：滥用 any 代替泛型
function getFirst(arr: any[]): any {
    return arr[0];
}

// ✅ 正确：用泛型保持类型信息
function getFirst<T>(arr: T[]): T | undefined {
    return arr[0];
}
```

```typescript
// ❌ 错误：泛型约束过宽（没有意义）
function process<T extends object>(data: T) { }

// ✅ 正确：精确约束
function process<T extends { id: number; name: string }>(data: T) {
    console.log(data.id, data.name);
}
```

```typescript
// ❌ 错误：泛型参数过多（超过 3 个可读性差）
function merge<A, B, C, D, E>(a: A, b: B, c: C, d: D, e: E) { }

// ✅ 正确：用接口或对象
interface MergeOptions { a: A; b: B; c: C; d: D; e: E }
function merge(opts: MergeOptions) { }
```

### 适用场景

| 场景 | 推荐方案 | 示例 |
|------|---------|------|
| 数组/集合操作 | 泛型函数 | `first<T>(arr: T[])` |
| 数据仓库/服务 | 泛型接口 | `Repository<T>` |
| 数据结构 | 泛型类 | `Stack<T>`, `Queue<T>` |
| 工具函数 | 泛型 + keyof | `getProperty(obj, key)` |
| API 响应 | 泛型包装 | `ApiResponse<T>` |

## L3 专家层：深入

### 泛型在 React 中的应用

```typescript
// 泛型组件 — 类型安全的列表渲染
interface ListProps<T> {
    items: T[];
    render: (item: T) => React.ReactNode;
    keyExtractor: (item: T) => string;
}

function List<T>({ items, render, keyExtractor }: ListProps<T>) {
    return (
        <ul>
            {items.map(item => (
                <li key={keyExtractor(item)}>{render(item)}</li>
            ))}
        </ul>
    );
}

// 使用
interface User { id: number; name: string }
<List<User>
    items={users}
    keyExtractor={u => String(u.id)}
    render={u => <span>{u.name}</span>}
/>
```

```typescript
// API 响应泛型
interface ApiResponse<T> {
    data: T;
    status: number;
    message: string;
}

async function fetchAPI<T>(url: string): Promise<ApiResponse<T>> {
    const res = await fetch(url);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return res.json();
}

// 使用
const { data: user } = await fetchAPI<User>('/api/users/1');
// user 的类型是 User，有完整的类型提示
```

### infer — 类型推断

```typescript
// 提取函数返回值类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;
type Result = ReturnType<() => string>;  // string

// 提取函数参数类型
type Parameters<T> = T extends (...args: infer P) => any ? P : never;
type Params = Parameters<(name: string, age: number) => void>;  // [string, number]

// 提取 Promise 的值
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;
type Value = UnwrapPromise<Promise<string>>;  // string

// 提取数组元素类型
type ElementType<T> = T extends (infer U)[] ? U : T;
type Item = ElementType<number[]>;  // number
```

```
泛型知识关联：

┌─────────────┐     ┌─────────────┐     ┌─────────────┐
│  泛型基础   │────→│  泛型约束   │────→│  条件类型   │
│  函数/      │     │  extends/   │     │  infer/     │
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **泛型** | 类型参数化，编写通用代码 | `<T>(x: T): T` |
| **泛型约束** | 限制泛型必须满足的条件 | `<T extends { length: number }>` |
| **keyof** | 获取对象键的联合类型 | `keyof User` → `'id' \| 'name'` |
| **索引访问** | 通过键访问属性的类型 | `User['name']` → `string` |
| **infer** | 在条件类型中推断未知类型 | `T extends Promise<infer U>` |

## 实践练习

```typescript
// 练习 1：实现 Pick
type MyPick<T, K extends keyof T> = {
    [P in K]: T[P];
};

type UserName = MyPick<User, 'name' | 'email'>;
// { name: string; email: string }

// 练习 2：实现 Omit
type MyOmit<T, K extends keyof T> = {
    [P in Exclude<keyof T, K>]: T[P];
};

type UserWithoutId = MyOmit<User, 'id'>;
// { name: string; email: string }

// 练习 3：类型安全的 event emitter
type EventMap = {
    'user:login': { userId: number };
    'user:logout': { userId: number };
    'error': { message: string };
};

class TypedEmitter<T extends Record<string, any>> {
    private listeners: { [K in keyof T]?: Array<(data: T[K]) => void> } = {};

    on<K extends keyof T>(event: K, listener: (data: T[K]) => void) {
        if (!this.listeners[event]) this.listeners[event] = [];
        this.listeners[event]!.push(listener);
    }

    emit<K extends keyof T>(event: K, data: T[K]) {
        this.listeners[event]?.forEach(fn => fn(data));
    }
}

const emitter = new TypedEmitter<EventMap>();
emitter.on('user:login', data => console.log(data.userId));  // ✅
// emitter.on('user:login', data => console.log(data.foo));  // ❌ foo 不存在
```

## 常见问题

### Q1：泛型和 any 有什么区别？

**any 丢失类型信息，泛型保持类型信息：**

```typescript
// any — 编译器不知道返回什么
function first(arr: any[]): any { return arr[0]; }
const x = first([1, 2, 3]);  // x 是 any，没有提示

// 泛型 — 编译器推断出具体类型
function first<T>(arr: T[]): T | undefined { return arr[0]; }
const y = first([1, 2, 3]);  // y 是 number，有完整提示
```

### Q2：什么时候需要显式指定泛型参数？

**大多数情况 TypeScript 能自动推断，以下情况需要显式指定：**

```typescript
// 自动推断
const result = first([1, 2, 3]);  // T = number

// 需要显式指定
const empty = first<number>([]);  // 空数组无法推断

// API 调用
const user = await fetchAPI<User>('/api/users/1');
```

### Q3：泛型约束 overextends 和类型断言选哪个？

**优先用泛型约束（安全），类型断言只在确定时使用：**

```typescript
// ✅ 泛型约束（编译器检查）
function getId<T extends { id: number }>(obj: T) {
    return obj.id;
}

// ❌ 类型断言（绕过检查）
function getIdUnsafe(obj: any) {
    return (obj as { id: number }).id;
}
```
