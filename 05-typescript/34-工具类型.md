## 13.1 Partial - 所有属性可选

### 用法

```typescript
interface User {
    id: number;
    name: string;
    email: string;
}

// 所有属性变为可选
type PartialUser = Partial<User>;

// 等同于
// { id?: number; name?: string; email?: string; }

const user: PartialUser = { name: 'Tom' };
```

### 实现原理

```typescript
type MyPartial<T> = {
    [P in keyof T]?: T[P];
};
```

## 13.3 Readonly - 所有属性只读

### 用法

```typescript
interface User {
    id: number;
    name: string;
}

// 所有属性变为只读
type ReadonlyUser = Readonly<User>;

// 等同于
// { readonly id: number; readonly name: string; }

const user: ReadonlyUser = { id: 1, name: 'Tom' };
// user.id = 2;  // ❌ 错误：只读
```

### 实现原理

```typescript
type MyReadonly<T> = {
    readonly [P in keyof T]: T[P];
};
```

## 13.6 Record - 键值对类型

### 用法

```typescript
type Role = 'admin' | 'user' | 'guest';

const permissions: Record<Role, string[]> = {
    admin: ['read', 'write', 'delete'],
    user: ['read', 'write'],
    guest: ['read']
};
```

### 实现原理

```typescript
type MyRecord<K extends keyof any, T> = {
    [P in K]: T;
};
```

## 13.7 Exclude / Extract

### Exclude - 排除类型

```typescript
type T = 'a' | 'b' | 'c' | 'd';

// 排除 'a' 和 'c'
type Excluded = Exclude<T, 'a' | 'c'>;

// 结果：'b' | 'd'
```

### Extract - 提取类型

```typescript
type T = 'a' | 'b' | 'c' | 'd';

// 提取 'a' 和 'c'
type Extracted = Extract<T, 'a' | 'c'>;

// 结果：'a' | 'c'
```

---

## 13.8 NonNullable - 排除 null/undefined

```typescript
type T = string | null | undefined | number;

// 排除 null 和 undefined
type NonNull = NonNullable<T>;

// 结果：string | number
```

## 13.9 ReturnType / Parameters

### ReturnType - 获取返回类型

```typescript
function getUser() {
    return { id: 1, name: 'Tom' };
}

type User = ReturnType<typeof getUser>;
// { id: number; name: string; }
```

### Parameters - 获取参数类型

```typescript
function createUser(name: string, age: number) {
    return { name, age };
}

type Params = Parameters<typeof createUser>;
// [string, number]
```

## 13.10 自定义工具类型

### DeepPartial - 深度可选

```typescript
type DeepPartial<T> = {
    [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

interface Company {
    name: string;
    address: {
        city: string;
        street: string;
    };
}

type PartialCompany = DeepPartial<Company>;
// { name?: string; address?: { city?: string; street?: string; } }
```

### DeepReadonly - 深度只读

```typescript
type DeepReadonly<T> = {
    readonly [P in keyof T]: T[P] extends object ? DeepReadonly<T[P]> : T[P];
};

interface Company {
    name: string;
    address: {
        city: string;
    };
}

type ReadonlyCompany = DeepReadonly<Company>;
```

## 13.11 实践练习

### 练习 1：使用工具类型

```typescript
interface User {
    id: number;
    name: string;
    email: string;
    age: number;
}

// 1. Partial - 创建用户时，部分字段可选
type CreateUser = Partial<User>;

// 2. Pick - 只获取用户基本信息
type UserBasicInfo = Pick<User, 'id' | 'name'>;

// 3. Omit - 排除敏感信息
type PublicUser = Omit<User, 'email'>;
```

### 练习 2：自定义工具类型

```typescript
// 提取函数类型
type FnReturn<T> = T extends (...args: any[]) => infer R ? R : never;

function getData() {
    return { id: 1, name: 'Tom' };
}

type Data = FnReturn<typeof getData>;  // { id: number; name: string }
```

### 练习 3：Record 应用

```typescript
type UserRole = 'admin' | 'editor' | 'viewer';

const rolePermissions: Record<UserRole, { canEdit: boolean; canDelete: boolean }> = {
    admin: { canEdit: true, canDelete: true },
    editor: { canEdit: true, canDelete: false },
    viewer: { canEdit: false, canDelete: false }
};
```

## 本章小结

| 工具类型 | 作用 | 示例 |
|----------|------|------|
| Partial | 所有属性可选 | `Partial<User>` |
| Required | 所有属性必填 | `Required<User>` |
| Readonly | 所有属性只读 | `Readonly<User>` |
| Pick | 选取属性 | `Pick<User, 'id'>` |
| Omit | 排除属性 | `Omit<User, 'pwd'>` |
| Record | 键值对 | `Record<K, V>` |
| Exclude | 排除类型 | `Exclude<A, B>` |
| Extract | 提取类型 | `Extract<A, B>` |
| NonNullable | 排除空值 | `NonNullable<T>` |
| ReturnType | 返回类型 | `ReturnType<fn>` |
| Parameters | 参数类型 | `Parameters<fn>` |
