## 6.1 类型守卫概述

### 什么是类型守卫

类型守卫（Type Guard）是在运行时检查变量类型的技术，它可以帮助 TypeScript 在特定作用域内收窄（narrow）联合类型。

```typescript
// 联合类型
function printId(id: number | string) {
    // 类型守卫 - 使用 typeof
    if (typeof id === 'number') {
        // 在这个 if 块内，id 是 number 类型
        console.log(id.toFixed(2));
    } else {
        // 在 else 块内，id 是 string 类型
        console.log(id.toUpperCase());
    }
}
```

### 类型收窄的方式

| 方式 | 说明 | 示例 |
|------|------|------|
| `typeof` | 检查基本类型 | `typeof x === 'string'` |
| `instanceof` | 检查构造函数实例 | `x instanceof Date` |
| `in` | 检查属性是否存在 | `'name' in x` |
| 类型谓词 | 自定义类型检查 | `x is string` |
| 相等性检查 | 检查值是否相等 | `x === null` |
| 真值检查 | 检查真假值 | `if (x)` |

## 6.2 typeof 类型守卫

### 基础用法

```typescript
function processValue(value: string | number | boolean): string {
    if (typeof value === 'string') {
        return `字符串：${value}`;
    } else if (typeof value === 'number') {
        return `数字：${value.toFixed(2)}`;
    } else {
        return `布尔值：${value ? 'true' : 'false'}`;
    }
}
```

### 实际应用

```typescript
type ValidationRule =
    | { type: 'required'; message: string }
    | { type: 'minLength'; value: number; message: string }
    | { type: 'pattern'; value: RegExp; message: string };

function validate(value: unknown, rules: ValidationRule[]): boolean {
    for (const rule of rules) {
        if (rule.type === 'required') {
            if (value === null || value === undefined || value === '') {
                return false;
            }
        } else if (rule.type === 'minLength') {
            if (typeof value !== 'string' || value.length < rule.value) {
                return false;
            }
        } else if (rule.type === 'pattern' && typeof value === 'string') {
            if (!rule.value.test(value)) {
                return false;
            }
        }
    }
    return true;
}
```

## 6.3 instanceof 守卫

### 基础用法

```typescript
class Bird {
    fly() {
        console.log('🐦 鸟在飞翔');
    }
}

class Fish {
    swim() {
        console.log('🐟 鱼在游泳');
    }
}

function move(animal: Bird | Fish) {
    if (animal instanceof Bird) {
        animal.fly();
    } else {
        animal.swim();
    }
}
```

### 内置对象检查

```typescript
// 日期处理
function formatDate(value: Date | string | number): string {
    if (value instanceof Date) {
        return value.toISOString();
    }
    return new Date(value).toISOString();
}

// 错误处理
function handleError(error: Error | string | unknown): void {
    if (error instanceof Error) {
        console.error('Error 对象:', error.message);
    } else if (typeof error === 'string') {
        console.error('字符串错误:', error);
    } else {
        console.error('未知错误:', error);
    }
}
```

## 6.4 in 操作符守卫

### 基础用法

```typescript
interface User {
    name: string;
    email: string;
}

interface Product {
    title: string;
    price: number;
}

function describe(value: User | Product): string {
    if ('name' in value) {
        return `用户：${value.name} (${value.email})`;
    } else {
        return `商品：${value.title} - ¥${value.price}`;
    }
}
```

### 可辨识联合

```typescript
type Shape =
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; side: number }
    | { kind: 'triangle'; base: number; height: number };

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'square':
            return shape.side ** 2;
        case 'triangle':
            return (shape.base * shape.height) / 2;
        default:
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}
```

## 6.5 类型谓词（Type Predicate）

### 自定义类型守卫

```typescript
// 类型谓词语法：parameterName is Type
function isString(value: unknown): value is string {
    return typeof value === 'string';
}

function process(value: unknown) {
    if (isString(value)) {
        // TypeScript 知道 value 是 string 类型
        console.log(value.toUpperCase());
    }
}
```

### 实用类型守卫函数

```typescript
// 检查是否为数字
function isNumber(value: unknown): value is number {
    return typeof value === 'number' && !isNaN(value);
}

// 检查是否为数组
function isArray(value: unknown): value is unknown[] {
    return Array.isArray(value);
}

// 检查是否为对象
function isObject(value: unknown): value is Record<string, unknown> {
    return typeof value === 'object' && value !== null && !Array.isArray(value);
}

// 使用
function processAny(value: unknown) {
    if (isNumber(value)) {
        console.log('数字:', value.toFixed(2));
    } else if (isArray(value)) {
        console.log('数组长度:', value.length);
    } else if (isObject(value)) {
        console.log('对象键:', Object.keys(value));
    }
}
```

## 6.6 完整性检查

### exhaustiveness checking

```typescript
type Status = 'pending' | 'loading' | 'success' | 'error';

function handleStatus(status: Status): string {
    switch (status) {
        case 'pending':
            return '等待中';
        case 'loading':
            return '加载中';
        case 'success':
            return '成功';
        case 'error':
            return '错误';
        default:
            // 如果新增类型，这里会报错
            const _exhaustive: never = status;
            return _exhaustive;
    }
}

// 如果新增 'cancelled' 类型，default 分支会报错
// 提醒你需要处理新类型
```

### 使用断言函数

```typescript
// 断言函数
function assertNever(value: never): never {
    throw new Error(`Unexpected value: ${value}`);
}

function handleStatus(status: Status): string {
    switch (status) {
        case 'pending':
            return '等待中';
        case 'loading':
            return '加载中';
        case 'success':
            return '成功';
        case 'error':
            return '错误';
        default:
            assertNever(status);
    }
}
```

## 6.7 实践练习：状态机

```typescript
// 定义状态
type ConnectionState =
    | { status: 'disconnected' }
    | { status: 'connecting'; progress: number }
    | { status: 'connected'; sessionId: string }
    | { status: 'error'; error: Error };

// 类型守卫
function isConnecting(state: ConnectionState): state is Extract<ConnectionState, { status: 'connecting' }> {
    return state.status === 'connecting';
}

function isConnected(state: ConnectionState): state is Extract<ConnectionState, { status: 'connected' }> {
    return state.status === 'connected';
}

// 状态处理
function handleConnection(state: ConnectionState): string {
    switch (state.status) {
        case 'disconnected':
            return '已断开连接';
        case 'connecting':
            return `连接中 ${state.progress}%`;
        case 'connected':
            return `已连接，会话 ID: ${state.sessionId}`;
        case 'error':
            return `连接错误：${state.error.message}`;
        default:
            const _exhaustive: never = state;
            return _exhaustive;
    }
}

// 使用
const state1: ConnectionState = { status: 'disconnected' };
const state2: ConnectionState = { status: 'connecting', progress: 50 };
const state3: ConnectionState = { status: 'connected', sessionId: 'abc123' };

console.log(handleConnection(state1));
console.log(handleConnection(state2));
console.log(handleConnection(state3));
```

## 6.8 常见问答

### Q1: 类型守卫和类型断言有什么区别？

**答：**
- 类型守卫：运行时检查，类型安全
- 类型断言：编译时提示，需要谨慎使用

```typescript
// 类型守卫
if (typeof value === 'string') {
    console.log(value.length);  // 安全
}

// 类型断言
console.log((value as string).length);  // 可能不安全
```

### Q2: 如何为联合类型编写类型守卫？

```typescript
type Result =
    | { success: true; data: string }
    | { success: false; error: Error };

function isSuccess(result: Result): result is Extract<Result, { success: true }> {
    return result.success === true;
}

// 使用
if (isSuccess(result)) {
    console.log(result.data);  // 类型收窄为 { success: true; data: string }
}
```
