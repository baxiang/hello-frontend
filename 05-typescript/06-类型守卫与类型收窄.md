## 6.2 typeof 类型守卫 - 检查基本类型

### 6.2.1 什么是 typeof？

### 6.2.2 基本用法

```typescript
// 联合类型：可能是字符串或数字
function processValue(value: string | number): string {
    // 使用 typeof 检查类型
    if (typeof value === 'string') {
        // 在这个分支，TypeScript 知道 value 是 string
        return `字符串：${value.toUpperCase()}`;
    } else {
        // 在这个分支，TypeScript 知道 value 是 number
        return `数字：${value.toFixed(2)}`;
    }
}

// 测试

console.log(processValue(123.456));  // 数字：123.46
```

### 6.2.4 实际应用：表单验证

**先理解什么是可辨识联合？**

```
┌─────────────────────────────────────────────────────────────┐
│  可辨识联合 = 用一个"标签"区分不同类型                        │
│                                                             │
│  就像：                                                     │
│  - 交通信号灯：红灯 / 绿灯 / 黄灯                           │
│  - 订单状态：待支付 / 已发货 / 已完成                        │
│  - 验证规则：必填 / 最小长度 / 格式验证                     │
│                                                             │
│  关键：每个类型都有一个共同的"type"属性                     │
│  通过检查这个 type 来确定具体是哪个类型                     │
└─────────────────────────────────────────────────────────────┘
```

**ValidationRule 解读：**

```typescript
// 这是一个"验证规则"的集合，有3种可能：
type ValidationRule =
    // 第1种：必填规则
    | { type: 'required'; message: string }
    //         ↑ 标签      ↑ 错误信息
    //         就像：这是"必填"类型的规则
    
    // 第2种：最小长度规则
    | { type: 'minLength'; value: number; message: string }
    //         ↑ 标签        ↑ 最小长度   ↑ 错误信息
    //         就像：这是"最小长度"类型的规则
    
    // 第3种：格式规则
    | { type: 'pattern'; value: RegExp; message: string };
    //         ↑ 标签        ↑ 正则表达式  ↑ 错误信息
    //         就像：这是"格式"类型的规则
```

**生活例子：**

```
想象一个"规则本"，里面有3种规则：

规则1：必填检查
{ type: 'required', message: '不能为空' }
  ↓
检查：输入框有没有值？

规则2：最小长度
{ type: 'minLength', value: 5, message: '至少5个字符' }
  ↓
检查：输入的字符数 >= 5？

规则3：格式验证
{ type: 'pattern', value: /^\d+$/, message: '必须是数字' }
  ↓
检查：输入是否符合正则表达式？
```

```typescript
type ValidationRule =
    | { type: 'required'; message: string }
    | { type: 'minLength'; value: number; message: string }
    | { type: 'pattern'; value: RegExp; message: string };

function validate(value: unknown, rules: ValidationRule[]): { valid: boolean; message?: string } {
    for (const rule of rules) {
        // required 规则
        if (rule.type === 'required') {
            if (value === null || value === undefined || value === '') {
                return { valid: false, message: rule.message };
            }
        }
        // minLength 规则
        else if (rule.type === 'minLength') {
            // 先检查是否是字符串
            if (typeof value !== 'string' || value.length < rule.value) {
                return { valid: false, message: rule.message };
            }
        }
        // pattern 规则
        else if (rule.type === 'pattern') {
            if (typeof value === 'string' && !rule.value.test(value)) {
                return { valid: false, message: rule.message };
            }
        }
    }
    return { valid: true };
}

// 使用
const result = validate('abc', [
    { type: 'required', message: '不能为空' },
    { type: 'minLength', value: 5, message: '至少5个字符' }
]);
console.log(result);  // { valid: false, message: '至少5个字符' }
```

## 6.3 instanceof 守卫 - 检查对象实例

### 6.3.1 什么是 instanceof？

```
┌─────────────────────────────────────────────────────────────┐
│  instanceof = 检查对象是不是某个"构造函数"创建的            │
│                                                             │
│  语法：object instanceof Constructor                        │
│                                                             │
│  就像：                                                     │
│  - 检查这辆车是不是 BMW 生产的                              │
│  - 检查这本书是不是某个出版社出版的                          │
│  - 检查这个人是不是学生                                     │
│                                                             │
│  适用场景：                                                 │
│  - 类（class）创建的对象                                    │
│  - 内置构造函数（Date, Array, Error 等）                   │
└─────────────────────────────────────────────────────────────┘
```

### 6.3.2 基本用法

```typescript
// 定义两个类
class Bird {
    fly() {
        console.log('🐦 鸟在飞翔');
    }
}

class Fish {
    swim() {
        console.log('🐟 鱼在游泳');
    }
}

// 处理函数
function move(animal: Bird | Fish) {
    // 使用 instanceof 检查
    if (animal instanceof Bird) {
        animal.fly();  // TypeScript 知道是 Bird
    } else {
        animal.swim(); // TypeScript 知道是 Fish
    }
}

// 测试
const bird = new Bird();
const fish = new Fish();

move(bird);  // 🐦 鸟在飞翔
move(fish);  // 🐟 鱼在游泳
```

### 6.3.3 instanceof vs typeof

```
┌─────────────────────────────────────────────────────────────┐
│                    typeof vs instanceof                      │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  typeof：                                                   │
│  - 检查基本类型                                             │
│  - typeof 'hello' === 'string'                            │
│  - typeof 123 === 'number'                                 │
│  - typeof [] === 'object'（数组也是 object！）              │
│                                                             │
│  instanceof：                                              │
│  - 检查对象实例                                             │
│  - [] instanceof Array                                     │
│  - new Date() instanceof Date                              │
│  - {} instanceof Object                                    │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 6.3.4 实际应用：日期处理

```typescript
// 日期处理函数
function formatDate(value: Date | string | number): string {
    if (value instanceof Date) {
        // 已经是 Date 对象
        return value.toISOString();
    } else {
        // 字符串或数字，转换为 Date
        return new Date(value).toISOString();
    }
}

// 测试
console.log(formatDate(new Date()));           // 2024...
console.log(formatDate('2024-01-01'));          // 2024...
console.logformatDate(1704067200000));         // 2024...
```

## 6.4 in 操作符守卫 - 检查属性是否存在

### 6.4.1 什么是 in？

```
┌─────────────────────────────────────────────────────────────┐
│  in = 检查对象里有没有某个属性                              │
│                                                             │
│  语法：'属性名' in 对象                                     │
│                                                             │
│  返回：true 或 false                                        │
│                                                             │
│  就像：                                                     │
│  - 检查房子里有没有钥匙                                      │
│  - 检查口袋里有没有手机                                      │
│  - 检查冰箱里有没有牛奶                                      │
│                                                             │
│  适用场景：                                                 │
│  - 两个对象类型有不同属性                                    │
│  - 检查可选属性是否存在                                      │
└─────────────────────────────────────────────────────────────┘
```

### 6.4.2 基本用法

```typescript
// 定义两个接口
interface User {
    name: string;
    email: string;
}

interface Product {
    title: string;
    price: number;
}

// 处理函数
function describe(value: User | Product): string {
    // 使用 in 检查属性
    if ('name' in value) {
        // 有 name 属性，说明是 User
        return `用户：${value.name} (${value.email})`;
    } else {
        // 没有 name 属性，说明是 Product
        return `商品：${value.title} - ¥${value.price}`;
    }
}

// 测试
const user: User = { name: '张三', email: 'zhang@example.com' };
const product: Product = { title: 'iPhone', price: 9999 };

console.log(describe(user));    // 用户：张三 (zhang@example.com)
console.log(describe(product)); // 商品：iPhone - ¥9999
```

### 6.4.3 流程图解

```
describe(value: User | Product)
              │
              ▼
       ┌─────────────┐
       │ 'name' in   │
       │    value    │
       └─────────────┘
         │       │
         是      否
         ▼       ▼
    ┌────────┐  ┌────────┐
    │  User  │  │Product│
    │ 分支   │  │ 分支  │
    └────────┘  └────────┘
```

### 6.4.4 可辨识联合 - 最佳实践

```
┌─────────────────────────────────────────────────────────────┐
│  可辨识联合 = 用一个"标签"属性区分不同类型                   │
│                                                             │
│  核心思想：                                                 │
│  - 每个类型都有一个共同的"标识"属性（通常是 kind 或 type） │
│  - 通过检查这个标识来区分类型                               │
│                                                             │
│  优点：                                                     │
│  - 代码更清晰                                              │
│  - 易于扩展                                                │
│  - 配合 switch 语句更优雅                                   │
└─────────────────────────────────────────────────────────────┘
```

```typescript
// 定义可辨识联合
type Shape =
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; side: number }
    | { kind: 'triangle'; base: number; height: number };

// 计算面积
function getArea(shape: Shape): number {
    // 使用 switch 检查 kind 属性
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'square':
            return shape.side ** 2;
        case 'triangle':
            return (shape.base * shape.height) / 2;
        default:
            // 完整性检查：确保处理了所有情况
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}

// 测试
const circle: Shape = { kind: 'circle', radius: 5 };
const square: Shape = { kind: 'square', side: 4 };
const triangle: Shape = { kind: 'triangle', base: 3, height: 4 };

console.log(getArea(circle));    // 78.54
console.log(getArea(square));    // 16
console.log(getArea(triangle));  // 6
```

## 6.5 类型谓词 - 自定义类型守卫

### 6.5.1 什么是类型谓词？

```
┌─────────────────────────────────────────────────────────────┐
│  类型谓词 = 自定义的"类型检查函数"                           │
│                                                             │
│  语法：function fn(x: unknown): x is Type                  │
│                                                             │
│  作用：                                                     │
│  - 告诉 TypeScript "这个变量确实是某个类型"                 │
│  - 可以封装复杂的类型检查逻辑                               │
│                                                             │
│  就像：                                                     │
│  - 自定义安检标准                                           │
│  - 专业资格认证                                             │
│  - 会员卡验证                                               │
└─────────────────────────────────────────────────────────────┘
```

### 6.5.2 基本语法

```typescript
// 定义类型谓词函数
// 语法：parameterName is Type
function isString(value: unknown): value is string {
    return typeof value === 'string';
}

// 使用
function process(value: unknown) {
    if (isString(value)) {
        // TypeScript 知道 value 是 string
        console.log(value.toUpperCase());
    }
}

// 测试
process('hello');  // HELLO
process(123);     // 不执行
```

### 6.5.3 对比：类型谓词 vs 类型断言

```
┌─────────────────────────────────────────────────────────────┐
│              类型谓词 vs 类型断言                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  类型谓词（推荐）：                                         │
│  function isString(x: unknown): x is string {               │
│      return typeof x === 'string';                         │
│  }                                                          │
│                                                             │
│  if (isString(value)) {                                    │
│      value.toUpperCase();  // ✅ 安全                      │
│  }                                                          │
│                                                             │
│  ─────────────────────────────────────────                 │
│                                                             │
│  类型断言（谨慎使用）：                                     │
│  (value as string).toUpperCase();  // ⚠️ 可能出错         │
│                                                             │
│  如果 value 不是字符串，运行时可能报错                      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 6.5.4 实用类型守卫函数

```typescript
// 检查是否为数字
function isNumber(value: unknown): value is number {
    return typeof value === 'number' && !isNaN(value);
}

// 检查是否为数组
function isArray(value: unknown): value is unknown[] {
    return Array.isArray(value);
}

// 检查是否为对象（非数组）
function isObject(value: unknown): value is Record<string, unknown> {
    return typeof value === 'object' && value !== null && !Array.isArray(value);
}

// 检查是否为有效日期
function isValidDate(value: unknown): value is Date {
    return value instanceof Date && !isNaN(value.getTime());
}

// 使用示例
function processAny(value: unknown): void {
    if (isNumber(value)) {
        console.log('数字:', value.toFixed(2));
    } else if (isArray(value)) {
        console.log('数组长度:', value.length);
    } else if (isObject(value)) {
        console.log('对象键:', Object.keys(value));
    } else if (isValidDate(value)) {
        console.log('日期:', value.toISOString());
    } else {
        console.log('其他类型');
    }
}

// 测试
processAny(123);                    // 数字: 123.00
processAny([1, 2, 3]);              // 数组长度: 3
processAny({ name: '张三' });       // 对象键: name
processAny(new Date());             // 日期: 2024...
processAny('hello');                // 其他类型
```

## 6.6 完整性检查 - 确保处理所有情况

### 6.6.1 什么是完整性检查？

```
┌─────────────────────────────────────────────────────────────┐
│  完整性检查 = 确保处理了所有可能的类型                       │
│                                                             │
│  目的：                                                     │
│  - 防止遗漏处理某些情况                                      │
│  - 新增类型时，编译器会报错提醒                             │
│                                                             │
│  就像：                                                     │
│  - 考试检查是否每道题都做了                                  │
│  - 清单检查是否每项都完成了                                  │
│  - 开关检查是否所有状态都处理了                              │
└─────────────────────────────────────────────────────────────┘
```

### 6.6.2 使用 never 进行完整性检查

```typescript
// 定义状态类型
type Status = 'pending' | 'loading' | 'success' | 'error';

// 处理函数
function handleStatus(status: Status): string {
    switch (status) {
        case 'pending':
            return '等待中';
        case 'loading':
            return '加载中';
        case 'success':
            return '成功';
        case 'error':
            return '错误';
        default:
            // 如果新增状态，这里会报错！
            const _exhaustive: never = status;
            return _exhaustive;
    }
}

// 测试
console.log(handleStatus('pending'));  // 等待中
console.log(handleStatus('loading'));  // 加载中
console.log(handleStatus('success'));  // 成功
console.log(handleStatus('error'));     // 错误
```

### 6.6.3 如果新增类型会怎样？

```typescript
// 假设我们新增了 'cancelled' 状态
type StatusNew = 'pending' | 'loading' | 'success' | 'error' | 'cancelled';

function handleStatusNew(status: StatusNew): string {
    switch (status) {
        case 'pending':
            return '等待中';
        case 'loading':
            return '加载中';
        case 'success':
            return '成功';
        case 'error':
            return '错误';
        // ⚠️ 忘记处理 'cancelled'
        default:
            // ❌ 编译错误！
            // TypeScript 会报错：Type '"cancelled"' is not assignable to type 'never'
            const _exhaustive: never = status;
            return _exhaustive;
    }
}
```

## 6.7 实践练习

### 练习 1：实现类型守卫

```typescript
// 定义联合类型
type Animal = 
    | { type: 'dog'; bark: () => void }
    | { type: 'cat'; meow: () => void }
    | { type: 'bird'; sing: () => void };

// 实现类型守卫
function isDog(animal: Animal): animal is { type: 'dog'; bark: () => void } {
    return animal.type === 'dog';
}

function isCat(animal: Animal): animal is { type: 'cat'; meow: () => void } {
    return animal.type === 'cat';
}

// 使用
function makeSound(animal: Animal): void {
    if (isDog(animal)) {
        animal.bark();
    } else if (isCat(animal)) {
        animal.meow();
    } else {
        animal.sing();
    }
}
```

### 练习 2：实现连接状态机

```typescript
// 定义状态
type ConnectionState =
    | { status: 'disconnected' }
    | { status: 'connecting'; progress: number }
    | { status: 'connected'; sessionId: string }
    | { status: 'error'; error: Error };

// 状态处理
function handleConnection(state: ConnectionState): string {
    switch (state.status) {
        case 'disconnected':
            return '已断开连接';
        case 'connecting':
            return `连接中 ${state.progress}%`;
        case 'connected':
            return `已连接，会话 ID: ${state.sessionId}`;
        case 'error':
            return `连接错误：${state.error.message}`;
        default:
            const _exhaustive: never = state;
            return _exhaustive;
    }
}

// 测试
const states: ConnectionState[] = [
    { status: 'disconnected' },
    { status: 'connecting', progress: 50 },
    { status: 'connected', sessionId: 'abc123' },
    { status: 'error', error: new Error('Network error') }
];

states.forEach(s => console.log(handleConnection(s)));
```

## 6.8 常见问答

### Q1: 类型守卫和类型断言有什么区别？

```typescript
// 类型守卫（安全）
if (typeof value === 'string') {
    console.log(value.length);  // ✅ TypeScript 知道是 string
}

// 类型断言（谨慎）
console.log((value as string).length);  // ⚠️ 假设 value 是 string，可能出错
```

### Q2: 什么时候用 typeof，什么时候用 instanceof？

```typescript
// 用 typeof：检查基本类型
if (typeof name === 'string') { }
if (typeof age === 'number') { }

// 用 instanceof：检查对象实例
if (date instanceof Date) { }
if (user instanceof User) { }
```

### Q3: 可辨识联合有什么优点？

```typescript
// 优点 1：代码清晰
type Shape = 
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; side: number };

// 优点 2：易于扩展
type Shape3D = Shape & { z: number }; // 添加 3D 支持

// 优点 3：完整性检查
function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'circle': return Math.PI * shape.radius ** 2;
        case 'square': return shape.side ** 2;
        // 忘记处理会报错
    }
}
```
