## 10.1 联合类型

### 什么是联合类型

联合类型表示值可以是多种类型之一：

```typescript
// 使用 | 分隔多种类型
let value: string | number;

value = 'hello';  // ✅
value = 123;      // ✅
value = true;     // ❌
```

### 基础示例

```typescript
function printId(id: string | number): void {
    console.log(`ID: ${id}`);
}

printId('abc123');  // ✅
printId(456);       // ✅
```

### 联合类型数组

```typescript
let arr: (string | number)[] = [1, 'a', 2, 'b'];
```

## 10.2 可辨识联合

### 什么是可辨识联合

通过公共属性（判别属性）区分联合类型：

```typescript
interface Square {
    kind: 'square';
    size: number;
}

interface Circle {
    kind: 'circle';
    radius: number;
}

type Shape = Square | Circle;

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'square':
            return shape.size * shape.size;
        case 'circle':
            return Math.PI * shape.radius * shape.radius;
    }
}
```

### 实际应用：状态管理

```typescript
interface LoadingState {
    status: 'loading';
}

interface SuccessState {
    status: 'success';
    data: string[];
}

interface ErrorState {
    status: 'error';
    error: Error;
}

type AsyncState = LoadingState | SuccessState | ErrorState;

function handleState(state: AsyncState): void {
    switch (state.status) {
        case 'loading':
            console.log('Loading...');
            break;
        case 'success':
            console.log('Data:', state.data);
            break;
        case 'error':
            console.log('Error:', state.error);
            break;
    }
}
```

## 10.3 交叉类型

### 什么是交叉类型

交叉类型表示同时具备多种类型：

```typescript
interface Person {
    name: string;
    age: number;
}

interface Employee {
    company: string;
    salary: number;
}

// 交叉类型：同时是 Person 和 Employee
type EmployeePerson = Person & Employee;

const employee: EmployeePerson = {
    name: 'Tom',
    age: 25,
    company: 'ABC',
    salary: 10000
};
```

### 实际应用：混入

```typescript
interface CanRun {
    run(): void;
}

interface CanSwim {
    swim(): void;
}

type Amphibian = CanRun & CanSwim;

class Frog implements Amphibian {
    run(): void {
        console.log('Running...');
    }
    swim(): void {
        console.log('Swimming...');
    }
}
```

## 10.4 联合 vs 交叉 对比

| 类型 | 符号 | 含义 | 记忆方法 |
|------|------|------|----------|
| 联合 | `\|` | 或（满足其一） | Union = 联合（合并） |
| 交叉 | `&` | 且（满足全部） | Intersection = 交叉（重叠） |

### 图示

```
联合类型 A | B：
┌───────┐     ┌───────┐
│   A   │ OR  │   B   │
└───────┘     └───────┘
   结果：可以是 A 或 B

交叉类型 A & B：
┌───────┐     ┌───────┐
│   A   │ AND │   B   │
└───────┘     └───────┘
   结果：同时是 A 和 B
```

### 示例对比

```typescript
// 联合：可以是 string 或 number
type StringOrNumber = string | number;
let value: StringOrNumber = 'hello';  // ✅
value = 123;                           // ✅

// 交叉：同时是 { a: string } 和 { b: number }
type Both = { a: string } & { b: number };
let obj: Both = { a: 'hello', b: 123 };  // ✅
```

## 10.5 实践练习

### 练习 1：联合类型

```typescript
// 参数可以是字符串或数字
function format(value: string | number): string {
    if (typeof value === 'string') {
        return value.toUpperCase();
    }
    return value.toFixed(2);
}

format('hello');  // 'HELLO'
format(123.456);  // '123.46'
```

### 练习 2：可辨识联合

```typescript
interface Circle {
    kind: 'circle';
    radius: number;
}

interface Rectangle {
    kind: 'rectangle';
    width: number;
    height: number;
}

interface Triangle {
    kind: 'triangle';
    base: number;
    height: number;
}

type Shape = Circle | Rectangle | Triangle;

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'rectangle':
            return shape.width * shape.height;
        case 'triangle':
            return 0.5 * shape.base * shape.height;
    }
}
```

### 练习 3：交叉类型

```typescript
interface Loggable {
    log(): void;
}

interface Serializable {
    serialize(): string;
}

type LoggableAndSerializable = Loggable & Serializable;

class User implements LoggableAndSerializable {
    name: string;
    
    constructor(name: string) {
        this.name = name;
    }
    
    log(): void {
        console.log(`User: ${this.name}`);
    }
    
    serialize(): string {
        return JSON.stringify({ name: this.name });
    }
}
```

## 本章小结

| 概念 | 符号 | 说明 |
|------|------|------|
| 联合类型 | `\|` | 值可以是多种类型之一 |
| 可辨识联合 | 通过 kind 属性区分 | 安全的联合类型处理 |
| 交叉类型 | `&` | 值同时具备多种类型 |
