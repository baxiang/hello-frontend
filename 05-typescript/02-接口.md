## 2.1 接口基础

### 什么是接口

```typescript
// 接口定义
interface User {
    id: number;
    name: string;
    email: string;
}

// 使用接口
const user: User = {
    id: 1,
    name: '张三',
    email: 'zhangsan@example.com'
};

// 错误：缺少属性
// const invalidUser: User = {
//     id: 1,
//     name: '张三'
//     // 错误：缺少 email
// };
```

### 接口描述对象形状

```typescript
// 接口只关注结构的形状，不关心实现
interface Point {
    x: number;
    y: number;
}

function printPoint(point: Point) {
    console.log(`x: ${point.x}, y: ${point.y}`);
}

// 字面量对象，只要有 x 和 y 就可以
const p = { x: 10, y: 20, z: 30 };
printPoint(p);  // OK，z 被忽略
```

## 2.2 可选属性

```typescript
interface Config {
    required: string;        // 必需属性
    optional?: number;       // 可选属性
    default?: boolean;       // 可选属性
}

// 使用
const config1: Config = {
    required: 'value'
};

const config2: Config = {
    required: 'value',
    optional: 42,
    default: true
};

// 错误：类型错误
// const config3: Config = {
//     required: 'value',
//     optional: 'string'  // 错误：应该是 number
// };
```

### 可选属性的实际应用

```typescript
interface CreateUserDTO {
    username: string;
    password: string;
    email?: string;
    phone?: string;
    avatar?: string;
}

async function createUser(data: CreateUserDTO) {
    // email、phone、avatar 是可选的
    const user = {
        id: Date.now(),
        username: data.username,
        password: data.password,
        email: data.email,
        phone: data.phone,
        avatar: data.avatar || '/default-avatar.png'
    };
    return user;
}
```

## 2.3 只读属性

```typescript
interface Person {
    readonly id: number;     // 只读
    name: string;
    readonly createdAt: Date; // 只读
}

const person: Person = {
    id: 1,
    name: '张三',
    createdAt: new Date()
};

// 修改只读属性 - 错误
// person.id = 2;  // 错误：不能修改
// person.createdAt = new Date();  // 错误

// 修改普通属性 - OK
person.name = '李四';  // OK
```

### ReadonlyArray

```typescript
// 只读数组
const numbers: readonly number[] = [1, 2, 3, 4, 5];

// 不能修改
// numbers.push(6);  // 错误
// numbers[0] = 10;  // 错误
// numbers.length = 3;  // 错误

// 可以读取
console.log(numbers[0]);  // 1
console.log(numbers.length);  // 5

// ReadonlyArray 泛型
const readonlyNums: ReadonlyArray<number> = [1, 2, 3];
```

### Readonly 工具类型

```typescript
interface Todo {
    id: number;
    title: string;
    completed: boolean;
}

const todo: Todo = {
    id: 1,
    title: '学习 TypeScript',
    completed: false
};

// 使用 Readonly 将所有属性变为只读
const readonlyTodo: Readonly<Todo> = todo;

// readonlyTodo.title = '新标题';  // 错误
```

## 2.4 函数类型接口

### 定义函数签名

```typescript
// 方式 1：接口定义函数类型
interface SearchFunc {
    (source: string, subString: string): boolean;
}

// 实现
const mySearch: SearchFunc = (src, sub) => {
    return src.includes(sub);
};

// 方式 2：完整接口
interface MathFunc {
    (a: number, b: number): number;
    description: string;
}

const add: MathFunc = (a, b) => a + b;
add.description = '加法函数';
```

### 函数参数接口

```typescript
interface GreetingSettings {
    greeting: string;
    duration?: number;
    color?: string;
}

function greet(person: string, settings: GreetingSettings) {
    const { greeting, duration = 1000, color = 'blue' } = settings;
    console.log(`${greeting}, ${person}`);
}

greet('张三', {
    greeting: '你好',
    duration: 2000
});
```

### 可调用接口

```typescript
interface Counter {
    (start: number): string;
    interval: number;
    reset(): void;
}

function createCounter(): Counter {
    const counter = ((start: number) => {
        return `Counting from ${start}`;
    }) as Counter;

    counter.interval = 1000;
    counter.reset = () => {
        console.log('重置计数器');
    };

    return counter;
}

const c = createCounter();
console.log(c(10));  // 调用
c.reset();           // 调用方法
```

## 2.5 接口继承

### 单继承

```typescript
// 基础接口
interface Shape {
    color: string;
    filled: boolean;
}

// 继承 Shape
interface Rectangle extends Shape {
    width: number;
    height: number;
}

const rect: Rectangle = {
    color: 'red',
    filled: true,
    width: 100,
    height: 50
};
```

### 多继承

```typescript
interface A {
    a: string;
}

interface B {
    b: number;
}

interface C extends A, B {
    c: boolean;
}

const obj: C = {
    a: 'hello',
    b: 42,
    c: true
};
```

### 接口继承类

```typescript
class Control {
    private x = 0;
    private y = 0;

    move(dx: number, dy: number) {
        this.x += dx;
        this.y += dy;
    }

    getPosition() {
        return { x: this.x, y: this.y };
    }
}

// 接口继承类（只继承实例成员，不继承私有成员）
interface Movable extends Control {
    speed: number;
}

class FastControl implements Movable {
    x = 0;
    y = 0;
    speed = 100;

    move(dx: number, dy: number) {
        this.x += dx * this.speed;
        this.y += dy * this.speed;
    }
}
```

## 2.6 实现接口

### implements 关键字

```typescript
interface Animal {
    name: string;
    makeSound(): void;
}

class Dog implements Animal {
    name: string;
    breed: string;

    constructor(name: string, breed: string) {
        this.name = name;
        this.breed = breed;
    }

    makeSound() {
        console.log('汪汪汪');
    }
}

class Cat implements Animal {
    name: string;

    constructor(name: string) {
        this.name = name;
    }

    makeSound() {
        console.log('喵喵喵');
    }
}
```

### 实现多个接口

```typescript
interface Flyable {
    fly(): void;
}

interface Swimmable {
    swim(): void;
}

// 同时实现多个接口
class Duck implements Flyable, Swimmable {
    fly() {
        console.log('鸭子在飞');
    }

    swim() {
        console.log('鸭子在游泳');
    }
}
```

### 类同时继承和实现

```typescript
interface Logger {
    log(message: string): void;
}

class Service {
    protected name: string;

    constructor(name: string) {
        this.name = name;
    }
}

class UserService extends Service implements Logger {
    constructor(name: string) {
        super(name);
    }

    log(message: string) {
        console.log(`[${this.name}] ${message}`);
    }
}
```

## 2.7 混合类型接口

```typescript
interface FullStack {
    // 可调用签名
    (framework: string): string;

    // 属性
    frontend: string[];
    backend: string[];

    // 方法
    learn(tech: string): void;
}

function createDeveloper(): FullStack {
    const dev = ((framework: string) => {
        return `Using ${framework}`;
    }) as FullStack;

    dev.frontend = ['React', 'Vue', 'Angular'];
    dev.backend = ['Node.js', 'Python', 'Go'];

    dev.learn = (tech: string) => {
        console.log(`Learning ${tech}`);
    };

    return dev;
}

const developer = createDeveloper();
console.log(developer('Next.js'));  // 调用
developer.learn('Rust');            // 方法
console.log(developer.frontend);    // 属性
```

## 2.8 接口与类型别名的区别

### 相同点

```typescript
// 接口
interface User1 {
    name: string;
    age: number;
}

// 类型别名
type User2 = {
    name: string;
    age: number;
};

// 使用上几乎一样
const user1: User1 = { name: '张三', age: 25 };
const user2: User2 = { name: '李四', age: 30 };
```

### 不同点

```typescript
// 1. 接口可以声明合并
interface Box {
    height: number;
}

interface Box {
    width: number;
}

const box: Box = { height: 100, width: 200 };  // OK

// 2. 接口可以被实现
interface Animal {
    name: string;
}

class Dog implements Animal {  // OK
    name = 'Dog';
}

// 类型别名不能被 implements
type Animal2 = { name: string };
// class Cat implements Animal2 {}  // 错误

// 3. 类型别名更灵活
type StringOrNumber = string | number;
type Maybe<T> = T | null;
type Point = [number, number];

// 4. 接口性能略好（在大型项目中）
```

### 选择建议

| 场景 | 推荐 |
|------|------|
| 定义对象结构 | 接口 |
| 需要声明合并 | 接口 |
| 需要被类实现 | 接口 |
| 联合类型 | 类型别名 |
| 泛型工具类型 | 类型别名 |
| 原始类型别名 | 类型别名 |

### 练习 2：表单验证

```typescript
// 验证规则接口
interface ValidationRule {
    required?: boolean;
    min?: number;
    max?: number;
    pattern?: RegExp;
    validator?: (value: any) => string | null;
}

// 字段配置接口
interface FormField {
    name: string;
    label: string;
    type: 'text' | 'number' | 'email' | 'password';
    rules?: ValidationRule;
}

// 验证结果接口
interface ValidationError {
    field: string;
    message: string;
}

interface ValidationResult {
    valid: boolean;
    errors: ValidationError[];
}

// 验证函数
function validateForm(
    values: Record<string, any>,
    fields: FormField[]
): ValidationResult {
    const errors: ValidationError[] = [];

    for (const field of fields) {
        const value = values[field.name];
        const rules = field.rules;

        if (!rules) continue;

        // 必填检查
        if (rules.required && !value) {
            errors.push({
                field: field.name,
                message: `${field.label}是必填的`
            });
            continue;
        }

        // 自定义验证器
        if (rules.validator) {
            const errorMsg = rules.validator(value);
            if (errorMsg) {
                errors.push({ field: field.name, message: errorMsg });
            }
        }
    }

    return {
        valid: errors.length === 0,
        errors
    };
}
```

### 练习 3：事件系统

```typescript
// 事件处理器接口
interface EventHandler<T = any> {
    (event: T): void;
}

// 事件总线接口
interface EventBus {
    on<T>(event: string, handler: EventHandler<T>): void;
    off<T>(event: string, handler: EventHandler<T>): void;
    emit<T>(event: string, data: T): void;
    once<T>(event: string, handler: EventHandler<T>): void;
}

// 实现
class SimpleEventBus implements EventBus {
    private handlers: Map<string, EventHandler[]> = new Map();

    on(event: string, handler: EventHandler) {
        if (!this.handlers.has(event)) {
            this.handlers.set(event, []);
        }
        this.handlers.get(event)!.push(handler);
    }

    off(event: string, handler: EventHandler) {
        const handlers = this.handlers.get(event);
        if (handlers) {
            const index = handlers.indexOf(handler);
            if (index > -1) {
                handlers.splice(index, 1);
            }
        }
    }

    emit(event: string, data: any) {
        const handlers = this.handlers.get(event);
        if (handlers) {
            handlers.forEach(handler => handler(data));
        }
    }

    once(event: string, handler: EventHandler) {
        const onceHandler = (data: any) => {
            handler(data);
            this.off(event, onceHandler);
        };
        this.on(event, onceHandler);
    }
}

// 使用
const bus = new SimpleEventBus();

bus.on('message', (data: string) => {
    console.log('收到消息:', data);
});

bus.emit('message', 'Hello TypeScript');
```

## 2.10 常见问答

### Q1: 接口和类型别名应该如何选择？

**答：**
- 定义对象结构、需要声明合并、需要被类实现 → 用接口
- 联合类型、元组、泛型工具 → 用类型别名

### Q2: 如何定义可选方法？

```typescript
interface Config {
    name: string;
    onSuccess?: (result: any) => void;
    onError?: (error: Error) => void;
}

function process(config: Config) {
    // 调用前检查
    if (config.onSuccess) {
        config.onSuccess('done');
    }
}
```

### Q3: 如何实现部分接口？

```typescript
interface AllFeatures {
    feature1(): void;
    feature2(): void;
    feature3(): void;
}

// 使用抽象类实现部分
abstract class BaseFeature implements Partial<AllFeatures> {
    feature1() {
        console.log('默认实现');
    }
}

class ConcreteFeature extends BaseFeature {
    feature2() {
        console.log('实现 feature2');
    }

    feature3() {
        console.log('实现 feature3');
    }
}
```
