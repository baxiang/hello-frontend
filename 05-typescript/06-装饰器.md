## 6.1 装饰器基础

### 什么是装饰器

```typescript
// 装饰器是一个函数，用于修改类、方法、属性或参数的行为
function log(target: any, key: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;

    descriptor.value = function(...args: any[]) {
        console.log(`调用 ${key}，参数：`, args);
        const result = originalMethod.apply(this, args);
        console.log(`返回：`, result);
        return result;
    };

    return descriptor;
}

class Calculator {
    @log
    add(a: number, b: number): number {
        return a + b;
    }
}

const calc = new Calculator();
calc.add(1, 2);
// 输出:
// 调用 add，参数：[1, 2]
// 返回：3
```

### 启用装饰器

```json
// tsconfig.json
{
    "compilerOptions": {
        "experimentalDecorators": true,
        "emitDecoratorMetadata": true  // 可选：启用反射元数据
    }
}
```

### 依赖注入

```typescript
// 服务容器
class Container {
    private services = new Map<string, any>();

    register(name: string, instance: any) {
        this.services.set(name, instance);
    }

    get<T>(name: string): T {
        return this.services.get(name);
    }

const container = new Container();

        this.logger.log(`创建用户：${name}`);
    }
}
```

## 6.3 方法装饰器

```typescript
function Log(

    key: string,
    descriptor: PropertyDescriptor
) {
    const original = descriptor.value;

    descriptor.value = function(...args: any[]) {
        console.log(`调用 ${key}`);

    }

```

### 权限验证装饰器

```typescript
function RequireAuth(roles: string[]) {
    return function(
        target: any,
        key: string,
        descriptor: PropertyDescriptor
    ) {
        const original = descriptor.value;

        descriptor.value = function(user: any, ...args: any[]) {
            if (!user || !roles.includes(user.role)) {
                throw new Error('没有权限');
            }
            return original.apply(this, [user, ...args]);
        };

        return descriptor;
    };
}

// 使用
class AdminController {
    @RequireAuth(['admin'])
    deleteUser(id: number) {
        console.log(`删除用户 ${id}`);
    }

    @RequireAuth(['admin', 'manager'])
    createReport(data: any) {
        console.log('创建报告');
    }
}
```

### 性能监控装饰器

```typescript
function PerformanceMonitor(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;

    descriptor.value = async function(...args: any[]) {
        const start = performance.now();
        try {
            return await original.apply(this, args);
        } finally {
            const end = performance.now();
            console.log(`${key} 执行时间：${(end - start).toFixed(2)}ms`);
        }
    };

    return descriptor;
}

// 使用
class DataService {
    @PerformanceMonitor
    async fetchData(): Promise<any[]> {
        await new Promise(r => setTimeout(r, 1000));
        return [{ id: 1, name: '数据' }];
    }
}
```

### 缓存装饰器

```typescript
function Cache(ttl: number = 5000) {
    const cache = new Map<string, { value: any; expires: number }>();

    return function(
        target: any,
        key: string,
        descriptor: PropertyDescriptor
    ) {
        const original = descriptor.value;

        descriptor.value = async function(...args: any[]) {
            const cacheKey = `${key}:${JSON.stringify(args)}`;
            const cached = cache.get(cacheKey);

            if (cached && cached.expires > Date.now()) {
                console.log('缓存命中');
                return cached.value;
            }

            const result = await original.apply(this, args);
            cache.set(cacheKey, {
                value: result,
                expires: Date.now() + ttl
            });

            return result;
        };

        return descriptor;
    };
}

// 使用
class ApiService {
    @Cache(10000)  // 10 秒缓存
    async getUser(id: number): Promise<any> {
        console.log('请求 API');
        await new Promise(r => setTimeout(r, 500));
        return { id, name: `用户${id}` };
    }
}
```

### 事务装饰器

```typescript
function Transaction() {
    return function(
        target: any,
        key: string,
        descriptor: PropertyDescriptor
    ) {
        const original = descriptor.value;

        descriptor.value = async function(...args: any[]) {
            // 开始事务
            console.log('开始事务');
            try {
                const result = await original.apply(this, args);
                console.log('提交事务');
                return result;
            } catch (error) {
                console.log('回滚事务');
                throw error;
            }
        };

        return descriptor;
    };
}

// 使用
class OrderService {
    @Transaction()
    async createOrder(data: any) {
        // 1. 创建订单
        // 2. 扣减库存
        // 3. 创建支付记录
        console.log('创建订单', data);
    }
}
```

### 可观察属性

```typescript
function Observable(target: any, key: string) {
    let value = target[key];
    const listeners = new Set<(newValue: any) => void>();

    const getter = function() {
        return value;
    };

    const setter = function(newValue: any) {
        const oldValue = value;
        value = newValue;
        listeners.forEach(listener => listener(newValue));
        console.log(`${key} 从 ${oldValue} 变更为 ${newValue}`);
    };

    Object.defineProperty(target, key, {
        get: getter,
        set: setter,
        enumerable: true,
        configurable: true
    });
}

// 使用
class State {
    @Observable
    count = 0;

    @Observable
    name = '';
}

const state = new State();
state.count = 1;    // 输出变更日志
state.name = 'TS';  // 输出变更日志
```

### 验证属性装饰器

```typescript
function Validate(validator: (value: any) => string | null) {
    return function(target: any, key: string) {
        let value = target[key];

        Object.defineProperty(target, key, {
            get: () => value,
            set: (newValue: any) => {
                const error = validator(newValue);
                if (error) {
                    throw new Error(`${key}: ${error}`);
                }
                value = newValue;
            },
            enumerable: true,
            configurable: true
        });
    };
}

// 使用
class Person {
    @Validate((v) => {
        if (!v) return '不能为空';
        if (v.length < 2) return '至少 2 个字符';
        return null;
    })
    name: string;

    @Validate((v) => {
        if (v < 0 || v > 150) return '年龄必须在 0-150 之间';
        return null;
    })
    age: number;

    constructor(name: string, age: number) {
        this.name = name;
        this.age = age;
    }
}

const person = new Person('张三', 25);
// person.name = 'A';  // 错误：至少 2 个字符
```

### 依赖注入参数装饰器

```typescript
// 服务容器
class Container {
    private services = new Map<string, any>();

    register(name: string, instance: any) {
        this.services.set(name, instance);
    }

    get(name: string): any {
        return this.services.get(name);
    }
}

const container = new Container();

// 服务定义
function Service(name?: string) {
    return function(constructor: Function) {
        const serviceName = name || constructor.name;
        container.register(serviceName, new constructor());
    };
}

// 参数装饰器
function Inject(name?: string) {
    return function(target: any, key: string, index: number) {
        const serviceName = name || Reflect.getMetadata('design:paramtypes', target, key)[index].name;
        const service = container.get(serviceName);
        if (!target.injectedParams) {
            target.injectedParams = [];
        }
        target.injectedParams[index] = service;
    };
}

// 使用
@Service()
class Logger {
    log(message: string) {
        console.log(message);
    }
}

@Service('userService')
class UserService {
    constructor(
        @Inject() private logger: Logger
    ) {}

    createUser(name: string) {
        this.logger.log(`创建用户：${name}`);
    }
}
```

## 6.6 装饰器组合

### 多个装饰器

```typescript
// 多个装饰器可以组合使用
function Log(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function(...args: any[]) {
        console.log(`[Log] 调用 ${key}`);
        return original.apply(this, args);
    };
    return descriptor;
}

function Timing(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = async function(...args: any[]) {
        const start = Date.now();
        const result = await original.apply(this, args);
        console.log(`[Timing] ${key} 耗时：${Date.now() - start}ms`);
        return result;
    };
    return descriptor;
}

class DataService {
    @Log
    @Timing
    async fetchData(): Promise<any> {
        await new Promise(r => setTimeout(r, 100));
        return { data: 'test' };
    }
}

// 输出顺序：
// [Log] 调用 fetchData
// [Timing] fetchData 耗时：100ms
```

### 装饰器工厂组合

```typescript
// 装饰器工厂
function LogWithPrefix(prefix: string) {
    return function(target: any, key: string, descriptor: PropertyDescriptor) {
        const original = descriptor.value;
        descriptor.value = function(...args: any[]) {
            console.log(`[${prefix}] ${key} 被调用`);
            return original.apply(this, args);
        };
        return descriptor;
    };
}

function ValidatePositive() {
    return function(target: any, key: string, descriptor: PropertyDescriptor) {
        const original = descriptor.value;
        descriptor.value = function(value: number) {
            if (value <= 0) {
                throw new Error('值必须大于 0');
            }
            return original.apply(this, arguments);
        };
        return descriptor;
    };
}

class Order {
    @LogWithPrefix('ORDER')
    @ValidatePositive()
    setQuantity(qty: number) {
        console.log(`设置数量：${qty}`);
    }
}
```

### 元数据装饰器

```typescript
import 'reflect-metadata';

// 创建装饰器工厂
function ApiEndpoint(path: string, method: string) {
    return function(target: any, key: string) {
        Reflect.defineMetadata('api:path', path, target, key);
        Reflect.defineMetadata('api:method', method, target, key);
    };
}

function ApiParam(name: string, type: any) {
    return function(target: any, key: string, index: number) {
        const params = Reflect.getMetadata('api:params', target, key) || [];
        params.push({ name, type });
        Reflect.defineMetadata('api:params', params, target, key);
    };
}

// 使用
class UserController {
    @ApiEndpoint('/users', 'GET')
    getAll() {
        return [{ id: 1, name: '张三' }];
    }

    @ApiEndpoint('/users/:id', 'GET')
    getById(
        @ApiParam('id', Number) id: number
    ) {
        return { id, name: '张三' };
    }
}

// 读取路由信息
function getRoutes(controller: any) {
    const routes: any[] = [];
    const proto = Object.getPrototypeOf(controller);

    for (const key of Object.getOwnPropertyNames(proto)) {
        const path = Reflect.getMetadata('api:path', proto, key);
        const method = Reflect.getMetadata('api:method', proto, key);

        if (path && method) {
            routes.push({ key, path, method });
        }
    }

    return routes;
}

const routes = getRoutes(new UserController());
console.log(routes);
```

## 6.8 实践练习

### 练习 1：简易 ORM 装饰器

```typescript
import 'reflect-metadata';

// 实体装饰器
function Entity(tableName: string) {
    return function(constructor: Function) {
        Reflect.defineMetadata('db:table', tableName, constructor);
    };
}

// 列装饰器
function Column(options: {
    type: string;
    primary?: boolean;
    generated?: boolean;
    nullable?: boolean;
}) {
    return function(target: any, key: string) {
        const columns = Reflect.getMetadata('db:columns', target.constructor) || [];
        columns.push({ key, ...options });
        Reflect.defineMetadata('db:columns', columns, target.constructor);
    };
}

// 使用
@Entity('users')
class User {
    @Column({ type: 'int', primary: true, generated: true })
    id!: number;

    @Column({ type: 'varchar', nullable: false })
    name!: string;

    @Column({ type: 'varchar', nullable: false })
    email!: string;
}

// 简单的 ORM 实现
class ORM {
    private tables = new Map<string, any[]>();

    getTableName(constructor: Function): string {
        return Reflect.getMetadata('db:table', constructor);
    }

    getColumns(constructor: Function): any[] {
        return Reflect.getMetadata('db:columns', constructor) || [];
    }

    async save<T extends { [key: string]: any }>(entity: T): Promise<T> {
        const constructor = entity.constructor as Function;
        const tableName = this.getTableName(constructor);
        const columns = this.getColumns(constructor);

        if (!this.tables.has(tableName)) {
            this.tables.set(tableName, []);
        }

        const data: any = {};
        for (const col of columns) {
            data[col.key] = (entity as any)[col.key];
        }

        const table = this.tables.get(tableName)!;
        table.push(data);

        return entity;
    }

    async findById<T>(constructor: Function, id: number): Promise<T | null> {
        const tableName = this.getTableName(constructor);
        const table = this.tables.get(tableName);

        if (!table) return null;

        return table.find(row => row.id === id) as T;
    }
}

// 使用
const orm = new ORM();

const user = new User();
user.name = '张三';
user.email = 'zhang@example.com';

orm.save(user);
```

### 练习 2：AOP 切面编程

```typescript
import 'reflect-metadata';

// 前置通知
function Before(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    const beforeFn = target[`before_${key}`];

    if (beforeFn) {
        descriptor.value = function(...args: any[]) {
            beforeFn.apply(this, args);
            return original.apply(this, args);
        };
    }

    return descriptor;
}

// 后置通知
function After(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    const afterFn = target[`after_${key}`];

    if (afterFn) {
        descriptor.value = function(...args: any[]) {
            const result = original.apply(this, args);
            afterFn.apply(this, [result, ...args]);
            return result;
        };
    }

    return descriptor;
}

// 异常捕获通知
function Around(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    const errorFn = target[`onError_${key}`];

    descriptor.value = async function(...args: any[]) {
        try {
            return await original.apply(this, args);
        } catch (error) {
            if (errorFn) {
                return errorFn.call(this, error, args);
            }
            throw error;
        }
    };

    return descriptor;
}

// 使用示例
class UserService {
    // 前置方法
    before_createUser(name: string) {
        console.log(`准备创建用户：${name}`);
    }

    // 后置方法
    after_createUser(result: any, name: string) {
        console.log(`用户创建成功：${result.id}`);
    }

    // 异常处理方法
    onError_createUser(error: Error, args: any[]) {
        console.error('创建用户失败:', error);
        return { error: error.message };
    }

    @Before
    @After
    @Around
    createUser(name: string) {
        if (!name) throw new Error('用户名不能为空');
        return { id: Date.now(), name };
    }
}

const service = new UserService();
service.createUser('张三');
```

### 练习 3：依赖注入容器

```typescript
import 'reflect-metadata';

// 存储服务的元数据
const SERVICE_META = Symbol('service');

// 服务装饰器
function Service<T>(constructor: new (...args: any[]) => T) {
    Reflect.defineMetadata(SERVICE_META, constructor, constructor);
}

// 依赖注入装饰器
function Inject<T>(token?: any) {
    return function(target: any, propertyKey: string | symbol) {
        const type = Reflect.getMetadata('design:type', target, propertyKey);
        const tokenToUse = token || type;

        const getter = function() {
            return Container.getInstance().get(tokenToUse);
        };

        Object.defineProperty(target, propertyKey, {
            get: getter,
            set: () => {},
            enumerable: true,
            configurable: true
        });
    };
}

// 容器类
class Container {
    private static instance: Container;
    private services = new Map<any, any>();

    static getInstance(): Container {
        if (!Container.instance) {
            Container.instance = new Container();
        }
        return Container.instance;
    }

    register<T>(token: new (...args: any[]) => T, instance: T): void {
        this.services.set(token, instance);
    }

    get<T>(token: new (...args: any[]) => T): T {
        if (this.services.has(token)) {
            return this.services.get(token);
        }

        // 自动创建实例
        const constructor = token as any;
        const params = Reflect.getMetadata('design:paramtypes', constructor) || [];
        const dependencies = params.map((param: any) => this.get(param));
        const instance = new constructor(...dependencies);

        this.services.set(token, instance);
        return instance;
    }

    clear(): void {
        this.services.clear();
    }
}

// 使用示例
@Service
class Logger {
    log(message: string) {
        console.log(`[LOG] ${message}`);
    }
}

@Service
class Database {
    connect() {
        console.log('数据库连接成功');
    }

    query(sql: string) {
        console.log(`执行 SQL: ${sql}`);
        return [];
    }
}

@Service
class UserService {
    @Inject()
    private logger!: Logger;

    @Inject()
    private database!: Database;

    getUser(id: number) {
        this.logger.log(`获取用户 ${id}`);
        return this.database.query('SELECT * FROM users');
    }
}

// 使用
const userService = Container.getInstance().get(UserService);
userService.getUser(1);
```

## 6.9 常见问答

### Q1: 装饰器的执行顺序是什么？

**答：**
1. 参数装饰器
2. 方法装饰器
3. 属性装饰器
4. 类装饰器

同一类型多个装饰器：从下往上执行

```typescript
function A() {
    console.log('A');
    return function() { console.log('A decorator'); };
}

function B() {
    console.log('B');
    return function() { console.log('B decorator'); };
}

@A()  // 先打印 A
@B()  // 后打印 B
class MyClass {}

// 装饰器执行：B -> A
```

```typescript
}
```

### Q3: 装饰器在哪些框架中使用？

**答：**
- **Angular**: 组件、服务、依赖注入
- **NestJS**: 控制器、服务、守卫、拦截器
- **TypeORM**: 实体、列、关系
- **inversify**: 依赖注入容器
