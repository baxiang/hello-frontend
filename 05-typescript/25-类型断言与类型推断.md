## 9.1 类型断言

### 什么是类型断言

告诉 TypeScript 一个值的具体类型：

```typescript
// 假设从 API 获取的数据
const data: unknown = { name: 'Tom', age: 25 };

// 断言为特定类型
const user = data as { name: string; age: number };
console.log(user.name);  // "Tom"
```

### 两种语法

```typescript
// 1. as 语法（推荐）
const value = someValue as string;

// 2. 尖括号语法（不推荐，与 JSX 冲突）
const value = <string>someValue;
```

### 使用场景

```typescript
// 1. 处理 unknown 类型
function process(data: unknown): void {
    const str = data as string;
    console.log(str.toUpperCase());
}

// 2. DOM 元素
const input = document.getElementById('myInput') as HTMLInputElement;
input.value = 'Hello';

// 3. 接口实现
interface User {
    name: string;
}

const obj = { name: 'Tom', age: 25 } as User;
```

## 9.2 非空断言

### 使用 !

确信值不是 null 或 undefined：

```typescript
function getName(name?: string): string {
    return name!.toUpperCase();
}

getName('Tom');  // 'TOM'
getName();      // ❌ 运行时会报错
```

### 实际应用

```typescript
// DOM 元素
const button = document.getElementById('submit')!;
button.addEventListener('click', () => {});

// 可选链 + 非空断言
const user = {
    profile: {
        name: 'Tom'
    }
};

console.log(user.profile!.name);  // "Tom"
```

### ⚠️ 谨慎使用

```typescript
// 危险：可能运行时出错
function getName(name?: string): string {
    return name!.toUpperCase();  // 如果 name 是 undefined，运行时报错
}

// 安全：先检查
function getName(name?: string): string {
    return name?.toUpperCase() ?? '';
}
```

## 9.3 类型推断

### 变量推断

```typescript
// 根据初始值推断类型
let x = 'hello';    // string
let y = 123;        // number
let z = true;       // boolean
```

### 函数推断

```typescript
// 返回类型会被推断
function add(a: number, b: number) {
    return a + b;  // 推断返回 number
}

// 泛型推断
function identity<T>(arg: T): T {
    return arg;
}

const num = identity(123);    // number
const str = identity('hello'); // string
```

### 上下文推断

```typescript
// 参数类型根据传入的函数推断
const numbers = [1, 2, 3];
numbers.forEach((num) => {
    // num 被推断为 number
    console.log(num.toFixed(2));
});
```

## 9.4 字面量类型

### 字符串字面量类型

```typescript
type Direction = 'up' | 'down' | 'left' | 'right';

const move: Direction = 'up';   // ✅
const invalid: Direction = 'forward';  // ❌
```

### 数字字面量类型

```typescript
type StatusCode = 200 | 404 | 500;

const code: StatusCode = 200;  // ✅
const invalid: StatusCode = 201;  // ❌
```

### 实际应用

```typescript
type Method = 'GET' | 'POST' | 'PUT' | 'DELETE';

function request(url: string, method: Method): void {
    console.log(`${method} ${url}`);
}

request('/api/users', 'GET');   // ✅
request('/api/users', 'PATCH');  // ❌
```

## 9.5 const 断言

### as const

将类型推断为最具体的字面量类型：

```typescript
// 不使用 as const
const roles = ['admin', 'user'];
// 类型：string[]

// 使用 as const
const roles = ['admin', 'user'] as const;
// 类型：readonly ['admin', 'user']
```

### 对比

```typescript
// 不使用 as const
const config = {
    port: 3000,
    host: 'localhost'
};
// 类型：{ port: number; host: string; }

// 使用 as const
const config = {
    port: 3000,
    host: 'localhost'
} as const;
// 类型：{ readonly port: 3000; readonly host: 'localhost'; }
```

### 实际应用

```typescript
// API 端点
const endpoints = {
    users: '/api/users',
    posts: '/api/posts'
} as const;

// 类型安全
endpoints.users;    // '/api/users'
endpoints.posts;   // '/api/posts'
// endpoints.admin  // ❌ 不存在
```

## 9.6 实践练习

### 练习 1：类型断言

```typescript
// 处理 unknown 数据
function processData(data: unknown): void {
    if (typeof data === 'string') {
        console.log(data.toUpperCase());
    } else if (typeof data === 'number') {
        console.log(data * 2);
    }
}

// 使用断言
function processUser(data: unknown): void {
    const user = data as { name: string; age: number };
    console.log(`${user.name} is ${user.age} years old`);
}
```

### 练习 2：字面量类型

```typescript
// 定义状态
type OrderStatus = 'pending' | 'processing' | 'completed' | 'cancelled';

function updateStatus(orderId: number, status: OrderStatus): void {
    console.log(`Order ${orderId} status: ${status}`);
}

updateStatus(1, 'pending');      // ✅
updateStatus(1, 'shipped');       // ❌
```

### 练习 3：const 断言

```typescript
// API 配置
const apiConfig = {
    baseUrl: 'https://api.example.com',
    timeout: 5000,
    retries: 3
} as const;

// 错误：不能修改只读属性
// apiConfig.baseUrl = 'other';  // ❌

// 正确：可以读取
console.log(apiConfig.baseUrl);  // 'https://api.example.com'
```

### 练习 4：非空断言

```typescript
// 获取 DOM 元素
function getInputValue(id: string): string {
    const input = document.getElementById(id) as HTMLInputElement;
    return input.value;
}

// 使用可选链更安全
function getInputValueSafe(id: string): string {
    const input = document.getElementById(id);
    return input?.value ?? '';
}
```

## 本章小结

| 概念 | 语法 | 说明 |
|------|------|------|
| 类型断言 | `value as Type` | 告诉 TS 具体类型 |
| 非空断言 | `value!` | 确信非 null/undefined |
| 字面量类型 | `'a' \| 'b'` | 特定值作为类型 |
| const 断言 | `as const` | 最严格的字面量类型 |
