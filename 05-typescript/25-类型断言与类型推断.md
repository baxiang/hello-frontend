## 9.1 类型断言

### 什么是类型断言

告诉 TypeScript 一个值的具体类型：

```typescript
// 假设从 API 获取的数据
const data: unknown = { name: 'Tom', age: 25 };

// 断言为特定类型
const user = data as { name: string; age: number };
console.log(user.name);  // "Tom"
```

### 两种语法

```typescript
// 1. as 语法（推荐）
const value = someValue as string;

// 2. 尖括号语法（不推荐，与 JSX 冲突）
const value = <string>someValue;
```

### 使用场景

```typescript
// 1. 处理 unknown 类型
function process(data: unknown): void {
    const str = data as string;
    console.log(str.toUpperCase());
}

// 2. DOM 元素
const input = document.getElementById('myInput') as HTMLInputElement;
input.value = 'Hello';

// 3. 接口实现
interface User {
    name: string;
}

const obj = { name: 'Tom', age: 25 } as User;
```

## 9.2 非空断言

### 使用 !

确信值不是 null 或 undefined：

```typescript
function getName(name?: string): string {
    return name!.toUpperCase();
}

getName('Tom');  // 'TOM'
getName();      // ❌ 运行时会报错
```

### 实际应用

```typescript
// DOM 元素
const button = document.getElementById('submit')!;
button.addEventListener('click', () => {});

// 可选链 + 非空断言
const user = {
    profile: {
        name: 'Tom'
    }
};

console.log(user.profile!.name);  // "Tom"
```

### ⚠️ 谨慎使用

```typescript
// 危险：可能运行时出错
function getName(name?: string): string {
    return name!.toUpperCase();  // 如果 name 是 undefined，运行时报错
}

// 安全：先检查
function getName(name?: string): string {
    return name?.toUpperCase() ?? '';
}
```

## 9.2 null 和 undefined 的区别（初学者必看）

### 最通俗的理解

```
undefined = "还没赋值"
null = "故意为空"

就像填表格：
- undefined = 表格空白，没填
- null = 表格填了"无"或"空"
```

### 具体例子

```typescript
// 1. 变量声明但没赋值 → undefined
let name;
console.log(name);  // undefined

// 2. 访问不存在的属性 → undefined
const user = {};
console.log(user.name);  // undefined

// 3. 访问数组不存在的索引 → undefined
const arr = [1, 2, 3];
console.log(arr[100]);  // undefined

// 4. 函数没有 return → undefined
function sayHello() {
    console.log('Hello');
}
const result = sayHello();  // undefined

// 5. 显式设置为 null
const user2 = {
    name: 'Tom',
    age: null  // 故意设置为 null，表示"年龄未知"
};
console.log(user2.age);  // null
```

### 什么时候返回 undefined？

```typescript
// 1. 变量声明未赋值
let a;
console.log(a);  // undefined

// 2. 对象没有这个属性
const user = { name: 'Tom' };
console.log(user.age);  // undefined（没有 age 属性）

// 3. 数组越界
const arr = [1, 2];
console.log(arr[5]);  // undefined

// 4. 函数没有返回值
function fn() {}
console.log(fn());  // undefined

// 5. 可选参数没传
function greet(name?: string) {
    console.log(name);  // undefined（没传参数）
}
greet();

// 6. 可选链访问不存在的属性
const user = { profile: null };
console.log(user.profile?.name);  // undefined（profile 存在但 name 不存在）
```

### 什么时候返回 null？

```typescript
// 1. 显式设置为 null
const user = {
    name: 'Tom',
    age: null  // 故意设置为 null
};
console.log(user.age);  // null

// 2. 某些 API 返回 null
// 比如查找元素没找到
const el = document.getElementById('notExist');
console.log(el);  // null

// 3. JSON.parse 解析空值
const data = JSON.parse('null');
console.log(data);  // null
```

### 核心区别

| 场景 | 返回 | 含义 |
|------|------|------|
| 变量声明未赋值 | undefined | 还没赋值 |
| 对象没有这个属性 | undefined | 根本没有这个属性 |
| 访问数组越界 | undefined | 超出范围 |
| 函数没有 return | undefined | 没有返回值 |
| 显式设置为 null | null | 故意设为空 |
| getElementById 没找到 | null | 找不到这个元素 |

### 为什么 `user.profile?.name` 返回 undefined？

```typescript
const user = {
    profile: null
};

// profile 存在，值是 null
// name 属性不存在
console.log(user.profile?.name);  // undefined

// 解释：
// user.profile = null（存在，但是 null）
// user.profile.name = 不存在（undefined）
// 所以返回 undefined
```

### 为什么 `user.address?.city` 返回 undefined？

```typescript
const user = {
    name: 'Tom'
    // 没有 address 属性
};

console.log(user.address?.city);  // undefined

// 解释：
// user.address = undefined（根本没有 address 属性）
// 可选链遇到 undefined 就停止，返回 undefined
```

### 实际开发技巧

#### 1. 如何区分？

```typescript
// 检查是 undefined 还是 null
function check(value: string | null | undefined) {
    if (value === undefined) {
        console.log('是 undefined - 还没赋值');
    } else if (value === null) {
        console.log('是 null - 故意设置为空');
    } else {
        console.log('有值:', value);
    }
}

// 使用
let a;
check(a);  // 是 undefined

let b = null;
check(b);  // 是 null

let c = 'hello';
check(c);  // 有值
```

#### 2. 同时检查 null 和 undefined

```typescript
// 方法一：==
if (value == null) {
    // 相当于 value === null || value === undefined
}

// 方法二：??
if (value ?? '默认值') {
    // null 或 undefined 时使用默认值
}

// 方法三：可选链 + 空值合并
const name = user.name ?? '匿名用户';
// user.name 是 null 或 undefined 时，使用 '匿名用户'
```

#### 3. 什么时候用 null？什么时候用 undefined？

```typescript
// 用 undefined 的场景：
let name;  // 还没赋值
const user = { age: undefined };  // 暂时没有年龄

// 用 null 的场景：
// - 明确表示"这里没有值"
const user = {
    name: 'Tom',
    spouse: null  // 没有配偶，明确表示
};

// 实际建议：在 TypeScript 中
// - 可选属性用 undefined
// - 明确表示"空"用 null
interface User {
    name: string;
    age?: number;      // 可选，不传就是 undefined
    spouse?: string | null;  // 可选，可能是 null
}
```

#### 4. 严格模式下不能随意赋值

```typescript
// TypeScript 严格模式下：
let name: string = null;  // ❌ 错误
let name: string | null = null;  // ✅ 正确

// 建议：使用可选属性
interface User {
    name?: string;  // 可选，等于 undefined
}
```

## 9.2.1 可选链（?.）和空值合并（??）

### 什么是 `?.` 可选链？

```
可选链 = 安全地访问深层属性

如果属性不存在，返回 undefined 而不是报错
```

```typescript
// 不用可选链（危险）
const user = {
    profile: {
        name: 'Tom'
    }
};

// 正常访问
console.log(user.profile.name);  // 'Tom'

// 如果 profile 不存在
const user2: any = {};
// console.log(user2.profile.name);  // ❌ 报错：Cannot read property 'name' of undefined

// 使用可选链（安全）
console.log(user2.profile?.name);  // undefined（不报错）
```

#### 实际应用

```typescript
// 场景：用户可能没有 profile
interface User {
    profile?: {
        name: string;
        email: string;
    };
}

const user: User = {};

// 不用可选链
console.log(user.profile && user.profile.name);  // undefined

// 用可选链（更简洁）
console.log(user.profile?.name);  // undefined

// 深层可选链
console.log(user.profile?.address?.city);  // undefined

// 方法调用
user.profile?.sayHello?.();  // 如果存在就调用，不存在就跳过
```

### 什么是 `??` 空值合并？

```
?? = 如果左边是 null/undefined，使用右边的值

注意：只对 null 和 undefined 有效，不包括 0、''、false
```

```typescript
// 基本用法
const name1 = null ?? '默认值';   // '默认值'
const name2 = undefined ?? '默认值'; // '默认值'
const name3 = '' ?? '默认值';      // ''（空字符串不是 null/undefined）
const name4 = 0 ?? '默认值';       // 0（0不是 null/undefined）
const name5 = false ?? '默认值';  // false（false不是 null/undefined）
```

#### 对比 `||` 或运算符

```typescript
// || 或运算符
const a = null || '默认值';   // '默认值'
const b = '' || '默认值';     // '默认值' （❌ 空字符串被当作 false）
const c = 0 || '默认值';      // '默认值' （❌ 0 被当作 false）
const d = false || '默认值';  // '默认值' （❌ false 被当作 false）

// ?? 空值合并（更精确）
const e = null ?? '默认值';   // '默认值'
const f = '' ?? '默认值';     // '' （✅ 只对 null/undefined 生效）
const g = 0 ?? '默认值';      // 0 （✅ 只对 null/undefined 生效）
const h = false ?? '默认值';  // false （✅ 只对 null/undefined 生效）
```

### 组合使用：`?.` + `??`

```typescript
// 完整示例
function getUserName(user?: { profile?: { name: string } }): string {
    // 如果 user、profile、name 任一不存在，返回 '匿名用户'
    return user?.profile?.name ?? '匿名用户';
}

// 测试
getUserName({ profile: { name: 'Tom' } });  // 'Tom'
getUserName({ profile: {} });                // '匿名用户'
getUserName({});                            // '匿名用户'
getUserName(undefined);                      // '匿名用户'
```

### 更多示例

```typescript
// 1. 处理 API 返回的数据
interface ApiResponse {
    data?: {
        user?: {
            name: string;
            email?: string;
        };
    };
}

const response: ApiResponse = {};
const name = response.data?.user?.name ?? '未知用户';
const email = response.data?.user?.email ?? '无邮箱';

// 2. 处理表单输入
const inputValue = (document.getElementById('input')?.value ?? '').trim();

// 3. 处理配置（0 或 false 是有效值）
const config = {
    timeout: 0,
    enabled: false
};

const timeout = config.timeout ?? 3000;    // 0 ✅
const enabled = config.enabled ?? true;    // false ✅
```

### 总结对比

| 语法 | 名称 | 说明 |
|------|------|------|
| `?.` | 可选链 | 安全访问深层属性，不报错 |
| `??` | 空值合并 | null/undefined 时使用默认值 |
| `||` | 或运算符 | falsy 值时使用默认值 |

### 最佳实践

```typescript
// ✅ 推荐：使用 ?? 处理可能为 null/undefined 的值
const name = user.name ?? '匿名用户';

// ✅ 推荐：使用 ?. 安全访问深层属性
const city = user.address?.city ?? '未知城市';

// ❌ 避免：|| 会把 0、''、false 当作 false
const count = inputCount || 1;  // 如果 inputCount 是 0，会变成 1

// ✅ 正确：使用 ??
const count = inputCount ?? 1;  // 如果 inputCount 是 0，保持 0
```

## 9.3 类型推断

### 变量推断

```typescript
// 根据初始值推断类型
let x = 'hello';    // string
let y = 123;        // number
let z = true;       // boolean
```

### 函数推断

```typescript
// 返回类型会被推断
function add(a: number, b: number) {
    return a + b;  // 推断返回 number
}

// 泛型推断
function identity<T>(arg: T): T {
    return arg;
}

const num = identity(123);    // number
const str = identity('hello'); // string
```

### 上下文推断

```typescript
// 参数类型根据传入的函数推断
const numbers = [1, 2, 3];
numbers.forEach((num) => {
    // num 被推断为 number
    console.log(num.toFixed(2));
});
```

## 9.4 字面量类型

### 字符串字面量类型

```typescript
type Direction = 'up' | 'down' | 'left' | 'right';

const move: Direction = 'up';   // ✅
const invalid: Direction = 'forward';  // ❌
```

### 数字字面量类型

```typescript
type StatusCode = 200 | 404 | 500;

const code: StatusCode = 200;  // ✅
const invalid: StatusCode = 201;  // ❌
```

### 实际应用

```typescript
type Method = 'GET' | 'POST' | 'PUT' | 'DELETE';

function request(url: string, method: Method): void {
    console.log(`${method} ${url}`);
}

request('/api/users', 'GET');   // ✅
request('/api/users', 'PATCH');  // ❌
```

## 9.5 const 断言

### as const

将类型推断为最具体的字面量类型：

```typescript
// 不使用 as const
const roles = ['admin', 'user'];
// 类型：string[]

// 使用 as const
const roles = ['admin', 'user'] as const;
// 类型：readonly ['admin', 'user']
```

### 对比

```typescript
// 不使用 as const
const config = {
    port: 3000,
    host: 'localhost'
};
// 类型：{ port: number; host: string; }

// 使用 as const
const config = {
    port: 3000,
    host: 'localhost'
} as const;
// 类型：{ readonly port: 3000; readonly host: 'localhost'; }
```

### 实际应用

```typescript
// API 端点
const endpoints = {
    users: '/api/users',
    posts: '/api/posts'
} as const;

// 类型安全
endpoints.users;    // '/api/users'
endpoints.posts;   // '/api/posts'
// endpoints.admin  // ❌ 不存在
```

## 9.6 实践练习

### 练习 1：类型断言

```typescript
// 处理 unknown 数据
function processData(data: unknown): void {
    if (typeof data === 'string') {
        console.log(data.toUpperCase());
    } else if (typeof data === 'number') {
        console.log(data * 2);
    }
}

// 使用断言
function processUser(data: unknown): void {
    const user = data as { name: string; age: number };
    console.log(`${user.name} is ${user.age} years old`);
}
```

### 练习 2：字面量类型

```typescript
// 定义状态
type OrderStatus = 'pending' | 'processing' | 'completed' | 'cancelled';

function updateStatus(orderId: number, status: OrderStatus): void {
    console.log(`Order ${orderId} status: ${status}`);
}

updateStatus(1, 'pending');      // ✅
updateStatus(1, 'shipped');       // ❌
```

### 练习 3：const 断言

```typescript
// API 配置
const apiConfig = {
    baseUrl: 'https://api.example.com',
    timeout: 5000,
    retries: 3
} as const;

// 错误：不能修改只读属性
// apiConfig.baseUrl = 'other';  // ❌

// 正确：可以读取
console.log(apiConfig.baseUrl);  // 'https://api.example.com'
```

### 练习 4：非空断言

```typescript
// 获取 DOM 元素
function getInputValue(id: string): string {
    const input = document.getElementById(id) as HTMLInputElement;
    return input.value;
}

// 使用可选链更安全
function getInputValueSafe(id: string): string {
    const input = document.getElementById(id);
    return input?.value ?? '';
}
```
