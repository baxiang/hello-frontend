## 5.1 类型守卫概述

### 什么是类型守卫

类型守卫（Type Guard）是在运行时检查变量类型的技术，它可以帮助 TypeScript 在特定作用域内收窄（narrow）联合类型。

```typescript
// 联合类型
function printId(id: number | string) {
    // 在这个作用域内，id 是 number | string
    // 我们只能访问共有的属性和方法

    // 类型守卫 - 使用 typeof
    if (typeof id === 'number') {
        // 在这个 if 块内，TypeScript 知道 id 是 number 类型
        console.log(id.toFixed(2));
    } else {
        // 在 else 块内，TypeScript 推断 id 是 string 类型
        console.log(id.toUpperCase());
    }
}
```

### 类型收窄的方式

TypeScript 提供了多种类型收窄的方式：

| 方式 | 说明 | 示例 |
|------|------|------|
| `typeof` | 检查基本类型 | `typeof x === 'string'` |
| `instanceof` | 检查构造函数实例 | `x instanceof Date` |
| `in` | 检查属性是否存在 | `'name' in x` |
| 类型谓词 | 自定义类型检查 | `x is string` |
| 相等性检查 | 检查值是否相等 | `x === null` |
| 真值检查 | 检查真假值 | `if (x)` |

## 5.2 typeof 类型守卫

### 基础用法

```typescript
// 检查联合类型
function processValue(value: string | number | boolean): string {
    if (typeof value === 'string') {
        // value: string
        return `字符串：${value}`;
    } else if (typeof value === 'number') {
        // value: number
        return `数字：${value.toFixed(2)}`;
    } else {
        // value: boolean
        return `布尔值：${value ? 'true' : 'false'}`;
    }
}

// 使用示例
console.log(processValue('hello'));    // 字符串：hello
console.log(processValue(42));         // 数字：42.00
console.log(processValue(true));       // 布尔值：true
```

### 实际应用：参数验证器

```typescript
type ValidationRule =
    | { type: 'required'; message: string }
    | { type: 'minLength'; value: number; message: string }
    | { type: 'maxLength'; value: number; message: string }
    | { type: 'pattern'; value: RegExp; message: string };

function validate(
    value: unknown,
    rules: ValidationRule[]
): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    for (const rule of rules) {
        // 使用 typeof 进行类型守卫
        if (rule.type === 'required') {
            if (value === null || value === undefined || value === '') {
                errors.push(rule.message);
            }
        } else if (rule.type === 'minLength' || rule.type === 'maxLength') {
            // 只有字符串才有 length 属性
            if (typeof value !== 'string') {
                errors.push('该字段必须是字符串');
                break;
            }
            if (rule.type === 'minLength' && value.length < rule.value) {
                errors.push(rule.message);
            }
            if (rule.type === 'maxLength' && value.length > rule.value) {
                errors.push(rule.message);
            }
        } else if (rule.type === 'pattern') {
            if (typeof value !== 'string') {
                errors.push('该字段必须是字符串');
                break;
            }
            if (!rule.value.test(value)) {
                errors.push(rule.message);
            }
        }
    }

    return { valid: errors.length === 0, errors };
}

// 使用示例
const result = validate('hello', [
    { type: 'required', message: '不能为空' },
    { type: 'minLength', value: 5, message: '最少 5 个字符' }
]);

console.log(result);  // { valid: true, errors: [] }
```

### typeof 的局限性

```typescript
// typeof 只能检查基本类型和字面量类型
type Result = {
    type: 'success';
    data: string;
} | {
    type: 'error';
    error: Error;
};

function handleResult(result: Result) {
    // typeof result.type 只能得到 'success' | 'error'
    // 无法直接推断出整个对象的类型

    // ❌ 错误：typeof 无法区分对象结构
    // if (typeof result === 'success') {}  // 无效

    // ✅ 正确：使用相等性检查 + 可辨识联合
    if (result.type === 'success') {
        // result: { type: 'success'; data: string }
        console.log(result.data);
    } else {
        // result: { type: 'error'; error: Error }
        console.error(result.error);
    }
}
```

## 5.3 instanceof 守卫

### 基础用法

```typescript
// 自定义类
class Bird {
    fly() {
        console.log('🐦 鸟在飞翔');
    }
}

class Fish {
    swim() {
        console.log('🐟 鱼在游泳');
    }
}

function move(animal: Bird | Fish) {
    if (animal instanceof Bird) {
        // animal: Bird
        animal.fly();
    } else {
        // animal: Fish
        animal.swim();
    }
}

// 使用
move(new Bird());  // 🐦 鸟在飞翔
move(new Fish());  // 🐟 鱼在游泳
```

### 内置对象检查

```typescript
// 日期处理
function formatDate(value: Date | string | number): string {
    if (value instanceof Date) {
        return value.toISOString();
    } else if (typeof value === 'string') {
        return new Date(value).toISOString();
    } else {
        return new Date(value).toISOString();
    }
}

// 错误处理
function handleError(error: Error | string | unknown): void {
    if (error instanceof Error) {
        console.error('Error 对象:', error.message);
        console.error('堆栈:', error.stack);
    } else if (typeof error === 'string') {
        console.error('字符串错误:', error);
    } else {
        console.error('未知错误:', error);
    }
}

// 数组类型检查
function processArray(value: string[] | ArrayBuffer | Uint8Array) {
    if (value instanceof ArrayBuffer) {
        console.log('ArrayBuffer，byteLength:', value.byteLength);
    } else if (value instanceof Uint8Array) {
        console.log('Uint8Array，length:', value.length);
    } else {
        console.log('字符串数组:', value);
    }
}
```

### 实际应用：HTTP 响应处理

```typescript
// 自定义错误类
class HttpError extends Error {
    constructor(
        public status: number,
        message: string
    ) {
        super(message);
        this.name = 'HttpError';
    }
}

class NetworkError extends Error {
    constructor(message: string) {
        super(message);
        this.name = 'NetworkError';
    }
}

async function fetchData(url: string): Promise<string> {
    try {
        const response = await fetch(url);

        if (!response.ok) {
            throw new HttpError(response.status, `HTTP ${response.status}`);
        }

        return await response.text();
    } catch (error) {
        if (error instanceof HttpError) {
            // HTTP 错误，可以访问 status
            console.error(`HTTP 错误 ${error.status}: ${error.message}`);

            // 根据状态码处理
            if (error.status === 401) {
                // 跳转登录
                window.location.href = '/login';
            } else if (error.status === 403) {
                // 权限不足
                alert('没有访问权限');
            }
        } else if (error instanceof NetworkError) {
            // 网络错误
            console.error('网络错误:', error.message);
        } else if (error instanceof Error) {
            // 其他 Error
            console.error('错误:', error.message);
        } else {
            // 未知错误
            console.error('未知错误:', error);
        }
        throw error;
    }
}
```

## 5.4 in 操作符守卫

### 基础用法

```typescript
interface User {
    name: string;
    email: string;
}

interface Product {
    title: string;
    price: number;
}

function describe(value: User | Product): string {
    // 使用 in 检查属性是否存在
    if ('name' in value) {
        // value: User
        return `用户：${value.name} (${value.email})`;
    } else {
        // value: Product
        return `商品：${value.title} - ¥${value.price}`;
    }
}

// 使用
console.log(describe({ name: '张三', email: 'zhang@example.com' }));
console.log(describe({ title: '手机', price: 5999 }));
```

### 检查多个属性

```typescript
type Shape =
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; side: number }
    | { kind: 'rectangle'; width: number; height: number }
    | { kind: 'triangle'; base: number; height: number };

function getArea(shape: Shape): number {
    // 使用 kind 字段进行区分（可辨识联合）
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'square':
            return shape.side ** 2;
        case 'rectangle':
            return shape.width * shape.height;
        case 'triangle':
            return (shape.base * shape.height) / 2;
        default:
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}

// 或者使用 in 进行复杂检查
function getPerimeter(shape: Shape): number {
    if ('radius' in shape) {
        // circle
        return 2 * Math.PI * shape.radius;
    } else if ('side' in shape) {
        // square
        return 4 * shape.side;
    } else if ('width' in shape && 'height' in shape) {
        if ('base' in shape) {
            // triangle
            return shape.base + shape.height + Math.sqrt(shape.base ** 2 + shape.height ** 2);
        }
        // rectangle
        return 2 * (shape.width + shape.height);
    }
    return 0;
}
```

## 5.5 类型谓词（Type Predicate）

### 自定义类型守卫

类型谓词是一种特殊的返回类型，语法为 `parameterName is Type`。

```typescript
// 类型谓词语法
function isString(value: unknown): value is string {
    return typeof value === 'string';
}

// 使用
function process(value: unknown) {
    if (isString(value)) {
        // TypeScript 知道 value 是 string 类型
        console.log(value.toUpperCase());
    }
}
```

### 实用类型守卫函数

```typescript
// 检查是否为数字
function isNumber(value: unknown): value is number {
    return typeof value === 'number' && !isNaN(value);
}

// 检查是否为数组
function isArray(value: unknown): value is unknown[] {
    return Array.isArray(value);
}

// 检查是否为对象（非数组、非 null）
function isObject(value: unknown): value is Record<string, unknown> {
    return typeof value === 'object' && value !== null && !Array.isArray(value);
}

// 检查是否为 null 或 undefined
function isNullOrUndefined(value: unknown): value is null | undefined {
    return value === null || value === undefined;
}

// 检查是否为空字符串
function isEmptyString(value: unknown): value is '' {
    return typeof value === 'string' && value.length === 0;
}

// 使用示例
function processAny(value: unknown) {
    if (isNumber(value)) {
        console.log('数字:', value.toFixed(2));
    } else if (isArray(value)) {
        console.log('数组长度:', value.length);
    } else if (isObject(value)) {
        console.log('对象键:', Object.keys(value));
    } else if (isNullOrUndefined(value)) {
        console.log('空值');
    } else if (isEmptyString(value)) {
        console.log('空字符串');
    } else {
        // value: never
        console.log('其他类型:', typeof value);
    }
}
```

### 泛型类型守卫

```typescript
// 泛型类型守卫
function isPresent<T>(value: T | null | undefined): value is T {
    return value !== null && value !== undefined;
}

// 使用
const items: (string | null | undefined)[] = ['a', null, 'b', undefined, 'c'];
const validItems = items.filter(isPresent);  // string[]

// 指定类型的类型守卫
function isType<T>(value: unknown, type: string): value is T {
    return typeof value === type;
}

// 使用
if (isType<string>(value, 'string')) {
    // value: string
    console.log(value.length);
}
```

### 属性存在性类型守卫

```typescript
interface ApiResponse {
    data?: {
        user: {
            id: number;
            name: string;
        };
    };
    error?: {
        code: number;
        message: string;
    };
}

// 使用 in 的类型守卫
function hasData(response: ApiResponse): response is ApiResponse & { data: NonNullable<ApiResponse['data']> } {
    return response.data !== undefined && response.data !== null;
}

function hasError(response: ApiResponse): response is ApiResponse & { error: NonNullable<ApiResponse['error']> } {
    return response.error !== undefined && response.error !== null;
}

// 使用
function handleResponse(response: ApiResponse) {
    if (hasData(response)) {
        // TypeScript 知道 data 存在
        console.log('用户:', response.data.user.name);
    } else if (hasError(response)) {
        // TypeScript 知道 error 存在
        console.error('错误:', response.error.message);
    }
}
```

## 5.6 可辨识联合（Discriminated Union）

### 基础概念

可辨识联合（也称为标签联合、代数数据类型）是一种设计模式，它使用一个共同的字面量类型字段来区分联合中的不同类型。

```typescript
// 可辨识联合的三个要素：
// 1. 联合类型
// 2. 共同的鉴别字段（discriminant）
// 3. 字面量类型值

interface Square {
    kind: 'square';      // 鉴别字段
    side: number;
}

interface Circle {
    kind: 'circle';      // 相同的鉴别字段
    radius: number;
}

interface Triangle {
    kind: 'triangle';    // 相同的鉴别字段
    base: number;
    height: number;
}

type Shape = Square | Circle | Triangle;

// 使用鉴别字段进行类型收窄
function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'square':
            return shape.side * shape.side;
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'triangle':
            return (shape.base * shape.height) / 2;
        default:
            // 完整性检查：如果新增类型，这里会报错
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}
```

### 实际应用：API 响应

```typescript
// API 响应类型
interface SuccessResponse<T> {
    status: 'success';
    data: T;
    timestamp: number;
}

interface ErrorResponse {
    status: 'error';
    error: {
        code: string;
        message: string;
        details?: Record<string, string>;
    };
}

interface LoadingResponse {
    status: 'loading';
    progress?: number;
}

type APIResponse<T> = SuccessResponse<T> | ErrorResponse | LoadingResponse;

// 处理响应
function handleApiResponse<T>(response: APIResponse<T>): void {
    switch (response.status) {
        case 'success':
            // response: SuccessResponse<T>
            console.log('成功:', response.data);
            console.log('时间戳:', new Date(response.timestamp));
            break;
        case 'error':
            // response: ErrorResponse
            console.error('错误代码:', response.error.code);
            console.error('错误消息:', response.error.message);
            if (response.error.details) {
                console.error('详细信息:', response.error.details);
            }
            break;
        case 'loading':
            // response: LoadingResponse
            console.log('加载中...', response.progress ? `${response.progress}%` : '');
            break;
        default:
            const _exhaustive: never = response;
    }
}

// 使用示例
const responses: APIResponse<string[]>[] = [
    { status: 'success', data: ['item1', 'item2'], timestamp: Date.now() },
    { status: 'error', error: { code: 'NOT_FOUND', message: '资源不存在' } },
    { status: 'loading', progress: 50 }
];

responses.forEach(handleApiResponse);
```

### 实际应用：Redux Action

```typescript
// Action 类型定义（使用可辨识联合）
interface IncrementAction {
    type: 'counter/increment';
    payload: { amount: number };
}

interface DecrementAction {
    type: 'counter/decrement';
    payload: { amount: number };
}

interface ResetAction {
    type: 'counter/reset';
}

interface SetValueAction {
    type: 'counter/set';
    payload: number;
}

type CounterAction =
    | IncrementAction
    | DecrementAction
    | ResetAction
    | SetValueAction;

// State 类型
interface CounterState {
    value: number;
}

// Reducer
function counterReducer(
    state: CounterState = { value: 0 },
    action: CounterAction
): CounterState {
    switch (action.type) {
        case 'counter/increment':
            return { value: state.value + action.payload.amount };
        case 'counter/decrement':
            return { value: state.value - action.payload.amount };
        case 'counter/reset':
            return { value: 0 };
        case 'counter/set':
            return { value: action.payload };
        default:
            // 完整性检查
            const _exhaustive: never = action;
            console.warn('Unknown action type:', _exhaustive);
            return state;
    }
}

// 使用
let state: CounterState = { value: 0 };
state = counterReducer(state, { type: 'counter/increment', payload: { amount: 5 } });
state = counterReducer(state, { type: 'counter/decrement', payload: { amount: 2 } });
state = counterReducer(state, { type: 'counter/set', payload: 100 });
state = counterReducer(state, { type: 'counter/reset' });
```

## 5.7 相等性检查和真值检查

### 相等性检查

```typescript
// 检查 null 和 undefined
function greet(name: string | null | undefined): string {
    if (name === null || name === undefined) {
        return '你好，访客';
    }
    // name: string
    return `你好，${name}`;
}

// 使用可选链和空值合并
function greet2(name: string | null | undefined): string {
    const displayName = name ?? '访客';
    return `你好，${displayName}`;
}

// 字面量联合类型的相等性检查
type Status = 'pending' | 'approved' | 'rejected';

function handleStatus(status: Status): void {
    if (status === 'pending') {
        console.log('等待审批...');
    } else if (status === 'approved') {
        console.log('已批准');
    } else {
        // status: 'rejected'
        console.log('已拒绝');
    }
}
```

### 真值检查

```typescript
// 真值检查收窄类型
function getLength(value: string | null | undefined): number {
    if (value) {
        // value: string（排除 null 和 undefined）
        return value.length;
    }
    return 0;
}

// 注意事项：真值检查会排除所有 falsy 值
function checkValue(value: string | 0 | null) {
    if (value) {
        // value: string（排除 0 和 null）
        // 但也会排除空字符串 ''
        return value;
    }
    return 'default';
}

// 使用更精确的检查
function checkValuePrecise(value: string | 0 | null) {
    if (value !== null && value !== undefined) {
        // value: string | 0
        return value;
    }
    return 'default';
}
```

## 5.8 条件类型（Conditional Types）

### 基础语法

条件类型允许我们根据类型关系来选择不同的类型。

```typescript
// 语法：T extends U ? X : Y
type IsString<T> = T extends string ? '是字符串' : '不是字符串';

type T0 = IsString<string>;   // '是字符串'
type T1 = IsString<number>;   // '不是字符串'
type T2 = IsString<'hello'>;  // '是字符串'（字面量类型也是字符串）

// 实际应用：判断是否为空对象
type IsEmptyObject<T> = keyof T extends never ? true : false;

type T3 = IsEmptyObject<{}>;           // true
type T4 = IsEmptyObject<{ a: 1 }>;     // false
```

### 分布式条件类型

当条件类型作用于联合类型时，会分布地应用。

```typescript
// 分布式条件类型
type ToArray<T> = T extends any ? T[] : never;

type T0 = ToArray<string>;                        // string[]
type T1 = ToArray<string | number>;               // string[] | number[]
type T2 = ToArray<string | number | boolean>;     // string[] | number[] | boolean[]

// 非分布式（使用元组包装）
type ToArrayNonDistributed<T> = [T] extends [any] ? T[] : never;

type T3 = ToArrayNonDistributed<string | number>;  // (string | number)[]
```

### infer 关键字

`infer` 用于在条件类型中声明一个待推断的类型变量。

```typescript
// 推断返回类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type T0 = MyReturnType<() => string>;              // string
type T1 = MyReturnType<(a: number) => Promise<number>>;  // Promise<number>

// 推断数组元素类型
type MyElementType<T> = T extends (infer U)[] ? U : never;

type T2 = MyElementType<string[]>;        // string
type T3 = MyElementType<(string | number)[]>;  // string | number

// 推断 Promise 值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

type T4 = UnwrapPromise<Promise<string>>;           // string
type T5 = UnwrapPromise<Promise<Promise<number>>>;  // number
type T6 = UnwrapPromise<string>;                    // string

// 推断函数第一个参数类型
type FirstParam<T> = T extends (arg: infer P, ...args: any[]) => any ? P : never;

type T7 = FirstParam<(name: string, age: number) => void>;  // string

// 推断函数参数元组
type ParamsTuple<T> = T extends (...args: infer P) => any ? P : never;

type T8 = ParamsTuple<(a: string, b: number) => void>;  // [string, number]
```

### 内置条件工具类型

```typescript
// Exclude<T, U> - 从 T 中排除可以赋值给 U 的类型
type T0 = Exclude<'a' | 'b' | 'c', 'a'>;           // 'b' | 'c'
type T1 = Exclude<string | number | null, null>;   // string | number

// Extract<T, U> - 从 T 中提取可以赋值给 U 的类型
type T2 = Extract<'a' | 'b' | 'c', 'a' | 'b'>;     // 'a' | 'b'
type T3 = Extract<string | number | null, number>; // number

// NonNullable<T> - 从 T 中排除 null 和 undefined
type T4 = NonNullable<string | null | undefined>;  // string
type T5 = NonNullable<string | number | null>;     // string | number

// 源码实现：
// type NonNullable<T> = T extends null | undefined ? never : T;
```

## 5.9 模板字面量类型（Template Literal Types）

### 基础用法

```typescript
// 字符串字面量联合
type Greeting = 'hello' | 'hi' | 'hey';
```

## 5.11 类型收窄综合应用

### 实际应用：API 客户端

```typescript
// 定义请求配置
interface RequestConfig {
    url: string;
    method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
    headers?: Record<string, string>;
    body?: unknown;
}

// 定义响应类型
interface SuccessResponse<T> {
    ok: true;
    data: T;
    status: number;
}

interface ErrorResponse {
    ok: false;
    error: {
        code: string;
        message: string;
    };
    status: number;
}

type Response<T> = SuccessResponse<T> | ErrorResponse;

// 类型守卫
function isSuccess<T>(response: Response<T>): response is SuccessResponse<T> {
    return response.ok === true;
}

// API 客户端类
class ApiClient {
    private baseURL: string;

    constructor(baseURL: string) {
        this.baseURL = baseURL;
    }

    async request<T>(config: RequestConfig): Promise<Response<T>> {
        try {
            const response = await fetch(`${this.baseURL}${config.url}`, {
                method: config.method || 'GET',
                headers: config.headers,
                body: config.body ? JSON.stringify(config.body) : undefined
            });

            const data = await response.json();

            if (response.ok) {
                return {
                    ok: true,
                    data: data as T,
                    status: response.status
                };
            } else {
                return {
                    ok: false,
                    error: data,
                    status: response.status
                };
            }
        } catch (error) {
            return {
                ok: false,
                error: {
                    code: 'NETWORK_ERROR',
                    message: error instanceof Error ? error.message : '未知错误'
                },
                status: 0
            };
        }
    }

    // 便捷方法
    async get<T>(url: string): Promise<Response<T>> {
        return this.request<T>({ url, method: 'GET' });
    }

    async post<T>(url: string, body?: unknown): Promise<Response<T>> {
        return this.request<T>({ url, method: 'POST', body });
    }
}

// 使用
async function example() {
    const api = new ApiClient('https://api.example.com');

    const response = await api.get<{ name: string; email: string }>('/user');

    if (isSuccess(response)) {
        // response: SuccessResponse<{ name: string; email: string }>
        console.log('用户:', response.data.name);
    } else {
        // response: ErrorResponse
        console.error('错误:', response.error.message);
    }
}
```

## 5.12 实践练习

### 练习 1：类型安全的事件系统

```typescript
// 定义事件映射
interface EventMap {
    'user:login': { userId: number; token: string };
    'user:logout': { userId: number };
    'data:update': { key: string; value: unknown };
    'error': { code: string; message: string };
}

// 事件处理器类型
type EventHandler<T> = (data: T) => void;

// 类型安全的事件总线
class TypedEventBus {
    private handlers = new Map<keyof EventMap, Set<EventHandler<any>>>();

    on<K extends keyof EventMap>(event: K, handler: EventHandler<EventMap[K]>): void {
        if (!this.handlers.has(event)) {
            this.handlers.set(event, new Set());
        }
        this.handlers.get(event)!.add(handler);
    }

    off<K extends keyof EventMap>(event: K, handler: EventHandler<EventMap[K]>): void {
        const set = this.handlers.get(event);
        if (set) {
            set.delete(handler);
        }
    }

    emit<K extends keyof EventMap>(event: K, data: EventMap[K]): void {
        const set = this.handlers.get(event);
        if (set) {
            set.forEach(handler => handler(data));
        }
    }
}

// 使用
const bus = new TypedEventBus();

bus.on('user:login', ({ userId, token }) => {
    console.log(`用户 ${userId} 登录`);
});

bus.on('error', ({ code, message }) => {
    console.error(`错误 ${code}: ${message}`);
});

bus.emit('user:login', { userId: 1, token: 'abc123' });
bus.emit('error', { code: 'UNKNOWN', message: '未知错误' });
```

## 5.13 常见问答

### Q1: typeof 和 instanceof 有什么区别？

| 特性 | typeof | instanceof |
|------|--------|------------|
| 检查对象 | 基本类型 | 构造函数实例 |
| 示例 | `typeof x === 'string'` | `x instanceof Date` |
| 适用 | string, number, boolean | Class, Array, Date, Error |

### Q2: 如何选择合适的类型守卫？

```typescript
// 基本类型 → typeof
if (typeof value === 'string') {}

// 类实例 → instanceof
if (value instanceof MyClass) {}

// 对象属性 → in
if ('property' in value) {}

// 自定义逻辑 → 类型谓词
function isValid(value: unknown): value is ValidType {
    // 自定义检查逻辑
    return true;
}
```

### Q3: 如何理解 infer？

`infer` 用于在条件类型中"捕获"待推断的类型。

```typescript
// 从函数类型中推断返回类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : any;

// 从数组类型中推断元素类型
type ElementType<T> = T extends (infer U)[] ? U : never;

// 从 Promise 中推断值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;
```
