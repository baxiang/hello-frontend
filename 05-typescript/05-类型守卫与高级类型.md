## 5.1 typeof 守卫

### 基础用法

```typescript
function printId(id: number | string) {
    if (typeof id === 'number') {
        // id: number
        console.log(id.toFixed(2));
    } else {
        // id: string
        console.log(id.toUpperCase());
    }
}

// 实际示例
function processValue(value: string | number | boolean) {
    if (typeof value === 'string') {
        return value.trim();
    } else if (typeof value === 'number') {
        return value.toFixed(2);
    } else {
        return value;
    }
}
```

### typeof 的局限性

```typescript
// typeof 只能检查基本类型
interface User {
    name: string;
}

interface Product {
    title: string;
}

function describe(value: User | Product) {
    // typeof 无法区分对象类型
    // if (typeof value === 'User') {}  // 无效

    // 使用 in 操作符
    if ('name' in value) {
        // value: User
        console.log(value.name);
    } else {
        // value: Product
        console.log(value.title);
    }
}
```

## 5.2 instanceof 守卫

### 基础用法

```typescript
class Bird {
    fly() {
        console.log('鸟在飞');
    }
}

class Fish {
    swim() {
        console.log('鱼在游');
    }
}

function move(animal: Bird | Fish) {
    if (animal instanceof Bird) {
        animal.fly();
    } else {
        animal.swim();
    }
}

// 实际示例：日期处理
function formatDate(value: Date | string | number): string {
    if (value instanceof Date) {
        return value.toISOString();
    } else if (typeof value === 'string') {
        return new Date(value).toISOString();
    } else {
        return new Date(value).toISOString();
    }
}
```

### 内置对象的 instanceof

```typescript
function processError(error: Error | string | unknown) {
    if (error instanceof Error) {
        console.log('Error 对象:', error.message);
    } else if (typeof error === 'string') {
        console.log('错误消息:', error);
    } else {
        console.log('未知错误:', error);
    }
}

function processArray(value: string[] | ArrayBuffer | Uint8Array) {
    if (value instanceof ArrayBuffer) {
        console.log('ArrayBuffer');
    } else if (value instanceof Uint8Array) {
        console.log('Uint8Array');
    } else {
        console.log('字符串数组:', value);
    }
}
```

## 5.3 类型谓词

### 自定义类型守卫

```typescript
// 类型谓词语法：parameterName is Type
function isString(value: unknown): value is string {
    return typeof value === 'string';
}

// 使用
function process(value: unknown) {
    if (isString(value)) {
        // value: string
        console.log(value.toUpperCase());
    }
}
```

### 实用的类型守卫

```typescript
// 检查是否为数字
function isNumber(value: unknown): value is number {
    return typeof value === 'number' && !isNaN(value);
}

// 检查是否为对象
function isObject(value: unknown): value is Record<string, unknown> {
    return typeof value === 'object' && value !== null && !Array.isArray(value);
}

// 检查是否为数组
function isArray<T>(value: unknown): value is T[] {
    return Array.isArray(value);
}

// 检查是否为空值
function isNullOrUndefined<T>(value: T | null | undefined): value is null | undefined {
    return value === null || value === undefined;
}

// 使用示例
function processValue(value: unknown) {
    if (isNumber(value)) {
        console.log('数字:', value.toFixed(2));
    } else if (isString(value)) {
        console.log('字符串:', value);
    } else if (isObject(value)) {
        console.log('对象:', Object.keys(value));
    } else if (isArray(value)) {
        console.log('数组:', value);
    }
}
```

### 可枚举的类型守卫

```typescript
// 检查是否为特定枚举值
type Status = 'pending' | 'success' | 'error';

function isPending(status: Status): status is 'pending' {
    return status === 'pending';
}

function isSuccess(status: Status): status is 'success' {
    return status === 'success';
}

// 使用
function handleStatus(status: Status) {
    if (isPending(status)) {
        // status: 'pending'
        console.log('处理中...');
    } else if (isSuccess(status)) {
        // status: 'success'
        console.log('成功!');
    } else {
        // status: 'error'
        console.log('出错!');
    }
}
```

## 5.4 可辨识联合

### 基础概念

```typescript
// 具有共同单例类型的联合
interface Square {
    kind: 'square';
    side: number;
}

interface Circle {
    kind: 'circle';
    radius: number;
}

interface Rectangle {
    kind: 'rectangle';
    width: number;
    height: number;
}

type Shape = Square | Circle | Rectangle;

// 使用 kind 字段进行区分
function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'square':
            return shape.side * shape.side;
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'rectangle':
            return shape.width * shape.height;
        default:
            // 完整性检查
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}
```

### 实际应用场景

```typescript
// API 响应类型
interface SuccessResponse<T> {
    status: 'success';
    data: T;
}

interface ErrorResponse {
    status: 'error';
    error: {
        code: string;
        message: string;
    };
}

interface LoadingResponse {
    status: 'loading';
}

type APIResponse<T> = SuccessResponse<T> | ErrorResponse | LoadingResponse;

// 使用
function handleResponse<T>(response: APIResponse<T>) {
    switch (response.status) {
        case 'success':
            console.log('数据:', response.data);
            break;
        case 'error':
            console.error('错误:', response.error.message);
            break;
        case 'loading':
            console.log('加载中...');
            break;
        default:
            const _exhaustive: never = response;
    }
}
```

### Redux Action 类型

```typescript
// Action 类型定义
interface IncrementAction {
    type: 'counter/increment';
    payload: { amount: number };
}

interface DecrementAction {
    type: 'counter/decrement';
    payload: { amount: number };
}

interface ResetAction {
    type: 'counter/reset';
}

type CounterAction = IncrementAction | DecrementAction | ResetAction;

// Reducer
interface CounterState {
    value: number;
}

function counterReducer(
    state: CounterState,
    action: CounterAction
): CounterState {
    switch (action.type) {
        case 'counter/increment':
            return { value: state.value + action.payload.amount };
        case 'counter/decrement':
            return { value: state.value - action.payload.amount };
        case 'counter/reset':
            return { value: 0 };
        default:
            const _exhaustive: never = action;
            return state;
    }
}
```

## 5.5 条件类型

### 基础语法

```typescript
// T extends U ? X : Y
type IsString<T> = T extends string ? true : false;

type T0 = IsString<string>;  // true
type T1 = IsString<number>;  // false
type T2 = IsString<'hello'>;  // true
```

### 分布式条件类型

```typescript
// 当 T 是联合类型时，条件类型会分布应用
type ToArray<T> = T extends any ? T[] : never;

type T0 = ToArray<string>;  // string[]
type T1 = ToArray<string | number>;  // string[] | number[]
type T2 = ToArray<string | number | boolean>;  // string[] | number[] | boolean[]

// 非分布式（使用元组包装）
type ToArrayNonDistributed<T> = [T] extends [any] ? T[] : never;

type T3 = ToArrayNonDistributed<string | number>;  // (string | number)[]
```

### infer 推断

```typescript
// 推断返回类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : any;

type T0 = ReturnType<() => string>;  // string
type T1 = ReturnType<(a: number) => Promise<number>>;  // Promise<number>

// 推断数组元素类型
type ElementType<T> = T extends (infer U)[] ? U : never;

type T2 = ElementType<string[]>;  // string
type T3 = ElementType<(string | number)[]>;  // string | number

// 推断 Promise 值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

type T4 = UnwrapPromise<Promise<string>>;  // string
type T5 = UnwrapPromise<Promise<Promise<number>>>;  // number

// 推断函数参数类型
type FirstParameter<T> = T extends (arg: infer P, ...args: any[]) => any ? P : never;

type T6 = FirstParameter<(name: string, age: number) => void>;  // string
```

### 复杂条件类型

```typescript
// 排除 null 和 undefined
type NonNullable<T> = T extends null | undefined ? never : T;

type T0 = NonNullable<string | null>;  // string
type T1 = NonNullable<string | number | null | undefined>;  // string | number

// Extract - 提取可赋值类型
type Extract<T, U> = T extends U ? T : never;

type T2 = Extract<'a' | 'b' | 'c', 'a' | 'b'>;  // 'a' | 'b'

// Exclude - 排除可赋值类型
type Exclude<T, U> = T extends U ? never : T;

type T3 = Exclude<'a' | 'b' | 'c', 'a'>;  // 'b' | 'c'

// 实际应用
type Status = 'pending' | 'success' | 'error' | null;
type ValidStatus = Exclude<Status, null>;  // 'pending' | 'success' | 'error'
```

## 5.6 模板字面量类型

### 基础用法

```typescript
// 字符串字面量联合
type EmailTemplate = 'welcome' | 'reset' | 'verify';
type EmailEvent = `${EmailTemplate}-email`;
// 'welcome-email' | 'reset-email' | 'verify-email'

// 多个联合组合
type Method = 'get' | 'post' | 'put' | 'delete';
type Resource = 'users' | 'posts' | 'comments';
type Endpoint = `${Method} /api/${Resource}`;
// 24 种组合
```

### 字符串工具类型

```typescript
// 内置工具
type T0 = Uppercase<'hello'>;  // 'HELLO'
type T1 = Lowercase<'HELLO'>;  // 'hello'
type T2 = Capitalize<'hello'>;  // 'Hello'
type T3 = Uncapitalize<'Hello'>;  // 'hello'

// 实际应用
type Events = 'click' | 'focus' | 'blur';
type ReactEvents = {
    [K in `on${Capitalize<Events>}`]: () => void;
};
// { onClick: () => void; onFocus: () => void; onBlur: () => void; }
```

### 模式匹配

```typescript
// 字符串结尾判断
type EndsWith<T extends string, U extends string> = T extends `${string}${U}`
    ? true
    : false;

type T0 = EndsWith<'hello.ts', '.ts'>;  // true
type T1 = EndsWith<'hello.js', '.ts'>;  // false

// 字符串开头判断
type StartsWith<T extends string, U extends string> = T extends `${U}${string}`
    ? true
    : false;

type T2 = StartsWith<'@vue/cli', '@'>;  // true

// 去除前缀
type RemovePrefix<T extends string, U extends string> = T extends `${U}${infer Rest}`
    ? Rest
    : T;

type T3 = RemovePrefix<'@vue/cli', '@'>;  // 'vue/cli'

// 去除后缀
type RemoveSuffix<T extends string, U extends string> = T extends `${infer Rest}${U}`
    ? Rest
    : T;

type T4 = RemoveSuffix<'hello.ts', '.ts'>;  // 'hello'
```

## 5.7 高级映射类型

### 键映射

```typescript
// 键名转换
type Camelize<T> = {
    [K in keyof T as K extends `${infer A}_${infer B}`
        ? `${A}${Capitalize<B>}`
        : K]: T[K];
};

interface UserInfo {
    user_name: string;
    user_age: number;
}

type T0 = Camelize<UserInfo>;
// { userName: string; userAge: number; }

// 过滤属性
type Getters<T> = {
    [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

interface Person {
    name: string;
    age: number;
}

type T1 = Getters<Person>;
// { getName: () => string; getAge: () => number; }

// 排除特定属性
type Mutable<T> = {
    -readonly [K in keyof T as T[K] extends Function ? never : K]: T[K];
};

interface Config {
    readonly id: number;
    update: () => void;
}

type T2 = Mutable<Config>;
// { id: number; }
```

### 值映射

```typescript
// 值转换为 Promise
type Async<T> = {
    [K in keyof T]: Promise<T[K]>;
};

interface API {
    getUser: (id: number) => { name: string };
    getPost: (id: number) => { title: string };
}

type T0 = Async<API>;
// { getUser: Promise<{ name: string }>; getPost: Promise<{ title: string }>; }

// 值转换为可选
type PartialValues<T> = {
    [K in keyof T]: T[K] | undefined;
};

// 深度转换
type DeepReadonly<T> = {
    readonly [K in keyof T]: T[K] extends object ? DeepReadonly<T[K]> : T[K];
};

interface Config {
    api: {
        url: string;
        timeout: number;
    };
}

type T1 = DeepReadonly<Config>;
// { readonly api: { readonly url: string; readonly timeout: number; } }
```

## 5.8 实践练习

### 练习 1：类型安全的路由配置

```typescript
// 定义路由类型
interface RouteConfig {
    path: string;
    name: string;
    component: string;
    meta?: {
        title?: string;
        requiresAuth?: boolean;
    };
}

// 路由守卫类型
type NavigationGuard = (to: RouteConfig, from: RouteConfig) => boolean | string;

// 路由类
class Router {
    private routes: RouteConfig[] = [];
    private guards: NavigationGuard[] = [];

    addRoute(route: RouteConfig): void {
        this.routes.push(route);
    }

    addGuard(guard: NavigationGuard): void {
        this.guards.push(guard);
    }

    navigate(to: string): boolean {
        const route = this.routes.find(r => r.path === to);
        if (!route) return false;

        for (const guard of this.guards) {
            const result = guard(route, { path: '/', name: 'home', component: '' });
            if (result === false) return false;
            if (typeof result === 'string') {
                return this.navigate(result);
            }
        }

        return true;
    }
}

// 使用
const router = new Router();

router.addRoute({
    path: '/home',
    name: 'home',
    component: 'HomePage'
});

router.addGuard((to, from) => {
    if (to.meta?.requiresAuth) {
        const isAuthenticated = true;  // 检查登录状态
        return isAuthenticated ? true : '/login';
    }
    return true;
});
```

### 练习 2：泛型状态管理

```typescript
// Action 类型
type Action<T extends string, P = void> = P extends void
    ? { type: T }
    : { type: T; payload: P };

// Reducer 类型
type Reducer<S, A> = (state: S, action: A) => S;

// Store 类型
interface Store<S, A> {
    getState: () => S;
    dispatch: (action: A) => void;
    subscribe: (listener: (state: S) => void) => () => void;
}

// 创建 Store 的函数
function createStore<S, A extends { type: string }>(
    reducer: Reducer<S, A>,
    initialState: S
): Store<S, A> {
    let state = initialState;
    const listeners: Array<(state: S) => void> = [];

    return {
        getState: () => state,
        dispatch: (action: A) => {
            state = reducer(state, action);
            listeners.forEach(listener => listener(state));
        },
        subscribe: (listener) => {
            listeners.push(listener);
            return () => {
                const index = listeners.indexOf(listener);
                if (index > -1) listeners.splice(index, 1);
            };
        }
    };
}

// 使用示例
interface TodoState {
    todos: { id: number; text: string; completed: boolean }[];
    filter: 'all' | 'active' | 'completed';
}

type TodoAction =
    | Action<'todo/add', { text: string }>
    | Action<'todo/toggle', { id: number }>
    | Action<'todo/setFilter', { filter: 'all' | 'active' | 'completed' }>;

function todoReducer(state: TodoState, action: TodoAction): TodoState {
    switch (action.type) {
        case 'todo/add':
            return {
                ...state,
                todos: [...state.todos, {
                    id: Date.now(),
                    text: action.payload.text,
                    completed: false
                }]
            };
        case 'todo/toggle':
            return {
                ...state,
                todos: state.todos.map(todo =>
                    todo.id === action.payload.id
                        ? { ...todo, completed: !todo.completed }
                        : todo
                )
            };
        case 'todo/setFilter':
            return { ...state, filter: action.payload.filter };
        default:
            return state;
    }
}

const store = createStore(todoReducer, {
    todos: [],
    filter: 'all'
});

store.dispatch({ type: 'todo/add', payload: { text: '学习 TypeScript' } });
```

### 练习 3：类型安全的表单验证

```typescript
// 验证结果类型
type ValidationResult = {
    valid: boolean;
    errors: Record<string, string>;
};

// 验证器类型
type Validator<T = any> = (value: T) => string | null;

// 内置验证器
const validators = {
    required: (message = '必填项'): Validator => (value) => {
        if (value === null || value === undefined || value === '') {
            return message;
        }
        return null;
    },

    minLength: (min: number, message?: string): Validator => (value) => {
        if (typeof value === 'string' && value.length < min) {
            return message ?? `最少${min}个字符`;
        }
        return null;
    },

    maxLength: (max: number, message?: string): Validator => (value) => {
        if (typeof value === 'string' && value.length > max) {
            return message ?? `最多${max}个字符`;
        }
        return null;
    },

    pattern: (pattern: RegExp, message: string): Validator => (value) => {
        if (typeof value === 'string' && !pattern.test(value)) {
            return message;
        }
        return null;
    },

    custom: <T>(fn: (value: T) => boolean, message: string): Validator<T> => (value) => {
        if (!fn(value)) {
            return message;
        }
        return null;
    }
};

// 表单配置类型
interface FormField<T = any> {
    name: string;
    value: T;
    validators: Validator<T>[];
}

interface FormConfig<T extends Record<string, any>> = {
    fields: {
        [K in keyof T]: FormField<T[K]>;
    };
};

// 表单验证类
class FormValidator<T extends Record<string, any>> {
    private config: FormConfig<T>;

    constructor(config: FormConfig<T>) {
        this.config = config;
    }

    validate(): ValidationResult {
        const errors: Record<string, string> = {};

        for (const [key, field] of Object.entries(this.config.fields)) {
            for (const validator of field.validators) {
                const error = validator(field.value);
                if (error) {
                    errors[key] = error;
                    break;
                }
            }
        }

        return {
            valid: Object.keys(errors).length === 0,
            errors
        };
    }

    validateField<K extends keyof T>(fieldName: K): string | null {
        const field = this.config.fields[fieldName];
        for (const validator of field.validators) {
            const error = validator(field.value);
            if (error) return error;
        }
        return null;
    }
}

// 使用示例
interface LoginForm {
    username: string;
    password: string;
    email: string;
}

const loginForm = new FormValidator<LoginForm>({
    fields: {
        username: {
            name: 'username',
            value: '',
            validators: [
                validators.required('请输入用户名'),
                validators.minLength(3, '用户名至少 3 个字符')
            ]
        },
        password: {
            name: 'password',
            value: '',
            validators: [
                validators.required('请输入密码'),
                validators.minLength(6, '密码至少 6 个字符')
            ]
        },
        email: {
            name: 'email',
            value: '',
            validators: [
                validators.required('请输入邮箱'),
                validators.pattern(/^[^\s@]+@[^\s@]+\.[^\s@]+$/, '邮箱格式不正确')
            ]
        }
    }
});

const result = loginForm.validate();
console.log(result);
```

## 5.9 常见问答

### Q1: 类型守卫和类型断言有什么区别？

**答：**
- 类型守卫：运行时检查，类型安全
- 类型断言：编译时提示，需要谨慎使用

```typescript
// 类型守卫（推荐）
function process(value: unknown) {
    if (typeof value === 'string') {
        console.log(value.length);
    }
}

// 类型断言（需要确保正确）
function process(value: unknown) {
    const str = value as string;
    console.log(str.length);  // 运行时可能出错
}
```

### Q2: 如何实现完整的类型检查？

```typescript
// 使用 never 和 exhaustive check
type Status = 'pending' | 'success' | 'error';

function handleStatus(status: Status) {
    switch (status) {
        case 'pending':
            break;
        case 'success':
            break;
        case 'error':
            break;
        default:
            const _exhaustive: never = status;
            // 如果 Status 新增类型，这里会报错
    }
}
```

### Q3: 条件类型中的 infer 怎么用？

```typescript
// 推断返回类型
```
