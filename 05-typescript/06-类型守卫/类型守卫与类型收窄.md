## L1 理解层：会用

### typeof 类型守卫

**最简示例（1-3行）：**

```typescript
function pad(value: string | number, width: number) {
    if (typeof value === 'string') return value.padStart(width);
    return String(value).padStart(width);
}
```

**详细示例：**

```typescript
// typeof 支持的基本类型
type Basic = string | number | boolean | undefined | object | function;

function process(value: string | number | boolean) {
    if (typeof value === 'string') {
        value.toUpperCase();  // ✅ TypeScript 知道这是 string
    } else if (typeof value === 'number') {
        value.toFixed(2);     // ✅ TypeScript 知道这是 number
    } else {
        // boolean
        console.log(value);
    }
}
```

---

### instanceof 类型守卫

```typescript
class Dog { bark() { return 'woof'; } }
class Cat { meow() { return 'meow'; } }

function sound(animal: Dog | Cat) {
    if (animal instanceof Dog) {
        animal.bark();  // ✅ TypeScript 知道这是 Dog
    } else {
        animal.meow();  // ✅ TypeScript 知道这是 Cat
    }
}
```

### in 类型守卫

```typescript
interface Fish { swim(): void; fins: number }
interface Bird { fly(): void; wings: number }

function move(animal: Fish | Bird) {
    if ('swim' in animal) {
        animal.swim();  // ✅ TypeScript 知道这是 Fish
    } else {
        animal.fly();   // ✅ TypeScript 知道这是 Bird
    }
}
```

### 类型谓词

**最简示例：**

```typescript
function isString(value: unknown): value is string {
    return typeof value === 'string';
}
```

**详细示例：**

```typescript
interface User { id: number; name: string }

function isUser(value: unknown): value is User {
    return (
        typeof value === 'object' &&
        value !== null &&
        'id' in value &&
        'name' in value &&
        typeof (value as any).id === 'number' &&
        typeof (value as any).name === 'string'
    );
}

// 使用
function processValue(value: unknown) {
    if (isUser(value)) {
        console.log(value.name);  // ✅ TypeScript 知道这是 User
    }
}
```

## L2 实践层：用好

### 类型守卫速查

| 守卫 | 适用 | 示例 |
|------|------|------|
| `typeof` | 基本类型 | `typeof x === 'string'` |
| `instanceof` | 类实例 | `x instanceof Dog` |
| `in` | 属性检查 | `'name' in x` |
| 类型谓词 | 自定义检查 | `function isUser(x): x is User` |
| 字面量判断 | 可辨识联合 | `shape.kind === 'circle'` |

### 反模式

```typescript
// ❌ 错误：用 any 绕过类型检查
function process(x: any) {
    if (typeof x === 'string') x.toUpperCase();
}

// ✅ 正确：用 unknown + 类型守卫
function process(x: unknown) {
    if (typeof x === 'string') x.toUpperCase();
}
```

```typescript
// ❌ 错误：类型谓词返回错误结果
function isUser(x: unknown): x is User {
    return true;  // 总是返回 true，不安全
}

// ✅ 正确：严格检查
function isUser(x: unknown): x is User {
    return typeof x === 'object' && x !== null && 'id' in x && 'name' in x;
}
```

### 适用场景

| 场景 | 推荐方案 | 示例 |
|------|---------|------|
| 基本类型判断 | `typeof` | `typeof x === 'string'` |
| 类实例判断 | `instanceof` | `x instanceof Error` |
| 接口/对象判断 | `in` | `'id' in obj` |
| 复杂验证 | 类型谓词 | `isUser(obj)` |
| 联合类型分支 | switch/case | `switch (shape.kind)` |

## L3 专家层：深入

### 可辨识联合的完整性检查

```typescript
type Shape =
    | { kind: 'circle'; radius: number }
    | { kind: 'square'; side: number }
    | { kind: 'triangle'; base: number; height: number };

function area(shape: Shape): number {
    switch (shape.kind) {
        case 'circle': return Math.PI * shape.radius ** 2;
        case 'square': return shape.side ** 2;
        case 'triangle': return (shape.base * shape.height) / 2;
        default:
            // 如果 Shape 新增类型但忘记处理，这里会报错
            const _exhaustive: never = shape;
            return _exhaustive;
    }
}
```

### 自定义类型守卫的陷阱

```typescript
// ❌ 陷阱：类型谓词不检查运行时类型
function isNumberArray(x: unknown): x is number[] {
    return Array.isArray(x);  // 没有检查元素类型！
}

// ✅ 正确：检查每个元素
function isNumberArray(x: unknown): x is number[] {
    return Array.isArray(x) && x.every(item => typeof item === 'number');
}
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **类型守卫** | 运行时检查类型的表达式 | `typeof x === 'string'` |
| **类型收窄** | 缩小变量的可能类型范围 | `if (typeof x === 'string')` |
| **类型谓词** | 自定义类型守卫的返回类型 | `x is User` |
| **可辨识联合** | 通过共同字段区分的联合 | `{ kind: 'A' } \| { kind: 'B' }` |
| **完整性检查** | 确保所有分支都被处理 | `const _: never = value` |

## 实践练习

```typescript
// 练习 1：API 响应类型守卫
interface SuccessResponse<T> { success: true; data: T }
interface ErrorResponse { success: false; error: string }

type ApiResponse<T> = SuccessResponse<T> | ErrorResponse;

function isSuccess<T>(response: ApiResponse<T>): response is SuccessResponse<T> {
    return response.success === true;
}

// 练习 2：类型安全的 JSON 解析
function safeParse(json: string): unknown {
    try {
        return JSON.parse(json);
    } catch {
        return null;
    }
}

// 练习 3：实现 isPlainObject
function isPlainObject(x: unknown): x is Record<string, unknown> {
    return (
        typeof x === 'object' &&
        x !== null &&
        Object.prototype.toString.call(x) === '[object Object]'
    );
}
```

## 常见问题

### Q1：类型谓词和类型断言有什么区别？

**类型谓词是运行时检查，类型断言是编译时告诉编译器"相信我"：**

```typescript
// 类型谓词 — 运行时检查
function isUser(x: unknown): x is User {
    return typeof x === 'object' && x !== null && 'id' in x;
}

// 类型断言 — 编译时告诉编译器
const user = value as User;  // 不检查，直接告诉编译器这是 User
```

### Q2：为什么需要 never 类型做完整性检查？

**如果联合类型新增成员但忘记处理，never 会让编译器报错：**

```typescript
type Status = 'pending' | 'success';

function handle(s: Status) {
    switch (s) {
        case 'pending': break;
        // 漏了 'success'
        default:
            const _: never = s;  // ❌ 报错：Type '"success"' is not assignable to 'never'
    }
}
```
