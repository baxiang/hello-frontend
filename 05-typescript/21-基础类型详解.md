## 5.1 基础类型一览

### 类型列表

```typescript
// 布尔
let isDone: boolean = false;

// 数字（整数、浮点数、进制）
let decimal: number = 6;

let octal: number = 0o744;

// 字符串

let list: number[] = [1, 2, 3];
```

## 5.2 数组类型

### 两种声明方式

```typescript
// 方式一：类型[]
let numbers: number[] = [1, 2, 3];
let strings: string[] = ['a', 'b', 'c'];
let booleans: boolean[] = [true, false];

// 方式二：Array<类型>
let numbers2: Array<number> = [1, 2, 3];
let strings2: Array<string> = ['a', 'b', 'c'];
```

### 混合类型数组

```typescript
// 允许 string 或 number
let mixed: (string | number)[] = [1, 'a', 2, 'b'];
```

## 5.3 元组类型

### 什么是元组

元组是固定长度、固定类型的数组：

```typescript
// 声明元组
let x: [string, number];

// 赋值
x = ['hello', 10];  // ✅
x = [10, 'hello'];  // ❌ 顺序必须一致

// 访问
console.log(x[0]);  // string
console.log(x[1]);  // number
```

### 实际应用

```typescript
// 用户信息 [姓名, 年龄, 是否VIP]
let user: [string, number, boolean] = ['Tom', 25, true];

// API 返回 [状态码, 消息]
function getStatus(): [number, string] {
    return [200, 'OK'];
}
```

### 可选元组元素

```typescript
// 第三个元素可选
let user: [string, number, boolean?] = ['Tom', 25];

// 访问可选元素
console.log(user[2]);  // undefined
```

## 5.4 枚举类型

### 基本用法

```typescript
enum Color {
    Red,     // 0
    Green,   // 1
    Blue     // 2
}

let c: Color = Color.Green;
console.log(c);  // 1
```

### 手动赋值

```typescript
enum Status {
    Success = 200,
    NotFound = 404,
    Error = 500
}

console.log(Status.Success);  // 200
```

### 字符串枚举

```typescript
enum Direction {
    Up = 'UP',
    Down = 'DOWN',
    Left = 'LEFT',
    Right = 'RIGHT'
}

let dir: Direction = Direction.Up;
console.log(dir);  // 'UP'
```

### 枚举反向映射

```typescript
enum Color {
    Red = 1,
    Green,
    Blue
}

console.log(Color[1]);  // 'Red'
console.log(Color[2]);  // 'Green'
```

## 5.5 any 类型

### 什么是 any

any 表示任意类型，关闭类型检查：

```typescript
let value: any = 'hello';
value = 123;
value = true;
value = { name: 'Tom' };

// 任何操作都不会报错
value.foo.bar.baz;  // 不报错
value();             // 不报错
```

### 使用场景

```typescript
// 1. 迁移 JavaScript 代码
let data: any = JSON.parse('{ "name": "Tom" }');

// 2. 动态内容
function process(input: any): any {
    return input;
}
```

### ⚠️ 谨慎使用

```typescript
// ❌ 不推荐
let a: any = 'hello';
a.toFixed(2);  // 运行时不报错，但无意义

// ✅ 推荐使用 unknown
let u: unknown = 'hello';
// u.toFixed(2);  // ❌ 编译错误
if (typeof u === 'number') {
    u.toFixed(2);  // ✅
}
```

## 5.6 unknown 类型

### 什么是 unknown

unknown 是类型安全的 any，使用前必须检查：

```typescript
let value: unknown = 'hello';

// ❌ 不能直接使用
value.toUpperCase();  // 错误

// ✅ 需要类型检查
if (typeof value === 'string') {
    value.toUpperCase();  // OK
}
```

### 与 any 对比

| 操作 | any | unknown |
|------|-----|---------|
| 赋值任意值 | ✅ | ✅ |
| 直接调用方法 | ✅ | ❌ |
| 类型检查后使用 | ✅ | ✅ |

```typescript
// any - 无检查
let a: any = 'hello';
a.toFixed(2);  // ❌ 运行时不报错，但无意义

// unknown - 安全
let u: unknown = 'hello';
// u.toFixed(2);  // ❌ 编译错误
if (typeof u === 'number') {
    u.toFixed(2);  // ✅
}
```

## 5.7 never 类型

### 什么是 never

never 表示永不返回的类型：

```typescript
// 抛出错误的函数
function error(message: string): never {
    throw new Error(message);
}

// 无限循环
function infiniteLoop(): never {
    while (true) {}
}
```

### 使用场景

#### 1. 类型穷尽检查

```typescript
function handleValue(value: string | number) {
    if (typeof value === 'string') {
        console.log('String:', value);
    } else if (typeof value === 'number') {
        console.log('Number:', value);
    } else {
        // value 是 never，表示所有情况已处理
        const _exhaustive: never = value;
    }
}
```

#### 2. 联合类型

```typescript
type Result = 'success' | 'error';

function handle(result: Result) {
    if (result === 'success') {
        console.log('OK');
    } else if (result === 'error') {
        console.log('Failed');
    } else {
        // 穷尽检查
        const _check: never = result;
    }
}
```

## 5.8 void 类型

### 什么是 void

void 表示无返回值：

```typescript
function log(message: string): void {
    console.log(message);
    // 没有 return 或 return undefined
}
```

### void vs never

```typescript
// void - 函数正常返回，但没有返回值
function fn1(): void { }

// never - 函数永不返回
function fn2(): never {
    throw new Error('error');
}
```

## 5.9 null 和 undefined

### 基础用法

```typescript
let n: null = null;
let u: undefined = undefined;

// 严格模式下，不能直接赋值
let name: string = null;  // ❌ 错误
let name2: string | null = null;  // ✅
```

### strictNullChecks

```json
{
    "strictNullChecks": true
}
```

开启后，需要显式处理 null/undefined：

```typescript
// 之前
let name: string = undefined;  // ✅

// 之后（strictNullChecks: true）
let name: string | undefined = undefined;  // ✅
```

## 5.10 object 类型

### 基本用法

```typescript
let obj: object = { name: 'Tom' };
let arr: object = [1, 2, 3];
let fn: object = function() {};

// 更具体的类型
let user: { name: string; age: number } = {
    name: 'Tom',
    age: 25
};
```

## 5.11 实践练习

### 练习 1：基础类型

```typescript
// 声明以下变量
let isActive: boolean = true;
let count: number = 42;
let name: string = 'Tom';
let numbers: number[] = [1, 2, 3];
```

### 练习 2：元组

```typescript
// 创建一个坐标元组
let coordinate: [number, number] = [100, 200];

// 创建一个用户信息元组 [名字, 年龄, 城市]
let userInfo: [string, number, string] = ['Tom', 25, 'Beijing'];
```

### 练习 3：枚举

```typescript
// 创建订单状态枚举
enum OrderStatus {
    Pending = 'pending',
    Processing = 'processing',
    Completed = 'completed',
    Cancelled = 'cancelled'
}

let status: OrderStatus = OrderStatus.Pending;
```

### 练习 4：any vs unknown

```typescript
// 使用 any（不推荐）
function parseAny(data: any): any {
    return data;
}

// 使用 unknown（推荐）
function parseUnknown(data: unknown): unknown {
    return data;
}
```

## 本章小结

| 类型 | 说明 | 示例 |
|------|------|------|
| boolean | 布尔值 | `true`, `false` |
| number | 数字 | `42`, `3.14` |
| string | 字符串 | `'hello'` |
| array | 数组 | `number[]` |
| tuple | 元组 | `[string, number]` |
| enum | 枚举 | `enum Color { Red }` |
| any | 任意类型 | 关闭检查 |
| unknown | 未知类型 | 安全版 any |
| never | 永不返回 | 错误处理 |
| void | 无返回值 | `(): void` |
| object | 对象 | `{ x: number }` |
