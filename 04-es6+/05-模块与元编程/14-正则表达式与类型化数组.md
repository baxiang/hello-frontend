## 14.1 正则表达式增强

### 14.1.1 dotAll（点号匹配换行）

```javascript
// 之前：点号不匹配换行符
const text = 'hello\nworld';
console.log(/hello.world/.test(text));  // false

// 现在：使用 s 标志
console.log(/hello.world/s.test(text));  // true
```

### 14.1.2 后行断言（Lookbehind）

```javascript
// 先行断言：(?=) 前面是
const str = '$100 $200 $300';

// 匹配 $ 后面是数字的
const matches = str.match(/\$\d+/g);
console.log(matches);  // ['$100', '$200', '$300']

// 后行断言：(?<=) 前面是
// 匹配数字前面是 $ 的
const regex = /(?<=\$)\d+/g;
console.log(str.match(regex));  // ['100', '200', '300']

// 负后行断言：(?<!) 前面不是
const str2 = '$100 ¥200 $300';
const regex2 = /(?<!\$)\d+/g;
console.log(str2.match(regex2));  // ['200', '300']
```

### 14.1.3 命名捕获组

```javascript
// 之前：数字索引
const date = '2024-01-15';
const match = date.match(/(\d{4})-(\d{2})-(\d{2})/);
console.log(match[1], match[2], match[3]);  // 2024 01 15

// 现在：命名捕获组
const regex = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/;
const namedMatch = date.match(regex);
console.log(namedMatch.groups.year);   // '2024'
console.log(namedMatch.groups.month);  // '01'
console.log(namedMatch.groups.day);   // '15'
```

## 14.2 类型化数组

### 14.2.1 ArrayBuffer

```javascript
// 创建 ArrayBuffer
const buffer = new ArrayBuffer(16);  // 16 字节
console.log(buffer.byteLength);  // 16

// 查看二进制数据
const view = new Uint8Array(buffer);
console.log(view.length);  // 16
```

### 14.2.2 TypedArray

```javascript
// Int8Array - 8位有符号整数
const int8 = new Int8Array([1, 2, 3]);
console.log(int8[0]);  // 1

// Uint8Array - 8位无符号整数
const uint8 = new Uint8Array([1, 2, 3]);
console.log(uint8[0]);  // 1

// Int16Array - 16位有符号整数
const int16 = new Int16Array([1000, 2000]);

// Float32Array - 32位浮点数
const float32 = new Float32Array([1.5, 2.5]);
console.log(float32[0]);  // 1.5
```

### 14.2.3 DataView

```javascript
// DataView - 灵活读取不同类型
const buffer = new ArrayBuffer(8);
const view = new DataView(buffer);

// 写入
view.setUint8(0, 255);
view.setUint16(2, 65535);
view.setFloat64(4, 3.14159);

// 读取
console.log(view.getUint8(0));    // 255
console.log(view.getUint16(2));   // 65535
console.log(view.getFloat64(4));  // 3.14159
```

## 14.3 实际应用场景

### 14.3.1 颜色处理

```javascript
// 解析十六进制颜色
function hexToRgb(hex) {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16)
    } : null;
}

console.log(hexToRgb('#ff0000'));  // { r: 255, g: 0, b: 0 }
```

### 14.3.2 二进制数据处理

```javascript
// 读取网络请求的二进制数据
async function fetchBinary(url) {
    const response = await fetch(url);
    const buffer = await response.arrayBuffer();
    const uint8 = new Uint8Array(buffer);
    
    // 处理每个字节
    for (let i = 0; i < uint8.length; i++) {
        console.log(`Byte ${i}: ${uint8[i]}`);
    }
}
```

## 14.4 常见问答

### Q1: dotAll 什么时候用？

```javascript
// 匹配多行 HTML 内容
const html = '<div>\n  <span>text</span>\n</div>';
const match = html.match(/<div>.*?<\/div>/s);
```

### Q2: 类型化数组和普通数组有什么区别？

```javascript
// 普通数组：可以存任意类型
const arr = [1, 'a', true];

// 类型化数组：只能存固定类型，更高效
const int8 = new Int8Array([1, 2, 3]);
```
