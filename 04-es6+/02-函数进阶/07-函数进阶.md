## 7.0 为什么要学函数进阶？

### 7.0.1 故事背景

ES6 对函数进行了重大增强：

```
┌─────────────────────────────────────────────────────────────┐
│                    ES6 函数新特性                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 参数默认值                                              │
│     → 不再需要检查参数是否为 undefined                      │
│                                                             │
│  2. rest 参数                                               │
│     → 更优雅地处理不确定数量的参数                          │
│                                                             │
│  3. 展开运算符                                              │
│     → 将数组展开为参数列表                                  │
│                                                             │
│  4. name 属性                                               │
│     → 获取函数的名称                                        │
│                                                             │
│  5. 箭头函数                                                │
│     → 更简洁的语法 + 词法 this 绑定                         │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 7.0.2 本章学习路径

```
第一步：参数默认值（让参数更灵活）
    ↓
第二步：rest 参数（处理不确定数量的参数）
    ↓
```

## 7.1 参数默认值详解

### 7.1.1 基本语法

```javascript
// ES5 方式：检查 undefined
function greet(name) {
    name = name || '访客';
    console.log('你好，' + name);
}

// ES6 默认参数
function greet(name = '访客') {
    console.log('你好，' + name);
}

greet();           // 你好，访客
greet('张三');     // 你好，张三
```

### 7.1.2 默认值触发条件

```javascript
// 只有 undefined 会触发默认值
function fn(x = '默认值') {
    console.log(x);
}

fn();        // '默认值'（没传参数）
fn(null);    // null（null 不会触发默认值！）
fn(undefined); // '默认值'
fn(0);       // 0（0 不会触发默认值！）
fn('');      // ''（空字符串不会触发默认值！）
```

### 7.1.3 默认值表达式

```javascript
// 默认值可以是表达式
function add(x, y = x * 2) {
    return x + y;
}

console.log(add(5));    // 15 (5 + 10)
console.log(add(5, 3)); // 8

// 默认值可以是函数调用
function getDefaultConfig() {
    return {
        timeout: 5000,
        retries: 3
    };
}

function fetchData(options = getDefaultConfig()) {
    console.log(options);
}

fetchData();  // { timeout: 5000, retries: 3 }
fetchData({ timeout: 10000 });  // { timeout: 10000, retries: 3 }
```

### 7.1.4 解构参数默认值

```javascript
// 对象解构参数 + 默认值
function createUser({ name = '匿名', age = 0, city = '未知' } = {}) {
    return { name, age, city };
}

createUser();  // { name: '匿名', age: 0, city: '未知' }
createUser({ name: '张三' });  // { name: '张三', age: 0, city: '未知' }
createUser({ name: '李四', age: 25 });  // { name: '李四', age: 25, city: '未知' }

// 数组解构参数 + 默认值
function getPosition([x = 0, y = 0] = []) {
    return { x, y };
}

getPosition();        // { x: 0, y: 0 }
getPosition([10]);    // { x: 10, y: 0 }
getPosition([10, 20]); // { x: 10, y: 20 }
```

## 7.2 rest 参数详解

### 7.2.1 什么是 rest 参数？

```
┌─────────────────────────────────────────────────────────────┐
│  rest 参数 = 剩余参数                                        │
│                                                             │
│  用 ... 参数名 的形式，将多余的参数收集到一个数组中          │
│                                                             │
│  就像：                                                     │
│  function fn(a, b, ...rest) {}                             │
│  fn(1, 2, 3, 4, 5)                                         │
│  → a = 1, b = 2, rest = [3, 4, 5]                          │
└─────────────────────────────────────────────────────────────┘
```

### 7.2.2 基本语法

```javascript
// 收集剩余参数
function sum(first, ...rest) {
    console.log('第一个:', first);  // 1
    console.log('其余:', rest);    // [2, 3, 4, 5]
    return rest.reduce((a, b) => a + b, first);
}

console.log(sum(1, 2, 3, 4, 5));  // 15
```

### 7.2.3 rest 参数 vs arguments

```javascript
// arguments（ES5）
function oldStyle() {
    console.log(arguments);  // 类数组对象
    console.log(Array.isArray(arguments));  // false
    
    // 需要转换
    const args = Array.from(arguments);
    console.log(args.map(x => x * 2));  // [2, 4, 6]
}

// rest 参数（ES6）
function newStyle(...args) {
    console.log(args);  // 真正的数组
    console.log(Array.isArray(args));  // true
    
    // 直接使用数组方法
    console.log(args.map(x => x * 2));  // [2, 4, 6]
}

oldStyle(1, 2, 3);
newStyle(1, 2, 3);
```

### 7.2.4 箭头函数和 rest 参数

```javascript
// 箭头函数没有 arguments
const fn1 = () => {
    // console.log(arguments);  // ReferenceError
};

// rest 参数可以用于箭头函数
const fn2 = (...args) => {
    console.log(args);  // OK
};

fn2(1, 2, 3);  // [1, 2, 3]
```

## 7.3 展开运算符在函数中的应用

### 7.3.1 函数调用展开

```javascript
// 将数组展开为参数
const numbers = [1, 2, 3];

// ES5 方式
console.log(Math.max.apply(null, numbers));  // 3

// ES6 展开
console.log(Math.max(...numbers));  // 3
console.log(Math.min(...numbers));  // 1
```

### 7.3.2 合并多个数组

```javascript
const arr1 = [1, 2];
const arr2 = [3, 4];
const arr3 = [5, 6];

// 合并并调用
console.log(...arr1, ...arr2, ...arr3);  // 1 2 3 4 5 6
```

### 7.3.3 rest 和展开配合使用

```javascript
// 收集然后展开
function applyAll(fn, ...args) {
    return fn(...args);  // 展开 args 作为参数
}

const add = (a, b, c) => a + b + c;
console.log(applyAll(add, 1, 2, 3));  // 6
```

## 7.4 函数属性详解

### 7.4.1 name 属性

```javascript
// 普通函数
function foo() {}
console.log(foo.name);  // 'foo'

// 匿名函数
const bar = function() {};
console.log(bar.name);  // 'bar'

// 箭头函数
const baz = () => {};
console.log(baz.name);  // 'baz'

// 有名字的函数表达式
const qux = function innerName() {};
console.log(qux.name);  // 'innerName'

// 类
class MyClass {}
console.log(MyClass.name);  // 'MyClass'

// bind
const bound = foo.bind({});
console.log(bound.name);  // 'bound foo'
```

### 7.4.2 length 属性

```javascript
// length 返回"必需参数"的个数
// 不包括 rest 参数和带默认值的参数

function f1(a, b, c) {}           // 3
function f2(a, b = 1, c) {}        // 1（a，b 有默认值不算）
function f3(a, ...rest) {}         // 1（rest 不算）
function f4(a = 1, b = 2, c = 3) {} // 0（所有都有默认值）

console.log(f1.length);  // 3
console.log(f2.length);  // 1
console.log(f3.length);  // 1
console.log(f4.length);  // 0
```

## 7.5 函数柯里化

### 7.5.1 什么是柯里化？

```
┌─────────────────────────────────────────────────────────────┐
│  柯里化（Currying）                                         │
│                                                             │
│  将接受多个参数的函数转换为接受单一参数的函数序列            │
│                                                             │
│  例子：                                                     │
│  add(a, b, c) → add(a)(b)(c)                               │
│                                                             │
│  就像：                                                     │
│  - 完整函数：一口吃掉一个汉堡                               │
│  - 柯里化：一小口一小口吃                                   │
└─────────────────────────────────────────────────────────────┘
```

### 7.5.2 基础示例

```javascript
// 普通函数
function add(a, b, c) {
    return a + b + c;
}
console.log(add(1, 2, 3));  // 6

// 柯里化版本
function curriedAdd(a) {
    return function(b) {
        return function(c) {
            return a + b + c;
        };
    };
}

console.log(curriedAdd(1)(2)(3));  // 6

// 使用箭头函数简化
const arrowCurriedAdd = a => b => c => a + b + c;
console.log(arrowCurriedAdd(1)(2)(3));  // 6
```

### 7.5.3 柯里化工具函数

```javascript
// 通用柯里化函数
function curry(fn) {
    return function curried(...args) {
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        }
        return function(...args2) {
            return curried.apply(this, args.concat(args2));
        };
    };
}

// 使用
function multiply(a, b, c) {
    return a * b * c;
}

const curriedMultiply = curry(multiply);
console.log(curriedMultiply(2)(3)(4));  // 24
console.log(curriedMultiply(2, 3)(4)); // 24
console.log(curriedMultiply(2)(3, 4)); // 24
```

### 7.5.4 实际应用

```javascript
// 柯里化：创建特定功能的函数
const multiply = x => y => z => x * y * z;

const multiplyBy2 = multiply(2);
const multiplyBy2And3 = multiply(2)(3);

console.log(multiplyBy2(3)(4));  // 24
console.log(multiplyBy2And3(4)); // 24

// 实际场景：API 请求
const createRequest = method => url => data => {
    console.log(`${method} ${url}`, data);
};

const get = createRequest('GET');
const post = createRequest('POST');

get('/api/users');
post('/api/users')({ name: '张三' });
```

## 7.6 函数组合

### 7.6.1 什么是函数组合？

```
┌─────────────────────────────────────────────────────────────┐
│  函数组合（Function Composition）                            │
│                                                             │
│  将多个函数串联起来，一个函数的输出作为下一个函数的输入       │
│                                                             │
│  例子：                                                     │
│  compose(f, g, h)(x) = h(g(f(x)))                          │
│                                                             │
│  就像：                                                     │
│  原材料 → 加工A → 加工B → 加工C → 产品                      │
└─────────────────────────────────────────────────────────────┘
```

### 7.6.2 compose 函数

```javascript
// 从右到左执行
const compose = (...fns) => {
    return (initialValue) => {
        return fns.reduceRight((acc, fn) => {
            return fn(acc);
        }, initialValue);
    };
};

// 使用
const add1 = x => x + 1;
const double = x => x * 2;
const square = x => x * x;

const composed = compose(square, double, add1);
console.log(composed(3));  // square(double(add1(3)))
                          // = square(double(4))
                          // = square(8)
                          // = 64
```

### 7.6.3 pipe 函数

```javascript
// pipe 和 compose 相反，从左到右执行
const pipe = (...fns) => {
    return (initialValue) => {
        return fns.reduce((acc, fn) => {
            return fn(acc);
        }, initialValue);
    };
};

// 使用
const piped = pipe(add1, double, square);
console.log(piped(3));  // add1(double(square(3)))
                       // = add1(double(9))
                       // = add1(18)
                       // = 19
```

## 7.7 实践练习

### 练习 1：参数默认值

```javascript
// 1.1 默认参数
function greet(name = '访客', greeting = '你好') {
    return `${greeting}，${name}！`;
}
console.log(greet());  // 你好，访客！
console.log(greet('张三', '欢迎'));  // 欢迎，张三！

// 1.2 解构参数默认值
function createUser({ name = '匿名', age = 0 } = {}) {
    return { name, age };
}
console.log(createUser({ name: '张三' }));  // { name: '张三', age: 0 }
```

### 练习 2：rest 参数

```javascript
// 2.1 求和函数
function sum(...numbers) {
    return numbers.reduce((acc, n) => acc + n, 0);
}
console.log(sum(1, 2, 3, 4, 5));  // 15

// 2.2 第一个参数 + 其余参数
function firstPlusRest(first, ...rest) {
    return first + rest.reduce((a, b) => a + b, 0);
}
console.log(firstPlusRest(1, 2, 3, 4));  // 10
```

### 练习 3：柯里化

```javascript
// 3.1 柯里化加法
const curriedAdd = a => b => c => a + b + c;
console.log(curriedAdd(1)(2)(3));  // 6

// 3.2 创建特定倍数的函数
const multiply = x => y => x * y;
const double = multiply(2);
const triple = multiply(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15
```

## 7.8 常见问答

### Q1: 参数默认值和 rest 参数可以同时使用吗？

```javascript
// 可以，但 rest 参数必须在最后
function fn(a, b = 1, ...rest) {
    console.log(a, b, rest);
}

fn(1);        // 1, 1, []
fn(1, 2);     // 1, 2, []
fn(1, 2, 3);  // 1, 2, [3]
fn(1, 2, 3, 4);  // 1, 2, [3, 4]
```

### Q2: 箭头函数可以设置默认参数吗？

```javascript
// 可以
const fn = (a = 1, b = 2) => a + b;
console.log(fn());      // 3
console.log(fn(5));    // 7
console.log(fn(5, 10)); // 15
```

### Q3: 柯里化和函数组合在实际开发中有什么用？

```javascript
// 柯里化：创建可复用的函数
const fetch = method => url => data => { /* ... */ };
const get = fetch('GET');
const post = fetch('POST');

// 函数组合：数据处理管道
const processData = pipe(
    validate,
    transform,
    save
);
```
