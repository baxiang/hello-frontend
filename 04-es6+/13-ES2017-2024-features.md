## 13.1 ES2017 新特性

### 13.1.1 async/await

```javascript
// Promise 方式
fetch('/api/data')
    .then(res => res.json())
    .then(data => console.log(data))
    .catch(err => console.error(err));

// async/await 方式（更简洁）
async function fetchData() {
    try {
        const res = await fetch('/api/data');
        const data = await res.json();
        console.log(data);
    } catch (err) {
        console.error(err);
    }
}
```

### 13.1.2 Object.values/entries

```javascript
const obj = { a: 1, b: 2, c: 3 };

// 获取所有值
console.log(Object.values(obj));  // [1, 2, 3]

// 获取所有键值对
console.log(Object.entries(obj));  // [['a', 1], ['b', 2], ['c', 3]]

// 遍历对象
for (const [key, value] of Object.entries(obj)) {
    console.log(`${key}: ${value}`);
}
```

## 13.2 ES2020 新特性

### 13.2.1 可选链（?.）

```javascript
const user = {
    profile: {
        name: '张三'
    }
};

// 传统方式
const city1 = user && user.profile && user.profile.address && user.profile.address.city;

// 可选链方式
const city2 = user?.profile?.address?.city;

// 数组可选链
const arr = [1, 2, 3];
console.log(arr?.[10]);  // undefined

// 方法可选链
const obj = {
    fn: () => 'hello'
};
console.log(obj.fn?.());  // 'hello'
console.log(obj.notExist?.());  // undefined
```

### 13.2.2 空值合并（??）

```javascript
// || 的问题
const value1 = 0 || 'default';  // 'default'（0 被当作假值）

// ?? 的解决方案
const value2 = 0 ?? 'default';   // 0（只有 null/undefined 用默认值）

// 实际应用
const config = {
    timeout: 0
};
const timeout = config.timeout ?? 5000;  // 0（保留）
```

### 13.2.3 BigInt

```javascript
// 大整数
const bigInt = 9007199254740991n;

// 运算
console.log(100n + 200n);  // 300n
console.log(5n * 3n);     // 15n

// 不能与普通数字混用
// console.log(5n + 5);  // 报错
console.log(5n + BigInt(5));  // 10n
```

## 13.3 ES2021 新特性

### 13.3.1 replaceAll

```javascript
// 替换所有
const str = 'hello world';
console.log(str.replaceAll('l', 'L'));  // 'heLLo worLd'

// 之前需要用正则
console.log(str.replace(/l/g, 'L'));  // 'heLLo worLd'
```

### 13.3.2 逻辑赋值

```javascript
// ||=
let a = 0;
a ||= 5;  // a = a || 5
console.log(a);  // 5

// &&=
let b = 1;
b &&= 5;  // b = b && 5
console.log(b);  // 5

// ??=
let c = null;
c ??= 5;  // c = c ?? 5
console.log(c);  // 5
```

## 13.4 ES2022 新特性

### 13.4.1 顶层 await

```javascript
// 之前：只能在 async 函数内使用 await
// 现在：模块顶层也可以使用

// data.js
export const data = await fetch('/api/data').then(r => r.json());

// main.js
import { data } from './data.js';
console.log(data);
```

### 13.4.2 类字段

```javascript
class Person {
    // 公共字段
    name = '张三';
    
    // 私有字段（ES2022）
    #age = 25;
    
    // 静态字段
    static species = 'human';
    
    getAge() {
        return this.#age;
    }
}
```

## 13.5 ES2023 新特性

### 13.5.1 数组新方法

```javascript
const arr = [1, 2, 3, 4, 5];

// findLast - 从后往前找
console.log(arr.findLast(n => n > 3));  // 5

// findLastIndex - 从后往前找索引
console.log(arr.findLastIndex(n => n > 3));  // 4

// toReversed - 返回反转后的新数组
console.log(arr.toReversed());  // [5, 4, 3, 2, 1]

// toSorted - 返回排序后的新数组
console.log(arr.toSorted((a, b) => b - a));  // [5, 4, 3, 2, 1]

// toSpliced - 返回修改后的新数组
console.log(arr.toSpliced(1, 2, 6, 7));  // [1, 6, 7, 4, 5]

// with - 返回修改指定位置后的新数组
console.log(arr.with(0, 100));  // [100, 2, 3, 4, 5]
```

## 13.6 常见问答

### Q1: 什么时候用 ?? 和 ||？

```javascript
// 用 ??：当 0 或空字符串是有效值时
const value = input ?? 'default';

// 用 ||：当 0 或空字符串应该被视为假值时
const value = input || 'default';
```

### Q2: 可选链可以替代所有判断吗？

```javascript
// 可选链用于读取
const value = obj?.prop?.nested;

// 不能用于赋值
// obj?.prop = 'value';  // 报错
```
