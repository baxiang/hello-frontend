## 5.1 let 和 const

### 基本用法

```javascript
// var 的问题
for (var i = 0; i < 5; i++) {
    setTimeout(() => console.log(i), 100);  // 5,5,5,5,5
}

// let 解决闭包问题
for (let i = 0; i < 5; i++) {
    setTimeout(() => console.log(i), 100);  // 0,1,2,3,4
}

// const 声明常量
const PI = 3.14159;
// PI = 3;  // 报错

// const 声明对象
const obj = { name: '张三' };
obj.name = '李四';  // 可以修改属性
// obj = {};  // 不能重新赋值

// 暂时性死区
if (true) {
    console.log(a);  // ReferenceError
    let a = 1;
}
```

## 5.2 箭头函数

### 语法

```javascript
// 基础语法
const add = (a, b) => a + b;

// 单个参数可省略括号
const double = x => x * 2;

// 无参数
const sayHi = () => console.log('Hi');

// 返回对象需加括号
const getUser = () => ({ name: '张三', age: 25 });

// 多语句
const process = (data) => {
    const result = data.map(item => item * 2);
    return result;
};
```

### this 绑定

```javascript
// 普通函数 this 指向调用者
// 箭头函数 this 指向定义时的上下文

function Timer() {
    this.seconds = 0;

    // 箭头函数 this 指向 Timer
    setInterval(() => {
        this.seconds++;
    }, 1000);
}

// 不适合用作构造函数
// const Person = (name) => { this.name = name; };  // 报错
// new Person('张三');

// 不适合需要动态 this 的场景
const obj = {
    name: '张三',
    // 错误用法
    // getName: () => this.name,  // undefined

    // 正确用法
    getName() {
        return this.name;
    }
};
```

## 5.3 解构赋值

```javascript
// 数组解构
const [a, b, ...rest] = [1, 2, 3, 4];
console.log(a, b, rest);  // 1, 2, [3, 4]

// 嵌套解构
const [x, [y]] = [1, [2]];

// 默认值
const [p = 1, q = 2] = [3];  // p=3, q=2

// 对象解构
const { name, age } = { name: '张三', age: 25 };

// 重命名
const { name: userName } = { name: '李四' };

// 默认值
const { gender = 'unknown' } = { name: '王五' };

// 函数参数解构
function printUser({ name, age, email = 'no-email' }) {
    console.log(name, age, email);
}

// 字符串解构
const [c1, c2] = 'hello';  // c1='h', c2='e'
```

## 5.5 模板字符串

```javascript
const name = '张三';
const age = 25;

// 插值
const str = `我的名字是${name}，今年${age}岁`;

// 多行
const html = `
    <div>
        <h3>${name}</h3>
        <p>${age}岁</p>
    </div>
`;

// 标签模板
function upper(strings, ...values) {
    return strings.reduce((result, str, i) => {
        return result + str + (values[i] || '').toUpperCase();
    }, '');
}

const result = upper`Hello ${name}`;  // "Hello 张三"

// 实用模板
const template = (strings, ...values) => {
    return strings.reduce((acc, str, i) => {
        return acc + str + (values[i] !== undefined ?
            `<span>${values[i]}</span>` : '');
    }, '');
};
```

## 5.6 类和继承

### class 基础

```javascript
class Person {
    // 构造函数
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    // 实例方法
    sayHi() {
        console.log(`Hi, I'm ${this.name}`);
    }

    // getter
    get info() {
        return `${this.name} (${this.age}岁)`;
    }

    // setter
    set age(value) {
        if (value < 0) throw new Error('年龄不能为负');
        this._age = value;
    }

    get age() {
        return this._age;
    }

    // 静态方法
    static create(name, age) {
        return new Person(name, age);
    }
}

const p = Person.create('张三', 25);
p.sayHi();
console.log(p.info);
```

### 继承

```javascript
class Student extends Person {
    constructor(name, age, grade) {
        super(name, age);  // 调用父类构造函数
        this.grade = grade;
    }

    // 重写父类方法
    sayHi() {
        super.sayHi();
        console.log(`我是${this.grade}年级的学生`);
    }

    // 子类方法
    study() {
        console.log(`${this.name}在学习`);
    }
}

const s = new Student('小明', 15, 9);
s.sayHi();
s.study();
```

## 5.7 模块化

### 导出

```javascript
// math.js

// 具名导出
export const PI = 3.14159;
export function add(a, b) { return a + b; }

// 默认导出
export default class Calculator {
    add(a, b) { return a + b; }
}

// 或者
const utils = { add, subtract };
export default utils;
```

### 导入

```javascript
// main.js

// 导入默认和具名
import Calculator, { PI, add } from './math.js';

// 重命名
import { add as plus } from './math.js';

// 全部导入
import * as Math from './math.js';
Math.add(1, 2);

// 只导入类型
import './styles.css';  // 只执行副作用

// 动态导入
async function loadModule() {
    const math = await import('./math.js');
    console.log(math.add(1, 2));
}
```

## 5.8 Promise 和 async/await

```javascript
// Promise 基础
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('成功');
    }, 1000);
});

promise.then(result => console.log(result));

// Promise 组合
Promise.all([p1, p2, p3]);  // 全部成功
Promise.race([p1, p2]);     // 第一个完成
Promise.allSettled([p1, p2]); // 等待全部完成
Promise.any([p1, p2]);      // 第一个成功

// async/await
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('请求失败:', error);
        throw error;
    }
}

// 并行执行
async function getAllData() {
    const [user, posts] = await Promise.all([
        fetch('/api/user').then(r => r.json()),
        fetch('/api/posts').then(r => r.json())
    ]);
    return { user, posts };
}
```

## 5.9 Map、Set 和 Symbol

### Map

```javascript
const map = new Map();
map.set('name', '张三');
map.set('age', 25);
map.set({ key: 'obj' }, '对象键');

console.log(map.get('name'));  // '张三'
console.log(map.size);         // 3

// 遍历
for (const [key, value] of map) {
    console.log(key, value);
}

map.forEach((value, key) => {
    console.log(key, value);
});
```

### Set

```javascript
const set = new Set([1, 2, 2, 3, 3, 3]);
console.log(set.size);  // 3

set.add(4);
set.delete(1);
console.log(set.has(2));  // true

// Set 运算
const a = new Set([1, 2, 3]);
const b = new Set([3, 4, 5]);

// 并集
const union = new Set([...a, ...b]);

// 交集
const intersection = new Set([...a].filter(x => b.has(x)));

// 差集
const difference = new Set([...a].filter(x => !b.has(x)));
```

### Symbol

```javascript
// 唯一值
const sym1 = Symbol('id');
const sym2 = Symbol('id');
console.log(sym1 === sym2);  // false

// 对象属性
const obj = {
    [sym1]: 'value',
    normal: 'normal'
};

// 全局 Symbol
const globalSym = Symbol.for('global');

// 内置 Symbol
const iterable = {
    [Symbol.iterator]() {
        return {
            next() {
                return { value: 'item', done: false };
            }
        };
    }
};
```

## 5.10 新特性概览

### ES2017 (ES8)

```javascript
// Object.values / Object.entries
const obj = { a: 1, b: 2, c: 3 };
Object.values(obj);   // [1, 2, 3]
Object.entries(obj);  // [['a',1],['b',2],['c',3]]

// String padding
'hello'.padStart(10, '*');   // '*****hello'
'hello'.padEnd(10, '*');     // 'hello*****'

// trailing commas
const arr = [1, 2, 3,];
const obj = { a: 1, b: 2, };
```

### ES2018 (ES9)

```javascript
// 对象展开
const obj1 = { a: 1, b: 2 };
const obj2 = { ...obj1, c: 3 };

// async 迭代
async function* asyncGenerator() {
    yield await Promise.resolve(1);
    yield await Promise.resolve(2);
}

// Promise.finally
fetch('/api/data')
    .then(res => res.json())
    .catch(err => console.error(err))
    .finally(() => console.log('请求完成'));
```

### ES2019 (ES10)

```javascript
// Object.fromEntries
const entries = [['a', 1], ['b', 2]];
const obj = Object.fromEntries(entries);  // { a: 1, b: 2 }

// Array flat/flatMap
[1, [2, [3, [4]]]].flat();      // [1, 2, [3, [4]]]
[1, [2, [3, [4]]]].flat(2);     // [1, 2, 3, [4]]
[1, [2, [3]]].flatMap(x => [x, x * 2]);

// String trimStart/trimEnd
'  hello  '.trimStart();  // 'hello  '
'  hello  '.trimEnd();    // '  hello'

// Optional catch
try {
    riskyOperation();
} catch {
    // 不使用 error
}
```

### ES2020 (ES11)

```javascript

const bigInt = 9007199254740991n;
const bigInt2 = BigInt(9007199254740991);

// Optional chaining
const city = user?.profile?.address?.city;
const first = items?.[0];
```

### ES2021 (ES12)


## 5.11 实践练习

### 练习 1：工具函数库

```javascript
// utils.js

// 防抖
export const debounce = (fn, delay = 300) => {
    let timer = null;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => fn.apply(this, args), delay);
    };
};

// 节流
export const throttle = (fn, limit = 300) => {
    let inThrottle = false;
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
};

// 深拷贝
export const deepClone = (obj, hash = new WeakMap()) => {
    if (obj === null || typeof obj !== 'object') return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof RegExp) return new RegExp(obj);
    if (hash.has(obj)) return hash.get(obj);

    const clone = Array.isArray(obj) ? [] : {};
    hash.set(obj, clone);
    return Object.assign(clone,
        Object.keys(obj).reduce((acc, key) => {
            acc[key] = deepClone(obj[key], hash);
            return acc;
        }, {})
    );
};

// 防抖搜索示例
export const createSearchHandler = (onSearch) => {
    return debounce(async (query) => {
        if (!query.trim()) return;
        const results = await onSearch(query);
        console.log('搜索结果:', results);
    }, 300);
};
```

### 练习 2：链式调用库

```javascript
class Chain {
    constructor(value) {
        this._value = value;
    }

    // 数组方法
    map(fn) {
        this._value = this._value.map(fn);
        return this;
    }

    filter(fn) {
        this._value = this._value.filter(fn);
        return this;
    }

    reduce(fn, initial) {
        this._value = this._value.reduce(fn, initial);
        return this;
    }

    // 字符串方法
    upper() {
        this._value = this._value.toUpperCase();
        return this;
    }

    lower() {
        this._value = this._value.toLowerCase();
        return this;
    }

    // 值操作
    valueOf() {
        return this._value;
    }

    tap(fn) {
        fn(this._value);
        return this;
    }
}

// 使用示例
const result = new Chain([1, 2, 3, 4, 5])
    .filter(x => x % 2 === 0)
    .map(x => x * 10)
    .reduce((sum, x) => sum + x, 0)
    .valueOf();

console.log(result);  // 60
```
