## 9.1 ES2017 (ES8)

### Object.values / Object.entries

```javascript
const obj = { a: 1, b: 2, c: 3 };

// Object.values
console.log(Object.values(obj));  // [1, 2, 3]

// Object.entries
console.log(Object.entries(obj));
// [['a', 1], ['b', 2], ['c', 3]]

// 遍历对象
for (const [key, value] of Object.entries(obj)) {
    console.log(`${key}: ${value}`);
}

// 与其他操作组合
const sum = Object.values(obj).reduce((a, b) => a + b, 0);
console.log(sum);  // 6
```

### String padding

```javascript
// padStart - 左侧填充
'5'.padStart(3, '0');      // '005'
'123'.padStart(5, '0');    // '00123'
'hello'.padStart(10);      // '     hello'

// padEnd - 右侧填充
'5'.padEnd(3, '0');        // '500'
'hello'.padEnd(10, '-');   // 'hello-----'

// 实用示例
// 格式化时间
function formatTime(date) {
    const h = String(date.getHours()).padStart(2, '0');
    const m = String(date.getMinutes()).padStart(2, '0');
    const s = String(date.getSeconds()).padStart(2, '0');
    return `${h}:${m}:${s}`;
}

// 格式化数字
function formatNumber(num, length = 3) {
    return String(num).padStart(length, '0');
}
```

### Object.getOwnPropertyDescriptors

```javascript
// 获取对象的完整描述
const obj = { name: '张三' };
Object.defineProperty(obj, 'age', {
    value: 25,
    writable: false,
    enumerable: false,
    configurable: false
});

const descriptors = Object.getOwnPropertyDescriptors(obj);
console.log(descriptors);
/*
{
    name: {
        value: '张三',
        writable: true,
        enumerable: true,
        configurable: true
    },
    age: {
        value: 25,
        writable: false,
        enumerable: false,
        configurable: false
    }
}
*/
```

## 9.2 ES2018 (ES9)

### 对象展开运算符

```javascript
// 展开对象
const defaults = { theme: 'light', lang: 'zh' };
const userPrefs = { theme: 'dark' };
const config = { ...defaults, ...userPrefs };
console.log(config);  // { theme: 'dark', lang: 'zh' }

// 复制对象
const original = { a: 1, b: 2 };
const copy = { ...original };

// 注意：后面的覆盖前面的
const obj1 = { a: 1, b: 1 };
const obj2 = { b: 2, c: 2 };
const merged = { ...obj1, ...obj2 };
console.log(merged);  // { a: 1, b: 2, c: 2 }
```

### Promise.finally

```javascript
fetchData()
    .then(result => console.log('成功:', result))
    .catch(error => console.error('失败:', error))
    .finally(() => {
        console.log('无论成功失败都会执行');
        // 常用于清理工作
        hideLoading();
    });

// 等价于
fetchData()
    .then(
        result => {
            hideLoading();
            return result;
        },
        error => {
            hideLoading();
            throw error;
        }
    );
```

### 异步迭代

```javascript
// 异步生成器
async function* asyncGenerator() {
    let i = 0;
    while (true) {
        yield await Promise.resolve(i++);
    }
}

// 使用 for await...of
async function consume() {
    const gen = asyncGenerator();
    for await (const value of gen) {
        console.log(value);
        if (value >= 3) break;
    }
}
```

## 9.3 ES2019 (ES10)

### Array.flat / flatMap

```javascript
// flat - 数组扁平化
const arr = [1, [2, [3, [4]]]];
console.log(arr.flat());      // [1, 2, [3, [4]]]
console.log(arr.flat(2));     // [1, 2, 3, [4]]
console.log(arr.flat(Infinity)); // [1, 2, 3, 4]

// flatMap - 先 map 再 flat
const sentences = ['Hello World', 'How are you'];
const words = sentences.flatMap(s => s.split(' '));
console.log(words);  // ['Hello', 'World', 'How', 'are', 'you']

// 过滤空值
const arr = ['a', '', 'b', '', 'c'];
const result = arr.flatMap(x => x ? [x] : []);
console.log(result);  // ['a', 'b', 'c']
```

### Object.fromEntries

```javascript
// 将键值对转换为对象
const entries = [['a', 1], ['b', 2], ['c', 3]];
const obj = Object.fromEntries(entries);
console.log(obj);  // { a: 1, b: 2, c: 3 }

// 与 Object.entries 配合使用
const obj = { a: 1, b: 2, c: 3 };
const doubled = Object.fromEntries(
    Object.entries(obj).map(([k, v]) => [k, v * 2])
);
console.log(doubled);  // { a: 2, b: 4, c: 6 }

// Map 转对象
const map = new Map([['a', 1], ['b', 2]]);
const obj = Object.fromEntries(map);
console.log(obj);  // { a: 1, b: 2 }
```

### String trimStart / trimEnd

```javascript
const str = '  hello  ';

// trimStart (别名 trimLeft)
console.log(str.trimStart());  // 'hello  '

// trimEnd (别名 trimRight)
console.log(str.trimEnd());    // '  hello'

// 实用示例
// 处理表单输入
const input = '  user@example.com  ';
const email = input.trim();

// 处理 URL
const url = '  https://example.com  ';
const cleanUrl = url.trim();
```

### Optional catch

```javascript
// 不需要使用 error 变量
try {
    riskyOperation();
} catch {
    // 静默处理
    console.log('出错了，但不需要 error 对象');
}

// 对比
try {
    riskyOperation();
} catch (error) {
    // error 未使用
    console.log('出错了');
}
```

## 9.4 ES2020 (ES11)

### Optional chaining

```javascript
// 安全访问嵌套属性
const user = { profile: { name: '张三' } };

// 传统方式
const name = user && user.profile && user.profile.name;

// Optional chaining
const name = user?.profile?.name;

// 调用函数
obj?.method?.();

// 访问数组
const first = arr?.[0];

// 示例
function getUserCity(user) {
    return user?.profile?.address?.city ?? '未知';
}
```

### Nullish coalescing

```javascript
// 只在 null 或 undefined 时使用默认值
const input = null;
const value = input ?? 'default';
console.log(value);  // 'default'

// 对比 ||
const count = 0;
console.log(count || 10);   // 10（0 被视为 falsy）
console.log(count ?? 10);   // 0（0 不是 null/undefined）

// 实用示例
function getConfig(config) {
    return {
        timeout: config.timeout ?? 5000,
        retries: config.retries ?? 3,
        debug: config.debug ?? false
    };
}
```

### BigInt

```javascript
// 创建 BigInt
const bigInt1 = 9007199254740991n;
const bigInt2 = BigInt(9007199254740991);

// 运算
console.log(bigInt1 + 1n);  // 9007199254740992n
console.log(bigInt1 * 2n);  // 18014398509481982n

// 注意：不能与 Number 混用
// const result = bigInt1 + 1;  // TypeError
const result = bigInt1 + BigInt(1);  // OK

// 转换
const num = Number(bigInt1);
const big = BigInt(num);
```

### Promise.allSettled

```javascript
// 等待所有 Promise 完成
const results = await Promise.allSettled([
    Promise.resolve('成功 1'),
    Promise.reject('失败 1'),
    Promise.resolve('成功 2')
]);

console.log(results);
/*
[
    { status: 'fulfilled', value: '成功 1' },
    { status: 'rejected', reason: '失败 1' },
    { status: 'fulfilled', value: '成功 2' }
]
*/

// 实用示例
const responses = await Promise.allSettled([
    fetch('/api/users'),
    fetch('/api/posts'),
    fetch('/api/comments')
]);

responses.forEach((result, i) => {
    if (result.status === 'fulfilled') {
        console.log(`请求 ${i} 成功`);
    } else {
        console.error(`请求 ${i} 失败:`, result.reason);
    }
});
```

### dynamic import

```javascript
// 动态导入
const module = await import('./module.js');
module.default();
module.namedFunction();

// 条件加载
async function loadModule(condition) {
    if (condition) {
        return import('./module-a.js');
    } else {
        return import('./module-b.js');
    }
}
```

### String.matchAll

```javascript
const str = 'Hello1 Hello2 Hello3';
const regex = /Hello\d/g;

// 传统方式
let match;
while ((match = regex.exec(str)) !== null) {
    console.log(match);
}

// matchAll
for (const match of str.matchAll(regex)) {
    console.log(match);
}

// 展开
const matches = [...str.matchAll(regex)];
console.log(matches);
```

## 9.5 ES2021 (ES12)

### String.replaceAll

```javascript
const str = 'hello world, hello javascript';

// 传统方式
const replaced = str.replace(/hello/gi, 'hi');

// replaceAll
const replaced = str.replaceAll('hello', 'hi');
console.log(replaced);  // 'hi world, hi javascript'
```

### Logical assignment

```javascript
// ||= 逻辑或赋值
a ||= b;  // 等价于 a || (a = b)

// &&= 逻辑与赋值
a &&= b;  // 等价于 a && (a = b)

// ??= 空值赋值
a ??= b;  // 等价于 a ?? (a = b)

// 实用示例
// 配置默认值
config.timeout ??= 5000;
config.retries ??= 3;

// 条件更新
user.isActive &&= newValue;
```

### Numeric separators

```javascript
// 提高可读性
const billion = 1_000_000_000;
const bytes = 0xFF_FF_FF_FF;
const binary = 0b1010_0001_1000_0101;

// 电话号
const phone = '138_0013_8000';
```

## 9.6 ES2022 (ES13)

### 顶层 await

```javascript
// 在 ES Module 顶层直接使用 await
const data = await fetch('/api/data').then(r => r.json());
export default data;

// 在模块中导入时使用
import { config } from await import('./config.js');
```

### 类字段

```javascript
class Person {
    // 实例字段
    name = '匿名';
    age = 0;

    // 静态字段
    static count = 0;
    static species = 'Homo sapiens';

    constructor(name, age) {
        this.name = name;
        this.age = age;
        Person.count++;
    }
}

// 私有字段
class Counter {
    #count = 0;

    increment() {
        this.#count++;
    }

    getCount() {
        return this.#count;
    }
}
```

### Array.at()

```javascript
const arr = [1, 2, 3, 4, 5];

// 传统方式获取最后一个元素
const last = arr[arr.length - 1];

// 使用 at()
const last = arr.at(-1);

// 支持负索引
console.log(arr.at(0));   // 1
console.log(arr.at(-1));  // 5
console.log(arr.at(-2));  // 4
```

### Object.hasOwn

```javascript
const obj = { name: '张三' };

// 传统方式
console.log(obj.hasOwnProperty('name'));  // true

// 新方式（推荐）
console.log(Object.hasOwn(obj, 'name'));  // true

// hasOwnProperty 的问题
const obj2 = { hasOwnProperty: () => false };
console.log(obj2.hasOwnProperty('name'));  // false（被覆盖）
console.log(Object.hasOwn(obj2, 'name'));  // true（正确）
```

### RegExp match indices

```javascript
const text = 'Contact: foo@bar.com';
const regex = /(\w+)@(\w+\.\w+)/d;  // 'd' flag for indices

const match = text.match(regex);
console.log(match.indices);
// [[10, 20], [10, 13], [14, 20]]
// 分别是整体、第一组、第二组的索引位置
```

## 9.7 实践练习

### 练习 1：配置合并工具

```javascript
function mergeConfig(...configs) {
    return configs.reduce((acc, config) => {
        if (!config) return acc;

        Object.entries(config).forEach(([key, value]) => {
            if (typeof value === 'object' && value !== null) {
                acc[key] = mergeConfig(acc[key] || {}, value);
            } else {
                acc[key] ??= value;  // 只填充未定义的值
            }
        });

        return acc;
    }, {});
}

// 使用
const defaultConfig = {
    api: { timeout: 5000, retries: 3 },
    features: { darkMode: false }
};

const userConfig = {
    api: { timeout: 10000 },
    features: { darkMode: true }
};

const config = mergeConfig(defaultConfig, userConfig);
```

### 练习 2：安全访问工具

```javascript
// 实现一个安全的嵌套访问函数
function safeGet(obj, path, defaultValue) {
    return path.split('.')
        .reduce((acc, key) => acc?.[key], obj) ?? defaultValue;
}

// 使用
const user = { profile: { address: { city: '北京' } } };
console.log(safeGet(user, 'profile.address.city'));  // '北京'
console.log(safeGet(user, 'profile.zipcode', '000000'));  // '000000'
```

### 练习 3：扁平化工具

```javascript
// 扁平化嵌套对象
function flatten(obj, prefix = '') {
    return Object.fromEntries(
        Object.entries(obj).flatMap(([key, value]) => {
            const newKey = prefix ? `${prefix}.${key}` : key;
            if (typeof value === 'object' && value !== null && !Array.isArray(value)) {
                return Object.entries(flatten(value, newKey));
            }
            return [[newKey, value]];
        })
    );
}

// 使用
const nested = {
    user: {
        name: '张三',
        address: {
            city: '北京',
            district: '朝阳'
        }
    }
};

console.log(flatten(nested));
/*
{
    'user.name': '张三',
    'user.address.city': '北京',
    'user.address.district': '朝阳'
}
*/
```
