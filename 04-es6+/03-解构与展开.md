# 解构赋值与展开运算符

## 学习目标
- 掌握数组和对象的解构赋值
- 理解嵌套解构和默认值
- 掌握展开运算符的使用
- 能够使用参数解构

---

## 3.1 数组解构

### 基础解构

```javascript
// 传统方式
const arr = [1, 2, 3];
const a = arr[0];
const b = arr[1];
const c = arr[2];

// 解构赋值
const [a, b, c] = [1, 2, 3];
console.log(a, b, c);  // 1, 2, 3

// 变量名可以任意
const [x, y, z] = [1, 2, 3];
console.log(x, y, z);  // 1, 2, 3
```

### 不完全解构

```javascript
// 跳过某些元素
const [first, , third] = [1, 2, 3, 4];
console.log(first, third);  // 1, 3

// 只取第一个
const [head] = [1, 2, 3];
console.log(head);  // 1

// 剩余元素
const [start, ...rest] = [1, 2, 3, 4, 5];
console.log(start);  // 1
console.log(rest);   // [2, 3, 4, 5]
```

### 默认值

```javascript
// 当值为 undefined 时使用默认值
const [a = 1, b = 2] = [3];
console.log(a, b);  // 3, 2

// null 不会触发默认值
const [c = 'default'] = [null];
console.log(c);  // null

// 默认值可以是表达式
const [d = 1 + 2] = [];
console.log(d);  // 3

// 默认值可以是函数
const getValue = () => {
    console.log('调用了 getValue');
    return 'default';
};
const [e = getValue()] = [undefined];  // 会调用
const [f = getValue()] = ['value'];    // 不会调用
```

## 3.2 对象解构

### 基础解构

```javascript
// 传统方式
const user = { name: '张三', age: 25 };
const name = user.name;
const age = user.age;

// 解构赋值
const { name, age } = { name: '张三', age: 25 };
console.log(name, age);  // '张三', 25
```

### 重命名变量

```javascript
// 变量名与属性名不同
const { name: userName, age: userAge } = { name: '张三', age: 25 };
console.log(userName);  // '张三'
console.log(userAge);   // 25

// 常用于 API 响应
const response = {
    code: 200,
    data: { id: 1, name: '张三' }
};
const { data: userData } = response;
console.log(userData);  // { id: 1, name: '张三' }
```

### 默认值

```javascript
// 属性不存在时使用默认值
const { name = '匿名用户', email = 'no-email' } = { age: 25 };
console.log(name);   // '匿名用户'
console.log(email);  // 'no-email'

// 默认值与重命名组合
const { name: userName = '访客', age = 0 } = { age: 25 };
console.log(userName);  // '访客'
console.log(age);       // 25
```

### 剩余属性

```javascript
const { name, ...rest } = {
    name: '张三',
    age: 25,
    city: '北京',
    email: 'zhang@example.com'
};

console.log(name);  // '张三'
console.log(rest);  // { age: 25, city: '北京', email: '...' }
```

## 3.3 嵌套解构

### 对象嵌套

```javascript
const user = {
    name: '张三',
    profile: {
        age: 25,
        address: {
            city: '北京',
            district: '朝阳'
        }
    }
};

// 嵌套解构
const {
    name,
    profile: {
        age,
        address: { city }
    }
} = user;

console.log(name);  // '张三'
console.log(age);   // 25
console.log(city);  // '北京'
```

### 数组嵌套

```javascript
const data = [1, [2, [3, [4]]]];

const [a, [b, [c, [d]]]] = data;
console.log(a, b, c, d);  // 1, 2, 3, 4
```

### 混合嵌套

```javascript
const config = {
    server: {
        hosts: ['host1', 'host2', 'host3'],
        port: 8080
    }
};

// 解构数组
const {
    server: {
        hosts: [primary, ...backups],
        port
    }
} = config;

console.log(primary);  // 'host1'
console.log(backups);  // ['host2', 'host3']
console.log(port);     // 8080
```

## 3.4 展开运算符

### 数组展开

```javascript
// 合并数组
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const merged = [...arr1, ...arr2];
console.log(merged);  // [1, 2, 3, 4, 5, 6]

// 在数组中间插入
const withInsert = [0, ...arr1, 7];
console.log(withInsert);  // [0, 1, 2, 3, 7]

// 复制数组（浅拷贝）
const copy = [...arr1];
console.log(copy);  // [1, 2, 3]
```

### 对象展开

```javascript
// 合并对象
const defaults = { theme: 'light', lang: 'zh' };
const userPrefs = { theme: 'dark' };
const config = { ...defaults, ...userPrefs };
console.log(config);  // { theme: 'dark', lang: 'zh' }

// 复制对象（浅拷贝）
const original = { a: 1, b: 2 };
const copy = { ...original };
console.log(copy);  // { a: 1, b: 2 }

// 注意：后面的属性覆盖前面的
const obj1 = { a: 1, b: 1 };
const obj2 = { b: 2, c: 2 };
const merged = { ...obj1, ...obj2 };
console.log(merged);  // { a: 1, b: 2, c: 2 }
```

### 函数调用展开

```javascript
// 展开数组作为参数
function sum(a, b, c) {
    return a + b + c;
}
const args = [1, 2, 3];
console.log(sum(...args));  // 6

// 对比 apply
console.log(sum.apply(null, args));  // 6
```

## 3.5 参数解构

### 函数参数解构

```javascript
// 对象参数解构
function printUser({ name, age, email = 'no-email' }) {
    console.log(`${name}, ${age}岁, ${email}`);
}

printUser({ name: '张三', age: 25 });
// '张三，25 岁，no-email'

// 数组参数解构
function printFirstTwo([first, second]) {
    console.log(first, second);
}
printFirstTwo([1, 2, 3]);  // 1, 2
```

### 带默认值的参数解构

```javascript
// 默认值
function createUser({ name = '匿名用户', role = 'user' } = {}) {
    return { name, role };
}

console.log(createUser());  // { name: '匿名用户', role: 'user' }
console.log(createUser({ name: '张三' }));  // { name: '张三', role: 'user' }

// 注意：= {} 是防止参数为 undefined
```

### Rest 参数

```javascript
// 剩余参数
function sum(...numbers) {
    return numbers.reduce((acc, cur) => acc + cur, 0);
}

console.log(sum(1, 2, 3, 4, 5));  // 15

// 与普通参数组合
function greet(greeting, ...names) {
    return names.map(name => `${greeting}, ${name}!`);
}

console.log(greet('Hello', '张三', '李四', '王五'));
// ['Hello, 张三!', 'Hello, 李四!', 'Hello, 王五!']
```

## 3.6 实用技巧

### 交换变量

```javascript
// 传统方式
let a = 1, b = 2;
let temp = a;
a = b;
b = temp;

// 解构方式
let a = 1, b = 2;
[a, b] = [b, a];
console.log(a, b);  // 2, 1
```

### 多值返回

```javascript
// 函数返回多个值
function getMinMax(numbers) {
    return [Math.min(...numbers), Math.max(...numbers)];
}

const [min, max] = getMinMax([3, 1, 4, 1, 5, 9]);
console.log(min, max);  // 1, 9

// 返回对象
function getUserInfo() {
    return { name: '张三', age: 25 };
}

const { name, age } = getUserInfo();
```

### 迭代器解构

```javascript
// Map 解构
const map = new Map([
    ['name', '张三'],
    ['age', 25]
]);

const [name, age] = map;
// 或者
const entries = [...map];
console.log(entries);  // [['name', '张三'], ['age', 25]]

// Set 解构
const set = new Set([1, 2, 3]);
const [a, b, c] = set;
console.log(a, b, c);  // 1, 2, 3
```

### 提取 JSON 数据

```javascript
// 从 API 响应中提取
const response = {
    data: {
        user: {
            id: 1,
            profile: {
                name: '张三',
                settings: { theme: 'dark' }
            }
        }
    }
};

// 深度解构
const {
    data: {
        user: {
            profile: {
                name,
                settings: { theme }
            }
        }
    }
} = response;

console.log(name);   // '张三'
console.log(theme);  // 'dark'
```

## 3.7 实践练习

### 练习 1：配置合并

```javascript
// 实现一个配置合并函数
function mergeConfig(defaults, overrides) {
    return { ...defaults, ...overrides };
}

const defaultConfig = {
    host: 'localhost',
    port: 3000,
    debug: false,
    timeout: 5000
};

const productionConfig = {
    host: 'api.example.com',
    debug: false
};

const config = mergeConfig(defaultConfig, productionConfig);
console.log(config);
// { host: 'api.example.com', port: 3000, debug: false, timeout: 5000 }
```

### 练习 2：数据处理

```javascript
// 处理用户数据
const users = [
    { id: 1, name: '张三', email: 'zhang@example.com', age: 25 },
    { id: 2, name: '李四', email: 'li@example.com', age: 30 },
    { id: 3, name: '王五', email: 'wang@example.com', age: 35 }
];

// 1. 提取用户名列表
const names = users.map(({ name }) => name);
console.log(names);  // ['张三', '李四', '王五']

// 2. 提取 id 和 name
const basicInfo = users.map(({ id, name }) => ({ id, name }));
console.log(basicInfo);

// 3. 按年龄分组
const byAgeGroup = users.reduce((acc, { age, ...user }) => {
    const group = age < 30 ? 'young' : 'mature';
    if (!acc[group]) acc[group] = [];
    acc[group].push(user);
    return acc;
}, {});
```

### 练习 3：URL 参数解析

```javascript
// 解析 URL 查询参数
function parseQueryString(qs) {
    const params = {};
    qs.replace(/[?&]?([^=&]+)=([^&]*)/gi, (match, key, value) => {
        params[decodeURIComponent(key)] = decodeURIComponent(value);
    });
    return params;
}

// 使用解构提取参数
const queryString = '?name=张三&age=25&city=北京';
const { name, age, city = '未知' } = parseQueryString(queryString);

console.log(name);  // '张三'
console.log(age);   // '25'
console.log(city);  // '北京'
```

## 3.8 常见问答

### Q1: 解构赋值的默认值什么时候生效？

**答：** 只有当值为 `undefined` 时才使用默认值。

```javascript
const [a = 1] = [undefined];  // a = 1
const [b = 1] = [null];       // b = null
const [c = 1] = [0];          // c = 0
const [d = 1] = [false];      // d = false
```

### Q2: 展开运算是深拷贝吗？

**答：** 不是，是浅拷贝。

```javascript
const original = { a: 1, b: { c: 2 } };
const copy = { ...original };

copy.b.c = 3;
console.log(original.b.c);  // 3（原对象也被修改了）
```

### Q3: 如何在解构时跳过某些属性？

```javascript
// 使用 rest 收集不需要的属性
const { password, ...safeUser } = {
    id: 1,
    name: '张三',
    password: 'secret'
};

console.log(safeUser);  // { id: 1, name: '张三' }
```

### Q4: 展开运算符可以用于字符串吗？

**答：** 可以，字符串是可迭代对象。

```javascript
const str = 'hello';
const chars = [...str];
console.log(chars);  // ['h', 'e', 'l', 'l', 'o']

// 或者
const [...spread] = 'world';
console.log(spread);  // ['w', 'o', 'r', 'l', 'd']
```
