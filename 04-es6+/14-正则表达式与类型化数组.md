## 12.0 通俗理解

### 正则表达式是什么？

```
正则表达式 = "匹配文本的规则"

就像：
- 找东西：符合规则的才匹配
- 验证：格式对不对
- 替换：找到并改掉
```

### 类型化数组是什么？

```
类型化数组 = "二进制数据"

就像：
- 普通数组：存任意类型
- 类型化数组：存固定类型（更高效）
- 用于：图片、音频、网络数据
```

### 通俗比喻

```
正则：搜索关键词
- /hello/ = 找"hello"
- /\d+/ = 找数字

类型化数组：原始二进制
- 处理图片：每个像素是 RGBA
- 处理音频：每个采样是数字
```

## 第一部分：正则表达式增强

### 12.1 点号匹配换行符（dotAll）

#### 问题背景

```javascript
// ES5 中，点号不匹配换行符
const text = 'hello\nworld';

// 无法匹配跨行内容
console.log(/hello.world/.test(text));  // false

// 传统解决方案：使用 [\s\S]
console.log(/hello[\s\S]world/.test(text));  // true
```

#### dotAll 模式（s 标志）

```javascript
// ES2018 新增 dotAll 模式
const text = 'hello\nworld';

// 使用 s 标志，点号匹配所有字符（包括换行）
console.log(/hello.world/s.test(text));  // true

// 匹配跨行内容
const html = '<div>\n  内容\n</div>';
const match = html.match(/<div>.*?<\/div>/s);
console.log(match[0]);  // '<div>\n  内容\n</div>'

// dotAll 属性
const regex = /hello.world/s;
console.log(regex.dotAll);  // true

const regex2 = /hello.world/;
console.log(regex2.dotAll);  // false
```

#### 实际应用场景

```javascript
// 提取 HTML 标签内容（可能跨行）
function extractTagContent(html, tagName) {
    const regex = new RegExp(`<${tagName}>(.*?)</${tagName}>`, 's');
    const match = html.match(regex);
    return match ? match[1] : null;
}

const html = `
<div>
    <p>第一段</p>
    <p>第二段</p>
</div>
`;

console.log(extractTagContent(html, 'div'));
// 输出包含换行的内容

// 提取注释（可能多行）
const commentRegex = /<!--[\s\S]*?-->/gs;
const comments = html.match(commentRegex);
```

### 12.2 后行断言（Lookbehind）

#### 先行断言 vs 后行断言

```javascript
// 先行断言（ES5 已有）
// 匹配后面跟着数字的美元符号
const regex1 = /\$(?=\d)/;
console.log(/price: $100/.match(regex1));  // ['$']

// 后行断言（ES2018 新增）
// 匹配前面有美元符号的数字
const regex2 = /(?<=\$)\d+/;
console.log(/price: $100/.match(regex2));  // ['100']
```

#### 后行肯定断言

```javascript
// 语法：(?<=pattern)expression

// 匹配价格数字（必须有$前缀）
const priceRegex = /(?<=\$)\d+(\.\d+)?/;

console.log('$99.99'.match(priceRegex));     // ['99.99']
console.log('€99.99'.match(priceRegex));     // null
console.log('99.99 USD'.match(priceRegex));  // null

// 提取文件名（不含扩展名）
const fileRegex = /(?<=\/)[^\/.]+(?=\.[^\/]+$)/;
const path = '/home/user/document.pdf';
console.log(path.match(fileRegex)[0]);  // 'document'
```

#### 后行否定断言

```javascript
// 语法：(?<!pattern)expression

// 匹配没有$前缀的数字
const regex = /(?<!\$)\d+/;

console.log('价格：100 元'.match(regex));    // ['100']
console.log('$100'.match(regex));           // null

// 匹配不是 http/https 的 URL
const urlRegex = /(?<!https?:)[^\s]+/g;
const text = '访问 http://example.com 或 ftp://files.com';
console.log(text.match(urlRegex));
// ['访问', 'http://example.com', '或', 'ftp://files.com']
```

### 12.3 命名捕获组

```javascript
// 传统捕获组
const dateRegex = /(\d{4})-(\d{2})-(\d{2})/;
const match = '2024-01-15'.match(dateRegex);
console.log(match[1]);  // '2024'（年份）
console.log(match[2]);  // '01'（月份）- 需要注释才知道含义

// 命名捕获组（ES2018）
const namedRegex = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/;
const namedMatch = '2024-01-15'.match(namedRegex);

console.log(namedMatch.groups.year);   // '2024'
console.log(namedMatch.groups.month);  // '01'
console.log(namedMatch.groups.day);    // '15'

// 在替换中使用命名捕获
const formatted = '2024-01-15'.replace(
    /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/,
    '$<month>/$<day>/$<year>'
);
console.log(formatted);  // '01/15/2024'

// 在正则中使用后向引用
const duplicateRegex = /(?<word>\w+)\s+\k<word>/;
console.log('hello hello'.match(duplicateRegex));
// ['hello hello', 'hello', groups: {word: 'hello'}]
```

### 12.4 Unicode 属性转义

```javascript
// 匹配中文字符
const chineseRegex = /\p{Script=Han}/u;
console.log(chineseRegex.test('中'));  // true
console.log(chineseRegex.test('A'));   // false

// 匹配任意Unicode字母
const letterRegex = /\p{Letter}/u;
console.log(letterRegex.test('A'));    // true
console.log(letterRegex.test('中'));   // true
console.log(letterRegex.test('1'));    // false

// 匹配数字（包括各种 Unicode 数字）
const numberRegex = /\p{Number}/u;
console.log(numberRegex.test('1'));      // true
console.log(numberRegex.test('①'));      // true
console.log(numberRegex.test('١'));      // true（阿拉伯语数字）

// 匹配 emoji
const emojiRegex = /\p{Emoji}/u;
console.log(emojiRegex.test('😀'));  // true

// 实用：移除所有 emoji
const text = '你好😀世界🌍';
const noEmoji = text.replace(/\p{Emoji}/gu, '');
console.log(noEmoji);  // '你好世界'
```

### 12.5 String.matchAll

```javascript
const text = 'Contact us at support@example.com or sales@example.com';
const emailRegex = /\w+@\w+\.\w+/g;

// 传统方式
let match;
const results1 = [];
while ((match = emailRegex.exec(text)) !== null) {
    results1.push(match);
}
console.log(results1);

// matchAll 方式（ES2020）
const results2 = [...text.matchAll(emailRegex)];
console.log(results2);

// 与命名捕获组配合
const namedEmailRegex = /(?<local>\w+)@(?<domain>\w+\.\w+)/g;
for (const match of text.matchAll(namedEmailRegex)) {
    console.log(`邮箱：${match.groups.local}@${match.groups.domain}`);
}
// 邮箱：support@example.com
// 邮箱：sales@example.com

// 转换为数组操作
const domains = [...text.matchAll(namedEmailRegex)]
    .map(m => m.groups.domain);
console.log(domains);  // ['example.com', 'example.com']
```

### 12.7 ArrayBuffer 基础

```javascript
// ArrayBuffer 是固定长度的二进制数据缓冲区
const buffer = new ArrayBuffer(16);  // 16 字节

console.log(buffer.byteLength);  // 16

// 不能直接操作 ArrayBuffer，需要使用视图
// const buffer[0] = 1;  // TypeError

// 创建视图
const view = new DataView(buffer);
view.setInt32(0, 42);  // 在偏移量 0 处写入 32 位整数

console.log(view.getInt32(0));  // 42

// 检查是否为 ArrayBuffer
console.log(buffer instanceof ArrayBuffer);  // true
```

### 12.8 TypedArray

```javascript
// TypedArray 是 ArrayBuffer 的视图，提供类型化访问

// 创建方式 1：指定长度
const int32 = new Int32Array(4);  // 4 个 32 位整数（16 字节）
int32[0] = 1;
int32[1] = 2;
console.log(int32.length);  // 4
console.log(int32.byteLength);  // 16

// 创建方式 2：从数组创建
const arr = new Uint8Array([1, 2, 3, 4, 5]);
console.log(arr);  // Uint8Array [1, 2, 3, 4, 5]

// 创建方式 3：从 ArrayBuffer 创建
const buffer = new ArrayBuffer(16);
const view = new Uint16Array(buffer);  // 8 个 16 位无符号整数

// 创建方式 4：从另一个 TypedArray 创建
const copy = new Uint8Array(arr);

// 常用 TypedArray 类型
// Int8Array, Uint8Array, Uint8ClampedArray - 8 位
// Int16Array, Uint16Array - 16 位
// Int32Array, Uint32Array - 32 位
// Float32Array, Float64Array - 浮点数

// 类似数组的方法
const typed = new Uint8Array([1, 2, 3, 4, 5]);
console.log(typed.map(x => x * 2));     // Uint8Array [2, 4, 6, 8, 10]
console.log(typed.filter(x => x > 2));  // Uint8Array [3, 4, 5]
console.log(typed.reduce((a, b) => a + b, 0));  // 15
```

### 12.9 DataView

```javascript
// DataView 提供对 ArrayBuffer 的低级读写，支持不同字节序

const buffer = new ArrayBuffer(16);
const view = new DataView(buffer);

// 写入不同类型的数据
view.setInt8(0, 42);           // 8 位有符号整数
view.setUint8(1, 255);         // 8 位无符号整数
view.setInt16(2, 1000);        // 16 位有符号整数
view.setInt32(4, 1000000);     // 32 位有符号整数
view.setFloat32(8, 3.14);      // 32 位浮点数
view.setFloat64(12, 2.718);    // 64 位浮点数

// 读取数据
console.log(view.getInt8(0));    // 42
console.log(view.getUint8(1));   // 255
console.log(view.getInt16(2));   // 1000
console.log(view.getFloat32(8)); // 3.14

// 字节序控制（大端/小端）
view.setInt16(0, 0x1234, false);  // 大端（默认）
view.setInt16(0, 0x1234, true);   // 小端

// 网络协议通常使用大端字节序
function writeNetworkInt16(view, offset, value) {
    view.setInt16(offset, value, false);  // false = 大端
}
```

### 12.10 实际应用场景

#### 场景 1：处理图片数据

```javascript
// 处理 PNG 文件头
async function readPngHeader(file) {
    const arrayBuffer = await file.arrayBuffer();
    const view = new DataView(arrayBuffer);

    // PNG 魔数：89 50 4E 47 0D 0A 1A 0A
    const magic = [
        view.getUint8(0), view.getUint8(1), view.getUint8(2), view.getUint8(3),
        view.getUint8(4), view.getUint8(5), view.getUint8(6), view.getUint8(7)
    ];

    const isPng = magic.every((b, i) =>
        b === [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A][i]
    );

    if (!isPng) {
        throw new Error('不是 PNG 文件');
    }

    // 读取宽度和高度（大端）
    const width = view.getUint32(16, false);
    const height = view.getUint32(20, false);

    return { width, height };
}
```

#### 场景 2：WebSocket 二进制消息

```javascript
// 发送二进制数据
function sendPosition(ws, x, y, z) {
    const buffer = new ArrayBuffer(12);  // 3 个 float32
    const view = new DataView(buffer);

    view.setFloat32(0, x, true);  // 小端
    view.setFloat32(4, y, true);
    view.setFloat32(8, z, true);

    ws.send(buffer);
}

// 接收二进制数据
ws.onmessage = (event) => {
    if (event.data instanceof ArrayBuffer) {
        const view = new DataView(event.data);
        const x = view.getFloat32(0, true);
        const y = view.getFloat32(4, true);
        const z = view.getFloat32(8, true);
        console.log('收到位置:', x, y, z);
    }
};
```

#### 场景 3：文件哈希计算

```javascript
// 使用 TypedArray 处理文件数据
async function computeHash(file) {
    const arrayBuffer = await file.arrayBuffer();
    const bytes = new Uint8Array(arrayBuffer);

    // 简单示例：XOR 哈希（实际应使用 crypto API）
    let hash = 0;
    for (let i = 0; i < bytes.length; i++) {
        hash = (hash ^ bytes[i]) & 0xFF;
    }

    return hash.toString(16).padStart(2, '0');
}
```

## 12.12 实践练习

### 练习 1：解析二进制协议

```javascript
// 模拟网络协议：
// 字节 0-1: 消息类型（uint16）
// 字节 2-3: 消息长度（uint16）
// 字节 4+:  数据内容

function parseMessage(buffer) {
    const view = new DataView(buffer);

    const messageType = view.getUint16(0, false);  // 大端
    const messageLength = view.getUint16(2, false);
    const data = new Uint8Array(buffer, 4, messageLength);

    return {
        type: messageType,
        length: messageLength,
        data: Array.from(data)
    };
}

// 创建测试消息
function createMessage(type, data) {
    const buffer = new ArrayBuffer(4 + data.length);
    const view = new DataView(buffer);

    view.setUint16(0, type, false);
    view.setUint16(2, data.length, false);

    new Uint8Array(buffer, 4).set(data);

    return buffer;
}

// 测试
const msgBuffer = createMessage(0x0102, [0x03, 0x04, 0x05]);
console.log(parseMessage(msgBuffer));
// { type: 258, length: 3, data: [3, 4, 5] }
```

### 练习 2：HTML 标签提取器

// 使用 matchAll 提取 HTML 标签
function extractTags(html) {
    const tagRegex = /<(?<tag>\w+)(?<attrs>[^>]*)>(?<content>.*?)<\/\k<tag>>/gs;
    const results = [];

    for (const match of html.matchAll(tagRegex)) {
        results.push({
            tagName: match.groups.tag,
            attributes: match.groups.attrs,
            content: match.groups.content
        });
    }

    return results;
}

const html = `

    <p>段落 1</p>
    <p>段落 2</p>
</div>
`;

console.log(extractTags(html));

### 练习 3：颜色值转换器

```javascript
// 十六进制颜色转 RGB
function hexToRgb(hex) {
    const result = /^#?(?<r>[a-f\d]{2})(?<g>[a-f\d]{2})(?<b>[a-f\d]{2})$/i.exec(hex);
    return result ? {
        r: parseInt(result.groups.r, 16),
        g: parseInt(result.groups.g, 16),
        b: parseInt(result.groups.b, 16)
    } : null;
}

console.log(hexToRgb('#FF5733'));  // { r: 255, g: 87, b: 51 }

// RGB 转十六进制
function rgbToHex(r, g, b) {
    return '#' + [r, g, b].map(x =>
        x.toString(16).padStart(2, '0')
    ).join('').toUpperCase();
}

console.log(rgbToHex(255, 87, 51));  // '#FF5733'
```

## 12.13 常见问答

### Q1: dotAll 模式有什么性能影响吗？

**答：** dotAll 模式本身不会带来明显的性能影响，但使用 `.*?` 等非贪婪匹配在长文本中可能较慢。

### Q2: 后行断言在所有浏览器都支持吗？

**答：** 后行断言是 ES2018 特性，现代浏览器（Chrome 62+、Firefox 78+、Safari 16.4+）都支持。如需支持旧浏览器，需要使用传统方式或 polyfill。

### Q3: TypedArray 和普通数组有什么区别？

**答：**
1. TypedArray 元素类型固定，只能存储指定类型的数字
2. TypedArray 有固定长度，创建后不能改变
3. TypedArray 的方法与普通数组类似，但返回值仍是 TypedArray
4. TypedArray 直接操作内存，性能更好

### Q4: 什么时候应该使用类型化数组？

**答：** 类型化数组适用于：
1. 处理二进制数据（文件、网络协议）
2. WebAudio、WebGL 等多媒体处理
3. 需要高性能数值计算的场景
4. 与原生 API 交互（如 Crypto API）
