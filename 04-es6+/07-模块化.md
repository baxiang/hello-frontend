## 7.1 导出（export）

### 具名导出

```javascript
// math.js

// 方式 1：声明时导出
export const PI = 3.14159;
export function add(a, b) {
    return a + b;
}

// 方式 2：先声明后导出
function subtract(a, b) {
    return a - b;
}
export { subtract };

// 方式 3：批量导出
function multiply(a, b) {
    return a * b;
}
function divide(a, b) {
    return a / b;
}
export { multiply, divide };
```

### 默认导出

```javascript
// calculator.js

// 方式 1：导出已存在的
function Calculator() {
    this.add = (a, b) => a + b;
}
export default Calculator;

// 方式 2：直接导出匿名
export default class {
    sayHi() {
        console.log('Hi');
    }
}

// 方式 3：导出表达式
export default {
    name: '默认导出',
    version: '1.0.0'
};
```

### 导出对比

```javascript
// 一个模块只能有一个默认导出
export default class A {}  // OK
// export default class B {}  // 错误

// 但可以有多个具名导出
export class C {}  // OK
export class D {}  // OK
```

## 7.2 导入（import）

### 基本导入

```javascript
// 导入具名导出
import { add, subtract } from './math.js';

// 导入默认导出
import Calculator from './calculator.js';

// 同时导入默认和具名
import Utils, { format, parse } from './utils.js';
```

### 重命名导入

```javascript
// 导入时重命名
import { add as plus, subtract as minus } from './math.js';

// 全部导入
import * as Math from './math.js';
Math.add(1, 2);
Math.PI;
```

### 只导入不绑定

```javascript
// 只执行模块副作用，不导入任何内容
import './polyfill.js';
import './styles.css';
```

## 7.3 动态导入

### 基础用法

```javascript
// 动态导入返回 Promise
button.addEventListener('click', async () => {
    const module = await import('./module.js');
    module.default();
    module.namedFunction();
});
```

### 条件加载

```javascript
// 根据条件加载不同模块
async function loadModule() {
    if (condition) {
        return import('./module-a.js');
    } else {
        return import('./module-b.js');
    }
}

// 使用
const module = await loadModule();
```

### 按需加载

```javascript
// 路由懒加载
const routes = [
    {
        path: '/home',
        component: () => import('./views/Home.vue')
    },
    {
        path: '/about',
        component: () => import('./views/About.vue')
    }
];

// 按需加载国际化
async function loadLocale(locale) {
    const messages = await import(`./locales/${locale}.js`);
    return messages.default;
}
```

## 7.4 模块重导出

### 重新导出

```javascript
// index.js - 统一出口

// 重新导出
export { add, subtract } from './math.js';
export { default as Calculator } from './calculator.js';

// 全部重新导出
export * from './math.js';
export * as Utils from './utils.js';

// 重命名后导出
export { format as formatDate } from './date.js';
```

### 聚合模块

```javascript
// components/index.js
export { Button } from './Button.js';
export { Input } from './Input.js';
export { Modal } from './Modal.js';

// 使用
import { Button, Input, Modal } from './components/index.js';
```

## 7.5 CommonJS vs ES6 Module

### 语法对比

```javascript
// CommonJS (Node.js)
// 导出
module.exports = { add, subtract };
// 或
exports.add = add;

// 导入
const { add, subtract } = require('./math.js');

// ES6 Module
// 导出
export { add, subtract };

// 导入
import { add, subtract } from './math.js';
```

### 核心区别

| 特性 | CommonJS | ES6 Module |
|------|----------|------------|
| 语法 | require/exports | import/export |
| 执行时机 | 运行时 | 编译时 |
| 输出 | 值的拷贝（可变） | 值的引用（只读） |
| this | 指向当前模块 | undefined |
| 静态分析 | ❌ | ✅ |
| 浏览器支持 | ❌ | ✅ |

### 循环依赖

```javascript
// a.js
export const foo = () => bar();
import { bar } from './b.js';

// b.js
export const bar = () => console.log('bar');

// ES6 Module 可以处理循环依赖
// CommonJS 会有问题
```

### 练习 1：模块组织

```javascript
// utils/index.js
export { formatDate } from './date.js';
export { formatCurrency } from './currency.js';
export { validateEmail, validatePhone } from './validator.js';

// utils/date.js
export function formatDate(date, format = 'YYYY-MM-DD') {
    // 实现
}

// utils/currency.js
export function formatCurrency(amount, currency = 'CNY') {
    // 实现
}

// utils/validator.js
export function validateEmail(email) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

export function validatePhone(phone) {
    return /^1[3-9]\d{9}$/.test(phone);
}

// 使用
import { formatDate, validateEmail } from './utils/index.js';
```

### 练习 2：插件系统

```javascript
// plugin-system.js
export class PluginManager {
    constructor() {
        this.plugins = [];
    }

    async register(pluginModule) {
        const plugin = await pluginModule;
        this.plugins.push(plugin.default);
    }

    async init(context) {
        for (const plugin of this.plugins) {
            await plugin.init?.(context);
        }
    }
}

// plugins/analytics.js
export default {
    name: 'analytics',
    init(context) {
        console.log('Analytics plugin initialized');
    },
    track(event) {
        console.log('Tracking:', event);
    }
};

// 使用
const manager = new PluginManager();
await manager.register(import('./plugins/analytics.js'));
await manager.init({ appId: '123' });
```

### 练习 3：配置模块

```javascript
// config.js
const defaultConfig = {
    api: {
        baseUrl: 'https://api.example.com',
        timeout: 5000
    },
    features: {
        darkMode: false,
        notifications: true
    }
};

export function createConfig(overrides = {}) {
    return {
        ...defaultConfig,
        ...overrides,
        features: {
            ...defaultConfig.features,
            ...overrides.features
        }
    };
}

export const config = createConfig();

// 使用
import { config, createConfig } from './config.js';

// 生产环境
const prodConfig = createConfig({
    api: { baseUrl: 'https://prod.api.com' },
    features: { darkMode: true }
});
```

## 7.7 常见问答

### Q1: export default 和具名导出有什么区别？

| 特性 | default | 具名 |
|------|---------|------|
| 数量 | 每模块 1 个 | 多个 |
| 导入命名 | 可自定义 | 必须匹配 |
| 重构友好 | ❌ | ✅ |
| Tree-shaking | ❌ | ✅ |

### Q2: 为什么 ES6 Module 更推荐？

**答：**
1. 静态分析，支持 Tree-shaking
2. 编译时检查错误
3. 浏览器原生支持
4. 异步加载

### Q3: 如何在 Node.js 中使用 ES6 Module？

```json
// package.json
{
    "type": "module"  // 启用 ES Module
}
```

```bash
# 或使用 .mjs 扩展名
node app.mjs
```

### Q4: import() 和 require() 可以混用吗？

**答：** 不可以，它们是不同模块系统的语法。
