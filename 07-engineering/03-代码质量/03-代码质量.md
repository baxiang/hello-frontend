## 3.1 ESLint 配置

### 基础配置

```javascript
// .eslintrc.js
module.exports = {
    // 环境
    env: {
        browser: true,
        es2021: true,
        node: true
    },

    // 扩展配置
    extends: [
        'eslint:recommended',
        'plugin:vue/vue3-recommended',
        '@vue/eslint-config-typescript'
    ],

    // 解析器
    parser: 'vue-eslint-parser',
    parserOptions: {
        ecmaVersion: 'latest',
        sourceType: 'module',
        parser: '@typescript-eslint/parser'
    },

    // 插件
    plugins: [
        'vue',
        '@typescript-eslint',
        'import'
    ],

    // 规则
    rules: {
        // 缩进
        'indent': ['error', 2],

        // 引号
        'quotes': ['error', 'single'],

        // 分号
        'semi': ['error', 'always'],

        // 尾随逗号
        'comma-dangle': ['error', 'never'],

        // 空格
        'space-before-function-paren': ['error', 'always'],

        // 箭头函数空格
        'arrow-spacing': ['error', { before: true, after: true }],

        // 无 console
        'no-console': process.env.NODE_ENV === 'production' ? 'warn' : 'off',

        // 无 debugger
        'no-debugger': process.env.NODE_ENV === 'production' ? 'warn' : 'off',

        // Vue 规则
        'vue/multi-word-component-names': 'off',
        'vue/singleline-html-element-content-newline': 'off',

        // 导入顺序
        'import/order': ['error', {
            groups: [
                'builtin',
                'external',
                'internal',
                'parent',
                'sibling',
                'index'
            ],
            'newlines-between': 'always'
        }]
    }
};
```

### 忽略文件

```json
// .eslintignore
node_modules/
dist/
build/
*.min.js
coverage/
```

## 3.2 Prettier 配置

### 配置文件

```json
// .prettierrc
{
  "semi": true,
  "singleQuote": true,
  "tabWidth": 2,
  "trailingComma": "none",
  "printWidth": 100,
  "bracketSpacing": true,
  "arrowParens": "always",
  "endOfLine": "lf",
  "htmlWhitespaceSensitivity": "ignore",
  "vueIndentScriptAndStyle": false
}
```

### 忽略文件

```json
// .prettierignore
node_modules
dist
build
coverage
*.min.js
```

### 与 ESLint 配合

```javascript
// 安装：npm install -D eslint-config-prettier

// .eslintrc.js 中添加
module.exports = {
    extends: [
        'eslint:recommended',
        'prettier'  // 放在最后，覆盖冲突的规则
    ]
};
```

## 3.3 Git Hooks

### Husky 配置

```bash
# 安装
npm install -D husky lint-staged

# 初始化
npx husky install

# 添加 pre-commit hook
npx husky add .husky/pre-commit "npx lint-staged"

# 添加 commit-msg hook
npx husky add .husky/commit-msg 'npx --no -- commitlint -e $1'
```

### lint-staged 配置

```json
// package.json
{
  "scripts": {
    "prepare": "husky install"
  },
  "lint-staged": {
    "*.{js,jsx,ts,tsx}": [
      "eslint --fix",
      "prettier --write"
    ],
    "*.{vue}": [
      "eslint --fix",
      "prettier --write"
    ],
    "*.{css,less,scss}": [
      "stylelint --fix",
      "prettier --write"
    ],
    "*.{json,md}": [
      "prettier --write"
    ]
  }
}
```

### commitlint 配置

```javascript
// commitlint.config.js
module.exports = {
    extends: ['@commitlint/config-conventional'],
    rules: {
        'type-enum': [
            2,
            'always',
            [
                'feat',     // 新功能
                'fix',      // 修复 bug
                'docs',     // 文档更新
                'style',    // 代码格式
                'refactor', // 重构
                'perf',     // 性能优化
                'test',     // 测试
                'build',    // 构建
                'ci',       // CI 配置
                'chore',    // 其他
                'revert'    // 回滚
            ]
        ]
    }
};
```

## 3.4 常见问题

### Q1: ESLint 和 Prettier 冲突怎么办？

```javascript
// 使用 eslint-config-prettier 禁用 ESLint 中与 Prettier 冲突的规则
module.exports = {
    extends: [
        'eslint:recommended',
        'plugin:vue/vue3-recommended',
        'prettier'  // 必须放在最后
    ]
};
```

### Q2: 如何跳过 pre-commit 检查？

```bash
# 使用 --no-verify 跳过
git commit --no-verify -m "message"
```

### Q3: 如何在 VS Code 中自动格式化？

```json
// .vscode/settings.json
{
  "editor.formatOnSave": true,
  "editor.defaultFormatter": "esbenp.prettier-vscode",
  "editor.codeActionsOnSave": {
    "source.fixAll.eslint": true
  }
}
```
