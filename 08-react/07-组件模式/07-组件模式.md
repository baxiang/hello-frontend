## L1 理解层：会用

### 高阶组件（HOC）

**最简示例（1-3行）：**

```tsx
const withAuth = (Component) => (props) => {
  const isAuthenticated = useAuth()
  return isAuthenticated ? <Component {...props} /> : <Login />
}
```

**详细示例：**

```tsx
// withAuth — 认证包装（给组件加上认证检查）
function withAuth<P extends object>(Component: React.ComponentType<P>) {
  return function AuthenticatedComponent(props: P) {
    const isAuthenticated = useAuth()
    if (!isAuthenticated) return <Login />
    return <Component {...props} />
  }
}

// 使用 — 一行代码给任何组件加上认证保护
const ProtectedDashboard = withAuth(Dashboard)
const ProtectedSettings = withAuth(Settings)
```

### Render Props

**最简示例（1-3行）：**

```tsx
<Mouse render={position => <p>位置：{position.x}, {position.y}</p>} />
```

**详细示例：**

```tsx
// Mouse — 追踪鼠标位置，把数据交给 render 函数
function Mouse({ render }: { render: (pos: { x: number; y: number }) => React.ReactNode }) {
  const [position, setPosition] = useState({ x: 0, y: 0 })

  useEffect(() => {
    const handleMove = (e: MouseEvent) => setPosition({ x: e.clientX, y: e.clientY })
    window.addEventListener('mousemove', handleMove)
    return () => window.removeEventListener('mousemove', handleMove)
  }, [])

  return render(position)
}

// 使用 — 你决定怎么显示数据
<Mouse render={({ x, y }) => <div>鼠标位置：{x}, {y}</div>} />
```

### Compound Components

**最简示例（1-3行）：**

```tsx
<Select value={val} onChange={setVal}>
  <Select.Option value="a">选项 A</Select.Option>
  <Select.Option value="b">选项 B</Select.Option>
</Select>
```

**详细示例：**

```tsx
// Select — 复合组件（父组件）
function Select({ value, onChange, children }: {
  value: string
  onChange: (v: string) => void
  children: React.ReactNode
}) {
  return (
    <div className="select">
      {React.Children.map(children, child =>
        React.cloneElement(child as React.ReactElement, { value, onChange })
      )}
    </div>
  )
}

// Option — 子组件
function Option({ value, onChange, children }: {
  value: string
  onChange: (v: string) => void
  children: React.ReactNode
}) {
  return (
    <div onClick={() => onChange(value)} className="option">
      {children}
    </div>
  )
}

// 组合 — Select.Option 作为 Select 的属性
Select.Option = Option

// 使用 — 语义清晰，父子组件自动配合
<Select value={selected} onChange={setSelected}>
  <Select.Option value="react">React</Select.Option>
  <Select.Option value="vue">Vue</Select.Option>
</Select>
```

## L2 实践层：用好

### 模式选择指南

| 模式 | 适用场景 | 示例 | 优缺点 |
|------|---------|------|--------|
| 自定义 Hook | 逻辑复用 | `useFetch`, `useLocalStorage` | ✅ 最常用 ✅ 代码简洁 |
| HOC | 组件包装 | `withAuth`, `withTheme` | ✅ 可组合 ⚠️ 嵌套深 |
| Render Props | 灵活渲染 | 鼠标追踪、滚动监听 | ✅ 灵活 ⚠️ 回调地狱 |
| Compound | 相关组件组合 | Select/Option、Tabs/Tab | ✅ 语义清晰 ⚠️ 实现复杂 |

### 自定义 Hooks

```tsx
// useLocalStorage — 持久化状态
function useLocalStorage<T>(key: string, initial: T) {
  const [value, setValue] = useState<T>(() => {
    const stored = localStorage.getItem(key)
    return stored ? JSON.parse(stored) : initial
  })

  useEffect(() => {
    localStorage.setItem(key, JSON.stringify(value))
  }, [key, value])

  return [value, setValue] as const
}

// useDebounce — 防抖值
function useDebounce<T>(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value)
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])
  return debounced
}

// 使用
const [search, setSearch] = useState('')
const debouncedSearch = useDebounce(search, 300)  // 300ms 后才更新
```

### 反模式：不要这样做

```tsx
// ❌ 错误：HOC 嵌套过深（回调地狱的组件版）
const App = withTheme(withAuth(withRouter(Dashboard)))

// ✅ 正确：用自定义 Hook 替代
function Dashboard() {
  const { theme } = useTheme()
  const { user } = useAuth()
  const params = useParams()
  return <div>...</div>
}
```

```tsx
// ❌ 错误：Render Props 嵌套过深
<Mouse>
  {pos => <Scroll>
    {scroll => <Fetch url="/api">
      {data => <div>{pos.x} {scroll.y} {data.name}</div>}
    </Fetch>}
  </Scroll>}
</Mouse>

// ✅ 正确：用自定义 Hook 替代
function MyComponent() {
  const pos = useMouse()
  const scroll = useScroll()
  const { data } = useFetch('/api')
  return <div>{pos.x} {scroll.y} {data.name}</div>
}
```

### 适用场景

| 场景 | 推荐方案 | 原因 |
|------|---------|------|
| 提取共享逻辑 | 自定义 Hook | 代码简洁，易测试 |
| 给组件加功能 | HOC | 可复用，可组合 |
| 数据提供者 | Render Props | 灵活控制渲染 |
| 关联组件 | Compound | 语义清晰，API 友好 |

## L3 专家层：深入

### Context + useReducer 模式

```tsx
// 状态管理模式 — 适合中等复杂度的全局状态
interface State { count: number }
type Action = { type: 'increment' } | { type: 'decrement' }

const CounterContext = createContext<{ state: State; dispatch: React.Dispatch<Action> } | null>(null)

function CounterProvider({ children }: { children: React.ReactNode }) {
  const [state, dispatch] = useReducer((s: State, a: Action) => {
    switch (a.type) {
      case 'increment': return { count: s.count + 1 }
      case 'decrement': return { count: s.count - 1 }
      default: return s
    }
  }, { count: 0 })

  return <CounterContext.Provider value={{ state, dispatch }}>{children}</CounterContext.Provider>
}

function useCounter() {
  const ctx = useContext(CounterContext)
  if (!ctx) throw new Error('useCounter must be used within CounterProvider')
  return ctx
}
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **高阶组件 HOC** | 接受组件返回新组件的函数 | `withAuth(Component)` |
| **Render Props** | 通过 render 函数传递数据的模式 | `<Mouse render={pos => ...}>` |
| **Compound Components** | 父子组件有默契的组合模式 | `<Select><Option>...</Option></Select>` |
| **自定义 Hook** | 以 use 开头的函数，复用状态逻辑 | `useFetch`, `useLocalStorage` |

## 实践练习

```tsx
// 练习 1：实现 useFetch Hook
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    fetch(url)
      .then(res => { if (!res.ok) throw new Error(`HTTP ${res.status}`); return res.json() })
      .then(data => { if (!cancelled) setData(data) })
      .catch(err => { if (!cancelled) setError(err) })
      .finally(() => { if (!cancelled) setLoading(false) })

    return () => { cancelled = true }
  }, [url])

  return { data, loading, error, refetch: () => {} }
}

// 练习 2：实现 Tabs 复合组件
function Tabs({ children, value, onChange }: {
  children: React.ReactNode
  value: string
  onChange: (v: string) => void
}) {
  return (
    <div>
      <div className="tabs">
        {React.Children.map(children, child =>
          React.cloneElement(child as React.ReactElement, { value, onChange })
        )}
      </div>
    </div>
  )
}

function Tab({ value, onChange, label, children }: {
  value: string
  onChange: (v: string) => void
  label: string
  children: React.ReactNode
}) {
  const isActive = value === label
  return (
    <div>
      <button onClick={() => onChange(label)} className={isActive ? 'active' : ''}>
        {label}
      </button>
      {isActive && <div className="tab-content">{children}</div>}
    </div>
  )
}

Tabs.Tab = Tab

// 使用
<Tabs value={active} onChange={setActive}>
  <Tabs.Tab label="首页"><HomeContent /></Tabs.Tab>
  <Tabs.Tab label="关于"><AboutContent /></Tabs.Tab>
</Tabs>
```

## 常见问题

### Q1：自定义 Hook 和 HOC 选哪个？

**优先用自定义 Hook（代码更简洁），需要包装整个组件时用 HOC：**

```tsx
// ✅ 自定义 Hook — 推荐
function Dashboard() {
  const { user } = useAuth()
  return user ? <Content /> : <Login />
}

// ⚠️ HOC — 适合需要统一包装的场景
const ProtectedDashboard = withAuth(Dashboard)
```

### Q2：什么时候用 Compound Components？

**当多个组件有紧密的父子关系，且需要共享状态时：**

```tsx
// ✅ Compound — 语义清晰
<Select value={val} onChange={setVal}>
  <Select.Option value="a">A</Select.Option>
</Select>

// ❌ Props — 配置复杂
<Select options={[{ value: 'a', label: 'A' }]} />
```

### Q3：Render Props 过时了吗？

**不是。但在自定义 Hook 普及后，Render Props 的使用场景变少了。适合需要灵活控制渲染的场景。**
