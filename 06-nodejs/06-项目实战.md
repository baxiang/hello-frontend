本项目是一个完整的博客后端 API 系统，包含以下功能：

- 评论系统
- 文件上传
- 分页查询
- 权限控制

```javascript
// src/models/User.js
const mongoose = require('mongoose');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

const userSchema = new mongoose.Schema({
    username: {
        type: String,
        required: [true, '用户名是必填项'],
        unique: true,
        trim: true,
        minlength: [3, '用户名至少 3 个字符'],
        maxlength: [30, '用户名最多 30 个字符']
    },
    email: {
        type: String,
        required: [true, '邮箱是必填项'],
        unique: true,
        lowercase: true,
        match: [/^\w+@[a-zA-Z_]+?\.[a-zA-Z]{2,3}$/, '请输入有效的邮箱']
    },
    password: {
        type: String,
        required: [true, '密码是必填项'],
        minlength: [6, '密码至少 6 个字符'],
        select: false
    },
    avatar: {
        type: String,
        default: ''
    },
    bio: {
        type: String,
        maxlength: [500, '简介最多 500 个字符']
    },
    role: {
        type: String,
        enum: ['user', 'admin'],
        default: 'user'
    },
    isActive: {
        type: Boolean,
        default: true
    }
}, {
    timestamps: true
});

// 索引
userSchema.index({ email: 1 });
userSchema.index({ username: 1 });

// 加密密码
userSchema.pre('save', async function(next) {
    if (!this.isModified('password')) return next();

    const salt = await bcrypt.genSalt(10);
    this.password = await bcrypt.hash(this.password, salt);
    next();
});

// 比较密码
userSchema.methods.comparePassword = async function(candidate) {
    return bcrypt.compare(candidate, this.password);
};

// 生成 token
userSchema.methods.generateToken = function() {
    return jwt.sign(
        { userId: this._id, role: this.role },
        process.env.JWT_SECRET,
        { expiresIn: process.env.JWT_EXPIRES_IN || '7d' }
    );
};

// 安全对象
userSchema.methods.toSafeObject = function() {
    const { _id, username, email, avatar, bio, role, createdAt } = this.toObject();
    return { _id, username, email, avatar, bio, role, createdAt };
};

module.exports = mongoose.model('User', userSchema);
```

### Post 模型

```javascript
// src/models/Post.js
const mongoose = require('mongoose');

const postSchema = new mongoose.Schema({
    title: {
        type: String,
        required: [true, '标题是必填项'],
        trim: true,
        maxlength: [200, '标题最多 200 个字符']
    },
    content: {
        type: String,
        required: [true, '内容是必填项']
    },
    excerpt: {
        type: String,
        maxlength: [500, '摘要最多 500 个字符']
    },
    cover: {
        type: String,
        default: ''
    },
    author: {
        type: mongoose.Schema.Types.ObjectId,
        ref: 'User',
        required: true
    },
    category: {
        type: mongoose.Schema.Types.ObjectId,
        ref: 'Category'
    },
    tags: [{
        type: String,
        trim: true
    }],
    status: {
        type: String,
        enum: ['draft', 'published'],
        default: 'draft'
    },
    views: {
        type: Number,
        default: 0
    },
    likes: [{
        type: mongoose.Schema.Types.ObjectId,
        ref: 'User'
    }]
}, {
    timestamps: true
});

// 索引
postSchema.index({ title: 'text', content: 'text' });
postSchema.index({ author: 1, createdAt: -1 });
postSchema.index({ category: 1, status: 1, createdAt: -1 });
postSchema.index({ tags: 1 });

// 虚拟属性 - 评论数
postSchema.virtual('commentCount', {
    ref: 'Comment',
    localField: '_id',
    foreignField: 'post',
    count: true
});

// 在 toJSON 中包含虚拟属性
postSchema.set('toJSON', {
    virtuals: true
});

// 静态方法 - 搜索
postSchema.statics.search = function(query, options = {}) {
    const { page = 1, limit = 10, author, category, status = 'published' } = options;

    const searchQuery = {
        status,
        ...author && { author },
        ...category && { category }
    };

    if (query) {
        searchQuery.$text = { $search: query };
    }

    return this.find(searchQuery)
        .populate('author', 'username avatar')
        .populate('category', 'name slug')
        .sort({ createdAt: -1 })
        .limit(limit)
        .skip((page - 1) * limit);
};

// 实例方法 - 增加阅读数
postSchema.methods.incrementViews = function() {
    this.views += 1;
    return this.save({ validateBeforeSave: false });
};

// 实例方法 - 点赞
postSchema.methods.toggleLike = function(userId) {
    if (this.likes.includes(userId)) {
        this.likes.pull(userId);
    } else {
        this.likes.push(userId);
    }
    return this.save();
};

module.exports = mongoose.model('Post', postSchema);
```

### 认证控制器

```javascript
// src/controllers/auth.js
const { ValidationError, UnauthorizedError, ConflictError } = require('../errors/AppError');
const User = require('../models/User');

// 注册
exports.register = async (req, res) => {
    const { username, email, password } = req.body;

    // 验证必填字段
    if (!username || !email || !password) {
        throw new ValidationError('请填写所有必填字段');
    }

    // 检查用户是否存在
    const existingUser = await User.findOne({
        $or: [{ email }, { username }]
    });

    if (existingUser) {
        throw new ConflictError('用户名或邮箱已被使用');
    }

    // 创建用户
    const user = await User.create({ username, email, password });
    const token = user.generateToken();

    res.status(201).json({
        message: '注册成功',
        token,
        user: user.toSafeObject()
    });
};

// 登录
exports.login = async (req, res) => {
    const { email, password } = req.body;

    // 验证必填字段
    if (!email || !password) {
        throw new ValidationError('请输入邮箱和密码');
    }

    // 查找用户
    const user = await User.findOne({ email }).select('+password');

    if (!user) {
        throw new UnauthorizedError('邮箱或密码错误');
    }

    // 检查用户状态
    if (!user.isActive) {
        throw new UnauthorizedError('账户已被禁用');
    }

    // 验证密码
    const isMatch = await user.comparePassword(password);

    if (!isMatch) {
        throw new UnauthorizedError('邮箱或密码错误');
    }

    // 生成 token
    const token = user.generateToken();

    res.json({
        message: '登录成功',
        token,
        user: user.toSafeObject()
    });
};

// 获取当前用户
exports.getMe = async (req, res) => {
    const user = await User.findById(req.user.userId);

    if (!user) {
        throw new UnauthorizedError('用户不存在');
    }

    res.json({
        user: user.toSafeObject()
    });
};

// 更新资料
exports.updateProfile = async (req, res) => {
    const { username, avatar, bio } = req.body;

    const user = await User.findByIdAndUpdate(
        req.user.userId,
        { username, avatar, bio },
        { new: true, runValidators: true }
    );

    if (!user) {
        throw new ValidationError('用户不存在');
    }

    res.json({
        message: '更新成功',
        user: user.toSafeObject()
    });
};

// 修改密码
exports.changePassword = async (req, res) => {
    const { currentPassword, newPassword } = req.body;

    if (!currentPassword || !newPassword) {
        throw new ValidationError('请输入当前密码和新密码');
    }

    const user = await User.findById(req.user.userId).select('+password');

    // 验证当前密码
    const isMatch = await user.comparePassword(currentPassword);

    if (!isMatch) {
        throw new ValidationError('当前密码错误');
    }

    // 更新密码
    user.password = newPassword;
    await user.save();

    res.json({
        message: '密码修改成功'
    });
};
```

### 文章控制器

```javascript
// src/controllers/post.js
const Post = require('../models/Post');
const { ValidationError, NotFoundError } = require('../errors/AppError');

// 获取文章列表
exports.getPosts = async (req, res) => {
    const {
        page = 1,
        limit = 10,
        category,
        tag,
        author,
        search
    } = req.query;

    const options = {
        page: parseInt(page),
        limit: parseInt(limit),
        status: req.query.status || 'published'
    };

    if (category) options.category = category;
    if (author) options.author = author;

    // 查询
    const posts = await Post.search(search, options);

    // 包含标签过滤
    let query = { status: options.status };
    if (tag) query.tags = tag;
    if (category) query.category = category;
    if (author) query.author = author;

    const total = await Post.countDocuments(query);

    res.json({
        posts,
        pagination: {
            currentPage: parseInt(page),
            totalPages: Math.ceil(total / parseInt(limit)),
            total
        }
    });
};

// 获取单篇文章
exports.getPost = async (req, res) => {
    const post = await Post.findById(req.params.id)
        .populate('author', 'username avatar bio')
        .populate('category', 'name slug')
        .populate({
            path: 'comments',
            match: { status: 'approved' },
            populate: { path: 'author', select: 'username avatar' }
        });

    if (!post) {
        throw new NotFoundError('文章不存在');
    }

    // 增加阅读数
    await post.incrementViews();

    res.json({ post });
};

// 创建文章
exports.createPost = async (req, res) => {
    const { title, content, excerpt, cover, category, tags, status } = req.body;

    const post = await Post.create({
        title,
        content,
        excerpt,
        cover,
        category,
        tags: tags ? (Array.isArray(tags) ? tags : tags.split(',')) : [],
        status,
        author: req.user.userId
    });

    res.status(201).json({
        message: '文章创建成功',
        post
    });
};

// 更新文章
exports.updatePost = async (req, res) => {
    const post = await Post.findOneAndUpdate(
        { _id: req.params.id, author: req.user.userId },
        req.body,
        { new: true, runValidators: true }
    );

    if (!post) {
        throw new NotFoundError('文章不存在或无权限');
    }

    res.json({
        message: '更新成功',
        post
    });
};

// 删除文章
exports.deletePost = async (req, res) => {
    const post = await Post.findOneAndDelete({
        _id: req.params.id,
        author: req.user.userId
    });

    if (!post) {
        throw new NotFoundError('文章不存在或无权限');
    }

    res.status(204).send();
};

// 点赞文章
exports.toggleLike = async (req, res) => {
    const post = await Post.findById(req.params.id);

    if (!post) {
        throw new NotFoundError('文章不存在');
    }

    await post.toggleLike(req.user.userId);

    res.json({
        message: '操作成功',
        likes: post.likes
    });
};
```

## 6.5 中间件

### 认证中间件

```javascript
// src/middleware/auth.js
const jwt = require('jsonwebtoken');
const { UnauthorizedError } = require('../errors/AppError');

function authMiddleware(req, res, next) {
    const authHeader = req.headers.authorization;

    if (!authHeader || !authHeader.startsWith('Bearer ')) {
        throw new UnauthorizedError('请先登录');
    }

    const token = authHeader.split(' ')[1];

    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.user = decoded;
        next();
    } catch (error) {
        if (error.name === 'TokenExpiredError') {
            throw new UnauthorizedError('登录已过期，请重新登录');
        }
        throw new UnauthorizedError('无效的登录凭证');
    }
}

module.exports = authMiddleware;
```

```javascript
            message: '接口不存在'
        }
    });
});

// 错误处理
app.use(errorHandler);

module.exports = app;
```

### server.js

```javascript
// server.js
const app = require('./app');
const mongoose = require('mongoose');
const config = require('./config');
const logger = require('./src/utils/logger');

const startServer = async () => {
    try {
        // 连接数据库
        await mongoose.connect(config.mongo.uri, config.mongo.options);
        logger.info('MongoDB 连接成功');

        // 启动服务器
        const server = app.listen(config.port, () => {
            logger.info(`服务器启动在 http://localhost:${config.port}`);
        });

        // 优雅关闭
        const gracefulShutdown = async (signal) => {
            logger.info(`收到信号 ${signal}`);
            server.close(async () => {
                await mongoose.connection.close();
                process.exit(0);
            });
        };

        process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
        process.on('SIGINT', () => gracefulShutdown('SIGINT'));

    } catch (error) {
        logger.error('服务器启动失败', { error });
        process.exit(1);
    }
};

startServer();
```
