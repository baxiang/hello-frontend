# fs 文件系统模块

> 读取、写入、复制、删除文件等操作

## 2.1 读取文件

### 异步读取

```javascript
const fs = require('fs').promises;

// 读取文件
async function readFile() {
    try {
        const content = await fs.readFile('file.txt', 'utf-8');
        console.log(content);
    } catch (err) {
        console.error('读取失败:', err);
    }
}

readFile();
```

### 同步读取

```javascript
const fs = require('fs');

// 同步读取（会阻塞）
const content = fs.readFileSync('file.txt', 'utf-8');
console.log(content);
```

### 读取二进制数据

```javascript
// 读取为 Buffer（不指定编码）
const buffer = fs.readFileSync('image.png');
console.log(buffer);  // <Buffer 89 50 4e 47 ...>

// 指定编码
const content = fs.readFileSync('file.txt', 'utf-8');
```

## 2.2 写入文件

### 写入文件（覆盖）

```javascript
const fs = require('fs').promises;

// 写入文本
await fs.writeFile('message.txt', 'Hello World');

// 写入 JSON
await fs.writeFile('data.json', JSON.stringify({ name: '张三' }));

// 指定编码
await fs.writeFile('message.txt', '你好', 'utf-8');
```

### 追加内容

```javascript
const fs = require('fs').promises;

// 追加到文件末尾
await fs.appendFile('log.txt', '新的日志内容\n');

// 如果文件不存在，会自动创建
await fs.appendFile('new.txt', '内容');
```

### 同步写入

```javascript
const fs = require('fs');

fs.writeFileSync('message.txt', 'Hello');
fs.appendFileSync('log.txt', 'new log\n');
```

## 2.3 文件信息

```javascript
const fs = require('fs').promises;

async function getFileInfo() {
    const stats = await fs.stat('file.txt');

    console.log('是文件:', stats.isFile());        // true
    console.log('是目录:', stats.isDirectory());  // false
    console.log('大小:', stats.size, '字节');
    console.log('创建时间:', stats.birthtime);
    console.log('修改时间:', stats.mtime);
    console.log('访问时间:', stats.atime);
}

getFileInfo();
```

## 2.4 目录操作

```javascript
const fs = require('fs').promises;

// 创建目录
await fs.mkdir('new-folder');

// 创建多层目录
await fs.mkdir('a/b/c', { recursive: true });

// 读取目录
const files = await fs.readdir('./src');
console.log(files);  // ['file1.js', 'file2.js', 'folder']

// 删除目录
await fs.rmdir('empty-folder');

// 递归删除（Node.js 14+）
await fs.rm('folder', { recursive: true });
```

## 2.5 检查文件是否存在

```javascript
const fs = require('fs').promises;

async function checkFile() {
    try {
        await fs.access('file.txt');
        console.log('文件存在');
    } catch {
        console.log('文件不存在');
    }
}

// 也可以检查读写权限
await fs.access('file.txt', fs.constants.R_OK | fs.constants.W_OK);
```

## 2.6 删除文件

```javascript
const fs = require('fs').promises;

// 删除文件
await fs.unlink('file-to-delete.txt');

// 删除多个文件
await Promise.all([
    fs.unlink('file1.txt'),
    fs.unlink('file2.txt')
]);
```

## 2.7 复制和移动

```javascript
const fs = require('fs').promises;

// 复制文件
await fs.copyFile('source.txt', 'destination.txt');

// 复制并覆盖
await fs.copyFile('source.txt', 'existing.txt');

// 移动/重命名

await fs.rename('file.txt', 'folder/file.txt');
```

## 2.8 监听文件变化

```javascript
const fs = require('fs');

// 监听文件变化
fs.watch('file.txt', (eventType, filename) => {
    console.log('事件类型:', eventType);  // 'change' 或 'rename'
    console.log('文件名:', filename);
});

// 更详细的监听
const watcher = fs.watch('./src', { recursive: true }, (eventType, filename) => {
    console.log(`${eventType}: ${filename}`);
});

// 停止监听
watcher.close();
```

## 2.9 完整示例：文件管理器

```javascript
const fs = require('fs').promises;
const path = require('path');

class FileManager {
    constructor(basePath) {
        this.basePath = basePath;
    }

    // 列出文件
    async listFiles(dir = '.') {
        const targetPath = path.join(this.basePath, dir);
        const files = await fs.readdir(targetPath);

        const fileList = await Promise.all(
            files.map(async (file) => {
                const filePath = path.join(targetPath, file);
                const stats = await fs.stat(filePath);
                return {
                    name: file,
                    isDirectory: stats.isDirectory(),
                    size: stats.size,
                    modified: stats.mtime
                };
            })
        );

        return fileList;
    }

    // 读取文件
    async readFile(filePath) {
        const fullPath = path.join(this.basePath, filePath);
        return await fs.readFile(fullPath, 'utf-8');
    }

    // 写入文件
    async writeFile(filePath, content) {
        const fullPath = path.join(this.basePath, filePath);
        await fs.writeFile(fullPath, content, 'utf-8');
    }

    // 创建目录
    async createDir(dirPath) {
        const fullPath = path.join(this.basePath, dirPath);
        await fs.mkdir(fullPath, { recursive: true });
    }

    // 删除文件
    async deleteFile(filePath) {
        const fullPath = path.join(this.basePath, filePath);
        await fs.unlink(fullPath);
    }

    // 复制文件
    async copyFile(source, destination) {
        const sourcePath = path.join(this.basePath, source);
        const destPath = path.join(this.basePath, destination);
        await fs.copyFile(sourcePath, destPath);
    }
}

// 使用
const fm = new FileManager('./my-folder');

async function main() {
    // 列出文件
    const files = await fm.listFiles();
    console.log(files);

    // 写入文件
    await fm.writeFile('test.txt', 'Hello World');

    // 读取文件
    const content = await fm.readFile('test.txt');
    console.log(content);
}

main();
```

## 2.10 回调风格 vs Promise 风格

```javascript
const fs = require('fs');

// 回调风格（旧）
fs.readFile('file.txt', 'utf-8', (err, data) => {
    if (err) throw err;
    console.log(data);
});

// Promise 风格（推荐）
const fsPromises = fs.promises;
await fsPromises.readFile('file.txt', 'utf-8');

// 简写
const fs2 = require('fs').promises;
await fs2.readFile('file.txt', 'utf-8');
```

## 2.11 常见问答

### Q1: 同步和异步怎么选？

```javascript
// 异步（推荐）：不阻塞主线程
const data = await fs.promises.readFile('file.txt');

// 同步（谨慎）：会阻塞主线程
const data = fs.readFileSync('file.txt');

// 场景选择：
// - 服务器响应：必须用异步
// - 启动脚本：可以用同步
// - 批量处理：大文件用异步
```

### Q2: 如何处理大文件？

```javascript
// 大文件不要一次性读取
// 使用流（stream）
const fs = require('fs');
const readStream = fs.createReadStream('big-file.txt');
const writeStream = fs.createWriteStream('copy.txt');

readStream.pipe(writeStream);
```
