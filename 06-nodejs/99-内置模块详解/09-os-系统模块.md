# os 系统模块

> 获取操作系统信息

## 什么是 os 模块？

```
┌─────────────────────────────────────────────────────────────┐
│  os = 操作系统信息模块                                       │
│                                                             │
│  就像：                                                     │
│  - 电脑的属性面板：查看系统信息                             │
│  - 任务管理器：查看 CPU、内存                               │
│  - 系统设置：查看操作系统版本                               │
│                                                             │
│  主要功能：                                                 │
│  - CPU 信息                                                 │
│  - 内存信息                                                 │
│  - 操作系统信息                                             │
│  - 网络信息                                                 │
│  - 路径分隔符                                               │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 9.1 系统信息

### os.platform() / os.type()

```javascript
const os = require('os');

// 平台
console.log(os.platform());  // darwin (Mac), win32 (Windows), linux
console.log(os.type());      // Darwin, Windows_NT, Linux

// 操作系统版本
console.log(os.release());  // 21.4.0, 10.0.19044, 5.4.0

// 主机名
console.log(os.hostname());  // zhangsandeMacBook-Pro
```

---

## 9.2 CPU 信息

### os.cpus()

```javascript
const os = require('os');

// CPU 信息
const cpus = os.cpus();
console.log('CPU 数量:', cpus.length);

cpus.forEach((cpu, index) => {
    console.log(`CPU ${index}:`, {
        model: cpu.model,
        speed: cpu.speed,  // MHz
        times: cpu.times
    });
});

// 整体信息
console.log('CPU 架构:', os.arch());  // x64, arm64
console.log('CPU 频率:', os.totalmem());  // 总内存
```

## 9.3 内存信息

```javascript
const os = require('os');

// 内存
console.log('总内存:', os.totalmem(), 'bytes', os.totalmem() / 1024 / 1024 / 1024, 'GB');
console.log('空闲内存:', os.freemem(), 'bytes', os.freemem() / 1024 / 1024 / 1024, 'GB');

// 计算使用率
const total = os.totalmem();
const free = os.freemem();
const used = total - free;
const percent = (used / total * 100).toFixed(2);

console.log(`内存使用率: ${percent}%`);
```

## 9.4 目录和路径

```javascript
const os = require('os');

// 路径分隔符
console.log('路径分隔符:', os.EOL);  // \n (Linux/Mac), \r\n (Windows)
console.log('路径分隔符:', os.platform() === 'win32' ? '\\' : '/');

// 临时目录
console.log('临时目录:', os.tmpdir());

// 用户目录
console.log('用户目录:', os.homedir());

// 用户信息
console.log('用户名:', os.userInfo());
```

## 9.5 网络信息

```javascript
const os = require('os');

// 网络接口
const interfaces = os.networkInterfaces();

for (const name of Object.keys(interfaces)) {
    console.log(`\n网络接口: ${name}`);
    interfaces[name].forEach(info => {
        console.log(`  ${info.family}: ${info.address}`);
    });
}

// 估算网络速度（简单示例）
const cpus = os.cpus();
let totalIdle = 0;
let totalTick = 0;

cpus.forEach(cpu => {
    for (const type in cpu.times) {
        totalTick += cpu.times[type];
    }
    totalIdle += cpu.times.idle;
});

console.log('CPU 空闲率:', (totalIdle / totalTick * 100).toFixed(2) + '%');
```

## 9.6 实际应用

### 系统监控

```javascript
const os = require('os');

function getSystemInfo() {
    return {
        platform: os.platform(),
        arch: os.arch(),
        cpuCount: os.cpus().length,
        totalMemory: (os.totalmem() / 1024 / 1024 / 1024).toFixed(2) + ' GB',
        freeMemory: (os.freemem() / 1024 / 1024 / 1024).toFixed(2) + ' GB',
        uptime: (os.uptime() / 60 / 60 / 24).toFixed(1) + ' 天',
        hostname: os.hostname()
    };
}

console.log(getSystemInfo());
```
