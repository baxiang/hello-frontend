## L1 理解层：会用

### Mongoose 基础

**最简示例（1-3行）：**

```javascript
await mongoose.connect('mongodb://localhost:27017/myapp');
const User = mongoose.model('User', new mongoose.Schema({ name: String, age: Number }));
const user = await User.create({ name: '张三', age: 25 });
```

**详细示例：**

```javascript
const mongoose = require('mongoose');

// 连接数据库
await mongoose.connect('mongodb://localhost:27017/myapp');

// 定义 Schema
const userSchema = new mongoose.Schema({
    name: { type: String, required: true, trim: true },
    email: { type: String, required: true, unique: true, lowercase: true },
    age: { type: Number, min: 0, max: 150 },
    role: { type: String, enum: ['user', 'admin'], default: 'user' },
    hobbies: [String],
    address: { city: String, district: String }
}, { timestamps: true });  // 自动添加 createdAt/updatedAt

// 创建模型
const User = mongoose.model('User', userSchema);
```

### CRUD 操作

**最简示例：**

```javascript
// Create
const user = await User.create({ name: '张三', age: 25 });

// Read
const users = await User.find({ age: { $gte: 18 } });
const user = await User.findById(id);

// Update
await User.findByIdAndUpdate(id, { age: 26 }, { new: true });

// Delete
await User.findByIdAndDelete(id);
```

**详细示例：**

```javascript
// Create — 创建
const user = new User({ name: '张三', email: 'zhang@example.com', age: 25 });
await user.save();

// Read — 查询
const all = await User.find();                           // 所有
const admins = await User.find({ role: 'admin' });       // 条件
const one = await User.findOne({ email: 'zhang@example.com' });
const byId = await User.findById('507f1...');            // 按 ID
const paged = await User.find().sort({ createdAt: -1 }).skip(0).limit(10);
const names = await User.find({}, 'name email');          // 选字段

// Update — 更新
await User.findByIdAndUpdate(id, { age: 26 }, { new: true, runValidators: true });
await User.updateOne({ _id: id }, { $set: { age: 27 } });

// Delete — 删除
await User.findByIdAndDelete(id);
await User.deleteMany({ role: 'guest' });
```

## L2 实践层：用好

### Schema 验证选项

| 选项 | 说明 | 示例 |
|------|------|------|
| `required` | 必填 | `required: [true, '必填']` |
| `minlength/maxlength` | 字符串长度 | `minlength: 2` |
| `min/max` | 数字范围 | `min: 0, max: 150` |
| `enum` | 枚举值 | `enum: ['user', 'admin']` |
| `unique` | 唯一索引 | `unique: true` |
| `default` | 默认值 | `default: Date.now` |
| `trim` | 去除首尾空格 | `trim: true` |
| `match` | 正则验证 | `match: /^\w+@\w+\.\w+$/` |

### 反模式：不要这样做

```javascript
// ❌ 错误：不处理验证错误
app.post('/users', async (req, res) => {
    const user = await User.create(req.body);  // 可能抛异常
    res.json(user);
});

// ✅ 正确：捕获验证错误
app.post('/users', async (req, res) => {
    try {
        const user = new User(req.body);
        await user.save();
        res.status(201).json(user);
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
});
```

### 关联查询

```javascript
// 定义关联
const articleSchema = new mongoose.Schema({
    title: String,
    author: { type: mongoose.Schema.Types.ObjectId, ref: 'User' }
});

const Article = mongoose.model('Article', articleSchema);

// 关联查询
const articles = await Article.find().populate('author', 'name email');
```

### 适用场景

| 场景 | 推荐方案 | 原因 |
|------|---------|------|
| 快速原型开发 | MongoDB | Schema 灵活，开发快 |
| 固定结构数据 | MySQL | 事务支持，数据一致性 |
| 文档型数据 | MongoDB | JSON 格式，嵌套方便 |
| 复杂关系查询 | MySQL | JOIN 能力强 |

## L3 专家层：深入

### 查询性能优化

```
优化策略：

1. 创建索引
   User.createIndex({ email: 1 })        // 单字段
   User.createIndex({ name: 1, age: -1 }) // 复合索引

2. 选择性返回字段
   User.find({}, 'name email')  // 不返回全部字段

3. 分页查询
   User.find().skip(100).limit(10)

4. 分析查询计划
   User.find().explain('executionStats')
```

### Schema vs Model

```
Schema（模板）     →     Model（实例）
定义结构和验证            提供 CRUD 操作
new mongoose.Schema()     mongoose.model('User', schema)
                          new User() → 文档
                          User.find() → 查询
```

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **Schema** | 定义文档结构和验证规则的模板 | `new mongoose.Schema({ name: String })` |
| **Model** | 基于 Schema 创建的数据操作对象 | `mongoose.model('User', schema)` |
| **Document** | 数据库中的一条记录 | `new User({ name: '张三' })` |
| **populate** | 填充关联文档 | `.populate('author')` |
| **索引** | 加速查询的数据结构 | `User.createIndex({ email: 1 })` |
| **CRUD** | 创建、读取、更新、删除 | Create/Read/Update/Delete |

## 实践练习

### 练习：用户管理 RESTful API

```javascript
const express = require('express');
const mongoose = require('mongoose');
const app = express();
app.use(express.json());

mongoose.connect('mongodb://localhost:27017/userdb');

const userSchema = new mongoose.Schema({
    name: { type: String, required: true },
    email: { type: String, required: true, unique: true },
    age: Number,
    role: { type: String, enum: ['user', 'admin'], default: 'user' }
}, { timestamps: true });

const User = mongoose.model('User', userSchema);

app.get('/users', async (req, res) => {
    const { page = 1, limit = 10 } = req.query;
    const users = await User.find()
        .skip((page - 1) * limit)
        .limit(Number(limit))
        .sort({ createdAt: -1 });
    res.json(users);
});

app.get('/users/:id', async (req, res) => {
    const user = await User.findById(req.params.id);
    user ? res.json(user) : res.status(404).json({ error: '不存在' });
});

app.post('/users', async (req, res) => {
    try {
        const user = new User(req.body);
        await user.save();
        res.status(201).json(user);
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
});

app.put('/users/:id', async (req, res) => {
    const user = await User.findByIdAndUpdate(req.params.id, req.body, { new: true, runValidators: true });
    user ? res.json(user) : res.status(404).json({ error: '不存在' });
});

app.delete('/users/:id', async (req, res) => {
    await User.findByIdAndDelete(req.params.id);
    res.status(204).send();
});

app.listen(3000);
```

## 常见问题

### Q1：MongoDB 和 MySQL 怎么选？

| 场景 | 推荐 | 原因 |
|------|------|------|
| 数据结构多变 | MongoDB | Schema 灵活 |
| 需要事务支持 | MySQL | ACID 事务 |
| 日志/内容管理 | MongoDB | 文档型存储 |
| 金融/订单系统 | MySQL | 数据一致性 |

### Q2：Schema 和 Model 有什么区别？

**Schema 是模板（定义结构和验证），Model 是操作接口（提供 CRUD）：**

```javascript
const userSchema = new mongoose.Schema({ name: String });  // 模板
const User = mongoose.model('User', userSchema);            // 操作接口
```

### Q3：如何优化 MongoDB 查询？

**三个关键：创建索引、选择返回字段、分页查询：**

```javascript
User.createIndex({ email: 1 });           // 索引
User.find({}, 'name email');              // 选字段
User.find().skip(100).limit(10);          // 分页
```
