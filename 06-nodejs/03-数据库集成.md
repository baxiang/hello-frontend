## 3.2 Mongoose ORM

### 安装和连接

```javascript
// 安装：npm install mongoose
const mongoose = require('mongoose');

// 连接数据库
mongoose.connect('mongodb://localhost:27017/myapp', {
    useNewUrlParser: true,
    useUnifiedTopology: true
});

// 监听连接事件
const db = mongoose.connection;

db.on('connected', () => {
    console.log('MongoDB 连接成功');
});

db.on('error', (err) => {
    console.error('MongoDB 连接错误:', err);
});

db.on('disconnected', () => {
    console.log('MongoDB 断开连接');
});

// 优雅关闭
process.on('SIGINT', async () => {
    await mongoose.connection.close();
    process.exit(0);
});
```

### 实例方法和静态方法

```javascript
const userSchema = new mongoose.Schema({
    name: String,
    email: String,
    password: String
});

// 实例方法
userSchema.methods.comparePassword = async function(candidatePassword) {
    const bcrypt = require('bcryptjs');
    return bcrypt.compare(candidatePassword, this.password);
};

userSchema.methods.toSafeObject = function() {
    const obj = this.toObject();
    delete obj.password;
    return obj;
};

// 静态方法
userSchema.statics.findByEmail = function(email) {
    return this.findOne({ email });
};

userSchema.statics.registerUser = async function(userData) {
    const existingUser = await this.findOne({ email: userData.email });

    if (existingUser) {
        throw new Error('邮箱已被使用');
    }

    return this.create(userData);
};

// 使用
const User = mongoose.model('User', userSchema);

// 实例方法
const user = await User.findByEmail('zhang@example.com');
const isMatch = await user.comparePassword('password123');
const safeUser = user.toSafeObject();

// 静态方法
const newUser = await User.registerUser({
    name: '张三',
    email: 'zhang@example.com',
    password: 'password123'
```

### 虚拟属性

```javascript
const userSchema = new mongoose.Schema({
    firstName: String,
    lastName: String,
    age: Number
});

// 虚拟属性 - 不会保存到数据库
userSchema.virtual('fullName').get(function() {
    return `${this.firstName} ${this.lastName}`;
});

userSchema.virtual('ageGroup').get(function() {
    if (this.age < 18) return '未成年';
    if (this.age < 60) return '成年';
    return '老年';
});

// 在 toJSON 中包含虚拟属性
userSchema.set('toJSON', {
    virtuals: true
});

// 使用
const user = await User.findOne();
console.log(user.fullName);    // 张三
console.log(user.ageGroup);    // 成年
```

### 中间件（Hooks）

```javascript
const userSchema = new mongoose.Schema({
    email: String,
    password: String
});

// 前置钩子 - 保存前加密密码
userSchema.pre('save', async function(next) {
    if (!this.isModified('password')) return next();

    const bcrypt = require('bcryptjs');
    this.password = await bcrypt.hash(this.password, 10);
    next();
});

// 后置钩子
userSchema.post('save', function(doc) {
    console.log('用户已保存:', doc.email);
});

// 删除前钩子
userSchema.pre('findOneAndDelete', async function(next) {
    const user = await this.model.findOne(this.getQuery());
    console.log('即将删除用户:', user.email);
    next();
});

// 使用后
const User = mongoose.model('User', userSchema);

const user = await User.create({
    email: 'zhang@example.com',
    password: 'plainpassword'  // 会自动加密
});
```

### CRUD 操作

```javascript
const mysql = require('mysql2/promise');
const pool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: 'password',
    database: 'myapp'
});

// 创建
async function createUser(name, email) {
    const [result] = await pool.execute(
        'INSERT INTO users (name, email) VALUES (?, ?)',
        [name, email]
    );
    return result.insertId;
}

// 读取
async function getUser(id) {
    const [rows] = await pool.execute(
        'SELECT * FROM users WHERE id = ?',
        [id]
    );
    return rows[0];
}

// 更新
async function updateUser(id, name, email) {
    await pool.execute(
        'UPDATE users SET name = ?, email = ? WHERE id = ?',
        [name, email, id]
    );
}

// 删除
async function deleteUser(id) {
    await pool.execute(
        'DELETE FROM users WHERE id = ?',
        [id]
    );
}

// 事务
async function transfer(fromId, toId, amount) {
    const connection = await pool.getConnection();

    try {
        await connection.beginTransaction();

        // 扣款
        await connection.execute(
            'UPDATE accounts SET balance = balance - ? WHERE id = ?',
            [amount, fromId]
        );

        // 存款
        await connection.execute(
            'UPDATE accounts SET balance = balance + ? WHERE id = ?',
            [amount, toId]
        );

        await connection.commit();
        console.log('转账成功');

    } catch (error) {
        await connection.rollback();
        console.error('转账失败:', error);
        throw error;

    } finally {
        connection.release();
    }
}
```

## 3.6 常见问答

### Q1: MongoDB 和 MySQL 如何选择？

**答：**

| 特性 | MongoDB | MySQL |
|------|---------|-------|
| 数据模型 | 文档型 | 关系型 |
| 数据结构 | JSON 文档，灵活 | 固定表结构 |
| 扩展性 | 水平扩展容易 | 垂直扩展为主 |
| 事务 | 4.0+ 支持 | 完整支持 |
| 适用场景 | 内容管理、日志、实时数据 | 金融、电商、需要事务 |

```javascript
    },
    async down(db) {
        await db.collection('users').dropIndex({ email: 1 });
    }
};
```
