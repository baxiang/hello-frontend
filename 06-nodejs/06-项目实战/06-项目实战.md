## L1 理解层：会用

### 项目结构

```
blog-api/
├── src/
│   ├── models/          # 数据模型
│   │   ├── User.js
│   │   └── Post.js
│   ├── controllers/     # 业务逻辑
│   │   ├── auth.js
│   │   └── post.js
│   ├── routes/          # 路由
│   │   ├── auth.js
│   │   └── post.js
│   └── middleware/      # 中间件
│       └── auth.js
├── app.js               # 入口
├── .env                 # 环境变量
└── package.json
```

### 核心代码

```javascript
// app.js
const express = require('express');
const mongoose = require('mongoose');
require('dotenv').config();

const app = express();
app.use(express.json());

// 路由
app.use('/api/auth', require('./src/routes/auth'));
app.use('/api/posts', require('./src/routes/post'));

// 错误处理
app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(err.status || 500).json({ error: err.message });
});

mongoose.connect(process.env.MONGO_URI)
    .then(() => app.listen(process.env.PORT || 3000));
```

## L2 实践层：用好

### API 设计

| 方法 | 路径 | 说明 | 认证 |
|------|------|------|:---:|
| POST | `/api/auth/register` | 注册 | ❌ |
| POST | `/api/auth/login` | 登录 | ❌ |
| GET | `/api/posts` | 文章列表 | ❌ |
| GET | `/api/posts/:id` | 文章详情 | ❌ |
| POST | `/api/posts` | 创建文章 | ✅ |
| PUT | `/api/posts/:id` | 更新文章 | ✅（作者） |
| DELETE | `/api/posts/:id` | 删除文章 | ✅（作者） |

### 关键代码示例

```javascript
// controllers/post.js
const Post = require('../models/Post');
const asyncHandler = require('../middleware/asyncHandler');

exports.getAll = asyncHandler(async (req, res) => {
    const { page = 1, limit = 10 } = req.query;
    const posts = await Post.find()
        .populate('author', 'name')
        .sort({ createdAt: -1 })
        .skip((page - 1) * limit)
        .limit(Number(limit));
    res.json(posts);
});

exports.create = asyncHandler(async (req, res) => {
    const post = await Post.create({ ...req.body, author: req.user.id });
    res.status(201).json(post);
});

exports.delete = asyncHandler(async (req, res) => {
    const post = await Post.findById(req.params.id);
    if (!post) return res.status(404).json({ error: '不存在' });
    if (post.author.toString() !== req.user.id) return res.status(403).json({ error: '无权限' });
    await post.deleteOne();
    res.status(204).send();
});
```

## 实践练习

### 完整博客 API

```javascript
// models/Post.js
const mongoose = require('mongoose');

const postSchema = new mongoose.Schema({
    title: { type: String, required: true, trim: true },
    content: { type: String, required: true },
    author: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true }
}, { timestamps: true });

module.exports = mongoose.model('Post', postSchema);
```

```javascript
// routes/post.js
const express = require('express');
const router = express.Router();
const auth = require('../middleware/auth');
const postCtrl = require('../controllers/post');

router.get('/', postCtrl.getAll);
router.get('/:id', postCtrl.getOne);
router.post('/', auth, postCtrl.create);
router.put('/:id', auth, postCtrl.update);
router.delete('/:id', auth, postCtrl.delete);

module.exports = router;
```

## 常见问题

### Q1：项目大了怎么组织代码？

**按功能模块分目录，不是按文件类型。例如 `users/` 下包含 model、controller、route。**

### Q2：如何处理权限？

**在控制器中检查 `post.author.toString() === req.user.id`，或用中间件统一处理。**
