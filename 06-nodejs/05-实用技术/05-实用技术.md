## L1 理解层：会用

### 文件上传

**最简示例（1-3行）：**

```javascript
const multer = require('multer');
const upload = multer({ dest: 'uploads/' });
app.post('/upload', upload.single('file'), (req, res) => res.json(req.file));
```

**详细示例：**

```javascript
const multer = require('multer');
const path = require('path');

const storage = multer.diskStorage({
    destination: (req, file, cb) => cb(null, 'uploads/'),
    filename: (req, file, cb) => cb(null, `${Date.now()}-${file.originalname}`)
});

const upload = multer({
    storage,
    limits: { fileSize: 5 * 1024 * 1024 },  // 5MB
    fileFilter: (req, file, cb) => {
        const allowed = /jpeg|jpg|png|gif|webp/;
        const ext = allowed.test(path.extname(file.originalname).toLowerCase());
        cb(null, ext);
    }
});

app.post('/upload', upload.single('avatar'), (req, res) => {
    res.json({ url: `/uploads/${req.file.filename}` });
});

app.post('/uploads', upload.array('photos', 5), (req, res) => {
    res.json(req.files.map(f => ({ url: `/uploads/${f.filename}` })));
});
```

### 日志记录

**最简示例：**

```javascript
const fs = require('fs');
const log = (level, msg) => {
    const line = `[${new Date().toISOString()}] [${level}] ${msg}\n`;
    fs.appendFileSync('app.log', line);
};
log('INFO', '应用启动');
```

**详细示例：**

```javascript
// 日志中间件
function logger(req, res, next) {
    const start = Date.now();
    res.on('finish', () => {
        const duration = Date.now() - start;
        const line = `${req.method} ${req.url} ${res.statusCode} ${duration}ms`;
        console.log(line);
    });
    next();
}

app.use(logger);
```

### 错误处理

**最简示例：**

```javascript
// 全局错误处理（必须 4 个参数）
app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(err.status || 500).json({ error: err.message });
});
```

**详细示例：**

```javascript
// 自定义错误类
class AppError extends Error {
    constructor(message, status = 500) {
        super(message);
        this.status = status;
    }
}

// 异步错误处理封装
const asyncHandler = fn => (req, res, next) => Promise.resolve(fn(req, res, next)).catch(next);

// 使用
app.get('/users/:id', asyncHandler(async (req, res) => {
    const user = await User.findById(req.params.id);
    if (!user) throw new AppError('用户不存在', 404);
    res.json(user);
}));

// 全局错误处理
app.use((err, req, res, next) => {
    const status = err.status || 500;
    res.status(status).json({
        error: err.name,
        message: err.message,
        ...(process.env.NODE_ENV === 'development' && { stack: err.stack })
    });
});
```

## L2 实践层：用好

### 环境变量配置

```javascript
// .env 文件
PORT=3000
MONGO_URI=mongodb://localhost:27017/myapp
JWT_SECRET=your-secret-key
NODE_ENV=development

// 使用
require('dotenv').config();
const PORT = process.env.PORT || 3000;
```

### 反模式：不要这样做

```javascript
// ❌ 错误：硬编码密钥
const SECRET = 'my-secret-key';

// ✅ 正确：用环境变量
const SECRET = process.env.JWT_SECRET;
```

```javascript
// ❌ 错误：不限制上传文件大小
const upload = multer({ dest: 'uploads/' });

// ✅ 正确：限制大小和类型
const upload = multer({ dest: 'uploads/', limits: { fileSize: 5 * 1024 * 1024 } });
```

### 日志分级

| 级别 | 用途 | 示例 |
|------|------|------|
| ERROR | 错误，需要立即处理 | 数据库连接失败 |

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **Multer** | Express 的文件上传中间件 | `upload.single('file')` |
| **dotenv** | 加载 .env 文件的库 | `require('dotenv').config()` |
| **中间件** | 处理请求的函数 | `app.use(logger)` |

## 实践练习

### 练习：完整项目结构

```javascript
// app.js
const express = require('express');
const multer = require('multer');
require('dotenv').config();

const app = express();
app.use(express.json());

// 日志中间件
app.use((req, res, next) => {
    const start = Date.now();
    res.on('finish', () => {
        console.log(`${req.method} ${req.url} ${res.statusCode} ${Date.now() - start}ms`);
    });
    next();
});

// 文件上传
const upload = multer({
    dest: 'uploads/',
    limits: { fileSize: 5 * 1024 * 1024 }
});
app.post('/upload', upload.single('file'), (req, res) => {
    res.json({ path: `/uploads/${req.file.filename}` });
});

// 404
app.use((req, res) => res.status(404).json({ error: '未找到' }));

// 全局错误处理
app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(err.status || 500).json({ error: err.message });
});

app.listen(process.env.PORT || 3000);
```

## 常见问题

### Q1：为什么错误处理中间件需要 4 个参数？

**Express 通过参数数量区分普通中间件和错误处理中间件：3 个参数 = 普通，4 个参数 = 错误处理。**

### Q2：生产环境如何管理密钥？

**用环境变量，不要写在代码里。部署时用 CI/CD 或云平台的环境变量功能注入。**
