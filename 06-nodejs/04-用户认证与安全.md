## 4.1 密码加密

### 为什么需要加密

```javascript
// 永远不要明文存储密码
// 错误的做法
const user = {
    username: '张三',
    password: '123456'  // 危险！
};

// 正确的做法
const user = {
    username: '张三',
    password: '$2a$10$N9qo8uLOickgx2ZMRZoMye...'  // 加密后
};
```

### 使用 bcrypt

```javascript
// 安装：npm install bcryptjs
const bcrypt = require('bcryptjs');

// 加密密码
async function hashPassword(password) {
    // 生成盐值
    const salt = await bcrypt.genSalt(10);
    // 加密
    const hash = await bcrypt.hash(password, salt);
    return hash;
}

// 或者使用简写
const hash = await bcrypt.hash('password123', 10);

// 验证密码
async function verifyPassword(password, hash) {
    const isMatch = await bcrypt.compare(password, hash);
    return isMatch;
}

// 使用示例
const password = 'mypassword123';
const hashedPassword = await hashPassword(password);

console.log('加密后:', hashedPassword);

const isValid = await verifyPassword('mypassword123', hashedPassword);
console.log('密码正确:', isValid);  // true

const isInvalid = await verifyPassword('wrongpassword', hashedPassword);
console.log('密码错误:', isInvalid);  // false
```

### 完整示例

```javascript
// utils/password.js
const bcrypt = require('bcryptjs');

const SALT_ROUNDS = 10;

// 加密密码
exports.hashPassword = async (password) => {
    return bcrypt.hash(password, SALT_ROUNDS);
};

// 验证密码
exports.comparePassword = async (password, hash) => {
    return bcrypt.compare(password, hash);
};

// 密码强度检查
exports.checkPasswordStrength = (password) => {
    const errors = [];

    if (password.length < 8) {
        errors.push('密码至少 8 个字符');
    }

    if (!/[A-Z]/.test(password)) {
        errors.push('密码需要包含大写字母');
    }

    if (!/[a-z]/.test(password)) {
        errors.push('密码需要包含小写字母');
    }

    if (!/[0-9]/.test(password)) {
        errors.push('密码需要包含数字');
    }

    return {
        isValid: errors.length === 0,
        errors
    };
};
```

## 4.2 JWT 认证

### JWT 基础

```javascript
// JWT 结构
// Header.Payload.Signature
// eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIiwiaWF0IjoxNTE2MjM5MDIyfQ.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c

// Header - 算法和令牌类型
{
    "alg": "HS256",
    "typ": "JWT"
}

// Payload - 负载数据
{
    "sub": "1234567890",
    "name": "John Doe",
    "iat": 1516239022,
    "exp": 1516242622
}

// Signature - 签名
HMACSHA256(
    base64UrlEncode(header) + "." + base64UrlEncode(payload),
    secret
)
```

### 使用 jsonwebtoken

```javascript
// 安装：npm install jsonwebtoken
const jwt = require('jsonwebtoken');

const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';
const JWT_EXPIRES_IN = '7d';

// 生成 token
function generateToken(payload) {
    return jwt.sign(payload, JWT_SECRET, {
        expiresIn: JWT_EXPIRES_IN
    });
}

// 验证 token
function verifyToken(token) {
    try {
        return jwt.verify(token, JWT_SECRET);
    } catch (error) {
        return null;
    }
}

// 解码 token（不验证）
function decodeToken(token) {
    return jwt.decode(token);
}

// 使用示例
const user = { id: 1, username: '张三', role: 'user' };

// 生成
const token = generateToken({
    userId: user.id,
    username: user.username,
    role: user.role
});

console.log('Token:', token);

// 验证
const decoded = verifyToken(token);
console.log('解码结果:', decoded);

// 检查是否过期
const isExpired = !decoded;
console.log('是否过期:', isExpired);
```

### Token 刷新

```javascript
// 双 Token 机制
const jwt = require('jsonwebtoken');

// 访问令牌（短期）
const accessTokenSecret = process.env.ACCESS_TOKEN_SECRET;
const accessTokenExpiresIn = '15m';

// 刷新令牌（长期）
const refreshTokenSecret = process.env.REFRESH_TOKEN_SECRET;
const refreshTokenExpiresIn = '7d';

// 生成访问令牌
function generateAccessToken(userId) {
    return jwt.sign({ userId }, accessTokenSecret, {
        expiresIn: accessTokenExpiresIn
    });
}

// 生成刷新令牌
function generateRefreshToken(userId) {
    return jwt.sign({ userId }, refreshTokenSecret, {
        expiresIn: refreshTokenExpiresIn
    });
}

// 刷新访问令牌
function refreshAccessToken(refreshToken) {
    try {
        const decoded = jwt.verify(refreshToken, refreshTokenSecret);
        return generateAccessToken(decoded.userId);
    } catch (error) {
        throw new Error('刷新令牌无效');
    }
}

// 使用
const tokens = {
    accessToken: generateAccessToken(userId),
    refreshToken: generateRefreshToken(userId)
};

// 当 accessToken 过期后，使用 refreshToken 获取新的 accessToken
const newAccessToken = refreshAccessToken(tokens.refreshToken);
```

## 4.3 认证中间件

### 基础认证

```javascript
// middleware/auth.js
const jwt = require('jsonwebtoken');

const JWT_SECRET = process.env.JWT_SECRET;

function authMiddleware(req, res, next) {
    // 获取 Authorization 头
    const authHeader = req.headers.authorization;

    // 检查头是否存在
    if (!authHeader) {
        return res.status(401).json({
            error: '未授权',
            message: '缺少认证令牌'
        });
    }

    // 检查格式
    if (!authHeader.startsWith('Bearer ')) {
        return res.status(401).json({
            error: '未授权',
            message: '令牌格式错误，应为 Bearer {token}'
        });
    }

    // 提取 token
    const token = authHeader.split(' ')[1];

    try {
        // 验证 token
        const decoded = jwt.verify(token, JWT_SECRET);

        // 将用户信息附加到请求
        req.user = decoded;

        next();
    } catch (error) {
        if (error.name === 'TokenExpiredError') {
            return res.status(401).json({
                error: '未授权',
                message: '令牌已过期'
            });
        }

        return res.status(401).json({
            error: '未授权',
            message: '令牌无效'
        });
    }
}

module.exports = authMiddleware;
```

### 角色授权

```javascript
// middleware/authorize.js
function authorize(...roles) {
    return (req, res, next) => {
        // 确保用户已认证
        if (!req.user) {
            return res.status(401).json({
                error: '未授权',
                message: '请先登录'
            });
        }

        // 检查角色
        if (!roles.includes(req.user.role)) {
            return res.status(403).json({
                error: '禁止访问',
                message: '权限不足'
            });
        }

        next();
    };
}

module.exports = authorize;

// 使用
const express = require('express');
const authMiddleware = require('./middleware/auth');
const authorize = require('./middleware/authorize');

const router = express.Router();

// 需要登录
router.get('/profile', authMiddleware, (req, res) => {
    res.json({ user: req.user });
});

// 需要管理员
router.get('/admin', authMiddleware, authorize('admin'), (req, res) => {
    res.json({ message: '管理员面板' });
});

// 需要管理员或编辑
router.post('/articles', authMiddleware, authorize('admin', 'editor'), (req, res) => {
    res.json({ message: '文章已创建' });
});
```

### 可选认证

```javascript
// middleware/optionalAuth.js
const jwt = require('jsonwebtoken');

function optionalAuth(req, res, next) {
    const authHeader = req.headers.authorization;

    if (!authHeader || !authHeader.startsWith('Bearer ')) {
        // 没有 token，继续请求
        return next();
    }

    try {
        const token = authHeader.split(' ')[1];
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.user = decoded;
    } catch (error) {
        // token 无效，但不阻止请求
    }

    next();
}

module.exports = optionalAuth;

// 使用 - 登录用户看到更多内容
router.get('/articles', optionalAuth, async (req, res) => {
    const articles = await Article.find({
        published: req.user ? { $in: [true, 'draft'] } : true
    });

    res.json(articles);
});
```

## 4.4 完整认证系统

### 用户模型

```javascript
// models/User.js
const mongoose = require('mongoose');
const bcrypt = require('bcryptjs');

const userSchema = new mongoose.Schema({
    username: {
        type: String,
        required: true,
        unique: true,
        trim: true,
        minlength: 3,
        maxlength: 30
    },
    email: {
        type: String,
        required: true,
        unique: true,
        lowercase: true,
        match: [/^\w+@[a-zA-Z_]+?\.[a-zA-Z]{2,3}$/, '请输入有效的邮箱']
    },
    password: {
        type: String,
        required: true,
        minlength: 6,
        select: false  // 查询时不返回
    },
    role: {
        type: String,
        enum: ['user', 'admin'],
        default: 'user'
    },
    avatar: String,
    bio: String,
    isActive: {
        type: Boolean,
        default: true
    }
}, { timestamps: true });

// 加密密码
userSchema.pre('save', async function(next) {
    if (!this.isModified('password')) return next();

    const salt = await bcrypt.genSalt(10);
    this.password = await bcrypt.hash(this.password, salt);
    next();
});

// 比较密码
userSchema.methods.comparePassword = async function(candidate) {
    return bcrypt.compare(candidate, this.password);
};

// 安全对象
userSchema.methods.toSafeObject = function() {
    const { _id, username, email, role, avatar, bio, createdAt } = this.toObject();
    return { _id, username, email, role, avatar, bio, createdAt };
};

// 生成 token
userSchema.methods.generateToken = function() {
    const jwt = require('jsonwebtoken');
    return jwt.sign(
        { userId: this._id, role: this.role },
        process.env.JWT_SECRET,
        { expiresIn: '7d' }
    );
};

module.exports = mongoose.model('User', userSchema);
```

### 认证路由

```javascript
// routes/auth.js
const express = require('express');
const jwt = require('jsonwebtoken');
const User = require('../models/User');
const authMiddleware = require('../middleware/auth');

const router = express.Router();

// 注册
router.post('/register', async (req, res) => {
    try {
        const { username, email, password } = req.body;

        // 验证必填字段
        if (!username || !email || !password) {
            return res.status(400).json({
                error: '缺少必填字段'
            });
        }

        // 检查用户是否存在
        const existingUser = await User.findOne({
            $or: [{ email }, { username }]
        });

        if (existingUser) {
            return res.status(400).json({
                error: '用户名或邮箱已被使用'
            });
        }

        // 创建用户
        const user = await User.create({
            username,
            email,
            password
        });

        // 生成 token
        const token = user.generateToken();

        res.status(201).json({
            token,
            user: user.toSafeObject()
        });

    } catch (error) {
        res.status(500).json({
            error: '注册失败',
            message: error.message
        });
    }
});

// 登录
router.post('/login', async (req, res) => {
    try {
        const { email, password } = req.body;

        // 验证必填字段
        if (!email || !password) {
            return res.status(400).json({
                error: '请输入邮箱和密码'
            });
        }

        // 查找用户（包含 password 字段）
        const user = await User.findOne({ email }).select('+password');

        if (!user) {
            return res.status(401).json({
                error: '邮箱或密码错误'
            });
        }

        // 检查用户状态
        if (!user.isActive) {
            return res.status(401).json({
                error: '账户已被禁用'
            });
        }

        // 验证密码
        const isMatch = await user.comparePassword(password);

        if (!isMatch) {
            return res.status(401).json({
                error: '邮箱或密码错误'
            });
        }

        // 生成 token
        const token = user.generateToken();

        res.json({
            token,
            user: user.toSafeObject()
        });

    } catch (error) {
        res.status(500).json({
            error: '登录失败',
            message: error.message
        });
    }
});

// 获取当前用户
router.get('/me', authMiddleware, async (req, res) => {
    try {
        const user = await User.findById(req.user.userId);

        if (!user) {
            return res.status(404).json({
                error: '用户不存在'
            });
        }

        res.json(user.toSafeObject());

    } catch (error) {
        res.status(500).json({
            error: '获取用户信息失败',
            message: error.message
        });
    }
});

// 更新资料
router.put('/me', authMiddleware, async (req, res) => {
    try {
        const { username, avatar, bio } = req.body;

        const user = await User.findByIdAndUpdate(
            req.user.userId,
            { username, avatar, bio },
            { new: true, runValidators: true }
        );

        if (!user) {
            return res.status(404).json({
                error: '用户不存在'
            });
        }

        res.json(user.toSafeObject());

    } catch (error) {
        res.status(500).json({
            error: '更新失败',
            message: error.message
        });
    }
});

// 修改密码
router.post('/change-password', authMiddleware, async (req, res) => {
    try {
        const { currentPassword, newPassword } = req.body;

        if (!currentPassword || !newPassword) {
            return res.status(400).json({
                error: '请输入当前密码和新密码'
            });
        }

        const user = await User.findById(req.user.userId).select('+password');

        // 验证当前密码
        const isMatch = await user.comparePassword(currentPassword);

        if (!isMatch) {
            return res.status(400).json({
                error: '当前密码错误'
            });
        }

        // 更新密码
        user.password = newPassword;
        await user.save();

        res.json({
            message: '密码修改成功'
        });

    } catch (error) {
        res.status(500).json({
            error: '修改密码失败',
            message: error.message
        });
    }
});

// 登出（可选，因为 JWT 是无状态的）
router.post('/logout', authMiddleware, async (req, res) => {
    // 可以将 token 加入黑名单
    res.json({ message: '登出成功' });
});

module.exports = router;
```

## 4.7 常见问答

### Q1: JWT 和 Session 有什么区别？

| 特性 | JWT | Session |
|------|-----|---------|
| 存储位置 | 客户端 | 服务器 |
| 扩展性 | 好（无状态） | 需要共享 session |
| 安全性 | 依赖签名 | 依赖 session ID |
| 适用场景 | API 认证 | 传统 Web 应用 |

```javascript
        user = await User.create({
            email: profile.emails[0].value,
            username: profile.displayName,
            avatar: profile.photos[0].value
        });
    }

    return done(null, user);
}));
```
