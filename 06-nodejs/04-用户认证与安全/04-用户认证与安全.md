## L1 理解层：会用

### 密码加密

**最简示例（1-3行）：**

```javascript
const hash = await bcrypt.hash('password123', 10);
await bcrypt.compare('password123', hash);  // true
```

**详细示例：**

```javascript
const bcrypt = require('bcryptjs');

// 加密（注册时）
async function hashPassword(plain) {
    const salt = await bcrypt.genSalt(10);  // 生成盐值
    return bcrypt.hash(plain, salt);
}

// 验证（登录时）
async function verifyPassword(plain, hash) {
    return bcrypt.compare(plain, hash);
}

// Mongoose 模型中使用
const userSchema = new mongoose.Schema({
    email: { type: String, required: true, unique: true },
    password: { type: String, required: true }
});

userSchema.pre('save', async function() {
    if (!this.isModified('password')) return;
    this.password = await bcrypt.hash(this.password, 10);
});

userSchema.methods.comparePassword = function(plain) {
    return bcrypt.compare(plain, this.password);
};
```

### JWT 认证

**语法结构图：**

```
JWT 结构：

header.payload.signature
  │       │         │
  │       │         └─ 签名（防止篡改）
  │       └─ 载荷（用户信息）
  └─ 头部（算法类型）

流程：
  登录 → 签发 JWT → 客户端保存 → 请求带 Authorization: Bearer <token> → 服务端验证
```

**最简示例：**

```javascript
const token = jwt.sign({ id: user._id }, SECRET, { expiresIn: '7d' });
const decoded = jwt.verify(token, SECRET);  // { id: '...', iat: ..., exp: ... }
```

**详细示例：**

```javascript
const jwt = require('jsonwebtoken');
const SECRET = process.env.JWT_SECRET || 'your-secret-key';

// 签发令牌（登录成功后）
function generateToken(user) {
    return jwt.sign(
        { id: user._id, role: user.role },
        SECRET,
        { expiresIn: '7d' }
    );
}

// 认证中间件
function authMiddleware(req, res, next) {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: '未授权' });

    try {
        const decoded = jwt.verify(token, SECRET);
        req.user = decoded;
        next();
    } catch (err) {
        res.status(401).json({ error: '令牌无效' });
    }
}

// 登录路由
app.post('/login', async (req, res) => {
    const user = await User.findOne({ email: req.body.email });
    if (!user || !await user.comparePassword(req.body.password)) {
        return res.status(401).json({ error: '账号或密码错误' });
    }
    const token = generateToken(user);
    res.json({ token });
});

// 受保护的路由
app.get('/profile', authMiddleware, (req, res) => {
    res.json({ id: req.user.id, role: req.user.role });
});
```

## L2 实践层：用好

### 安全要点速查

| 威胁 | 防护方法 | 示例 |
|------|---------|------|
| 密码泄露 | bcrypt 加密 | `bcrypt.hash(password, 10)` |
| 未授权访问 | JWT 中间件 | `app.get('/api', auth, handler)` |
| XSS 攻击 | 输出转义 | `he.escape(userInput)` |
| SQL/NoSQL 注入 | 参数化查询 | `User.find({ _id: id })` |
| CSRF | SameSite Cookie | `res.cookie('token', t, { sameSite: 'strict' })` |

### 反模式：不要这样做

```javascript
// ❌ 错误：明文存储密码
const user = new User({ password: req.body.password });

// ✅ 正确：用 bcrypt 加密
user.password = await bcrypt.hash(req.body.password, 10);
```

```javascript
// ❌ 错误：JWT 密钥硬编码
jwt.sign(payload, 'my-secret');

// ✅ 正确：用环境变量
jwt.sign(payload, process.env.JWT_SECRET);
```

```javascript
// ❌ 错误：不验证令牌过期
jwt.verify(token, SECRET);  // 默认会检查过期

// ✅ 正确：明确设置过期时间
jwt.sign(payload, SECRET, { expiresIn: '7d' });
```

### JWT 最佳实践

| 实践 | 说明 |
|------|------|
| 密钥用环境变量 | 不要硬编码在代码中 |
| 设置合理过期时间 | 一般 7 天，敏感操作更短 |
| Refresh Token | 用短寿命 Access + 长寿命 Refresh |
| 不在 JWT 存敏感数据 | 只存 ID 和角色，不存密码/邮箱 |

## 术语定义

| 术语 | 定义 | 示例 |
|------|------|------|
| **bcrypt** | 不可逆密码加密算法 | `bcrypt.hash(password, salt)` |
| **盐值（Salt）** | 随机字符串，增加破解难度 | `bcrypt.genSalt(10)` |
| **JWT** | JSON Web Token，自包含的身份令牌 | `eyJhbG...` |
| **Bearer Token** | HTTP 认证方案，请求头携带令牌 | `Authorization: Bearer <token>` |
| **XSS** | 跨站脚本攻击，注入恶意代码 | `<script>alert(1)</script>` |
| **CSRF** | 跨站请求伪造，冒充用户操作 | 伪造表单提交 |

## 实践练习

### 练习：完整认证系统

```javascript
const express = require('express');
const jwt = require('jsonwebtoken');
const bcrypt = require('bcryptjs');
const mongoose = require('mongoose');

const app = express();
app.use(express.json());

const SECRET = process.env.JWT_SECRET || 'change-me';

// 用户模型
const userSchema = new mongoose.Schema({
    email: { type: String, required: true, unique: true, lowercase: true },
    password: { type: String, required: true },
    role: { type: String, enum: ['user', 'admin'], default: 'user' }
}, { timestamps: true });

userSchema.pre('save', async function() {
    if (!this.isModified('password')) return;
    this.password = await bcrypt.hash(this.password, 12);
});

userSchema.methods.comparePassword = function(plain) {
    return bcrypt.compare(plain, this.password);
};

const User = mongoose.model('User', userSchema);

// 注册
app.post('/register', async (req, res) => {
    try {
        const user = await User.create(req.body);
        const token = jwt.sign({ id: user._id, role: user.role }, SECRET, { expiresIn: '7d' });
        res.status(201).json({ token });
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
});

// 登录
app.post('/login', async (req, res) => {
    const user = await User.findOne({ email: req.body.email });
    if (!user || !await user.comparePassword(req.body.password)) {
        return res.status(401).json({ error: '账号或密码错误' });
    }
    const token = jwt.sign({ id: user._id, role: user.role }, SECRET, { expiresIn: '7d' });
    res.json({ token });
});

// 认证中间件
const auth = (req, res, next) => {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: '未授权' });
    try {
        req.user = jwt.verify(token, SECRET);
        next();
    } catch { res.status(401).json({ error: '令牌无效' }); }
};

app.get('/profile', auth, async (req, res) => {
    const user = await User.findById(req.user.id).select('-password');
    res.json(user);
});

mongoose.connect('mongodb://localhost:27017/auth-app');
app.listen(3000);
```

## 常见问题

### Q1：JWT 和 Session 认证有什么区别？

| 特性 | JWT | Session |
|------|-----|---------|
| 存储位置 | 客户端 | 服务端 |
| 扩展性 | 好（无状态） | 需要共享存储 |
| 撤销困难 | 是（需黑名单） | 容易 |
| 适用场景 | API、微服务 | 传统 Web |

### Q2：JWT 泄露了怎么办？

**JWT 无法撤销，所以：**
1. 设置短过期时间
2. 用 Refresh Token 机制
3. 维护令牌黑名单

### Q3：bcrypt 的 cost factor 设多少？

**推荐 10-12。越高越安全但越慢。生产环境根据服务器性能调整。**
